192{
193 DDI_CHK_NULL(data, "nullptr data", VA_STATUS_ERROR_INVALID_PARAMETER);
194
195 // Assume only one SPS here, modify when enable multiple SPS support
196 PCODEC_AVC_ENCODE_SEQUENCE_PARAMS seqParams = (PCODEC_AVC_ENCODE_SEQUENCE_PARAMS)(m_encodeCtx->pSeqParams);
197 VAEncMiscParameterFrameRate *encMiscParamFR = (VAEncMiscParameterFrameRate *)data;
198 DDI_CHK_NULL(seqParams, "nullptr seqParams", VA_STATUS_ERROR_INVALID_PARAMETER);
199
200 uint32_t numerator = (encMiscParamFR->framerate & 0xffff) * 100; ;
201 auto denominator = (encMiscParamFR->framerate >> 16)&0xfff;
202 if(denominator == 0)
203 {
204 denominator = 1;
205 }
206
207 // Pass frame rate value to seqParams and set BRC reset flag and bNewSeq to true while dynamic BRC occures
208 seqParams->FramesPer100Sec = (uint16_t)(numerator / denominator);
209 if(m_previousFRper100sec != 0)
210 {
211 if(m_previousFRper100sec != seqParams->FramesPer100Sec)
212 {
213 seqParams->bResetBRC = 0x1;
214 m_encodeCtx->bNewSeq = true;
215 }
216 }
217 m_previousFRper100sec = seqParams->FramesPer100Sec;
218
219 return VA_STATUS_SUCCESS;

222VAStatus DdiEncodeAvc::ParseMiscParamRC(void *data)
223{
224 DDI_CHK_NULL(data, "nullptr data", VA_STATUS_ERROR_INVALID_PARAMETER);
225
226 CODECHAL_ENCODE_AVC_VUI_PARAMS *vuiParam = (CODECHAL_ENCODE_AVC_VUI_PARAMS *)m_encodeCtx->pVuiParams;
227
228 // Assume only one SPS here, modify when enable multiple SPS support
229 PCODEC_AVC_ENCODE_SEQUENCE_PARAMS seqParams = (PCODEC_AVC_ENCODE_SEQUENCE_PARAMS)(m_encodeCtx->pSeqParams) + current_seq_parameter_set_id;
230 PCODEC_AVC_ENCODE_PIC_PARAMS picParams = (PCODEC_AVC_ENCODE_PIC_PARAMS)(m_encodeCtx->pPicParams) + current_pic_parameter_set_id;
231 VAEncMiscParameterRateControl *encMiscParamRC = (VAEncMiscParameterRateControl *)data;
232 DDI_CHK_NULL(vuiParam, "nullptr vuiParam", VA_STATUS_ERROR_INVALID_PARAMETER);
233 DDI_CHK_NULL(seqParams, "nullptr seqParams", VA_STATUS_ERROR_INVALID_PARAMETER);
234 DDI_CHK_NULL(picParams, "nullptr picParams", VA_STATUS_ERROR_INVALID_PARAMETER);
235
236 seqParams->TargetBitRate = encMiscParamRC->bits_per_second;
237 vuiParam->bit_rate_value_minus1[0] = MOS_ROUNDUP_SHIFT(encMiscParamRC->bits_per_second, 6 + vuiParam->bit_rate_scale) - 1;
238 seqParams->MBBRC = encMiscParamRC->rc_flags.bits.mb_rate_control;
239
240 // Assuming picParams are sent before MiscParams
241 picParams->ucMinimumQP = encMiscParamRC->min_qp;
242 picParams->ucMaximumQP = encMiscParamRC->max_qp;
243 if (picParams->ucMaximumQP == 0 && picParams->ucMinimumQP)
244 picParams->ucMaximumQP = 51;
245
246 if ((VA_RC_CBR == m_encodeCtx->uiRCMethod) || ((VA_RC_CBR | VA_RC_MB) == m_encodeCtx->uiRCMethod))
247 {
248 seqParams->MaxBitRate = seqParams->TargetBitRate;
249 seqParams->MinBitRate = seqParams->TargetBitRate;
250 vuiParam->cbr_flag = 0x1;
251 if (m_encodeCtx->uiTargetBitRate != seqParams->TargetBitRate)
252 {
253 if (m_encodeCtx->uiTargetBitRate)
254 {
255 seqParams->bResetBRC = 0x1;
256 m_encodeCtx->bNewSeq = true;
257 }
258 m_encodeCtx->uiTargetBitRate = seqParams->TargetBitRate;
259 m_encodeCtx->uiMaxBitRate = seqParams->TargetBitRate;
260 }
261 }
262 else if (VA_RC_ICQ == m_encodeCtx->uiRCMethod)
263 {
264 seqParams->ICQQualityFactor = encMiscParamRC->ICQ_quality_factor;
265 }
266 else if (VA_RC_AVBR == m_encodeCtx->uiRCMethod)
267 {
268 seqParams->AVBRAccuracy = encMiscParamRC->target_percentage;
269 seqParams->AVBRConvergence = encMiscParamRC->window_size;
270 }
271 else
272 {
273 seqParams->MaxBitRate = seqParams->TargetBitRate;
274 seqParams->TargetBitRate = (uint32_t)((uint64_t)seqParams->TargetBitRate * encMiscParamRC->target_percentage / 100);
275 seqParams->MinBitRate = seqParams->TargetBitRate;
276 vuiParam->cbr_flag = 0x0;
277
278 if (VA_RC_QVBR == m_encodeCtx->uiRCMethod)
279 {
280 seqParams->ICQQualityFactor = encMiscParamRC->quality_factor;
281 }
282
283 if ((m_encodeCtx->uiTargetBitRate != seqParams->TargetBitRate) ||
284 (m_encodeCtx->uiMaxBitRate != seqParams->MaxBitRate))
285 {
286 if ((m_encodeCtx->uiTargetBitRate != 0) && (m_encodeCtx->uiMaxBitRate != 0))
287 {
288 seqParams->bResetBRC = 0x1;
289 m_encodeCtx->bNewSeq = true;
290 }
291 m_encodeCtx->uiTargetBitRate = seqParams->TargetBitRate;
292 m_encodeCtx->uiMaxBitRate = seqParams->MaxBitRate;
293 }
294 }
295 //if RateControl method is VBR/CBR, we can set MBBRC to enable or disable
296 if (VA_RC_CQP != m_encodeCtx->uiRCMethod)
297 {
298 if (encMiscParamRC->rc_flags.bits.mb_rate_control <= mbBrcDisabled)
299 seqParams->MBBRC = encMiscParamRC->rc_flags.bits.mb_rate_control;
300 }
301
302#ifndef ANDROID
303 seqParams->FrameSizeTolerance = static_cast<ENCODE_FRAMESIZE_TOLERANCE>(encMiscParamRC->rc_flags.bits.frame_tolerance_mode);
304#endif
305
306#if VA_CHECK_VERSION(1, 10, 0)
307 if (m_encodeCtx->bVdencActive &&
308 (m_encodeCtx->uiRCMethod & (VA_RC_VBR | VA_RC_QVBR | VA_RC_VCM)))
309 {
310 picParams->TargetFrameSize = encMiscParamRC->target_frame_size;
311 // to force VC scenario for TCBRC
312 seqParams->bAutoMaxPBFrameSizeForSceneChange = encMiscParamRC->target_frame_size != 0;
313 }
314#endif
315
316 return VA_STATUS_SUCCESS;
317}

482VAStatus DdiEncodeAvc::ParseMiscParameterRIR(void *data)
483{
484 DDI_CHK_NULL(data, "nullptr data", VA_STATUS_ERROR_INVALID_PARAMETER);
485
486 PCODEC_AVC_ENCODE_PIC_PARAMS picParams = (PCODEC_AVC_ENCODE_PIC_PARAMS)(m_encodeCtx->pPicParams) + current_pic_parameter_set_id;
487 PCODEC_AVC_ENCODE_SEQUENCE_PARAMS seqParams = (PCODEC_AVC_ENCODE_SEQUENCE_PARAMS)(m_encodeCtx->pSeqParams) + current_seq_parameter_set_id;
488 VAEncMiscParameterRIR *vaEncMiscParamRIR = (VAEncMiscParameterRIR *)data;
489 DDI_CHK_NULL(picParams, "nullptr picParams", VA_STATUS_ERROR_INVALID_PARAMETER);
490 DDI_CHK_NULL(seqParams, "nullptr seqParams", VA_STATUS_ERROR_INVALID_PARAMETER);
491
492 picParams->EnableRollingIntraRefresh = vaEncMiscParamRIR->rir_flags.value & 0x3; //only lower two bits are valid
493
494#ifdef Android
495 //Convert row based rolling I to suqare region based rolling I before MSDK is ready to support square region based rolling I
496 if (ROLLING_I_ROW == picParams->EnableRollingIntraRefresh)
497 {
498 if (GFX_IS_PRODUCT(m_encodeCtx->pMediaCtx->platform, IGFX_BROADWELL))
499 {
500 picParams->EnableRollingIntraRefresh = ROLLING_I_SQUARE;
501 }
502 vaEncMiscParamRIR->intra_insert_size *= (seqParams->FrameWidth / CODECHAL_ENCODE_AVC_ROI_WIDTH_SCALE_FACTOR);
503 // From the experiment result, -6 could be the proper setting
504 vaEncMiscParamRIR->qp_delta_for_inserted_intra = -6;
505 }
506#endif
507 // Set for all frames since pic type is not known yet. Disable in slice params if its I or B type.
508 switch (picParams->EnableRollingIntraRefresh)
509 {
510 case ROLLING_I_COLUMN:
511 picParams->IntraRefreshMBx = (uint8_t)vaEncMiscParamRIR->intra_insertion_location;
512 picParams->IntraRefreshMBNum = (uint8_t)vaEncMiscParamRIR->intra_insertion_location;
513 picParams->IntraRefreshUnitinMB = (uint8_t)vaEncMiscParamRIR->intra_insert_size;
514 break;
515 case ROLLING_I_ROW:
516 picParams->IntraRefreshMBy = (uint8_t)vaEncMiscParamRIR->intra_insertion_location;
517 picParams->IntraRefreshMBNum = (uint8_t)vaEncMiscParamRIR->intra_insertion_location;
518 picParams->IntraRefreshUnitinMB = (uint8_t)vaEncMiscParamRIR->intra_insert_size;
519 break;
520 case ROLLING_I_SQUARE:
521 picParams->IntraRefreshUnitinMB = (uint8_t)GFX_UF_ROUND(sqrt((double)vaEncMiscParamRIR->intra_insert_size));
522 break;
523 case ROLLING_I_DISABLED:
524 break;
525 default:
526 return MOS_STATUS_INVALID_PARAMETER;
527 }
528 picParams->IntraRefreshQPDelta = vaEncMiscParamRIR->qp_delta_for_inserted_intra;
529
530 // UMD tracks the MBx/MBy for the intra square

875 VADriverContextP ctx,
876 VAContextID context,
877 VABufferID *buffers,
878 int32_t numBuffers)
879{
880 DDI_FUNCTION_ENTER();
881
882 DDI_CHK_NULL(ctx, "nullptr ctx", VA_STATUS_ERROR_INVALID_CONTEXT);
883 PDDI_MEDIA_CONTEXT mediaCtx = DdiMedia_GetMediaContext(ctx);
884 DDI_CHK_NULL(mediaCtx, "nullptr mediaCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
885
886 // assume the VAContextID is encoder ID
887 DDI_CHK_NULL(m_encodeCtx, "nullptr m_encodeCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
888
889 uint32_t numSlices = 0;
890 VAStatus vaStatus = VA_STATUS_SUCCESS;
891
892 DDI_MEDIA_BUFFER *buf;
893 void *data;
894 uint32_t dataSize;
895 for (int32_t i = 0; i < numBuffers; i++)
896 {
897 buf = DdiMedia_GetBufferFromVABufferID(mediaCtx, buffers[i]);
898 DDI_CHK_NULL(buf, "Invalid buffer.", VA_STATUS_ERROR_INVALID_BUFFER);
899 if (buf->uiType == VAEncMacroblockDisableSkipMapBufferType)
900 {
901 DdiMedia_MediaBufferToMosResource(buf, &(m_encodeCtx->resPerMBSkipMapBuffer));
902 m_encodeCtx->bMbDisableSkipMapEnabled = true;
903 continue;
904 }
905 dataSize = buf->iSize;
906 // can use internal function instead of DdiMedia_MapBuffer here?
907 DdiMedia_MapBuffer(ctx, buffers[i], &data);
908
909 DDI_CHK_NULL(data, "nullptr data", VA_STATUS_ERROR_INVALID_BUFFER);
910
911 switch (buf->uiType)
912 {
913 case VAIQMatrixBufferType:
914 case VAQMatrixBufferType:
915 DDI_CHK_STATUS(Qmatrix(data), VA_STATUS_ERROR_INVALID_BUFFER);
916 break;
917
918 case VAEncSequenceParameterBufferType:
919 DDI_CHK_STATUS(ParseSeqParams(data), VA_STATUS_ERROR_INVALID_BUFFER);
920 m_encodeCtx->bNewSeq = true;
921 break;
922
923 case VAEncPictureParameterBufferType:
924 DDI_CHK_STATUS(ParsePicParams(mediaCtx, data), VA_STATUS_ERROR_INVALID_BUFFER);
925
926 DDI_CHK_STATUS(
927 AddToStatusReportQueue((void *)m_encodeCtx->resBitstreamBuffer.bo),
928 VA_STATUS_ERROR_INVALID_BUFFER);
929 break;
930
931 case VAEncSliceParameterBufferType:
932 numSlices = buf->uiNumElements;
933 DDI_CHK_STATUS(ParseSlcParams(mediaCtx, data, numSlices), VA_STATUS_ERROR_INVALID_BUFFER);
934 break;
935
936 case VAEncPackedHeaderParameterBufferType:
937 DDI_CHK_STATUS(ParsePackedHeaderParams(data), VA_STATUS_ERROR_INVALID_BUFFER);
938 break;
939
940 case VAEncPackedHeaderDataBufferType:
941 DDI_CHK_STATUS(ParsePackedHeaderData(data), VA_STATUS_ERROR_INVALID_BUFFER);
942 break;
943
944 case VAEncMiscParameterBufferType:
945 DDI_CHK_STATUS(ParseMiscParams(data), VA_STATUS_ERROR_INVALID_BUFFER);
946 break;
947
948 case VAEncQPBufferType:
949 DdiMedia_MediaBufferToMosResource(buf, &m_encodeCtx->resMBQpBuffer);
950 m_encodeCtx->bMBQpEnable = true;
951 break;
952
953 default:
954 if(m_encodeCtx->pCpDdiInterface)
955 {
956 vaStatus = m_encodeCtx->pCpDdiInterface->RenderCencPicture(ctx, context, buf, data);
957 }
958 else
959 {
960 vaStatus = VA_STATUS_ERROR_UNSUPPORTED_BUFFERTYPE;
961 }
962 break;
963 }
964 DdiMedia_UnmapBuffer(ctx, buffers[i]);
965 }
966
967 DDI_FUNCTION_EXIT(vaStatus);
968 return vaStatus;
969}
