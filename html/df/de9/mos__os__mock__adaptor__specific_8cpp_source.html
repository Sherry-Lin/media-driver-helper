60MOS_STATUS MosMockAdaptorSpecific::InitializeSkuWaTable(PMOS_CONTEXT context)
61{
62 MOS_OS_CHK_NULL_RETURN(context);
63 MOS_OS_CHK_NULL_RETURN(m_pPlatform);
64 auto userSettingPtr = MosInterface::MosGetUserSettingInstance(context);
65
66 LinuxDriverInfo drvInfo = {18, 3, 0, 23172, 3, 1, 0, 1, 0, 0, 1, 0};
67 if (HWInfoGetLinuxDrvInfo(context->fd, &drvInfo) != MOS_STATUS_SUCCESS)
68 {
69 return MOS_STATUS_INVALID_HANDLE;
70 }
71
72 drvInfo.devId = m_pPlatform->usDeviceID;
73 drvInfo.devRev = m_pPlatform->usRevId;
74
75 GfxDeviceInfo *devInfo = GetDeviceInfo(drvInfo.devId);
76 if (devInfo == nullptr)
77 {
78 MOS_OS_ASSERTMESSAGE("Failed to get the device info for Device id: %x\n", drvInfo.devId);
79 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
80 }
81 /* Initialize Platform Info */
82 m_pGtSystemInfo->SliceCount = devInfo->SliceCount;
83 m_pGtSystemInfo->SubSliceCount = devInfo->SubSliceCount;
84 m_pGtSystemInfo->EUCount = devInfo->EUCount;
85
86 if (devInfo->InitMediaSysInfo &&
87 devInfo->InitMediaSysInfo(devInfo, m_pGtSystemInfo))
88 {
89 MOS_OS_NORMALMESSAGE("Init Media SystemInfo\n");
90 }
91 else
92 {
93 MOS_OS_ASSERTMESSAGE("Failed to Init Gt System Info\n");
94 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
95 }
96
97 unsigned int maxNengine = 0;
98 if((m_pGtSystemInfo->VDBoxInfo.NumberOfVDBoxEnabled == 0)
99 || (m_pGtSystemInfo->VEBoxInfo.NumberOfVEBoxEnabled == 0))
100 {
101 if (mos_query_engines_count(context->bufmgr, &maxNengine) || (maxNengine == 0))
102 {
103 MOS_OS_ASSERTMESSAGE("Failed to query engines count.\n");
104 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
105 }
106 }
107
108 if (m_pGtSystemInfo->VDBoxInfo.NumberOfVDBoxEnabled == 0)
109 {
110 unsigned int nengine = maxNengine;
111 struct i915_engine_class_instance *uengines = nullptr;
112 uengines = (struct i915_engine_class_instance *)MOS_AllocAndZeroMemory(nengine * sizeof(struct i915_engine_class_instance));
113 MOS_OS_CHK_NULL_RETURN(uengines);
114 if (mos_query_engines(context->bufmgr, I915_ENGINE_CLASS_VIDEO, 0, &nengine,uengines) == 0)
115 {
116 m_pGtSystemInfo->VDBoxInfo.NumberOfVDBoxEnabled = nengine;
117 }
118 else
119 {
120 MOS_OS_ASSERTMESSAGE("Failed to query vdbox engine\n");
121 MOS_SafeFreeMemory(uengines);
122 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
123 }
124 for (int i=0; i<nengine; i++)
125 {
126 m_pGtSystemInfo->VDBoxInfo.Instances.VDBoxEnableMask |= 1 << uengines[i].engine_instance;
127 }
128 MOS_SafeFreeMemory(uengines);
129 }
130
131 if (m_pGtSystemInfo->VEBoxInfo.NumberOfVEBoxEnabled == 0)
132 {
133 unsigned int nengine = maxNengine;
134 struct i915_engine_class_instance *uengines = nullptr;
135 uengines = (struct i915_engine_class_instance *)MOS_AllocAndZeroMemory(nengine * sizeof(struct i915_engine_class_instance));
136 MOS_OS_CHK_NULL_RETURN(uengines);
137 if (mos_query_engines(context->bufmgr,I915_ENGINE_CLASS_VIDEO_ENHANCE,0,&nengine,uengines) == 0)
138 {
139 MOS_OS_ASSERT(nengine <= maxNengine);
140 m_pGtSystemInfo->VEBoxInfo.NumberOfVEBoxEnabled = nengine;
141 }
142 else
143 {
144 MOS_OS_ASSERTMESSAGE("Failed to query vebox engine\n");
145 MOS_SafeFreeMemory(uengines);
146 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
147 }
148 MOS_SafeFreeMemory(uengines);
149 }
150
151 uint32_t platformKey = devInfo->productFamily;
152 LinuxDeviceInit *devInit = GetDeviceInit(platformKey);
153
154 if (devInit && devInit->InitMediaFeature &&
155 devInit->InitMediaWa &&
156 devInit->InitMediaFeature(devInfo, m_pSkuTable, &drvInfo, userSettingPtr) &&
157 devInit->InitMediaWa(devInfo, m_pWaTable, &drvInfo))
158 {
159 MOS_OS_NORMALMESSAGE("Init Media SKU/WA info successfully\n");
160 }
161 else
162 {
163 MOS_OS_ASSERTMESSAGE("Failed to Init SKU/WA Info\n");
164 return MOS_STATUS_PLATFORM_NOT_SUPPORTED;
165 }
166
167 uint32_t devExtKey = platformKey + MEDIA_EXT_FLAG;
168 LinuxDeviceInit *devExtInit = GetDeviceInit(devExtKey);
169
170 /* The initializationof Ext SKU/WA is optional. So skip the check of return value */
171 if (devExtInit && devExtInit->InitMediaFeature &&
172 devExtInit->InitMediaWa &&
173 devExtInit->InitMediaFeature(devInfo, m_pSkuTable, &drvInfo, userSettingPtr) &&
174 devExtInit->InitMediaWa(devInfo, m_pWaTable, &drvInfo))
175 {
176 MOS_OS_NORMALMESSAGE("Init Media SystemInfo successfully\n");
177 }
178
179 /* disable it on Linux */
180 MEDIA_WR_SKU(m_pSkuTable, FtrPerCtxtPreemptionGranularityControl, 0);
181 MEDIA_WR_SKU(m_pSkuTable, FtrMediaThreadGroupLevelPreempt, 0);
182 MEDIA_WR_SKU(m_pSkuTable, FtrMediaMidBatchPreempt, 0);
183 MEDIA_WR_SKU(m_pSkuTable, FtrMediaMidThreadLevelPreempt, 0);
184 MEDIA_WR_SKU(m_pSkuTable, FtrGpGpuThreadGroupLevelPreempt, 0);
185 MEDIA_WR_SKU(m_pSkuTable, FtrGpGpuMidBatchPreempt, 0);
186 MEDIA_WR_SKU(m_pSkuTable, FtrGpGpuMidThreadLevelPreempt, 0);
187
188 return MOS_STATUS_SUCCESS;
189}
