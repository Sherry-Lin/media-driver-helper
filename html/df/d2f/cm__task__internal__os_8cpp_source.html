41 if(!m_cmDevice->IsVtuneLogOn())

44 }
45
46 if (m_taskProfilingInfo.kernelCount == 0 ||
47 m_taskProfilingInfo.kernelNames == nullptr)

49 //Skip ETW Write since information is not filled

53
54 //Get Complete Time

59 m_taskEvent->GetHWEndTime(&m_taskProfilingInfo.hwEndTime);

61 if (ftrace == nullptr)
