320MOS_STATUS MosOcaInterfaceSpecific::AddResourceToDumpList(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_CONTEXT mosCtx, MOS_RESOURCE &resource, MOS_HW_COMMAND hwCmdType, uint32_t locationInCmd, uint32_t offsetInRes)
321{
322 uint32_t i = 0;
323 MOS_MEMCOMP_STATE resMmcMode = {};
324 if (!m_isOcaEnabled || !m_isInitialized)
325 {
326 return MOS_STATUS_SUCCESS;
327 }
328
329 if (!m_ocaBufContextList[ocaBufHandle].is1stLevelBBStarted)
330 {
331 MOS_OS_ASSERTMESSAGE("AddResourceToDumpList is called before oca buffer initialized!");
332 MosOcaInterfaceSpecific::OnOcaError(mosCtx, MOS_STATUS_UNINITIALIZED, __FUNCTION__, __LINE__);
333 return MOS_STATUS_UNINITIALIZED;
334 }
335
336 if (nullptr == m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList)
337 {
338 MOS_OS_NORMALMESSAGE("OCA resource info dump not enabled.");
339 return MOS_STATUS_SUCCESS;
340 }
341
342 if (nullptr == resource.pGmmResInfo || Mos_ResourceIsNull(&resource))
343 {
344 return MOS_STATUS_SUCCESS;
345 }
346
347 if (hwCmdType >= MOS_HW_COMMAND_MAX || MOS_HW_COMMAND_MAX > 64)
348 {
349 MosOcaInterfaceSpecific::OnOcaError(mosCtx, MOS_STATUS_UNINITIALIZED, __FUNCTION__, __LINE__);
350 return MOS_STATUS_INVALID_PARAMETER;
351 }
352
353 for (i = 0; i < m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount; ++i)
354 {
355 if (m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].allocationHandle == resource.bo->handle)
356 {
357 if (offsetInRes > m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes)
358 {
359 // Only update resource info for the largest offsetInRes case.
360 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].hwCmdType = (uint32_t)hwCmdType;
361 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes = offsetInRes;
362 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].locationInCmd = locationInCmd;
363 }
364 return MOS_STATUS_SUCCESS;
365 }
366 }
367
368 if (m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount >= m_ocaBufContextList[ocaBufHandle].logSection.resInfo.maxResInfoCount)
369 {
370 // Not reture error but record the resource count skipped.
371 ++m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped;
372 MOS_OS_NORMALMESSAGE("The resource is skipped to be dumpped to oca buffer.");
373 return MOS_STATUS_SUCCESS;
374 }
375
376 MosStreamState steamState = {};
377 steamState.osDeviceContext = mosCtx->m_osDeviceContext;
378 if (MOS_FAILED(MosInterface::GetMemoryCompressionMode(&steamState, &resource, resMmcMode)))
379 {
380 resMmcMode = MOS_MEMCOMP_DISABLED;
381 }
382 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gfxAddress = resource.bo->offset64;
383 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeAllocation = resource.pGmmResInfo->GetSizeAllocation();
384 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeSurface = resource.pGmmResInfo->GetSizeSurface();
385 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeSurfacePhy = resource.pGmmResInfo->GetSizeSurface();
386 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeMainSurface = resource.pGmmResInfo->GetSizeMainSurface();
387 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].allocationHandle = resource.bo->handle;
388 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].hwCmdType = (uint32_t)hwCmdType;
389 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].locationInCmd = locationInCmd;
390 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes = offsetInRes;
391 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].pitch = (uint32_t)resource.pGmmResInfo->GetRenderPitch();
392 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].width = (uint32_t)resource.pGmmResInfo->GetBaseWidth();
393 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].height = (uint32_t)resource.pGmmResInfo->GetBaseHeight();
394 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].format = (uint32_t)resource.Format;
395 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmFormat = (uint32_t)resource.pGmmResInfo->GetResourceFormat();
396 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmTileMode = (uint32_t)resource.pGmmResInfo->GmmGetTileMode();
397 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmClient = (uint32_t)resource.pGmmResInfo->GetClientType();
398 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmResUsageType = (uint32_t)resource.pGmmResInfo->GetCachePolicyUsage();
399 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].mmcMode = (uint32_t)resMmcMode;
400 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].mmcHint = (uint32_t)resource.pGmmResInfo->GetMmcHint(0);
401 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].auxYOffset = resource.pGmmResInfo->GetUnifiedAuxSurfaceOffset(GMM_AUX_Y_CCS);
402 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].auxUVOffset = resource.pGmmResInfo->GetUnifiedAuxSurfaceOffset(GMM_AUX_UV_CCS);
403 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].auxCCSOffset = resource.pGmmResInfo->GetUnifiedAuxSurfaceOffset(GMM_AUX_CCS);
404 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].auxCCOffset = resource.pGmmResInfo->GetUnifiedAuxSurfaceOffset(GMM_AUX_CC);
405
406 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].flags.isLocalOnly = resource.pGmmResInfo->GetResFlags().Info.LocalOnly;
407 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].flags.isNonLocalOnly = resource.pGmmResInfo->GetResFlags().Info.NonLocalOnly;
408 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].flags.isNotLockable = resource.pGmmResInfo->GetResFlags().Info.NotLockable;
409 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].flags.isShared = resource.pGmmResInfo->GetResFlags().Info.Shared;

411 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].flags.isRenderTarget = resource.pGmmResInfo->GetResFlags().Gpu.RenderTarget;
412
413 ++m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount;
414
415 return MOS_STATUS_SUCCESS;

418void MosOcaInterfaceSpecific::AddResourceInfoToLogSection(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_CONTEXT mosCtx)
419{
420 if (!m_ocaBufContextList[ocaBufHandle].is1stLevelBBStarted)
421 {
422 return;
423 }
424 if (0 == m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount)
425 {
426 return;
427 }
428 if (!IsLogSectionEnabled(ocaBufHandle))
429 {
430 return;
431 }
432
433 MOS_OCA_LOG_HEADER_RESOURCE_INFO header = {};
434 header.header.type = MOS_OCA_LOG_TYPE_RESOURCE_INFO;
435 header.header.headerSize = sizeof(MOS_OCA_LOG_HEADER_RESOURCE_INFO);
436 header.header.dataSize = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount * sizeof(MOS_OCA_RESOURCE_INFO);
437 header.resCount = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount;
438 header.resCountSkipped = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped;
439
440 MOS_STATUS status = DumpDataBlock(ocaBufHandle, (PMOS_OCA_LOG_HEADER)&header, m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList);
441 if (MOS_FAILED(status))
442 {
443 MosOcaInterfaceSpecific::OnOcaError(mosCtx, status, __FUNCTION__, __LINE__);
444 }
445 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount = 0;
446 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped = 0;
447 return;
448}

463MOS_STATUS MosOcaInterfaceSpecific::DumpDataBlock(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_CONTEXT mosCtx, PMOS_OCA_LOG_HEADER pHeader, void *pData)

483MOS_STATUS MosOcaInterfaceSpecific::DumpDataBlock(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_OCA_LOG_HEADER pHeader, void *pData)
484{
485 if (pHeader->headerSize < sizeof(MOS_OCA_LOG_HEADER) ||
486 pHeader->type >= MOS_OCA_LOG_TYPE_COUNT || pHeader->type <= MOS_OCA_LOG_TYPE_INVALID ||
487 (pHeader->dataSize > 0 && nullptr == pData) || (pHeader->dataSize == 0 && pData))
488 {
489 return MOS_STATUS_INVALID_PARAMETER;
490 }
491
492 uint32_t copySize = pHeader->headerSize + pHeader->dataSize;
493 if (m_ocaBufContextList[ocaBufHandle].logSection.offset + copySize > m_ocaLogSectionSizeLimit)
494 {
495 return MOS_STATUS_NOT_ENOUGH_BUFFER;
496 }
497
498 MOS_OS_CHK_STATUS_RETURN(InsertData(ocaBufHandle, (uint8_t *)pHeader, pHeader->headerSize));
499 if (pHeader->dataSize > 0)
500 {
501 MOS_OS_CHK_STATUS_RETURN(InsertData(ocaBufHandle, (uint8_t *)pData, pHeader->dataSize));
502 }
503
504 return MOS_STATUS_SUCCESS;
505}

507MOS_STATUS MosOcaInterfaceSpecific::InsertData(MOS_OCA_BUFFER_HANDLE ocaBufHandle, uint8_t *p, uint32_t size)
508{
509 if (nullptr == p || 0 == size)
510 {
511 return MOS_STATUS_INVALID_PARAMETER;
512 }
513
514 uint32_t sizeCpy = std::min(m_ocaLogSectionSizeLimit - m_ocaBufContextList[ocaBufHandle].logSection.offset, size);
515 MOS_OS_CHK_STATUS_RETURN(MOS_SecureMemcpy((uint8_t *)m_ocaBufContextList[ocaBufHandle].logSection.base + m_ocaBufContextList[ocaBufHandle].logSection.offset, sizeCpy, p, sizeCpy));
516 m_ocaBufContextList[ocaBufHandle].logSection.offset += sizeCpy;
517 return MOS_STATUS_SUCCESS;
518}

535void MosOcaInterfaceSpecific::InitLogSection(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_RESOURCE resCmdBuf)
536{
537 MOS_LINUX_BO *boCmdBuf = resCmdBuf->bo;
538 if (boCmdBuf->size <= OCA_LOG_SECTION_SIZE_MAX)
539 {
540 MOS_OS_NORMALMESSAGE("1st level BB size is not larger than OCA_LOG_SECTION_SIZE_MAX.");
541 return;
542 }
543
544 uint64_t *logSectionBase = (uint64_t*)((char *)boCmdBuf->virt + boCmdBuf->size - OCA_LOG_SECTION_SIZE_MAX);
545 uint64_t magicNum = *logSectionBase;
546 if (OCA_LOG_SECTION_MAGIC_NUMBER != magicNum)
547 {
548 MOS_OS_NORMALMESSAGE("Log section not exists in current 1st level BB.");
549 return;
550 }
551
552 m_ocaBufContextList[ocaBufHandle].logSection.base = logSectionBase;
553 // Reserve an uint64 for magic number;
554 m_ocaBufContextList[ocaBufHandle].logSection.offset = sizeof(uint64_t);
555 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount = 0;
556 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped = 0;
557}

575void MosOcaInterfaceSpecific::OnOcaError(PMOS_CONTEXT mosCtx, MOS_STATUS status, const char *functionName, uint32_t lineNumber)
576{
577 if (MOS_SUCCEEDED(status))
578 {
579 return;
580 }
581
582 MOS_OS_ASSERTMESSAGE("%s failed at line %d with status = %d!", functionName, lineNumber, status);
583
584 MOS_TraceEventExt(EVENT_OCA_ERROR,
585 EVENT_TYPE_INFO,
586 &status,
587 sizeof(uint32_t),
588 nullptr,
589 0);
590
591 if (MOS_SUCCEEDED(s_ocaStatus))
592 {
593 MediaUserSettingSharedPtr userSettingPtr = nullptr;
594
595 s_ocaStatus = status;
596 s_lineNumForOcaErr = lineNumber;
597
598 userSettingPtr = MosInterface::MosGetUserSettingInstance(mosCtx);
599 ReportUserSetting(
600 userSettingPtr,
601 __MEDIA_USER_FEATURE_VALUE_OCA_STATUS,
602 status,
603 MediaUserSetting::Group::Device);
604
605 ReportUserSetting(
606 userSettingPtr,
607 __MEDIA_USER_FEATURE_VALUE_OCA_ERROR_HINT,
608 lineNumber,
609 MediaUserSetting::Group::Device);
610 }
611}

639void MosOcaInterfaceSpecific::Initialize(PMOS_CONTEXT mosContext)
640{
641 MosOcaInterfaceSpecific::InitOcaErrorHandler(mosContext);
642 if (m_isInitialized == false)
643 {
644 MediaUserSettingSharedPtr userSettingPtr = MosInterface::MosGetUserSettingInstance(mosContext);
645 // read isOcaEnabled from reg key
646 int32_t value = 0;
647 ReadUserSetting(
648 userSettingPtr,
649 value,
650 __MEDIA_USER_FEATURE_VALUE_ENABLE_UMD_OCA,
651 MediaUserSetting::Group::Device);
652
653 m_isOcaEnabled = value;
654 if (!m_isOcaEnabled)
655 {
656 return;
657 }
658 m_config.maxResInfoCount = OCA_MAX_RESOURCE_INFO_COUNT_MAX;
659 m_resInfoPool = MOS_NewArray(MOS_OCA_RESOURCE_INFO, m_config.maxResInfoCount * MAX_NUM_OF_OCA_BUF_CONTEXT);
660 if (nullptr == m_resInfoPool)
661 {
662 MOS_DeleteArray(m_resInfoPool);
663 return;
664 }
665 MosUtilities::MosZeroMemory(m_resInfoPool, sizeof(MOS_OCA_RESOURCE_INFO)*m_config.maxResInfoCount * MAX_NUM_OF_OCA_BUF_CONTEXT);
666 for (int i = 0; i < MAX_NUM_OF_OCA_BUF_CONTEXT; ++i)
667 {
668 m_ocaBufContextList[i].logSection.resInfo.resInfoList = m_resInfoPool + i * m_config.maxResInfoCount;
669 }
670 if (nullptr != m_ocaMutex)
671 {
672 return;
673 }
674
675 m_ocaMutex = MosUtilities::MosCreateMutex();
676 if (nullptr == m_ocaMutex)
677 {
678 return;
679 }
680
681 m_isInitialized = true;
682 }
