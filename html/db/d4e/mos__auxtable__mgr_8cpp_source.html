49static int AllocateCb(void *bufMgr, size_t size, size_t alignment, void **bo, void **cpuAddr, uint64_t *gpuAddr)
50{
51 int ret = 0;
52 MOS_BUFMGR *bm;
53 void *userptr;
54 mos_linux_bo *lbo;
55
56 if (bufMgr == nullptr)
57 return -EINVAL;
58
59 bm = (MOS_BUFMGR *)bufMgr;
60 userptr = MOS_AlignedAllocMemory(size, alignment);
61 if (userptr == nullptr)
62 return -ENOMEM;
63
64 lbo = mos_bo_alloc_userptr(bm, "GmmAuxPageTable", userptr, TILING_NONE, size, size, 0);
65 if (lbo == nullptr)
66 {
67 MOS_FreeMemory(userptr);
68 return -EAGAIN;
69 }
70
71 mos_bo_set_softpin(lbo);
72
73 *bo = lbo;
74 *cpuAddr = userptr;
75 *gpuAddr = (uint64_t)lbo->offset64;
76
77 return 0;
78}

175MOS_STATUS AuxTableMgr::MapResource(GMM_RESOURCE_INFO *gmmResInfo, MOS_LINUX_BO *bo)
176{
177 if (gmmResInfo == nullptr || bo == nullptr)
178 {
179 return MOS_STATUS_NULL_POINTER;
180 }
181
182 GMM_RESOURCE_FLAG flags = gmmResInfo->GetResFlags();
183 if ((flags.Info.MediaCompressed || flags.Info.RenderCompressed) &&
184 (flags.Gpu.MMC && flags.Gpu.CCS) &&
185 (bo->aux_mapped == false))
186 {
187 int ret = 0;
188
189 ret = mos_bo_set_softpin(bo);
190 if (ret != 0)
191 {
192 MOS_OS_ASSERTMESSAGE("Aux mapping failed: softpin failed");
193 return MOS_STATUS_UNKNOWN;
194 }
195
196 GMM_DDI_UPDATEAUXTABLE updateReq = {};
197
198 updateReq.BaseResInfo = gmmResInfo;
199 updateReq.BaseGpuVA = bo->offset64;
200 updateReq.Map = 1;
201 if (GMM_SUCCESS != ((GMM_PAGETABLE_MGR *)m_gmmPageTableMgr)->UpdateAuxTable(&updateReq))
202 {
203 MOS_OS_ASSERTMESSAGE("update AuxTable failed");
204 return MOS_STATUS_UNKNOWN;
205 }
206 bo->aux_mapped = true;
207 }
208 return MOS_STATUS_SUCCESS;
209}

211MOS_STATUS AuxTableMgr::UnmapResource(GMM_RESOURCE_INFO *gmmResInfo, MOS_LINUX_BO *bo)
212{
213 if (gmmResInfo && bo && bo->aux_mapped)
214 {
215 GMM_DDI_UPDATEAUXTABLE updateReq = {};
216
217 MOS_OS_ASSERT(gmmResInfo->GetResFlags().Info.MediaCompressed ||
218 gmmResInfo->GetResFlags().Info.RenderCompressed);
219
220 updateReq.BaseResInfo = gmmResInfo;
221 updateReq.BaseGpuVA = bo->offset64;
222 updateReq.Map = 0;
223 ((GMM_PAGETABLE_MGR*)m_gmmPageTableMgr)->UpdateAuxTable(&updateReq);
224 bo->aux_mapped = false;
225 }
226 return MOS_STATUS_SUCCESS;
227}

229MOS_STATUS AuxTableMgr::EmitAuxTableBOList(MOS_LINUX_BO *cmd_bo)
230{
231 // Retrieve aux table bo list from GMM and emit it to exec buffer bo list
232 int boCnt = ((GMM_PAGETABLE_MGR*)m_gmmPageTableMgr)->GetNumOfPageTableBOs(AUXTT);
233 if (boCnt <= 0) {
234 return MOS_STATUS_SUCCESS;
235 }
236
237 mos_linux_bo **boList = (MOS_LINUX_BO **)MOS_AllocAndZeroMemory(boCnt * sizeof(mos_linux_bo*));
238 if (boList == nullptr)
239 return MOS_STATUS_NO_SPACE;
240
241 ((GMM_PAGETABLE_MGR*)m_gmmPageTableMgr)->GetPageTableBOList(AUXTT, boList);
242 for (int i=0; i<boCnt; i++)
243 {
244 int ret = mos_bo_add_softpin_target(cmd_bo, boList[i], false);
245 if (ret != 0)
246 {
247 MOS_OS_ASSERTMESSAGE("Error patching alloc_bo = 0x%x, cmd_bo = 0x%x.",
248 (uintptr_t)boList[i],
249 (uintptr_t)cmd_bo);
250 return MOS_STATUS_UNKNOWN;
251 }
252 }
253 MOS_FreeMemory(boList);
254
255 return MOS_STATUS_SUCCESS;
256}
