599VAStatus
600VpUpdateProcMirrorState(PVPHAL_SURFACE pVpHalSrcSurf, uint32_t mirror_state)
601{
602 VP_DDI_FUNCTION_ENTER;
603 DDI_CHK_NULL(pVpHalSrcSurf, "Null pVpHalSrcSurf.", VA_STATUS_ERROR_INVALID_PARAMETER);
604
605 if(mirror_state > VA_MIRROR_VERTICAL)
606 {
607 VP_DDI_ASSERTMESSAGE("VpUpdateProcMirrorState mirror_state = %d is out of range.", mirror_state);
608 VP_DDI_ASSERTMESSAGE("VpUpdateProcMirrorState reset mirror_state to VA_MIRROR_NONE.");
609 mirror_state = VA_MIRROR_NONE;
610 }
611
612 // Rotation must be a valid angle
613 switch(pVpHalSrcSurf->Rotation)
614 {
615 case VPHAL_ROTATION_IDENTITY:
616 if(mirror_state == VA_MIRROR_HORIZONTAL)
617 {
618 pVpHalSrcSurf->Rotation = VPHAL_MIRROR_HORIZONTAL;
619 }
620 else if(mirror_state == VA_MIRROR_VERTICAL)
621 {
622 pVpHalSrcSurf->Rotation = VPHAL_MIRROR_VERTICAL;
623 }
624 break;
625 case VPHAL_ROTATION_90:
626 if(mirror_state == VA_MIRROR_HORIZONTAL)
627 {
628 pVpHalSrcSurf->Rotation = VPHAL_ROTATE_90_MIRROR_HORIZONTAL;
629 }
630 else if(mirror_state == VA_MIRROR_VERTICAL)
631 {
632 pVpHalSrcSurf->Rotation = VPHAL_ROTATE_90_MIRROR_VERTICAL;
633 }
634 break;
635 case VPHAL_ROTATION_180:
636 if(mirror_state == VA_MIRROR_HORIZONTAL)
637 {
638 pVpHalSrcSurf->Rotation = VPHAL_MIRROR_VERTICAL;
639 }
640 else if(mirror_state == VA_MIRROR_VERTICAL)
641 {
642 pVpHalSrcSurf->Rotation = VPHAL_MIRROR_HORIZONTAL;
643 }
644 break;
645 case VPHAL_ROTATION_270:
646 if(mirror_state == VA_MIRROR_HORIZONTAL)
647 {
648 pVpHalSrcSurf->Rotation = VPHAL_ROTATE_90_MIRROR_VERTICAL;
649 }
650 else if(mirror_state == VA_MIRROR_VERTICAL)
651 {
652 pVpHalSrcSurf->Rotation = VPHAL_ROTATE_90_MIRROR_HORIZONTAL;
653 }
654 break;
655 default:
656 VP_DDI_ASSERTMESSAGE("VpUpdateProcMirrorState Unexpected Invalid Rotation = %d.", pVpHalSrcSurf->Rotation);
657 return VA_STATUS_ERROR_INVALID_PARAMETER;
658
659 }
660
661 return VA_STATUS_SUCCESS;
662}

671bool IsProcmpEnable(PVPHAL_SURFACE pVpHalSrcSurf)
672{
673 DDI_CHK_NULL(pVpHalSrcSurf, "Null pVpHalSrcSurf.", VA_STATUS_ERROR_INVALID_PARAMETER);
674
675 if ((pVpHalSrcSurf->pProcampParams && pVpHalSrcSurf->pProcampParams->bEnabled) &&
676 (pVpHalSrcSurf->pProcampParams->fContrast == 1 && pVpHalSrcSurf->pProcampParams->fHue == 0 && pVpHalSrcSurf->pProcampParams->fSaturation == 1) &&
677 !pVpHalSrcSurf->pBlendingParams && !pVpHalSrcSurf->pLumaKeyParams && (!pVpHalSrcSurf->pIEFParams || !pVpHalSrcSurf->pIEFParams->bEnabled) &&
678 !pVpHalSrcSurf->pDeinterlaceParams && (!pVpHalSrcSurf->pDenoiseParams || (!pVpHalSrcSurf->pDenoiseParams->bEnableChroma && !pVpHalSrcSurf->pDenoiseParams->bEnableLuma)) &&
679 (!pVpHalSrcSurf->pColorPipeParams || (!pVpHalSrcSurf->pColorPipeParams->bEnableACE && !pVpHalSrcSurf->pColorPipeParams->bEnableSTE && !pVpHalSrcSurf->pColorPipeParams->bEnableTCC)) &&
680 !pVpHalSrcSurf->pHDRParams)
681 {
682 return true;
683 }
684
685 return false;
686}

3755DdiVp_SetProcPipelineBlendingParams(
3756 PDDI_VP_CONTEXT pVpCtx,
3757 uint32_t uiSurfIndex,
3758 VAProcPipelineParameterBuffer* pPipelineParam)
3759{
3760 PVPHAL_RENDER_PARAMS pVpHalRenderParams;
3761 PVPHAL_SURFACE pSrc;
3762 PVPHAL_SURFACE pTarget;
3763 bool bGlobalAlpha;
3764 bool bPreMultAlpha;
3765
3766 const VABlendState * blend_state = pPipelineParam->blend_state;
3767 bGlobalAlpha = false;
3768 bPreMultAlpha = false;
3769
3770 VP_DDI_FUNCTION_ENTER;
3771 DDI_CHK_NULL(pVpCtx, "Null pVpCtx.", VA_STATUS_ERROR_INVALID_CONTEXT);
3772 DDI_CHK_NULL(pPipelineParam,
3773 "Null pPipelineParam.",
3774 VA_STATUS_ERROR_INVALID_BUFFER);
3775
3776 // initialize
3777 pVpHalRenderParams = VpGetRenderParams(pVpCtx);
3778 DDI_CHK_NULL(pVpHalRenderParams,
3779 "Null pVpHalRenderParams.",
3780 VA_STATUS_ERROR_INVALID_PARAMETER);
3781
3782 pSrc = pVpHalRenderParams->pSrc[uiSurfIndex];
3783 DDI_CHK_NULL(pSrc, "Null pSrc.", VA_STATUS_ERROR_INVALID_SURFACE);
3784
3785 if(nullptr == pVpHalRenderParams->pCompAlpha)
3786 {
3787 pVpHalRenderParams->pCompAlpha = (PVPHAL_ALPHA_PARAMS)MOS_AllocAndZeroMemory(sizeof(VPHAL_ALPHA_PARAMS));
3788 DDI_CHK_NULL(pVpHalRenderParams->pCompAlpha, "Null pCompAlpha.", VA_STATUS_ERROR_ALLOCATION_FAILED);
3789 }
3790
3791 // So far vaapi does not have alpha fill mode API.
3792 // And for blending support, we use VPHAL_ALPHA_FILL_MODE_NONE by default.
3793 pTarget = pVpHalRenderParams->pTarget[0];
3794 DDI_CHK_NULL(pTarget, "Null pTarget.", VA_STATUS_ERROR_INVALID_SURFACE);
3795
3796 //For surface with alpha, we need to bypass SFC that could change the output alpha.
3797 if (hasAlphaInSurface(pSrc) &&
3798 hasAlphaInSurface(pTarget))
3799 {
3800 pVpHalRenderParams->pCompAlpha->fAlpha = 0.0f;
3801 pVpHalRenderParams->pCompAlpha->AlphaMode = VPHAL_ALPHA_FILL_MODE_SOURCE_STREAM;
3802 }
3803 else
3804 {
3805 pVpHalRenderParams->pCompAlpha->fAlpha = 1.0f;
3806 pVpHalRenderParams->pCompAlpha->AlphaMode = VPHAL_ALPHA_FILL_MODE_NONE;
3807 }
3808
3809 // First, no Blending
3810 if(!blend_state)
3811 {
3812 if (pSrc->pBlendingParams)
3813 {
3814 pSrc->pBlendingParams->BlendType = BLEND_NONE;
3815 pSrc->pBlendingParams->fAlpha = 1.0;
3816 }
3817
3818 if (pSrc->pLumaKeyParams)
3819 {
3820 pSrc->pLumaKeyParams->LumaLow = 0;
3821 pSrc->pLumaKeyParams->LumaHigh = 0;
3822 }
3823
3824 return VA_STATUS_SUCCESS;
3825 }
3826
3827 // Then, process all blending types
3828
3829 if(blend_state->flags & VA_BLEND_GLOBAL_ALPHA)
3830 bGlobalAlpha = true;
3831
3832 if(blend_state->flags & VA_BLEND_PREMULTIPLIED_ALPHA)
3833 bPreMultAlpha = true;
3834
3835 if(nullptr == pSrc->pBlendingParams)
3836 {
3837 pSrc->pBlendingParams = (PVPHAL_BLENDING_PARAMS)MOS_AllocAndZeroMemory(sizeof(VPHAL_BLENDING_PARAMS));
3838 DDI_CHK_NULL(pSrc->pBlendingParams, "Null pBlendingParams.", VA_STATUS_ERROR_ALLOCATION_FAILED);
3839 }
3840
3842 // Surfaces contains alpha value, will be devide into premultiplied and non-premultiplied:
3843 // For premultiplied surface, will support:
3844 // a. per-pixel alpha blending;
3845 // b. per-pixel plus per-plane alpha blending;
3846 // For non-promultilied surface, will support:
3847 // a. per-plane alpha blending;
3848 // b. per-pixel alpha blending;
3849 // c. NOT support per-pixel plus per-plane alpha blending;
3850 // Surfaces NOT contains alpha value, will not be devided into premultiplied or non-premultiplied:
3851 // Will only support per-plane alpha blending.
3853 if(hasAlphaInSurface(pSrc))
3854 {
3855 // For premultiplied surface, we just support four blending types until now
3856 if(bPreMultAlpha && !bGlobalAlpha) // Do per-pixel alpha blending for premultiplied surface.
3857 {
3858 pSrc->pBlendingParams->BlendType = BLEND_PARTIAL;
3859 }
3860 else if(bPreMultAlpha && bGlobalAlpha) // Do per-pixel plus per-plane alpha blending For premultiplied surface.
3861 {
3862 if(blend_state->global_alpha < 1.0f)
3863 {
3864 pSrc->pBlendingParams->BlendType = BLEND_CONSTANT_PARTIAL;
3865 }
3866 else
3867 {
3868 // Optimization when per-plane alpha value is 1.0
3869 pSrc->pBlendingParams->BlendType = BLEND_PARTIAL;
3870 }
3871 pSrc->pBlendingParams->fAlpha = blend_state->global_alpha;
3872 }
3873 else if(!bPreMultAlpha && bGlobalAlpha) // Do per-plane alpha blending for non-premultiplied surface, not do per-pixel blending
3874 {
3875 pSrc->pBlendingParams->BlendType = BLEND_CONSTANT;
3876 pSrc->pBlendingParams->fAlpha = blend_state->global_alpha;
3877 VP_DDI_ASSERTMESSAGE("BLEND_CONSTANT do not support alpha calculating.\n");
3878 }
3879 else if(!bPreMultAlpha && !bGlobalAlpha) // Do per-pixel alpha blending for non-premultiplied surface
3880 {
3881 pSrc->pBlendingParams->BlendType = BLEND_SOURCE;
3882 VP_DDI_ASSERTMESSAGE("BLEND_SOURCE do not support alpha calculating.\n");
3883 }
3884
3885 // Not support per-plane plus per-pixel alpha blending for non-premultiplied surface.
3886 }
3887 else
3888 {
3889 if(bGlobalAlpha) // Do per-plane alpha blending for surface which not contain alpha value
3890 {
3891 if(blend_state->global_alpha < 1.0f)
3892 {
3893 pSrc->pBlendingParams->BlendType = BLEND_CONSTANT;
3894 pSrc->pBlendingParams->fAlpha = blend_state->global_alpha;
3895 VP_DDI_ASSERTMESSAGE("BLEND_CONSTANT do not support alpha calculating.\n");
3896 }
3897 else
3898 {
3899 pSrc->pBlendingParams->BlendType = BLEND_PARTIAL;
3900 VP_DDI_NORMALMESSAGE("Because BLEND_CONSTANT do not support alpha calculating, use BLEND_PARTIAL instead.\n");
3901 }
3902
3903 // bPreMultAlpha should not be true for surfaces which do not contain alpha value,
3904 // but inorder to make the driver more usable, choose the most reasonable blending mode, but print out a message.
3905 if(bPreMultAlpha)
3906 {
3907 VP_DDI_NORMALMESSAGE("Should not set VA_BLEND_PREMULTIPLIED_ALPHA for surface which do not contain alpha value.\n");
3908 }
3909 }
3910 else
3911 {
3912 pSrc->pBlendingParams->BlendType = BLEND_NONE;
3913 pSrc->pBlendingParams->fAlpha = 1.0;
3914
3915 // bPreMultAlpha should not be true for surfaces which do not contain alpha value,
3916 // but inorder to make the driver more usable, choose the most reasonable blending mode, but print out a message.
3917 if(bPreMultAlpha)
3918 {
3919 VP_DDI_NORMALMESSAGE("Should not set VA_BLEND_PREMULTIPLIED_ALPHA for surface which do not contain alpha value.\n");
3920 }
3921 }
3922 }
3923
3924 if(blend_state->flags & VA_BLEND_LUMA_KEY)
3925 {
3926 if(nullptr == pSrc->pLumaKeyParams)
3927 {
3928 pSrc->pLumaKeyParams = (PVPHAL_LUMAKEY_PARAMS)MOS_AllocAndZeroMemory(sizeof(VPHAL_LUMAKEY_PARAMS));
3929 DDI_CHK_NULL(pSrc->pLumaKeyParams, "Null pLumaKeyParams.", VA_STATUS_ERROR_ALLOCATION_FAILED);
3930 }
3931 pSrc->pLumaKeyParams->LumaLow = (int16_t)(pPipelineParam->blend_state->min_luma * 255);
3932 pSrc->pLumaKeyParams->LumaHigh = (int16_t)(pPipelineParam->blend_state->max_luma * 255);
3933 }
3934
3935 return VA_STATUS_SUCCESS;
3936}
