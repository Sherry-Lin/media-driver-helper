68MOS_STATUS OsContextSpecific::LockSemaphore(int32_t semid)
69{
70 struct sembuf op[2];
71 op[0].sem_num = 0; // wait for [0] to be 0
72 op[0].sem_op = 0;
73 op[0].sem_flg = 0;
74 op[1].sem_num = 0;
75 op[1].sem_op = 1; // increment
76 op[1].sem_flg = SEM_UNDO;
77
78 if (semid < 0)
79 {
80 return MOS_STATUS_UNKNOWN;
81 }
82
83 if (semop(semid, op, 2) < 0)
84 {
85 return MOS_STATUS_UNKNOWN;
86 }
87
88 return MOS_STATUS_SUCCESS;
89}

91MOS_STATUS OsContextSpecific::UnLockSemaphore(int32_t semid)
92{
93 struct sembuf op;
94 op.sem_num = 0;
95 op.sem_op = -1; // decrement back to 0
96 op.sem_flg = SEM_UNDO;
97
98 if (semid < 0)
99 {
100 return MOS_STATUS_UNKNOWN;
101 }
102
103 if (semop(semid, &op, 1) < 0)
104 {
105 return MOS_STATUS_UNKNOWN;
106 }
107
108 return MOS_STATUS_SUCCESS;

148MOS_STATUS OsContextSpecific::ConnectCreateShm(long key, uint32_t size, int32_t *pShmid, void* *ppShm)
149{
150 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
151 MOS_OS_CHK_NULL_RETURN(pShmid);
152 MOS_OS_CHK_NULL_RETURN(ppShm);
153
154 struct shmid_ds buf;
155 int32_t shmid = 0;
156 key_t key_value = (key_t)key;
157 void *shmptr = nullptr;
158 MOS_ZeroMemory(&buf, sizeof(buf));
159
160 shmid = shmget(key_value, size, IPC_CREAT | 0666);
161 if (shmid < 0)
162 {
163 return MOS_STATUS_UNKNOWN;
164 }
165
166 shmptr = shmat(shmid, 0, 0);
167 if (shmptr == MOS_LINUX_SHM_INVALID)
168 {
169 DetachDestroyShm(shmid, shmptr);
170 return MOS_STATUS_UNKNOWN;
171 }
172
173 if (shmctl(shmid, IPC_STAT, &buf) < 0)
174 {
175 // can't get any status info
176 DetachDestroyShm(shmid, shmptr);
177 return MOS_STATUS_UNKNOWN;
178 }
179
180 *ppShm = shmptr;
181 *pShmid = shmid;
182
183 return MOS_STATUS_SUCCESS;
184}

201 return MOS_STATUS_UNKNOWN;

212 {
213 return MOS_STATUS_UNKNOWN;
214 }

216 return MOS_STATUS_SUCCESS;

219MOS_STATUS OsContextSpecific::ConnectCreateSemaphore(long key, int32_t *pSemid)
220{
221 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
222 MOS_OS_CHK_NULL_RETURN(pSemid);
223
224 int32_t semid = 0;
225 struct sembuf sop;
226 struct semid_ds buf;
227 key_t key_value = (key_t)key;
228 int32_t val = 0;
229
230 MOS_ZeroMemory(&sop, sizeof(sop));
231 MOS_ZeroMemory(&buf, sizeof(buf));
232
233 semid = semget(key, 1, IPC_CREAT | IPC_EXCL | 0666);
234
235 if (semid != MOS_LINUX_IPC_INVALID_ID)
236 {
237 // initialize it to 0
238 if (semctl(semid, 0, SETVAL, val) == -1)
239 {
240 return MOS_STATUS_UNKNOWN;
241 }
242
243 // Perform a "no-op" semaphore operation - changes sem_otime
244 // so other processes can see we've initialized the set.
245 sop.sem_num = 0;
246 sop.sem_op = 0; //Wait for value to equal 0
247 sop.sem_flg = 0;
248 if (semop(semid, &sop, 1) == -1)
249 {
250 return MOS_STATUS_UNKNOWN;
251 }
252 }
253 else
254 {
255 // errno EEXIST
256 semid = semget(key, 1, 0666);
257 if (semid == MOS_LINUX_IPC_INVALID_ID)
258 {
259 return MOS_STATUS_UNKNOWN;
260 }
261 }
262
263 *pSemid = semid;
264
265 return MOS_STATUS_SUCCESS;
266}

268MOS_STATUS OsContextSpecific::CreateIPC()
269{
270 MOS_STATUS eStatus;
271
272 m_semId = MOS_LINUX_IPC_INVALID_ID;
273 m_shmId = MOS_LINUX_IPC_INVALID_ID;
274 m_shm = MOS_LINUX_SHM_INVALID;
275 if (m_apoMosEnabled)
276 {
277 return MOS_STATUS_SUCCESS;
278 }
279
280 struct semid_ds buf;
281 MOS_ZeroMemory(&buf, sizeof(buf));
282 //wait and retry till to get a valid semphore
283 for(int i = 0; i < MOS_LINUX_SEM_MAX_TRIES; i++)
284 {
285 ConnectCreateSemaphore(m_dualVdboxKey, &m_semId);
286
287 if (m_semId == MOS_LINUX_IPC_INVALID_ID)
288 {
289 return MOS_STATUS_UNKNOWN;
290 }
291 //check wether the semid is initialized or not
292 if (semctl(m_semId, 0, IPC_STAT, &buf) == -1)
293 {
294 return MOS_STATUS_UNKNOWN;
295 }
296 if (buf.sem_otime != 0)
297 {
298 break;
299 }
300
301 MosUtilities::MosSleep(1); //wait and retry
302 }
303
304 LockSemaphore(m_semId);
305 eStatus = ConnectCreateShm(m_dualVdboxKey, sizeof(VDBOX_WORKLOAD), &(m_shmId), &(m_shm));
306 UnLockSemaphore(m_semId);
307 MOS_CHK_STATUS_SAFE(eStatus);
308
309finish:
310 return eStatus;
311}

313void OsContextSpecific::DestroyIPC()
314{
315 if (m_apoMosEnabled)
316 {
317 return;
318 }
319
320 if (MOS_LINUX_IPC_INVALID_ID != m_semId)
321 {
322 int16_t iAttachedNum = 0;
323
324 if (MOS_LINUX_IPC_INVALID_ID != m_shmId)
325 {
326 LockSemaphore(m_semId);
327 iAttachedNum = ShmAttachedNumber(m_shmId);
328
329 DetachDestroyShm(m_shmId, m_shm);
330 m_shmId = MOS_LINUX_IPC_INVALID_ID;
331 m_shm = MOS_LINUX_SHM_INVALID;
332
333 if (iAttachedNum)
334 {
335 --iAttachedNum;
336 }
337 UnLockSemaphore(m_semId);
338 }
339 }

409MOS_STATUS OsContextSpecific::Init(PMOS_CONTEXT pOsDriverContext)
410{
411 uint32_t iDeviceId = 0;
412 MOS_STATUS eStatus;
413 uint32_t i = 0;
414
415 MOS_OS_FUNCTION_ENTER;
416
417 eStatus = MOS_STATUS_SUCCESS;
418
419 if (GetOsContextValid() == false)
420 {
421 if( nullptr == pOsDriverContext ||
422 nullptr == pOsDriverContext->bufmgr ||
423 0 >= pOsDriverContext->fd )
424 {
425 MOS_OS_ASSERT(false);
426 return MOS_STATUS_INVALID_HANDLE;
427 }
428 m_apoMosEnabled = pOsDriverContext->m_apoMosEnabled;
429 m_bufmgr = pOsDriverContext->bufmgr;
430 m_gpuContextMgr = static_cast<GpuContextMgr *>(pOsDriverContext->m_gpuContextMgr);
431 m_cmdBufMgr = static_cast<CmdBufMgr *>(pOsDriverContext->m_cmdBufMgr);
432 m_fd = pOsDriverContext->fd;
433 MOS_SecureMemcpy(&m_perfData, sizeof(PERF_DATA), pOsDriverContext->pPerfData, sizeof(PERF_DATA));
434 mos_bufmgr_gem_enable_reuse(pOsDriverContext->bufmgr);
435 m_pGmmClientContext = pOsDriverContext->pGmmClientContext;
436 m_auxTableMgr = pOsDriverContext->m_auxTableMgr;
437
438 // DDI layer can pass over the DeviceID.
439 iDeviceId = pOsDriverContext->iDeviceId;
440 if (0 == iDeviceId)
441 {
442 PLATFORM platformInfo;
443 MEDIA_FEATURE_TABLE skuTable;
444 MEDIA_WA_TABLE waTable;
445 MEDIA_SYSTEM_INFO gtSystemInfo;
446
447 MOS_ZeroMemory(&platformInfo, sizeof(platformInfo));
448 MOS_ZeroMemory(&skuTable, sizeof(skuTable));
449 MOS_ZeroMemory(&waTable, sizeof(waTable));
450 MOS_ZeroMemory(&gtSystemInfo, sizeof(gtSystemInfo));
451 eStatus = HWInfo_GetGfxInfo(pOsDriverContext->fd, pOsDriverContext->bufmgr, &platformInfo, &skuTable, &waTable, &gtSystemInfo, pOsDriverContext->m_userSettingPtr);
452 if (eStatus != MOS_STATUS_SUCCESS)
453 {
454 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Sku/Wa/GtSystemInfo initialization");
455 return eStatus;
456 }
457
458 MOS_SecureMemcpy(&m_platformInfo, sizeof(PLATFORM), &platformInfo, sizeof(PLATFORM));
459 MOS_SecureMemcpy(&m_gtSystemInfo, sizeof(MEDIA_SYSTEM_INFO), &gtSystemInfo, sizeof(MEDIA_SYSTEM_INFO));
460
461 pOsDriverContext->iDeviceId = platformInfo.usDeviceID;
462 m_skuTable = skuTable;
463 m_waTable = waTable;
464
465 pOsDriverContext->SkuTable = skuTable;
466 pOsDriverContext->WaTable = waTable;
467 pOsDriverContext->gtSystemInfo = gtSystemInfo;
468 pOsDriverContext->platform = platformInfo;
469
470 MOS_OS_NORMALMESSAGE("DeviceID was created DeviceID = %d, platform product %d", iDeviceId, platformInfo.eProductFamily);
471 }
472 else
473 {
474 // pOsDriverContext's parameters were passed by CmCreateDevice.
475 // Get SkuTable/WaTable/systemInfo/platform from OSDriver directly.
476 MOS_SecureMemcpy(&m_platformInfo, sizeof(PLATFORM), &(pOsDriverContext->platform), sizeof(PLATFORM));
477 MOS_SecureMemcpy(&m_gtSystemInfo, sizeof(MEDIA_SYSTEM_INFO), &(pOsDriverContext->gtSystemInfo), sizeof(MEDIA_SYSTEM_INFO));
478
479 m_skuTable = pOsDriverContext->SkuTable;
480 m_waTable = pOsDriverContext->WaTable;
481 }
482
483 m_use64BitRelocs = true;
484 m_useSwSwizzling = pOsDriverContext->bSimIsActive || MEDIA_IS_SKU(&m_skuTable, FtrUseSwSwizzling);
485 m_tileYFlag = MEDIA_IS_SKU(&m_skuTable, FtrTileY);
486
487 MOS_TraceEventExt(EVENT_GPU_CONTEXT_CREATE, EVENT_TYPE_START,
488 &eStatus, sizeof(eStatus), nullptr, 0);
489 if (!Mos_Solo_IsEnabled(nullptr) && MEDIA_IS_SKU(&m_skuTable,FtrContextBasedScheduling))
490 {
491 m_intelContext = mos_gem_context_create_ext(pOsDriverContext->bufmgr,0);
492 if (m_intelContext)
493 {
494 m_intelContext->vm = mos_gem_vm_create(pOsDriverContext->bufmgr);
495 if (m_intelContext->vm == nullptr)
496 {
497 MOS_OS_ASSERTMESSAGE("Failed to create vm.\n");
498 return MOS_STATUS_UNKNOWN;
499 }
500 }
501 }
502 else //use legacy context create ioctl for pre-gen11 platforms
503 {
504 m_intelContext = mos_gem_context_create(pOsDriverContext->bufmgr);
505 if (m_intelContext)
506 {
507 m_intelContext->vm = nullptr;
508 }
509 }
510 MOS_TraceEventExt(EVENT_GPU_CONTEXT_CREATE, EVENT_TYPE_END,
511 &m_intelContext, sizeof(void *),
512 &eStatus, sizeof(eStatus));
513
514 if (m_intelContext == nullptr)
515 {
516 MOS_OS_ASSERTMESSAGE("Failed to create drm intel context");
517 return MOS_STATUS_UNKNOWN;
518 }
519
520 m_isAtomSOC = IS_ATOMSOC(iDeviceId);
521
522 #ifndef ANDROID
523
524 if ((m_gtSystemInfo.VDBoxInfo.IsValid) && (m_gtSystemInfo.VDBoxInfo.NumberOfVDBoxEnabled > 1))
525 {
526 m_kmdHasVCS2 = true;
527 }
528 else
529 {
530 m_kmdHasVCS2 = false;
531 }
532
533 if (m_kmdHasVCS2)
534 {
535 eStatus = CreateIPC();
536 if (eStatus != MOS_STATUS_SUCCESS)
537 {
538 MOS_OS_ASSERTMESSAGE("Fatal error - create IPC failed");
539 return eStatus;
540 }
541 }
542
543 eStatus = CreateSSEUIPC();
544 if (eStatus != MOS_STATUS_SUCCESS)
545 {
546 MOS_OS_ASSERTMESSAGE("Fatal error - Failed to create shared memory for SSEU configuration.");
547 return eStatus;
548 }
549 #endif
550
551 m_transcryptedKernels = nullptr;
552 m_transcryptedKernelsSize = 0;
553
554 // For Media Memory compression
555 m_mediaMemDecompState = pOsDriverContext->ppMediaMemDecompState;
556 m_memoryDecompress = pOsDriverContext->pfnMemoryDecompress;
557 m_mediaMemCopy = pOsDriverContext->pfnMediaMemoryCopy;
558 m_mediaMemCopy2D = pOsDriverContext->pfnMediaMemoryCopy2D;
559 m_mosContext = pOsDriverContext;
560
561 m_noParsingAssistanceInKmd = true;
562 m_numNalUnitBytesIncluded = MOS_NAL_UNIT_LENGTH - MOS_NAL_UNIT_STARTCODE_LENGTH;
563
564 // Init reset count for the context
565 uint32_t dwResetCount = 0;
566 mos_get_reset_stats(m_intelContext, &dwResetCount, nullptr, nullptr);
567 m_gpuResetCount = dwResetCount;
568 m_gpuActiveBatch = 0;
569 m_gpuPendingBatch = 0;
570
571 m_usesPatchList = true;
572 m_usesGfxAddress = false;
573
574 m_inlineCodecStatusUpdate = true;
575
576 SetOsContextValid(true);
577 }
578 return eStatus;
579}

581void OsContextSpecific::Destroy()
582{
583 MOS_OS_FUNCTION_ENTER;
584
585 if (GetOsContextValid() == true)
586 {
587 // APO MOS will destory each stream's GPU context at different place
588 if (!m_apoMosEnabled)
589 {
590 for (auto i = 0; i < MOS_GPU_CONTEXT_MAX; i++)
591 {
592 if (m_GpuContextHandle[i] != MOS_GPU_CONTEXT_INVALID_HANDLE)
593 {
594 if (m_gpuContextMgr == nullptr)
595 {
596 MOS_OS_ASSERTMESSAGE("GpuContextMgr is null when destroy GpuContext");
597 break;
598 }
599 auto gpuContext = m_gpuContextMgr->GetGpuContext(m_GpuContextHandle[i]);
600 if (gpuContext == nullptr)
601 {
602 MOS_OS_ASSERTMESSAGE("cannot find the gpuContext corresponding to the active gpuContextHandle");
603 continue;
604 }
605 m_gpuContextMgr->DestroyGpuContext(gpuContext);
606 }
607 }
608 }
609
610 #ifndef ANDROID
611 if (m_kmdHasVCS2)
612 {
613 DestroyIPC();
614 }
615 DestroySSEUIPC();
616 #endif
617 m_skuTable.reset();
618 m_waTable.reset();
619 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_START,
620 &m_intelContext, sizeof(void *), nullptr, 0);
621 if (m_intelContext && m_intelContext->vm)
622 {
623 mos_gem_vm_destroy(m_intelContext->bufmgr, m_intelContext->vm);
624 }
625 if (m_intelContext)
626 {
627 mos_gem_context_destroy(m_intelContext);
628 }
629 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_END,
630 nullptr, 0, nullptr, 0);
631 SetOsContextValid(false);
632 }
