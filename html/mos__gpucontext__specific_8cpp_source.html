92

105

108 m_commandBufferSize = COMMAND_BUFFER_SIZE;
109
110 m_nextFetchIndex = 0;
111

141 m_createOptionEnhanced = (MOS_GPUCTX_CREATOPTIONS_ENHANCED*)MOS_AllocAndZeroMemory(sizeof(MOS_GPUCTX_CREATOPTIONS_ENHANCED));

143 m_createOptionEnhanced->SSEUValue = createOption->SSEUValue;
144
145 if (typeid(*createOption) == typeid(MOS_GPUCTX_CREATOPTIONS_ENHANCED))

179 return MOS_STATUS_UNKNOWN;

182 MOS_OS_CHK_NULL_RETURN(engine_map);
183
184 if (GpuNode == MOS_GPU_NODE_3D)
185 {
186 __u16 engine_class = I915_ENGINE_CLASS_RENDER;
187 __u64 caps = 0;
188
189 if (mos_query_engines(osInterface->pOsContext->bufmgr, engine_class, caps, &nengine, engine_map))
190 {
191 MOS_OS_ASSERTMESSAGE("Failed to query engines.\n");
192 MOS_SafeFreeMemory(engine_map);
193 return MOS_STATUS_UNKNOWN;
194 }
195
196 if (mos_set_context_param_load_balance(m_i915Context[0], engine_map, nengine))
197 {
198 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
199 MOS_SafeFreeMemory(engine_map);
200 return MOS_STATUS_UNKNOWN;
201 }
202
203 if (createOption->SSEUValue != 0)
204 {
205 struct drm_i915_gem_context_param_sseu sseu;
206 MOS_ZeroMemory(&sseu, sizeof(sseu));
207 sseu.flags = I915_CONTEXT_SSEU_FLAG_ENGINE_INDEX;
208 sseu.engine.engine_instance = m_i915ExecFlag;
209
210 if (mos_get_context_param_sseu(m_i915Context[0], &sseu))
211 {
212 MOS_OS_ASSERTMESSAGE("Failed to get sseu configuration.");
213 MOS_SafeFreeMemory(engine_map);
214 return MOS_STATUS_UNKNOWN;
215 }
216
217 if (mos_hweight8(sseu.subslice_mask) > createOption->packed.SubSliceCount)
218 {
219 sseu.subslice_mask = mos_switch_off_n_bits(sseu.subslice_mask,
220 mos_hweight8(sseu.subslice_mask)-createOption->packed.SubSliceCount);
221 }
222
223 if (mos_set_context_param_sseu(m_i915Context[0], sseu))
224 {
225 MOS_OS_ASSERTMESSAGE("Failed to set sseu configuration.");
226 MOS_SafeFreeMemory(engine_map);
227 return MOS_STATUS_UNKNOWN;
228 }
229 }
230 }
231 else if (GpuNode == MOS_GPU_NODE_COMPUTE)
232 {
233 __u16 engine_class = 4; //To change later when linux define the name
234 __u64 caps = 0;
235
236 if (mos_query_engines(osInterface->pOsContext->bufmgr, engine_class, caps, &nengine, engine_map))
237 {
238 MOS_OS_ASSERTMESSAGE("Failed to query engines.\n");
239 MOS_SafeFreeMemory(engine_map);

241 }
242
243#if (_DEBUG || _RELEASE_INTERNAL)
244 SelectEngineInstanceByUser(engine_map, &nengine, m_engineInstanceSelect, GpuNode);
245#endif

247 {
248 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
249 MOS_SafeFreeMemory(engine_map);

251 }
252 }

257 __u64 caps = 0;

273 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
274 MOS_SafeFreeMemory(engine_map);
275 return MOS_STATUS_UNKNOWN;
276 }
277
278 if (nengine >= 2)
279 {
280 int i;
281 //master queue
282 m_i915Context[1] = mos_gem_context_create_shared(osInterface->pOsContext->bufmgr,
283 osInterface->pOsContext->intel_context,
284 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE);
285 if (m_i915Context[1] == nullptr)
286 {
287 MOS_OS_ASSERTMESSAGE("Failed to create master context.\n");
288 MOS_SafeFreeMemory(engine_map);
289 return MOS_STATUS_UNKNOWN;
290 }
291 m_i915Context[1]->pOsContext = osInterface->pOsContext;
292
293 if (mos_set_context_param_load_balance(m_i915Context[1], engine_map, 1))
294 {
295 MOS_OS_ASSERTMESSAGE("Failed to set master context bond extension.\n");
296 MOS_SafeFreeMemory(engine_map);
297 return MOS_STATUS_UNKNOWN;
298 }
299
300 //slave queue
301 for (i=1; i<nengine; i++)
302 {
303 m_i915Context[i+1] = mos_gem_context_create_shared(osInterface->pOsContext->bufmgr,
304 osInterface->pOsContext->intel_context,
305 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE);
306 if (m_i915Context[i+1] == nullptr)
307 {
308 MOS_OS_ASSERTMESSAGE("Failed to create slave context.\n");
309 MOS_SafeFreeMemory(engine_map);

311 }

330 }

333 {

340 for(i = 1; i < nengine; i++)
341 {

345 0); // I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE not allowed for parallel submission

348 MOS_OS_ASSERTMESSAGE("Failed to set parallel extension since discontinuous logical engine.\n");
349 mos_gem_context_destroy(m_i915Context[i]);

351 break;
352 }
353 }
354 }
355 }
356 }
357 else if (GpuNode == MOS_GPU_NODE_BLT)
358 {
359 __u16 engine_class = I915_ENGINE_CLASS_COPY;
360 __u64 caps = 0;
361
362 if (mos_query_engines(osInterface->pOsContext->bufmgr, engine_class, caps, &nengine, engine_map))
363 {
364 MOS_OS_ASSERTMESSAGE("Failed to query engines.\n");
365 MOS_SafeFreeMemory(engine_map);
366 return MOS_STATUS_UNKNOWN;
367 }
368
369 if (mos_set_context_param_load_balance(m_i915Context[0], engine_map, nengine))
370 {
371 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
372 MOS_SafeFreeMemory(engine_map);
373 return MOS_STATUS_UNKNOWN;
374 }
375 }
376 else
377 {
378 MOS_OS_ASSERTMESSAGE("Unknown engine class.\n");
379 MOS_SafeFreeMemory(engine_map);
380 return MOS_STATUS_UNKNOWN;
381 }
382 MOS_SafeFreeMemory(engine_map);
383 MOS_TraceEventExt(EVENT_GPU_CONTEXT_CREATE, EVENT_TYPE_END,
384 m_i915Context, sizeof(void *),
385 &nengine, sizeof(nengine));
386 }
387 return MOS_STATUS_SUCCESS;

390void GpuContextSpecific::Clear()
391{
392 MOS_OS_FUNCTION_ENTER;
393
394 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_START,
395 m_i915Context, sizeof(void *), nullptr, 0);
396 // hanlde the status buf bundled w/ the specified gpucontext
397 if (m_statusBufferResource)
398 {
399 if (m_statusBufferResource->Unlock(m_osContext) != MOS_STATUS_SUCCESS)
400 {
401 MOS_OS_ASSERTMESSAGE("failed to unlock the status buf bundled w/ the specified gpucontext");
402 }
403 m_statusBufferResource->Free(m_osContext, 0);
404 MOS_Delete(m_statusBufferResource);
405 }
406 MOS_FreeMemAndSetNull(m_statusBufferMosResource);
407
408 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
409
410 if (m_cmdBufMgr)
411 {
412 for (auto& curCommandBuffer : m_cmdBufPool)
413 {
414 auto curCommandBufferSpecific = static_cast<CommandBufferSpecific *>(curCommandBuffer);
415 if (curCommandBufferSpecific == nullptr)
416 continue;
417 curCommandBufferSpecific->waitReady(); // wait ready and return to comamnd buffer manager.
418 m_cmdBufMgr->ReleaseCmdBuf(curCommandBuffer);
419 }
420 }
421
422 m_cmdBufPool.clear();
423
424 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
425 MosUtilities::MosDestroyMutex(m_cmdBufPoolMutex);
426 m_cmdBufPoolMutex = nullptr;
427 MOS_SafeFreeMemory(m_commandBuffer);
428 MOS_SafeFreeMemory(m_allocationList);
429 MOS_SafeFreeMemory(m_patchLocationList);
430 MOS_SafeFreeMemory(m_attachedResources);
431 MOS_SafeFreeMemory(m_writeModeList);
432 MOS_SafeFreeMemory(m_createOptionEnhanced);
433
434 for (int i=0; i<MAX_ENGINE_INSTANCE_NUM; i++)
435 {
436 if (m_i915Context[i])
437 {
438 mos_gem_context_destroy(m_i915Context[i]);
439 m_i915Context[i] = nullptr;
440 }
441 }
442 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_END,
443 nullptr, 0, nullptr, 0);
444}

446MOS_STATUS GpuContextSpecific::RegisterResource(
447 PMOS_RESOURCE osResource,
448 bool writeFlag)
449{
450 MOS_OS_FUNCTION_ENTER;
451
452 MOS_OS_CHK_NULL_RETURN(osResource);
453
454 MOS_OS_CHK_NULL_RETURN(m_attachedResources);
455
456 PMOS_RESOURCE registeredResources = m_attachedResources;
457 uint32_t allocationIndex = 0;
458
459 for ( allocationIndex = 0; allocationIndex < m_resCount; allocationIndex++, registeredResources++)
460 {
461 if (osResource->bo == registeredResources->bo)
462 {
463 break;
464 }
465 }
466
467 // Allocation list to be updated
468 if (allocationIndex < m_maxNumAllocations)
469 {
470 // New buffer
471 if (allocationIndex == m_resCount)
472 {
473 m_resCount++;
474 }
475
476 // Set allocation
477 if (m_gpuContext >= MOS_GPU_CONTEXT_MAX)
478 {
479 MOS_OS_ASSERTMESSAGE("Gpu context exceeds max.");
480 return MOS_STATUS_UNKNOWN;
481 }
482
483 osResource->iAllocationIndex[m_gpuContext] = (allocationIndex);
484 m_attachedResources[allocationIndex] = *osResource;
485 m_writeModeList[allocationIndex] |= writeFlag;
486 m_allocationList[allocationIndex].hAllocation = &m_attachedResources[allocationIndex];
487 m_allocationList[allocationIndex].WriteOperation |= writeFlag;
488 m_numAllocations = m_resCount;
489 }
490 else
491 {
492 MOS_OS_ASSERTMESSAGE("Reached max # registrations.");
493 return MOS_STATUS_UNKNOWN;
494 }
495
496 return MOS_STATUS_SUCCESS;
497}

536MOS_STATUS GpuContextSpecific::GetCommandBuffer(
537 PMOS_COMMAND_BUFFER comamndBuffer,
538 uint32_t flags)
539{
540 MOS_OS_FUNCTION_ENTER;
541
542 MOS_OS_CHK_NULL_RETURN(comamndBuffer);
543 MOS_OS_CHK_NULL_RETURN(m_cmdBufMgr);
544 MOS_OS_CHK_NULL_RETURN(m_commandBuffer);
545
546 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
547 CommandBuffer* cmdBuf = nullptr;
548
549 uint32_t secondaryIdx = flags;
550 bool isPrimaryCmdBuffer = (secondaryIdx == 0);
551 bool hasSecondaryCmdBuffer = (!isPrimaryCmdBuffer &&
552 (m_secondaryCmdBufs.count(secondaryIdx) != 0));
553
554 bool needToAlloc = ((isPrimaryCmdBuffer && m_cmdBufFlushed) ||
555 (!isPrimaryCmdBuffer && !hasSecondaryCmdBuffer));
556
557 if (needToAlloc)
558 {
559 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
560 if (m_cmdBufPool.size() < MAX_CMD_BUF_NUM)
561 {
562 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
563 if (cmdBuf == nullptr)
564 {
565 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
566 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
567 return MOS_STATUS_NULL_POINTER;
568 }
569 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
570 {
571 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
572 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
573 return eStatus;
574 }
575 m_cmdBufPool.push_back(cmdBuf);
576 }
577 else if (m_cmdBufPool.size() == MAX_CMD_BUF_NUM && m_nextFetchIndex < m_cmdBufPool.size())
578 {
579 auto cmdBufOld = m_cmdBufPool[m_nextFetchIndex];
580 auto cmdBufSpecificOld = static_cast<CommandBufferSpecific *>(cmdBufOld);
581 if (cmdBufSpecificOld == nullptr)
582 {
583 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
584 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
585 return MOS_STATUS_NULL_POINTER;
586 }
587 cmdBufSpecificOld->waitReady();
588 cmdBufSpecificOld->UnBindToGpuContext();
589 m_cmdBufMgr->ReleaseCmdBuf(cmdBufOld); // here just return old command buffer to available pool
590
591 //pick up new comamnd buffer
592 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
593 if (cmdBuf == nullptr)
594 {
595 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
596 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
597 return MOS_STATUS_NULL_POINTER;
598 }
599 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
600 {
601 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
602 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
603 return eStatus;
604 }
605 m_cmdBufPool[m_nextFetchIndex] = cmdBuf;
606 }
607 else
608 {
609 MOS_OS_ASSERTMESSAGE("Command buffer bool size exceed max.");
610 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
611 return MOS_STATUS_UNKNOWN;
612 }
613 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
614
615 // util now, we got new command buffer from CmdBufMgr, next step to fill in the input command buffer
616 MOS_OS_CHK_STATUS_RETURN(cmdBuf->GetResource()->ConvertToMosResource(&comamndBuffer->OsResource));
617 comamndBuffer->pCmdBase = (uint32_t *)cmdBuf->GetLockAddr();
618 comamndBuffer->pCmdPtr = (uint32_t *)cmdBuf->GetLockAddr();
619 comamndBuffer->iOffset = 0;
620 comamndBuffer->iRemaining = cmdBuf->GetCmdBufSize();
621 comamndBuffer->iCmdIndex = m_nextFetchIndex;
622 comamndBuffer->iVdboxNodeIndex = MOS_VDBOX_NODE_INVALID;
623 comamndBuffer->iVeboxNodeIndex = MOS_VEBOX_NODE_INVALID;
624 comamndBuffer->is1stLvlBB = true;
625 comamndBuffer->Attributes.pAttriVe = nullptr;
626
627 // zero comamnd buffer
628 MOS_ZeroMemory(comamndBuffer->pCmdBase, comamndBuffer->iRemaining);
629 comamndBuffer->iSubmissionType = SUBMISSION_TYPE_SINGLE_PIPE;
630 MOS_ZeroMemory(&comamndBuffer->Attributes,sizeof(comamndBuffer->Attributes));
631
632 if (isPrimaryCmdBuffer)
633 {
634 // update command buffer relared filed in GPU context
635 m_cmdBufFlushed = false;
636
637 // keep a copy in GPU context
638 MOS_SecureMemcpy(m_commandBuffer, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
639 }
640 else
641 {
642 PMOS_COMMAND_BUFFER tempCmdBuf = (PMOS_COMMAND_BUFFER)MOS_AllocAndZeroMemory(sizeof(MOS_COMMAND_BUFFER));
643 MOS_OS_CHK_NULL_RETURN(tempCmdBuf);
644 m_secondaryCmdBufs[secondaryIdx] = tempCmdBuf;
645 MOS_SecureMemcpy(tempCmdBuf, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
646 }
647
648 // Command buffers are treated as cyclical buffers, the CB after the just submitted one
649 // has the minimal fence value that we should wait
650 m_nextFetchIndex++;
651 if (m_nextFetchIndex >= MAX_CMD_BUF_NUM)
652 {
653 m_nextFetchIndex = 0;
654 }
655 }
656 else
657 {
658 // current command buffer still active, directly copy to comamndBuffer
659 if (isPrimaryCmdBuffer)
660 {
661 MOS_SecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_commandBuffer, sizeof(MOS_COMMAND_BUFFER));
662 }
663 else
664 {
665 MOS_SecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_secondaryCmdBufs[secondaryIdx], sizeof(MOS_COMMAND_BUFFER));
666 }
667 }
668
669 if (isPrimaryCmdBuffer)
670 {
671 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_commandBuffer->OsResource, false));
672 }
673 else
674 {
675 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_secondaryCmdBufs[secondaryIdx]->OsResource, false));
676 }
677
678 return MOS_STATUS_SUCCESS;
679}

861
862MOS_STATUS GpuContextSpecific::SubmitCommandBuffer(
863 PMOS_INTERFACE osInterface,
864 PMOS_COMMAND_BUFFER cmdBuffer,
865 bool nullRendering)
866{
867 MOS_OS_FUNCTION_ENTER;
868
869 MOS_TraceEventExt(EVENT_MOS_BATCH_SUBMIT, EVENT_TYPE_START, nullptr, 0, nullptr, 0);
870
871 MOS_OS_CHK_NULL_RETURN(osInterface);
872 PMOS_CONTEXT osContext = osInterface->pOsContext;
873 MOS_OS_CHK_NULL_RETURN(osContext);
874 MOS_OS_CHK_NULL_RETURN(cmdBuffer);
875 MOS_OS_CHK_NULL_RETURN(m_patchLocationList);
876
877 MOS_GPU_NODE gpuNode = OSKMGetGpuNode(m_gpuContext);
878 uint32_t execFlag = gpuNode;
879 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
880 int32_t ret = 0;
881 bool scalaEnabled = false;
882 auto it = m_secondaryCmdBufs.begin();
883
884 // Command buffer object DRM pointer
885 m_cmdBufFlushed = true;
886 auto cmd_bo = cmdBuffer->OsResource.bo;
887
888 // Map Resource to Aux if needed
889 MapResourcesToAuxTable(cmd_bo);
890 for(auto it : m_secondaryCmdBufs)
891 {
892 MapResourcesToAuxTable(it.second->OsResource.bo);
893 }
894
895 if (m_secondaryCmdBufs.size() >= 2)
896 {
897 scalaEnabled = true;
898 cmdBuffer->iSubmissionType = SUBMISSION_TYPE_MULTI_PIPE_MASTER;
899 }
900
901 std::vector<PMOS_RESOURCE> mappedResList;
902 std::vector<MOS_LINUX_BO *> skipSyncBoList;
903
904 // Now, the patching will be done, based on the patch list.
905 for (uint32_t patchIndex = 0; patchIndex < m_currentNumPatchLocations; patchIndex++)
906 {
907 auto currentPatch = &m_patchLocationList[patchIndex];
908 MOS_OS_CHK_NULL_RETURN(currentPatch);
909
910 auto tempCmdBo = currentPatch->cmdBo == nullptr ? cmd_bo : currentPatch->cmdBo;
911
912 // Following are for Nested BB buffer, if it's nested BB, we need to ensure it's locked.
913 if (tempCmdBo != cmd_bo)
914 {
915 bool isSecondaryCmdBuf = false;
916 it = m_secondaryCmdBufs.begin();
917 while(it != m_secondaryCmdBufs.end())
918 {
919 if (it->second->OsResource.bo == tempCmdBo)
920 {
921 isSecondaryCmdBuf = true;
922 break;
923 }
924 it++;
925 }
926
927 for(auto allocIdx = 0; allocIdx < m_numAllocations && (!isSecondaryCmdBuf); allocIdx++)
928 {
929 auto tempRes = (PMOS_RESOURCE)m_allocationList[allocIdx].hAllocation;
930 if (tempCmdBo == tempRes->bo)
931 {
932 GraphicsResource::LockParams param;
933 param.m_writeRequest = true;
934 tempRes->pGfxResource->Lock(m_osContext, param);
935 mappedResList.push_back(tempRes);
936 break;
937 }
938 }
939 }
940
941 // This is the resource for which patching will be done
942 auto resource = (PMOS_RESOURCE)m_allocationList[currentPatch->AllocationIndex].hAllocation;
943 MOS_OS_CHK_NULL_RETURN(resource);
944
945 // For now, we'll assume the system memory's DRM bo pointer
946 // is NULL. If nullptr is detected, then the resource has been
947 // placed inside the command buffer's indirect state area.
948 // We'll simply set alloc_bo to the command buffer's bo pointer.
949 MOS_OS_ASSERT(resource->bo);
950
951 auto alloc_bo = (resource->bo) ? resource->bo : tempCmdBo;
952
953 MOS_OS_CHK_STATUS_RETURN(osInterface->osCpInterface->PermeatePatchForHM(
954 tempCmdBo->virt,
955 currentPatch,
956 resource));
957
958 uint64_t boOffset = alloc_bo->offset64;
959 if (alloc_bo != tempCmdBo)
960 {
961 auto item_ctx = osContext->contextOffsetList.begin();
962 for (; item_ctx != osContext->contextOffsetList.end(); item_ctx++)
963 {
964 if (item_ctx->intel_context == osContext->intel_context && item_ctx->target_bo == alloc_bo)
965 {
966 boOffset = item_ctx->offset64;
967 break;
968 }
969 }
970 }
971
972 MOS_OS_CHK_NULL_RETURN(tempCmdBo->virt);
973 if (osContext->bUse64BitRelocs)
974 {
975 *((uint64_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
976 boOffset + currentPatch->AllocationOffset;
977 }
978 else
979 {
980 *((uint32_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
981 boOffset + currentPatch->AllocationOffset;
982 }
983
984 if (scalaEnabled)
985 {
986 it = m_secondaryCmdBufs.begin();
987 while(it != m_secondaryCmdBufs.end())
988 {
989 if (it->second->OsResource.bo == tempCmdBo &&
990 it->second->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE &&
991 !mos_gem_bo_is_exec_object_async(alloc_bo))
992 {
993 skipSyncBoList.push_back(alloc_bo);
994 break;
995 }
996 it++;
997 }
998 }
999 else if (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE &&
1000 !mos_gem_bo_is_exec_object_async(alloc_bo))
1001 {
1002 skipSyncBoList.push_back(alloc_bo);
1003 }
1004
1005#if (_DEBUG || _RELEASE_INTERNAL)
1006 {
1007 uint32_t evtData[] = {alloc_bo->handle, currentPatch->uiWriteOperation, currentPatch->AllocationOffset};
1008 MOS_TraceEventExt(EVENT_MOS_BATCH_SUBMIT, EVENT_TYPE_INFO,
1009 evtData, sizeof(evtData),
1010 &boOffset, sizeof(boOffset));
1011 }
1012#endif
1013 if(mos_gem_bo_is_softpin(alloc_bo))
1014 {
1015 if (alloc_bo != tempCmdBo)
1016 {
1017 ret = mos_bo_add_softpin_target(tempCmdBo, alloc_bo, currentPatch->uiWriteOperation);
1018 }
1019 }
1020 else
1021 {
1022 // This call will patch the command buffer with the offsets of the indirect state region of the command buffer
1023 ret = mos_bo_emit_reloc2(
1024 tempCmdBo, // Command buffer
1025 currentPatch->PatchOffset, // Offset in the command buffer
1026 alloc_bo, // Allocation object for which the patch will be made.
1027 currentPatch->AllocationOffset, // Offset to the indirect state
1028 I915_GEM_DOMAIN_RENDER, // Read domain
1029 (currentPatch->uiWriteOperation) ? I915_GEM_DOMAIN_RENDER : 0x0, // Write domain
1030 boOffset);
1031 }
1032
1033 if (ret != 0)
1034 {
1035 MOS_OS_ASSERTMESSAGE("Error patching alloc_bo = 0x%x, cmd_bo = 0x%x.",
1036 (uintptr_t)alloc_bo,
1037 (uintptr_t)tempCmdBo);
1038 return MOS_STATUS_UNKNOWN;
1039 }
1040 }
1041
1042 for(auto res: mappedResList)
1043 {
1044 res->pGfxResource->Unlock(m_osContext);
1045 }
1046 mappedResList.clear();
1047
1048 if (scalaEnabled)
1049 {
1050 it = m_secondaryCmdBufs.begin();
1051 while(it != m_secondaryCmdBufs.end())
1052 {
1053 //Add Batch buffer End Command
1054 uint32_t batchBufferEndCmd = MI_BATCHBUFFER_END;
1055 if (MOS_FAILED(Mos_AddCommand(
1056 it->second,
1057 &batchBufferEndCmd,
1058 sizeof(uint32_t))))
1059 {
1060 MOS_OS_ASSERTMESSAGE("Inserting BB_END failed!");
1061 return MOS_STATUS_UNKNOWN;
1062 }
1063 it++;
1064 }
1065 }
1066 else
1067 {
1068 //Add Batch buffer End Command
1069 uint32_t batchBufferEndCmd = MI_BATCHBUFFER_END;
1070 if (MOS_FAILED(Mos_AddCommand(
1071 cmdBuffer,
1072 &batchBufferEndCmd,
1073 sizeof(uint32_t))))
1074 {
1075 MOS_OS_ASSERTMESSAGE("Inserting BB_END failed!");
1076 return MOS_STATUS_UNKNOWN;
1077 }
1078 }
1079
1080 // Now, we can unmap the video command buffer, since we don't need CPU access anymore.
1081 MOS_OS_CHK_NULL_RETURN(cmdBuffer->OsResource.pGfxResource);
1082 cmdBuffer->OsResource.pGfxResource->Unlock(m_osContext);
1083
1084 it = m_secondaryCmdBufs.begin();
1085 while(it != m_secondaryCmdBufs.end())
1086 {
1087 MOS_OS_CHK_NULL_RETURN(it->second->OsResource.pGfxResource);
1088 it->second->OsResource.pGfxResource->Unlock(m_osContext);
1089
1090 it++;
1091 }
1092
1093 int32_t perfData;
1094 if (osContext->pPerfData != nullptr)
1095 {
1096 perfData = *(int32_t *)(osContext->pPerfData);
1097 }
1098 else
1099 {
1100 perfData = 0;
1101 }
1102
1103 drm_clip_rect_t *cliprects = nullptr;
1104 int32_t num_cliprects = 0;
1105 int32_t DR4 = osContext->uEnablePerfTag ? perfData : 0;
1106
1107 //Since CB2 command is not supported, remove it and set cliprects to nullprt as default.
1108 if ((gpuNode == MOS_GPU_NODE_VIDEO || gpuNode == MOS_GPU_NODE_VIDEO2) &&
1109 (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_SINGLE_PIPE_MASK))
1110 {
1111 if (osContext->bKMDHasVCS2)
1112 {
1113 if (osContext->bPerCmdBufferBalancing && osInterface->pfnGetVdboxNodeId)
1114 {
1115 execFlag = GetVcsExecFlag(osInterface, cmdBuffer, gpuNode);
1116 }
1117 else if (gpuNode == MOS_GPU_NODE_VIDEO)
1118 {
1119 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING1;
1120 }
1121 else if (gpuNode == MOS_GPU_NODE_VIDEO2)
1122 {
1123 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING2;
1124 }
1125 else if ((gpuNode == MOS_GPU_NODE_BLT))
1126 {
1127 execFlag = I915_EXEC_BLT;
1128 }
1129 else
1130 {
1131 MOS_OS_ASSERTMESSAGE("Invalid gpuNode.");
1132 }
1133 }
1134 else
1135 {
1136 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING1;
1137 }
1138 }
1139
1140#if (_DEBUG || _RELEASE_INTERNAL)
1141
1142 MOS_LINUX_BO *bad_cmd_bo = nullptr;
1143 MOS_LINUX_BO *nop_cmd_bo = nullptr;
1144 uint32_t dwComponentTag = 0;
1145 uint32_t dwCallType = 0;
1146
1147 //dwComponentTag 3: decode,5: vpp,6: encode
1148 //dwCallType 8: PAK(CODECHAL_ENCODE_PERFTAG_CALL_PAK_ENGINE)
1149 // 34: PREENC
1150 // 5: VPP
1151 dwComponentTag = (perfData & 0xF000) >> 12;
1152 dwCallType = (perfData & 0xFC) >> 2;
1153
1154 if (osInterface->bTriggerCodecHang &&
1155 (dwComponentTag == 3 || (dwComponentTag == 6 && dwCallType == 8) ||
1156 (dwComponentTag == 6 && dwCallType == 34) ||
1157 (dwComponentTag == 5 && dwCallType == 5)))
1158 {
1159 bad_cmd_bo = Mos_GetBadCommandBuffer_Linux(osInterface);
1160 if (bad_cmd_bo)
1161 {
1162 ret = mos_bo_mrb_exec(bad_cmd_bo,
1163 4096,
1164 nullptr,
1165 0,
1166 0,
1167 execFlag);
1168 }
1169 else
1170 {
1171 MOS_OS_ASSERTMESSAGE("Mos_GetBadCommandBuffer_Linux failed!");
1172 }
1173 }
1174 else if (osInterface->bTriggerVPHang == true)
1175 {
1176 bad_cmd_bo = Mos_GetBadCommandBuffer_Linux(osInterface);
1177
1178 if (bad_cmd_bo)
1179 {
1180 ret = mos_bo_mrb_exec(bad_cmd_bo,
1181 4096,
1182 nullptr,
1183 0,
1184 0,
1185 execFlag);
1186 }
1187 else
1188 {
1189 MOS_OS_ASSERTMESSAGE("Mos_GetBadCommandBuffer_Linux failed!");
1190 }
1191
1192 osInterface->bTriggerVPHang = false;
1193 }
1194
1195 nop_cmd_bo = nullptr;
1196 if (nullRendering == true)
1197 {
1198 nop_cmd_bo = Mos_GetNopCommandBuffer_Linux(osInterface);
1199
1200 if (nop_cmd_bo)
1201 {
1202 ret = mos_bo_mrb_exec(nop_cmd_bo,
1203 4096,
1204 nullptr,
1205 0,
1206 0,
1207 execFlag);
1208 }
1209 else
1210 {
1211 MOS_OS_ASSERTMESSAGE("Mos_GetNopCommandBuffer_Linux failed!");
1212 }
1213 }
1214
1215#endif //(_DEBUG || _RELEASE_INTERNAL)
1216
1217 if (gpuNode != I915_EXEC_RENDER &&
1218 osInterface->osCpInterface->IsTearDownHappen())
1219 {
1220 // skip PAK command when CP tear down happen to avoid of GPU hang
1221 // conditonal batch buffer start PoC is in progress
1222 }
1223 else if (nullRendering == false)
1224 {
1225 if (osInterface->ctxBasedScheduling && m_i915Context[0] != nullptr)
1226 {
1227 if (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASK)
1228 {
1229 if (scalaEnabled && !osInterface->bGucSubmission)
1230 {
1231 uint32_t secondaryIndex = 0;
1232 it = m_secondaryCmdBufs.begin();
1233 while(it != m_secondaryCmdBufs.end())
1234 {
1235 if (it->second->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE)
1236 {
1237 if(execFlag == MOS_GPU_NODE_VE)
1238 {
1239 // decode excluded since init in other place
1240 it->second->iSubmissionType |= (secondaryIndex << SUBMISSION_TYPE_MULTI_PIPE_SLAVE_INDEX_SHIFT);
1241 secondaryIndex++;
1242 }
1243 }
1244
1245 ret = SubmitPipeCommands(it->second,
1246 it->second->OsResource.bo,
1247 osContext,
1248 skipSyncBoList,
1249 execFlag,
1250 DR4);
1251 it++;
1252 }
1253 }
1254 else if(scalaEnabled && osInterface->bGucSubmission)
1255 {
1256 ret = ParallelSubmitCommands(m_secondaryCmdBufs,
1257 osContext,
1258 execFlag,
1259 DR4);
1260 }
1261 else
1262 {
1263 ret = SubmitPipeCommands(cmdBuffer,
1264 cmd_bo,
1265 osContext,
1266 skipSyncBoList,
1267 execFlag,
1268 DR4);
1269 }
1270 }
1271 else
1272 {
1273 ret = mos_gem_bo_context_exec2(cmd_bo,
1274 m_commandBufferSize,
1275 m_i915Context[0],
1276 cliprects,
1277 num_cliprects,
1278 DR4,
1279 m_i915ExecFlag,
1280 nullptr);
1281 }
1282 }
1283 else
1284 {
1285 ret = mos_gem_bo_context_exec2(cmd_bo,
1286 m_commandBufferSize,
1287 osContext->intel_context,
1288 cliprects,
1289 num_cliprects,
1290 DR4,
1291 execFlag,
1292 nullptr);
1293 }
1294 if (ret != 0)
1295 {
1296 eStatus = MOS_STATUS_UNKNOWN;
1297 }
1298 }
1299
1300 if (eStatus != MOS_STATUS_SUCCESS)
1301 {
1302 MOS_OS_ASSERTMESSAGE("Command buffer submission failed!");
1303 }
1304
1305 MOS_DEVULT_FuncCall(pfnUltGetCmdBuf, cmdBuffer);
1306
1307#if MOS_COMMAND_BUFFER_DUMP_SUPPORTED
1308pthread_mutex_lock(&command_dump_mutex);
1309if (osInterface->bDumpCommandBuffer)
1310 {
1311 if (scalaEnabled)
1312 {
1313 it = m_secondaryCmdBufs.begin();
1314 while(it != m_secondaryCmdBufs.end())
1315 {
1316 mos_bo_map(it->second->OsResource.bo, 0);
1317 osInterface->pfnDumpCommandBuffer(osInterface, it->second);
1318 mos_bo_unmap(it->second->OsResource.bo);
1319 it++;
1320 }
1321 }
1322 else
1323 {
1324 mos_bo_map(cmd_bo, 0);
1325 osInterface->pfnDumpCommandBuffer(osInterface, cmdBuffer);
1326 mos_bo_unmap(cmd_bo);
1327 }
1328 }
1329 pthread_mutex_unlock(&command_dump_mutex);
1330#endif // MOS_COMMAND_BUFFER_DUMP_SUPPORTED
1331
1332#if (_DEBUG || _RELEASE_INTERNAL)
1333 if (bad_cmd_bo)
1334 {
1335 mos_bo_wait_rendering(bad_cmd_bo);
1336 mos_bo_unreference(bad_cmd_bo);
1337 }
1338 if (nop_cmd_bo)
1339 {
1340 mos_bo_unreference(nop_cmd_bo);
1341 }
1342#endif //(_DEBUG || _RELEASE_INTERNAL)
1343
1344 //clear command buffer relocations to fix memory leak issue
1345 mos_gem_bo_clear_relocs(cmd_bo, 0);
1346 it = m_secondaryCmdBufs.begin();
1347 while(it != m_secondaryCmdBufs.end())
1348 {
1349 mos_gem_bo_clear_relocs(it->second->OsResource.bo, 0);
1350 MOS_FreeMemory(it->second);
1351 it++;
1352 }
1353 m_secondaryCmdBufs.clear();
1354 skipSyncBoList.clear();
1355
1356 // Reset resource allocation
1357 m_numAllocations = 0;
1358 MOS_ZeroMemory(m_allocationList, sizeof(ALLOCATION_LIST) * m_maxNumAllocations);
1359 m_currentNumPatchLocations = 0;
1360 MOS_ZeroMemory(m_patchLocationList, sizeof(PATCHLOCATIONLIST) * m_maxNumAllocations);
1361 m_resCount = 0;
1362

1371 MOS_LINUX_BO *cmdBo,
1372 PMOS_CONTEXT osContext,
1373 const std::vector<MOS_LINUX_BO *> &skipSyncBoList,
1374 uint32_t execFlag,
1375 int32_t dr4)
1376{
1377 int32_t ret = 0;
1378 int fence = -1;
1379 unsigned int fenceFlag = 0;
1380
1381 MOS_LINUX_CONTEXT *queue = m_i915Context[0];
1382 bool isVeboxSubmission = false;
1383

1392 }

1400 {

1402 }

1404 {

1406 return -1;
1407 }
1408
1409 if (isVeboxSubmission)
1410 {
1411 queue = m_i915Context[cmdBuffer->iVeboxNodeIndex + 1];
1412 }
1413
1414 for(auto bo: skipSyncBoList)
1415 {
1416 mos_bo_set_exec_object_async(cmdBo, bo);
1417 }
1418 }
1419
1420 //Keep FE and BE0 running on same engine for VT decode
1421 if((cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_ALONE)
1422 || (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASTER))
1423 {
1424 if(cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASTER)
1425 {
1426 //Only master pipe needs fence out flag
1427 fenceFlag = I915_EXEC_FENCE_OUT;
1428 }
1429 queue = m_i915Context[1];
1430 }
1431
1432 ret = mos_gem_bo_context_exec2(cmdBo,
1433 cmdBo->size,
1434 queue,
1435 nullptr,
1436 0,
1437 dr4,
1438 execFlag | fenceFlag,
1439 &fence);
1440
1441 if(cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASTER)
1442 {
1443 osContext->submit_fence = fence;
1444 }
1445
1446 if(cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_FLAGS_LAST_PIPE)
1447 {
1448 close(fence);
1449 }

1493 execFlag | fenceFlag,

1501 {
1502 cmdBos[numBos++] = it->second->OsResource.bo;
1503
1504 if(it->second->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_FLAGS_LAST_PIPE)

1510 fenceFlag = I915_EXEC_FENCE_IN;

1514 numBos,
1515 queue,
1516 nullptr,
1517 0,
1518 dr4,
1519 execFlag | fenceFlag,
1520 &fence);

1522 for(int i = 0; i < numBos; i++)
1523 {
1524 cmdBos[i] = nullptr;
1525 }
1526 numBos = 0;
1527
1528 if(-1 != fence)
1529 {
1530 close(fence);
1531 }
1532 }
1533 }
1534
1535 it++;
1536 }
