32bool MediaLibvaApoDecision::InitDdiApoState(int32_t devicefd, MediaUserSettingSharedPtr userSettingPtr)
33{
34 DDI_FUNCTION_ENTER();
35
36 bool apoMosEnabled = SetupApoMosSwitch(devicefd, userSettingPtr);
37 bool apoDdiEnabled = SetupApoDdiSwitch(devicefd, userSettingPtr);
38 if (!apoMosEnabled || !apoDdiEnabled)
39 {
40 return false;
41 }
42
43 PRODUCT_FAMILY eProductFamily = IGFX_UNKNOWN;
44 HWInfo_GetGfxProductFamily(devicefd, eProductFamily);
45
46 if (eProductFamily < IGFX_METEORLAKE)
47 {
48 apoDdiEnabled = false;
49 }
50
51 // mediaSoloEnabled is not used due to mos context is not ready for solo
52 bool mediaSoloEnabled = SetupMediaSoloSwitch();
53 if (mediaSoloEnabled)
54 {
55 return false;
56 }
57 return apoDdiEnabled && apoMosEnabled;
58}
