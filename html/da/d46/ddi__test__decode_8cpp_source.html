56void MediaDecodeDdiTest::DecodeExecute(DecTestData *pDecData, Platform_t platform)
57{
58 VAConfigID config_id;
59 VAContextID context_id;
60 VASurfaceStatus surface_status;
61
62 // So far we still use DeviceConfigTable to find the platform, as the libdrm mock use this.
63 // If we want to use vector Platforms, we would use vector in libdrm too.
64 int ret = m_driverLoader.InitDriver(platform);
65 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
66 << ", Failed function = m_driverLoader.InitDriver" << endl;
67
68 // The attribute only use RCType and FEI function type in createconfig.
69 ret = m_driverLoader.m_ctx.vtable->vaCreateConfig(&m_driverLoader.m_ctx,
70 pDecData->GetFeatureID().profile, pDecData->GetFeatureID().entrypoint,
71 (VAConfigAttrib *)&(pDecData->GetConfAttrib()[0]), pDecData->GetConfAttrib().size(), &config_id);
72 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
73 << ", Failed function = m_driverLoader.m_ctx.vtable->vaCreateConfig" << endl;
74
75 vector<VASurfaceID> &resources = pDecData->GetResources();
76 ret = m_driverLoader.m_ctx.vtable->vaCreateSurfaces2(&m_driverLoader.m_ctx, VA_RT_FORMAT_YUV420,
77 pDecData->GetWidth(), pDecData->GetHeight(), &resources[0], resources.size(), nullptr, 0);
78 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
79 << ", Failed function = m_driverLoader.m_ctx.vtable->vaCreateSurfaces2" << endl;
80
81 ret = m_driverLoader.m_ctx.vtable->vaCreateContext(&m_driverLoader.m_ctx, config_id, pDecData->GetWidth(),
82 pDecData->GetHeight(),VA_PROGRESSIVE, &resources[0], resources.size(), &context_id);
83 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
84 << ", Failed function = m_driverLoader.m_ctx.vtable->vaCreateContext" << endl;
85
86 for (int i = 0; i < pDecData->m_num_frames; i++)
87 {
88 // As BeginPicture would reset some parameters, so it should be called before RenderPicture.
89 ret = m_driverLoader.m_ctx.vtable->vaBeginPicture(&m_driverLoader.m_ctx, context_id, resources[0]);
90 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
91 << ", Failed function = m_driverLoader.m_ctx.vtable->vaBeginPicture" << endl;
92
93 vector<vector<CompBufConif>> &compBufs = pDecData->GetCompBuffers();
94 for (int j = 0; j < compBufs[i].size(); j++)
95 {
96 ret = m_driverLoader.m_ctx.vtable->vaCreateBuffer(&m_driverLoader.m_ctx, context_id,
97 compBufs[i][j].bufType, compBufs[i][j].bufSize, 1, compBufs[i][j].pData, &compBufs[i][j].bufID);
98 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
99 << ", Failed function = m_driverLoader.m_ctx.vtable->vaCreateBuffer" << endl;
100 }
101
102 pDecData->UpdateCompBuffers(i);
103 for (int j = 0; j < compBufs[i].size(); j++)
104 {
105 // In RenderPicture, it suppose all needed buffer has been created already.
106 ret = m_driverLoader.m_ctx.vtable->vaRenderPicture(&m_driverLoader.m_ctx,
107 context_id, &compBufs[i][j].bufID, 1);
108 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
109 << ", Failed function = m_driverLoader.m_ctx.vtable->vaRenderPicture" << endl;
110 }
111
112 ret = m_driverLoader.m_ctx.vtable->vaEndPicture(&m_driverLoader.m_ctx, context_id);
113 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
114 << ", Failed function = m_driverLoader.m_ctx.vtable->vaEndPicture" << endl;
115
116 do
117 {
118 ret = m_driverLoader.m_ctx.vtable->vaQuerySurfaceStatus(
119 &m_driverLoader.m_ctx, resources[0], &surface_status);
120 } while (surface_status != VASurfaceReady);
121
122 for (int j = 0; j < compBufs[i].size(); j++)
123 {
124 ret = m_driverLoader.m_ctx.vtable->vaDestroyBuffer(&m_driverLoader.m_ctx, compBufs[i][j].bufID);
125 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
126 << ", Failed function = m_driverLoader.m_ctx.vtable->vaDestroyBuffer" << endl;
127 }
128 }
129
130 ret = m_driverLoader.m_ctx.vtable->vaDestroySurfaces(&m_driverLoader.m_ctx, &resources[0], resources.size());
131 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
132 << ", Failed function = m_driverLoader.m_ctx.vtable->vaDestroySurfaces" << endl;
133
134 ret = m_driverLoader.m_ctx.vtable->vaDestroyContext(&m_driverLoader.m_ctx, context_id);
135 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
136 << ", Failed function = m_driverLoader.m_ctx.vtable->vaDestroyContext" << endl;
137
138 ret = m_driverLoader.m_ctx.vtable->vaDestroyConfig(&m_driverLoader.m_ctx, config_id);
139 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
140 << ", Failed function = m_driverLoader.m_ctx.vtable->vaDestroyConfig" << endl;
141
142 ret = m_driverLoader.CloseDriver();
143 EXPECT_EQ(VA_STATUS_SUCCESS, ret) << "Platform = " << g_platformName[platform]
144 << ", Failed function = m_driverLoader.CloseDriver" << endl;
145}
