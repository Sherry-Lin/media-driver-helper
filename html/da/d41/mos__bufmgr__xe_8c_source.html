596static int
597__mos_get_default_alignment_xe(struct mos_bufmgr *bufmgr)
598{
599 MOS_DRM_CHK_NULL_RETURN_VALUE(bufmgr, -EINVAL)
600 struct mos_xe_bufmgr_gem *bufmgr_gem = (struct mos_xe_bufmgr_gem *)bufmgr;
601 struct mos_xe_device *dev = &bufmgr_gem->xe_device;
602 int fd = bufmgr_gem->fd;
603 MOS_DRM_CHK_XE_DEV(dev, mem_regions, __mos_query_mem_regions_xe, -ENODEV)
604 struct drm_xe_query_mem_regions *mem_regions = dev->mem_regions;
605 uint16_t mem_class;
606
607 for (int i = 0; i < mem_regions->num_mem_regions; i++)
608 {
609 if (DRM_XE_MEM_REGION_CLASS_SYSMEM == mem_regions->mem_regions[i].mem_class)
610 {
611 mem_class = MOS_XE_MEM_CLASS_SYSMEM;
612 }
613 else if (DRM_XE_MEM_REGION_CLASS_VRAM == mem_regions->mem_regions[i].mem_class)
614 {
615 mem_class = MOS_XE_MEM_CLASS_VRAM;
616 }
617 else
618 {
619 MOS_DRM_ASSERTMESSAGE("Unsupported mem class");
620 return -EINVAL;
621 }
622
623 if (bufmgr_gem->default_alignment[mem_class] < mem_regions->mem_regions[i].min_page_size)
624 {
625 bufmgr_gem->default_alignment[mem_class] = mem_regions->mem_regions[i].min_page_size;
626 }
627 }
628
629 return 0;
630}

635static int
636mos_query_uc_version_xe(struct mos_bufmgr *bufmgr, struct mos_drm_uc_version *version)
637{
638 int ret = 0;
639 struct mos_xe_bufmgr_gem *bufmgr_gem = (struct mos_xe_bufmgr_gem *)bufmgr;
640 int fd = bufmgr_gem->fd;
641 struct mos_xe_device *dev = &bufmgr_gem->xe_device;
642
643 if (bufmgr && version && version->uc_type < UC_TYPE_MAX)
644 {
648 if (dev->uc_versions[version->uc_type].uc_type != version->uc_type)
649 {
650 struct drm_xe_device_query query;
651 memclear(query);
652 query.size = sizeof(struct drm_xe_query_uc_fw_version);
653 query.query = DRM_XE_DEVICE_QUERY_UC_FW_VERSION;
654 memclear(dev->uc_versions[version->uc_type]);
655 dev->uc_versions[version->uc_type].uc_type = version->uc_type;
656 query.data = (uintptr_t)&dev->uc_versions[version->uc_type];
657
658 ret = drmIoctl(bufmgr_gem->fd, DRM_IOCTL_XE_DEVICE_QUERY,
659 &query);
660 if (ret)
661 {
662 memclear(dev->uc_versions[version->uc_type]);
663 dev->uc_versions[version->uc_type].uc_type = UC_TYPE_INVALID;
664 MOS_DRM_ASSERTMESSAGE("Failed to query UC version, uc type: %d, errno: %d", version->uc_type, ret);
665 return ret;
666 }
667 }
668
669 version->major_version = dev->uc_versions[version->uc_type].major_ver;
670 version->minor_version = dev->uc_versions[version->uc_type].minor_ver;
671 }
672
673 return ret;
674}

1406static struct mos_linux_bo *
1407mos_bo_alloc_tiled_xe(struct mos_bufmgr *bufmgr,
1408 struct mos_drm_bo_alloc_tiled *alloc_tiled)
1409{
1410 struct mos_xe_bufmgr_gem *bufmgr_gem = (struct mos_xe_bufmgr_gem *) bufmgr;
1411 unsigned long size, stride;
1412 uint32_t tiling;
1413
1414 uint32_t alignment = bufmgr_gem->default_alignment[MOS_XE_MEM_CLASS_SYSMEM];
1415
1416 if (bufmgr_gem->has_vram &&
1417 (MOS_MEMPOOL_VIDEOMEMORY == alloc_tiled->ext.mem_type || MOS_MEMPOOL_DEVICEMEMORY == alloc_tiled->ext.mem_type))
1418 {
1419 alignment = bufmgr_gem->default_alignment[MOS_XE_MEM_CLASS_VRAM];
1420 }
1421
1422 do {
1423 unsigned long aligned_y, height_alignment;
1424
1425 tiling = alloc_tiled->ext.tiling_mode;
1426
1427 /* If we're tiled, our allocations are in 8 or 32-row blocks,
1428 * so failure to align our height means that we won't allocate
1429 * enough pages.
1430 *
1431 * If we're untiled, we still have to align to 2 rows high
1432 * because the data port accesses 2x2 blocks even if the
1433 * bottom row isn't to be rendered, so failure to align means
1434 * we could walk off the end of the GTT and fault. This is
1435 * documented on 965, and may be the case on older chipsets
1436 * too so we try to be careful.
1437 */
1438 aligned_y = alloc_tiled->y;
1439 height_alignment = 2;
1440
1441 if (TILING_X == tiling)
1442 height_alignment = 8;
1443 else if (TILING_Y == tiling)
1444 height_alignment = 32;
1445 aligned_y = ALIGN(alloc_tiled->y, height_alignment);
1446
1447 stride = alloc_tiled->x * alloc_tiled->cpp;
1448 stride = __mos_bo_tile_pitch_xe(bufmgr_gem, stride, &alloc_tiled->ext.tiling_mode);
1449 size = stride * aligned_y;

1546 bufmgr_gem->m_lock.lock();
1547 ret = drmPrimeFDToHandle(bufmgr_gem->fd, prime_fd, &handle);
1548 if (ret)
1549 {
1550 MOS_DRM_ASSERTMESSAGE("create_from_prime: failed to obtain handle from fd: %s", strerror(errno));
1551 bufmgr_gem->m_lock.unlock();

1553 }
1554
1555 /*
1556 * See if the kernel has already returned this buffer to us. Just as
1557 * for named buffers, we must not create two bo's pointing at the same
1558 * kernel object
1559 */
1560 for (list = bufmgr_gem->named.next; list != &bufmgr_gem->named; list = list->next)
1561 {

1565 mos_bo_reference_xe(&bo_gem->bo);

1589 * later, we can lseek on the prime fd to get the size. Older
1590 * kernels will just fail, in which case we fall back to the
1591 * provided (estimated or guess size). */
1592 ret = lseek(prime_fd, 0, SEEK_END);

1594 bo_gem->bo.size = ret;
1595 else

1600 * Note, currectly there is no cpu_caching and pat_index for external-imported bo, hard code for it.

1602 */

1605
1606 bo_gem->gem_handle = handle;
1607 atomic_set(&bo_gem->ref_count, 1);
1608
1609 memcpy(bo_gem->name, "prime", sizeof("prime"));
1610 bo_gem->mem_region = MEMZONE_PRIME;
1611
1612 DRMLISTADDTAIL(&bo_gem->name_list, &bufmgr_gem->named);
1613 bufmgr_gem->m_lock.unlock();

1616
1617 ret = mos_vm_bind_sync_xe(bufmgr_gem->fd,
1618 bufmgr_gem->vm_id,
1619 bo_gem->gem_handle,
1620 0,

1622 bo_gem->bo.size,
1623 bo_gem->pat_index,
1624 DRM_XE_VM_BIND_OP_MAP);
1625 if (ret)
1626 {
1627 MOS_DRM_ASSERTMESSAGE("mos_vm_bind_sync_xe ret: %d", ret);
1628 mos_bo_free_xe(&bo_gem->bo);
1629 return nullptr;
1630 }
1631 else
1632 {
1633 bo_gem->bo.vm_id = bufmgr_gem->vm_id;
1634 }
1635
