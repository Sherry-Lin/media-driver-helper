205VAStatus DdiEncodeHevc::RenderPicture(VADriverContextP ctx, VAContextID context, VABufferID *buffers, int32_t numBuffers)
206{
207 VAStatus vaStatus = VA_STATUS_SUCCESS;
208
209 DDI_CODEC_FUNC_ENTER;
210
211 DDI_CODEC_CHK_NULL(ctx, "nullptr context", VA_STATUS_ERROR_INVALID_CONTEXT);
212
213 DDI_MEDIA_CONTEXT *mediaCtx = GetMediaContext(ctx);
214 DDI_CODEC_CHK_NULL(mediaCtx, "nullptr mediaCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
215
216 DDI_CODEC_CHK_NULL(m_encodeCtx, "nullptr m_encodeCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
217
218 for (int32_t i = 0; i < numBuffers; i++)
219 {
220 DDI_MEDIA_BUFFER *buf = MediaLibvaCommonNext::GetBufferFromVABufferID(mediaCtx, buffers[i]);
221 DDI_CODEC_CHK_NULL(buf, "Invalid buffer.", VA_STATUS_ERROR_INVALID_BUFFER);
222 if (buf->uiType == VAEncMacroblockDisableSkipMapBufferType)
223 {
224 MediaLibvaCommonNext::MediaBufferToMosResource(buf, &(m_encodeCtx->resPerMBSkipMapBuffer));
225 m_encodeCtx->bMbDisableSkipMapEnabled = true;
226 continue;
227 }
228 uint32_t dataSize = buf->iSize;
229 // can use internal function instead of MapBuffer here?
230 void *data = nullptr;
231 MediaLibvaInterfaceNext::MapBuffer(ctx, buffers[i], &data);
232
233 DDI_CODEC_CHK_NULL(data, "nullptr data.", VA_STATUS_ERROR_INVALID_BUFFER);
234
235 switch (buf->uiType)
236 {
237 case VAQMatrixBufferType:
238 DDI_CHK_STATUS(Qmatrix(data), VA_STATUS_ERROR_INVALID_BUFFER);
239 break;
240
241 case VAEncSequenceParameterBufferType:
242 DDI_CHK_STATUS(ParseSeqParams(data), VA_STATUS_ERROR_INVALID_BUFFER);
243 m_encodeCtx->bNewSeq = true;
244 break;
245
246 case VAEncPictureParameterBufferType:
247 DDI_CHK_STATUS(ParsePicParams(mediaCtx, data), VA_STATUS_ERROR_INVALID_BUFFER);
248 DDI_CHK_STATUS(
249 AddToStatusReportQueue((void *)m_encodeCtx->resBitstreamBuffer.bo),
250 VA_STATUS_ERROR_INVALID_BUFFER);
251 break;
252
253 case VAEncSliceParameterBufferType:
254 {
255 uint32_t numSlices = buf->uiNumElements;
256 DDI_CHK_STATUS(ParseSlcParams(mediaCtx, data, numSlices), VA_STATUS_ERROR_INVALID_BUFFER);
257 break;
258 }
259
260 case VAEncPackedHeaderParameterBufferType:
261 vaStatus = ParsePackedHeaderParams(data);
262 break;
263
264 case VAEncPackedHeaderDataBufferType:
265 vaStatus = ParsePackedHeaderData(data);
266 break;
267
268 case VAEncMiscParameterBufferType:
269 DDI_CHK_STATUS(ParseMiscParams(data), VA_STATUS_ERROR_INVALID_BUFFER);
270 break;
271
272 case VAEncQPBufferType:
273 MediaLibvaCommonNext::MediaBufferToMosResource(buf, &m_encodeCtx->resMBQpBuffer);
274 m_encodeCtx->bMBQpEnable = true;
275 break;
276
277 default:
278 DDI_CODEC_ASSERTMESSAGE("not supported buffer type.");
279 break;
280 }
281 MediaLibvaInterfaceNext::UnmapBuffer(ctx, buffers[i]);
282 }
283
284 return vaStatus;
285}

287VAStatus DdiEncodeHevc::EncodeInCodecHal(uint32_t numSlices)
288{
289 DDI_CODEC_CHK_NULL(m_encodeCtx, "nullptr m_encodeCtx", VA_STATUS_ERROR_INVALID_PARAMETER);
290 DDI_CODEC_CHK_NULL(m_encodeCtx->pCodecHal, "nullptr m_encodeCtx->pCodecHal", VA_STATUS_ERROR_INVALID_PARAMETER);
291
292 DDI_CODEC_RENDER_TARGET_TABLE *rtTbl = &(m_encodeCtx->RTtbl);
293
294 EncoderParams encodeParams;
295 MOS_ZeroMemory(&encodeParams, sizeof(encodeParams));
296
297 encodeParams.ExecCodecFunction = CODECHAL_FUNCTION_ENC_VDENC_PAK;
298
299 // Raw Surface
300 MOS_SURFACE rawSurface;
301 MOS_ZeroMemory(&rawSurface, sizeof(rawSurface));
302 rawSurface.dwOffset = 0;
303
304 MediaLibvaCommonNext::MediaSurfaceToMosResource(rtTbl->pCurrentRT, &(rawSurface.OsResource));
305
306 // Recon Surface
307 MOS_SURFACE reconSurface;
308 MOS_ZeroMemory(&reconSurface, sizeof(reconSurface));
309 reconSurface.dwOffset = 0;
310
311 MediaLibvaCommonNext::MediaSurfaceToMosResource(rtTbl->pCurrentReconTarget, &(reconSurface.OsResource));
312
313 //clear registered recon/ref surface flags
314 DDI_CODEC_CHK_RET(ClearRefList(&m_encodeCtx->RTtbl, true), "ClearRefList failed!");
315
316 // Bitstream surface
317 MOS_RESOURCE bitstreamSurface;
318 MOS_ZeroMemory(&bitstreamSurface, sizeof(bitstreamSurface));
319 bitstreamSurface = m_encodeCtx->resBitstreamBuffer; // in render picture
320 bitstreamSurface.Format = Format_Buffer;
321
322 encodeParams.psRawSurface = &rawSurface;
323 encodeParams.psReconSurface = &reconSurface;
324 encodeParams.presBitstreamBuffer = &bitstreamSurface;
325
326 MOS_SURFACE mbQpSurface;
327 if (m_encodeCtx->bMBQpEnable)
328 {
329 // MBQp surface
330 MOS_ZeroMemory(&mbQpSurface, sizeof(mbQpSurface));
331 mbQpSurface.Format = Format_Buffer_2D;
332 mbQpSurface.dwOffset = 0;
333 mbQpSurface.OsResource = m_encodeCtx->resMBQpBuffer;
334
335 encodeParams.psMbQpDataSurface = &mbQpSurface;
336 encodeParams.bMbQpDataEnabled = true;
337 }
338
339 PCODEC_HEVC_ENCODE_SEQUENCE_PARAMS hevcSeqParams = (PCODEC_HEVC_ENCODE_SEQUENCE_PARAMS)((uint8_t *)m_encodeCtx->pSeqParams);
340 if (m_encodeCtx->bNewSeq)
341 {
342 hevcSeqParams->TargetUsage = m_encodeCtx->targetUsage;
343 }
344 encodeParams.pSeqParams = m_encodeCtx->pSeqParams;
345 encodeParams.pVuiParams = m_encodeCtx->pVuiParams;
346 encodeParams.pPicParams = m_encodeCtx->pPicParams;
347 encodeParams.pSliceParams = m_encodeCtx->pSliceParams;
348 encodeParams.pIQMatrixBuffer = m_encodeCtx->pQmatrixParams;
349
350 // Sequence data
351 encodeParams.bNewSeq = m_encodeCtx->bNewSeq;
352
353 // VUI
354 encodeParams.bNewVuiData = m_encodeCtx->bNewVuiData;
355
356 // Slice level data
357 encodeParams.dwNumSlices = numSlices;
358
359 // IQmatrix params
360 encodeParams.bNewQmatrixData = m_encodeCtx->bNewQmatrixData;
361 encodeParams.bPicQuant = m_encodeCtx->bPicQuant;
362 encodeParams.ppNALUnitParams = m_encodeCtx->ppNALUnitParams;
363 encodeParams.uiNumNalUnits = m_encodeCtx->indexNALUnit;
364 encodeParams.pSeiData = m_encodeCtx->pSEIFromApp;
365 encodeParams.pSeiParamBuffer = m_encodeCtx->pSEIFromApp->pSEIBuffer;
366 encodeParams.dwSEIDataOffset = 0;
367
368 // whether driver need to pack slice header
369 if (m_encodeCtx->bHavePackedSliceHdr)
370 {
371 encodeParams.bAcceleratorHeaderPackingCaps = false;
372 }
373 else
374 {
375 encodeParams.bAcceleratorHeaderPackingCaps = true;
376 }
377
378 encodeParams.pBSBuffer = m_encodeCtx->pbsBuffer;
379 encodeParams.pSlcHeaderData = (void *)m_encodeCtx->pSliceHeaderData;
380
381 MOS_STATUS status = m_encodeCtx->pCodecHal->Execute(&encodeParams);
382 if (MOS_STATUS_SUCCESS != status)
383 {
384 DDI_CODEC_ASSERTMESSAGE("DDI:Failed in Codechal!");
385 return VA_STATUS_ERROR_ENCODING_ERROR;
386 }
387
388 return VA_STATUS_SUCCESS;
389}

443
444 hevcSeqParams->RateControlMethod = VARC2HalRC(m_encodeCtx->uiRCMethod);
445
446 hevcSeqParams->TargetBitRate = MOS_ROUNDUP_DIVIDE(seqParams->bits_per_second, CODECHAL_ENCODE_BRC_KBPS);
447 hevcSeqParams->MaxBitRate = MOS_ROUNDUP_DIVIDE(seqParams->bits_per_second, CODECHAL_ENCODE_BRC_KBPS);
448 hevcSeqParams->MinBitRate = MOS_ROUNDUP_DIVIDE(seqParams->bits_per_second, CODECHAL_ENCODE_BRC_KBPS);
449 // if didn't setting FrameRate, set to 30 by default, can be overwritten by misc paramter
450 if (!hevcSeqParams->FrameRate.Numerator) {
451 hevcSeqParams->FrameRate.Numerator = 3000;
452 hevcSeqParams->FrameRate.Denominator = 100;
453 }
454
455 // set default same as application, can be overwritten by HRD params
456 hevcSeqParams->InitVBVBufferFullnessInBit = (uint32_t)seqParams->bits_per_second;
457 hevcSeqParams->VBVBufferSizeInBit = (uint32_t)seqParams->bits_per_second << 1;
458
459 hevcSeqParams->scaling_list_enable_flag = seqParams->seq_fields.bits.scaling_list_enabled_flag;
460 hevcSeqParams->sps_temporal_mvp_enable_flag = seqParams->seq_fields.bits.sps_temporal_mvp_enabled_flag;

462 hevcSeqParams->amp_enabled_flag = seqParams->seq_fields.bits.amp_enabled_flag;
463 hevcSeqParams->SAO_enabled_flag = seqParams->seq_fields.bits.sample_adaptive_offset_enabled_flag;
464 hevcSeqParams->pcm_enabled_flag = seqParams->seq_fields.bits.pcm_enabled_flag;
465 hevcSeqParams->pcm_loop_filter_disable_flag = seqParams->seq_fields.bits.pcm_loop_filter_disabled_flag;
466 hevcSeqParams->LowDelayMode = seqParams->seq_fields.bits.low_delay_seq;
467 hevcSeqParams->HierarchicalFlag = seqParams->seq_fields.bits.hierachical_flag;

470 seqParams->log2_min_luma_coding_block_size_minus3;
471 hevcSeqParams->log2_min_coding_block_size_minus3 = seqParams->log2_min_luma_coding_block_size_minus3;
472 hevcSeqParams->log2_max_transform_block_size_minus2 = seqParams->log2_diff_max_min_transform_block_size +
473 seqParams->log2_min_transform_block_size_minus2;
474 hevcSeqParams->log2_min_transform_block_size_minus2 = seqParams->log2_min_transform_block_size_minus2;

476 hevcSeqParams->max_transform_hierarchy_depth_inter = seqParams->max_transform_hierarchy_depth_inter;

954VAStatus DdiEncodeHevc::ParsePackedHeaderData(void *ptr)
955{
956 DDI_CODEC_CHK_NULL(m_encodeCtx, "nullptr m_encodeCtx.", VA_STATUS_ERROR_INVALID_PARAMETER);
957 DDI_CODEC_CHK_NULL(ptr, "nullptr ptr.", VA_STATUS_ERROR_INVALID_PARAMETER);
958
959 BSBuffer *bsBuffer = m_encodeCtx->pbsBuffer;
960 DDI_CODEC_CHK_NULL(bsBuffer, "nullptr bsBuffer.", VA_STATUS_ERROR_INVALID_PARAMETER);
961
962 if (m_encodeCtx->indexNALUnit == 0 && m_encodeCtx->uiSliceHeaderCnt == 0)
963 {
964 bsBuffer->pCurrent = bsBuffer->pBase;
965 bsBuffer->SliceOffset = 0;
966 bsBuffer->BitOffset = 0;
967 bsBuffer->BitSize = 0;
968 }
969
970 uint32_t hdrDataSize = 0;
971 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
972 if (true == m_encodeCtx->bLastPackedHdrIsSlice)
973 {
974 hdrDataSize = (m_encodeCtx->pSliceHeaderData[m_encodeCtx->uiSliceHeaderCnt].BitSize + 7) / 8;
975
976 eStatus = MOS_SecureMemcpy(bsBuffer->pCurrent, bsBuffer->BufferSize - bsBuffer->SliceOffset, (uint8_t *)ptr, hdrDataSize);
977 if (MOS_STATUS_SUCCESS != eStatus)
978 {
979 DDI_CODEC_ASSERTMESSAGE("DDI:packed slice header size is too large to be supported!");
980 return VA_STATUS_ERROR_INVALID_PARAMETER;
981 }
982
983 m_encodeCtx->pSliceHeaderData[m_encodeCtx->uiSliceHeaderCnt].SliceOffset = bsBuffer->pCurrent - bsBuffer->pBase;
984
985 // correct SkipEmulationByteCount
986 // according to LibVA principle, one packed header buffer should only contain one NALU,
987 // so when has_emulation_bytes is 0, SkipEmulationByteCount only needs to skip the NALU start codes
988 if (m_encodeCtx->pSliceHeaderData[m_encodeCtx->uiSliceHeaderCnt].SkipEmulationByteCount != hdrDataSize)
989 {
990 uint32_t startCodesOffset = 0;
991 uint32_t startCodesLength = 0;
992 VAStatus vaSts = VA_STATUS_SUCCESS;
993 vaSts = FindNalUnitStartCodes((uint8_t *)ptr, hdrDataSize, &startCodesOffset, &startCodesLength);
994 if (VA_STATUS_SUCCESS != vaSts)
995 {
996 DDI_CODEC_ASSERTMESSAGE("DDI: packed slice header doesn't include NAL unit start codes!");
997 return vaSts;
998 }
999 m_encodeCtx->pSliceHeaderData[m_encodeCtx->uiSliceHeaderCnt].SkipEmulationByteCount = MOS_MIN(15, (startCodesOffset + startCodesLength));
1000 }
1001
1002 m_encodeCtx->uiSliceHeaderCnt++;
1003 m_encodeCtx->bLastPackedHdrIsSlice = false;
1004 }
1005 else
1006 {
1007 // copy sps and pps header data
1008 hdrDataSize = m_encodeCtx->ppNALUnitParams[m_encodeCtx->indexNALUnit]->uiSize;
1009
1010 eStatus = MOS_SecureMemcpy(bsBuffer->pCurrent, bsBuffer->BufferSize - bsBuffer->SliceOffset, (uint8_t *)ptr, hdrDataSize);
1011 if (MOS_STATUS_SUCCESS != eStatus)
1012 {
1013 DDI_CODEC_ASSERTMESSAGE("DDI:packed header size is too large to be supported!");
1014 return VA_STATUS_ERROR_INVALID_PARAMETER;
1015 }
1016
1017 // correct uiSkipEmulationCheckCount
1018 // according to LibVA principle, one packed header buffer should only contain one NALU,
1019 // so when has_emulation_bytes is 0, uiSkipEmulationCheckCount only needs to skip the NALU start codes
1020 if (m_encodeCtx->ppNALUnitParams[m_encodeCtx->indexNALUnit]->uiSkipEmulationCheckCount != hdrDataSize)
1021 {
1022 uint32_t startCodesOffset = 0;
1023 uint32_t startCodesLength = 0;
1024 VAStatus vaSts = VA_STATUS_SUCCESS;
1025 vaSts = FindNalUnitStartCodes((uint8_t *)ptr, hdrDataSize, &startCodesOffset, &startCodesLength);
1026 if (VA_STATUS_SUCCESS != vaSts)
1027 {
1028 DDI_CODEC_ASSERTMESSAGE("DDI: packed header doesn't include NAL unit start codes!");
1029 return vaSts;
1030 }
1031 m_encodeCtx->ppNALUnitParams[m_encodeCtx->indexNALUnit]->uiSkipEmulationCheckCount = MOS_MIN(15, (startCodesOffset + startCodesLength));
1032 }
1033
1034 m_encodeCtx->ppNALUnitParams[m_encodeCtx->indexNALUnit]->uiOffset = bsBuffer->pCurrent - bsBuffer->pBase;
1035 m_encodeCtx->indexNALUnit++;
1036 }
1037
1038 bsBuffer->pCurrent += hdrDataSize;
1039 bsBuffer->SliceOffset += hdrDataSize;
1040 bsBuffer->BitSize += hdrDataSize * 8;
1041
1042 return VA_STATUS_SUCCESS;
1043}
