375/*
376 * Round a given pitch up to the minimum required for X tiling on a
377 * given chip. We use 512 as the minimum to allow for a later tiling
378 * change.
379 */
380static unsigned long
381mos_gem_bo_tile_pitch(struct mos_bufmgr_gem *bufmgr_gem,
382 unsigned long pitch, uint32_t *tiling_mode)
383{
384 unsigned long tile_width;
385 unsigned long i;
386
387 /* If untiled, then just align it so that we can do rendering
388 * to it with the 3D engine.
389 */
390 if (*tiling_mode == I915_TILING_NONE)
391 return ALIGN(pitch, 64);
392
393 if (*tiling_mode == I915_TILING_X
394 || (IS_915(bufmgr_gem->pci_device)
395 && *tiling_mode == I915_TILING_Y))
396 tile_width = 512;
397 else
398 tile_width = 128;
399
400 /* 965 is flexible */
401 return ROUND_UP_TO(pitch, tile_width);
402}

491static void
492mos_add_validate_buffer(struct mos_linux_bo *bo)
493{
494 struct mos_bufmgr_gem *bufmgr_gem = (struct mos_bufmgr_gem *) bo->bufmgr;
495 struct mos_bo_gem *bo_gem = (struct mos_bo_gem *) bo;
496 int index;
497 struct drm_i915_gem_exec_object *exec_objects;
498 struct mos_linux_bo **exec_bos;
499
500 if (bo_gem->validate_index != -1)
501 return;
502
503 /* Extend the array of validation entries as necessary. */
504 if (bufmgr_gem->exec_count == bufmgr_gem->exec_size) {
505 int new_size = bufmgr_gem->exec_size * 2;
506
507 if (new_size == 0)
508 new_size = 5;
509
510 exec_objects = (struct drm_i915_gem_exec_object *)realloc(bufmgr_gem->exec_objects,
511 sizeof(*bufmgr_gem->exec_objects) * new_size);
512 if (!exec_objects)
513 return;
514
515 bufmgr_gem->exec_objects = exec_objects;
516
517 exec_bos = (struct mos_linux_bo **)realloc(bufmgr_gem->exec_bos,
518 sizeof(*bufmgr_gem->exec_bos) * new_size);
519 if (!exec_bos)
520 return;
521
522 bufmgr_gem->exec_bos = exec_bos;
523 bufmgr_gem->exec_size = new_size;
524 }
525
526 index = bufmgr_gem->exec_count;
527 bo_gem->validate_index = index;
528 /* Fill in array entry */
529 bufmgr_gem->exec_objects[index].handle = bo_gem->gem_handle;
530 bufmgr_gem->exec_objects[index].relocation_count = bo_gem->reloc_count;
531 bufmgr_gem->exec_objects[index].relocs_ptr = (uintptr_t) bo_gem->relocs;
532 bufmgr_gem->exec_objects[index].alignment = bo->align;
533 bufmgr_gem->exec_objects[index].offset = 0;
534 bufmgr_gem->exec_bos[index] = bo;
535 bufmgr_gem->exec_count++;
536}

713static void
714mos_bo_gem_set_in_aperture_size(struct mos_bufmgr_gem *bufmgr_gem,
715 struct mos_bo_gem *bo_gem,
716 unsigned int alignment)
717{
718 unsigned int size;
719
720 assert(!bo_gem->used_as_reloc_target);
721
722 /* The older chipsets are far-less flexible in terms of tiling,
723 * and require tiled buffer to be size aligned in the aperture.
724 * This means that in the worst possible case we will need a hole
725 * twice as large as the object in order for it to fit into the
726 * aperture. Optimal packing is for wimps.
727 */
728 size = bo_gem->bo.size;
729
730 bo_gem->reloc_tree_size = size + alignment;
731}

874drm_export struct mos_linux_bo *
875mos_gem_bo_alloc_internal(struct mos_bufmgr *bufmgr,
876 const char *name,
877 unsigned long size,
878 unsigned long flags,
879 uint32_t tiling_mode,
880 unsigned long stride,
881 unsigned int alignment,
882 int mem_type,
883 unsigned int pat_index,
884 bool cpu_cacheable)
885{
886 MOS_UNUSED(pat_index);
887 MOS_UNUSED(cpu_cacheable);
888 struct mos_bufmgr_gem *bufmgr_gem = (struct mos_bufmgr_gem *) bufmgr;
889 struct mos_bo_gem *bo_gem;
890 unsigned int page_size = getpagesize();
891 int ret;
892 struct mos_gem_bo_bucket *bucket;
893 bool alloc_from_cache;
894 unsigned long bo_size;
895 bool for_render = false;
896
897 if (flags & BO_ALLOC_FOR_RENDER)
898 for_render = true;
899
900 /* Round the allocated size up to a power of two number of pages. */
901 bucket = mos_gem_bo_bucket_for_size(bufmgr_gem, size);
902
903 /* If we don't have caching at this size, don't actually round the
904 * allocation up.
905 */
906 if (bucket == nullptr) {
907 bo_size = size;
908 if (bo_size < page_size)
909 bo_size = page_size;
910 } else {
911 bo_size = bucket->size;
912 }
913 if(GetDrmMode())//libdrm_mock
914 {
915 pthread_mutex_lock(&bufmgr_gem->lock);
916
917 bo_gem = (struct mos_bo_gem *)calloc(1, sizeof(*bo_gem));
918 if (!bo_gem)
919 return nullptr;
920
921 bo_gem->bo.size = bo_size;
922 bo_gem->bo.handle = -1;
923 bo_gem->bo.bufmgr = bufmgr;
924 bo_gem->bo.align = alignment;
925#ifdef __cplusplus
926 bo_gem->bo.virt = malloc(bo_size);
927 bo_gem->mem_virtual = bo_gem->bo.virt;
928#else
929 bo_gem->bo.virtual = malloc(bo_size);
930 bo_gem->mem_virtual = bo_gem->bo.virtual;
931#endif
932
933 atomic_set(&bo_gem->refcount, 1);
934 pthread_mutex_unlock(&bufmgr_gem->lock);
935
936 return &bo_gem->bo;
937 }
938
939 pthread_mutex_lock(&bufmgr_gem->lock);
940 /* Get a buffer out of the cache if available */
941retry:
942 alloc_from_cache = false;
943 if (bucket != nullptr && !DRMLISTEMPTY(&bucket->head)) {
944 if (for_render) {
945 /* Allocate new render-target BOs from the tail (MRU)
946 * of the list, as it will likely be hot in the GPU
947 * cache and in the aperture for us.
948 */
949 bo_gem = DRMLISTENTRY(struct mos_bo_gem,
950 bucket->head.prev, head);
951 DRMLISTDEL(&bo_gem->head);
952 alloc_from_cache = true;
953 bo_gem->bo.align = alignment;
954 } else {
955 assert(alignment == 0);
956 /* For non-render-target BOs (where we're probably
957 * going to map it first thing in order to fill it
958 * with data), check if the last BO in the cache is
959 * unbusy, and only reuse in that case. Otherwise,
960 * allocating a new buffer is probably faster than
961 * waiting for the GPU to finish.
962 */
963 bo_gem = DRMLISTENTRY(struct mos_bo_gem,
964 bucket->head.next, head);
965 if (!mos_gem_bo_busy(&bo_gem->bo)) {
966 alloc_from_cache = true;
967 DRMLISTDEL(&bo_gem->head);
968 }
969 }
970
971 if (alloc_from_cache) {
972 if (!mos_gem_bo_madvise_internal
973 (bufmgr_gem, bo_gem, I915_MADV_WILLNEED)) {
974 mos_gem_bo_free(&bo_gem->bo);
975 mos_gem_bo_cache_purge_bucket(bufmgr_gem,
976 bucket);
977 goto retry;
978 }
979
980 if (mos_gem_bo_set_tiling_internal(&bo_gem->bo,
981 tiling_mode,
982 stride)) {
983 mos_gem_bo_free(&bo_gem->bo);
984 goto retry;
985 }
986 }
987 }
988 pthread_mutex_unlock(&bufmgr_gem->lock);
989
990 if (!alloc_from_cache) {
991 struct drm_i915_gem_create create;
992
993 bo_gem = (struct mos_bo_gem *)calloc(1, sizeof(*bo_gem));
994 if (!bo_gem)
995 return nullptr;
996
997 bo_gem->bo.size = bo_size;
998
999 memclear(create);
1000 create.size = bo_size;
1001
1002 ret = drmIoctl(bufmgr_gem->fd,
1003 DRM_IOCTL_I915_GEM_CREATE,
1004 &create);
1005 bo_gem->gem_handle = create.handle;
1006 bo_gem->bo.handle = bo_gem->gem_handle;
1007 if (ret != 0) {
1008 free(bo_gem);
1009 return nullptr;
1010 }
1011 bo_gem->bo.bufmgr = bufmgr;
1012 bo_gem->bo.align = alignment;
1013
1014 bo_gem->tiling_mode = I915_TILING_NONE;
1015 bo_gem->swizzle_mode = I915_BIT_6_SWIZZLE_NONE;
1016 bo_gem->stride = 0;
1017
1018 /* drm_intel_gem_bo_free calls DRMLISTDEL() for an uninitialized
1019 list (vma_list), so better set the list head here */
1020 DRMINITLISTHEAD(&bo_gem->name_list);
1021 if (mos_gem_bo_set_tiling_internal(&bo_gem->bo,
1022 tiling_mode,
1023 stride)) {
1024 mos_gem_bo_free(&bo_gem->bo);
1025 return nullptr;
1026 }
1027 }
1028
1029 bo_gem->name = name;
1030 atomic_set(&bo_gem->refcount, 1);
1031 bo_gem->validate_index = -1;
1032 bo_gem->reloc_tree_fences = 0;
1033 bo_gem->used_as_reloc_target = false;
1034 bo_gem->has_error = false;
1035 bo_gem->reusable = true;
1036 bo_gem->use_48b_address_range = bufmgr_gem->bufmgr.bo_use_48b_address_range ? true : false;
1037
1038 mos_bo_gem_set_in_aperture_size(bufmgr_gem, bo_gem, alignment);
1039
1040 if (bufmgr_gem->use_softpin)
1041 {
1042 mos_bo_set_softpin(&bo_gem->bo);
1043 }
1044
1045 MOS_DBG("bo_create: buf %d (%s) %ldb\n",
1046 bo_gem->gem_handle, bo_gem->name, size);
1047
1048 return &bo_gem->bo;
1049}

1082static struct mos_linux_bo *
1083mos_gem_bo_alloc_tiled(struct mos_bufmgr *bufmgr, const char *name,
1084 int x, int y, int cpp, uint32_t *tiling_mode,
1085 unsigned long *pitch, unsigned long flags,
1086 int mem_type, unsigned int pat_index, bool cpu_cacheable)
1087{
1088 struct mos_bufmgr_gem *bufmgr_gem = (struct mos_bufmgr_gem *)bufmgr;
1089 unsigned long size, stride;
1090 uint32_t tiling;
1091
1092 do {
1093 unsigned long aligned_y, height_alignment;
1094
1095 tiling = *tiling_mode;
1096
1097 /* If we're tiled, our allocations are in 8 or 32-row blocks,
1098 * so failure to align our height means that we won't allocate
1099 * enough pages.
1100 *
1101 * If we're untiled, we still have to align to 2 rows high
1102 * because the data port accesses 2x2 blocks even if the
1103 * bottom row isn't to be rendered, so failure to align means
1104 * we could walk off the end of the GTT and fault. This is
1105 * documented on 965, and may be the case on older chipsets
1106 * too so we try to be careful.
1107 */
1108 aligned_y = y;
1109 height_alignment = 2;
1110
1111 if (tiling == I915_TILING_X
1112 || (IS_915(bufmgr_gem->pci_device)
1113 && tiling == I915_TILING_Y))
1114 height_alignment = 8;
1115 else if (tiling == I915_TILING_Y)
1116 height_alignment = 32;
1117 aligned_y = ALIGN(y, height_alignment);
1118
1119 stride = x * cpp;
1120 stride = mos_gem_bo_tile_pitch(bufmgr_gem, stride, tiling_mode);
1121 size = stride * aligned_y;
1122 size = mos_gem_bo_tile_size(bufmgr_gem, size, tiling_mode);
1123 } while (*tiling_mode != tiling);
1124 *pitch = stride;
1125
1126 if (tiling == I915_TILING_NONE)
1127 stride = 0;
1128 return mos_gem_bo_alloc_internal(bufmgr, name, size, flags,
1129 tiling, stride, 0, mem_type, pat_index, cpu_cacheable);
1130}

1215static bool
1216has_userptr(struct mos_bufmgr_gem *bufmgr_gem)
1217{
1218 int ret;
1219 void *ptr;
1220 long pgsz;
1221 struct drm_i915_gem_userptr userptr;
1222
1223 pgsz = sysconf(_SC_PAGESIZE);
1224 assert(pgsz > 0);
1225
1226 ret = posix_memalign(&ptr, pgsz, pgsz);
1227 if (ret) {
1228 MOS_DBG("Failed to get a page (%ld) for userptr detection!\n",
1229 pgsz);
1230 return false;
1231 }
1232
1233 memclear(userptr);
1234 userptr.user_ptr = (__u64)(unsigned long)ptr;
1235 userptr.user_size = pgsz;
1236
1237retry:
1238 ret = drmIoctl(bufmgr_gem->fd, DRM_IOCTL_I915_GEM_USERPTR, &userptr);
1239 if (ret) {
1240 if (errno == ENODEV && userptr.flags == 0) {
1241 userptr.flags = I915_USERPTR_UNSYNCHRONIZED;
1242 goto retry;
1243 }
1244 free(ptr);
1245 return false;
1246 }
1247
1248 /* We don't release the userptr bo here as we want to keep the
1249 * kernel mm tracking alive for our lifetime. The first time we
1250 * create a userptr object the kernel has to install a mmu_notifer
1251 * which is a heavyweight operation (e.g. it requires taking all
1252 * mm_locks and stop_machine()).
1253 */
1254
1255 bufmgr_gem->userptr_active.ptr = ptr;
1256 bufmgr_gem->userptr_active.handle = userptr.handle;
1257
1258 return true;
1259}

1285static struct mos_linux_bo *
1286mos_bufmgr_bo_gem_create_from_name(struct mos_bufmgr *bufmgr,
1287 const char *name,
1288 unsigned int handle)
1289{
1290 struct mos_bufmgr_gem *bufmgr_gem = (struct mos_bufmgr_gem *) bufmgr;
1291 struct mos_bo_gem *bo_gem;
1292 int ret;
1293 struct drm_gem_open open_arg;
1294 struct drm_i915_gem_get_tiling get_tiling;
1295 drmMMListHead *list;
1296
1297 /* At the moment most applications only have a few named bo.
1298 * For instance, in a DRI client only the render buffers passed
1299 * between X and the client are named. And since X returns the
1300 * alternating names for the front/back buffer a linear search
1301 * provides a sufficiently fast match.
1302 */
1303 pthread_mutex_lock(&bufmgr_gem->lock);
1304 for (list = bufmgr_gem->named.next;
1305 list != &bufmgr_gem->named;
1306 list = list->next) {
1307 bo_gem = DRMLISTENTRY(struct mos_bo_gem, list, name_list);
1308 if (bo_gem->global_name == handle) {
1309 mos_gem_bo_reference(&bo_gem->bo);
1310 pthread_mutex_unlock(&bufmgr_gem->lock);
1311 return &bo_gem->bo;
1312 }
1313 }
1314
1315 memclear(open_arg);
1316 open_arg.name = handle;
1317 ret = drmIoctl(bufmgr_gem->fd,
1318 DRM_IOCTL_GEM_OPEN,
1319 &open_arg);
1320 if (ret != 0) {
1321 MOS_DBG("Couldn't reference %s handle 0x%08x: %s\n",
1322 name, handle, strerror(errno));
1323 pthread_mutex_unlock(&bufmgr_gem->lock);
1324 return nullptr;
1325 }
1326 /* Now see if someone has used a prime handle to get this
1327 * object from the kernel before by looking through the list
1328 * again for a matching gem_handle
1329 */
1330 for (list = bufmgr_gem->named.next;
1331 list != &bufmgr_gem->named;
1332 list = list->next) {
1333 bo_gem = DRMLISTENTRY(struct mos_bo_gem, list, name_list);
1334 if (bo_gem->gem_handle == open_arg.handle) {
1335 mos_gem_bo_reference(&bo_gem->bo);
1336 pthread_mutex_unlock(&bufmgr_gem->lock);
1337 return &bo_gem->bo;
1338 }
1339 }
1340
1341 bo_gem = (struct mos_bo_gem *)calloc(1, sizeof(*bo_gem));
1342 if (!bo_gem) {
1343 pthread_mutex_unlock(&bufmgr_gem->lock);
1344 return nullptr;
1345 }
1346
1347 bo_gem->bo.size = open_arg.size;
1348 bo_gem->bo.offset = 0;
1349 bo_gem->bo.offset64 = 0;
1350#if defined(__cplusplus)
1351 bo_gem->bo.virt = nullptr;
1352#else
1353 bo_gem->bo.virtual = nullptr;
1354#endif
1355 bo_gem->bo.bufmgr = bufmgr;
1356 bo_gem->name = name;
1357 atomic_set(&bo_gem->refcount, 1);
1358 bo_gem->validate_index = -1;
1359 bo_gem->gem_handle = open_arg.handle;
1360 bo_gem->bo.handle = open_arg.handle;
1361 bo_gem->global_name = handle;
1362 bo_gem->reusable = false;
1363 bo_gem->use_48b_address_range = bufmgr_gem->bufmgr.bo_use_48b_address_range ? true : false;
1364
1365 memclear(get_tiling);
1366 get_tiling.handle = bo_gem->gem_handle;
1367 ret = drmIoctl(bufmgr_gem->fd,
1368 DRM_IOCTL_I915_GEM_GET_TILING,
1369 &get_tiling);
1370 if (ret != 0) {
1371 mos_gem_bo_unreference(&bo_gem->bo);
1372 pthread_mutex_unlock(&bufmgr_gem->lock);
1373 return nullptr;
1374 }
1375 bo_gem->tiling_mode = get_tiling.tiling_mode;
1376 bo_gem->swizzle_mode = get_tiling.swizzle_mode;
1377 /* XXX stride is unknown */
1378 mos_bo_gem_set_in_aperture_size(bufmgr_gem, bo_gem, 0);
1379
1380 if (bufmgr_gem->use_softpin)
1381 {
1382 mos_bo_set_softpin(&bo_gem->bo);
1383 }
1384
1385 DRMLISTADDTAIL(&bo_gem->name_list, &bufmgr_gem->named);
1386 pthread_mutex_unlock(&bufmgr_gem->lock);
1387 MOS_DBG("bo_create_from_handle: %d (%s)\n", handle, bo_gem->name);
1388
1389 return &bo_gem->bo;
1390}

1482drm_export void
1483mos_gem_bo_unreference_final(struct mos_linux_bo *bo, time_t time)
1484{
1485 struct mos_bufmgr_gem *bufmgr_gem = (struct mos_bufmgr_gem *) bo->bufmgr;
1486 struct mos_bo_gem *bo_gem = (struct mos_bo_gem *) bo;
1487 struct mos_gem_bo_bucket *bucket;
1488 int i;
1489
1490 /* Unreference all the target buffers */
1491 for (i = 0; i < bo_gem->reloc_count; i++) {
1492 if (bo_gem->reloc_target_info[i].bo != bo) {
1493 mos_gem_bo_unreference_locked_timed(bo_gem->
1494 reloc_target_info[i].bo,
1495 time);
1496 }
1497 }
1498 for (i = 0; i < bo_gem->softpin_target_count; i++)
1499 mos_gem_bo_unreference_locked_timed(bo_gem->softpin_target[i].bo,
1500 time);

1503 bo_gem->softpin_target_count = 0;
1504 bo_gem->exec_async = false;

1507 bo_gem->gem_handle, bo_gem->name);
1508 bo_gem->pad_to_size = 0;
1509
1510 /* release memory associated with this object */
1511 if (bo_gem->reloc_target_info) {
1512 free(bo_gem->reloc_target_info);
1513 bo_gem->reloc_target_info = nullptr;
1514 }
1515 if (bo_gem->relocs) {
1516 free(bo_gem->relocs);
1517 bo_gem->relocs = nullptr;
1518 }
1519 if (bo_gem->softpin_target) {
1520 free(bo_gem->softpin_target);
1521 bo_gem->softpin_target = nullptr;
1522 bo_gem->softpin_target_size = 0;
1523 }
1524 if(GetDrmMode()){
1525 mos_gem_bo_free(bo);
1526 return;
1527 }
1528
1529 /* Clear any left-over mappings */
1530 if (bo_gem->map_count) {
1531 MOS_DBG("bo freed with non-zero map-count %d\n", bo_gem->map_count);
1532 bo_gem->map_count = 0;
1533 mos_gem_bo_mark_mmaps_incoherent(bo);
1534 }
1535
1536 DRMLISTDEL(&bo_gem->name_list);
1537
1538 bucket = mos_gem_bo_bucket_for_size(bufmgr_gem, bo->size);
1539 /* Put the buffer into our internal cache for reuse if we can. */

1542 I915_MADV_DONTNEED)) {
1543 bo_gem->free_time = time;
1544
1545 bo_gem->name = nullptr;
1546 bo_gem->validate_index = -1;

1548 DRMLISTADDTAIL(&bo_gem->head, &bucket->head);
1549 } else {
1550 mos_gem_bo_free(bo);
1551 }
1552}
