70void HalOcaInterface::On1stLevelBBStart(MOS_COMMAND_BUFFER &cmdBuffer, MOS_CONTEXT &mosContext,
71 uint32_t gpuContextHandle, MhwMiInterface &mhwMiInterface, MHW_MI_MMIOREGISTERS &mmioRegisters,
72 uint32_t offsetOf1stLevelBB, bool bUseSizeOfCmdBuf, uint32_t sizeOf1stLevelBB)
73{
74 MosInterface::SetObjectCapture(&cmdBuffer.OsResource);
75 MOS_STATUS status = MOS_STATUS_SUCCESS;
76 MosOcaInterface *pOcaInterface = &MosOcaInterfaceSpecific::GetInstance();
77 MOS_OCA_BUFFER_HANDLE ocaBufHandle = 0;
78 uint64_t ocaBase = 0;
79
80 if (nullptr == pOcaInterface || !((MosOcaInterfaceSpecific*)pOcaInterface)->IsOcaEnabled())
81 {
82 return;
83 }
84 PMOS_MUTEX mutex = pOcaInterface->GetMutex();
85 if (mutex == nullptr)
86 {
87 OnOcaError(&mosContext, MOS_STATUS_NULL_POINTER, __FUNCTION__, __LINE__);
88 return;
89 }
90 ocaBufHandle = GetOcaBufferHandle(cmdBuffer, mosContext);
91 if (ocaBufHandle != MOS_OCA_INVALID_BUFFER_HANDLE)
92 {
93 // will come here when On1stLevelBBStart being called twice without On1stLevelBBEnd being called.
94 OnOcaError(&mosContext, MOS_STATUS_INVALID_PARAMETER, __FUNCTION__, __LINE__);
95 return;
96 }
97 else
98 {
99 MosOcaAutoLock autoLock(mutex);
100 ocaBufHandle = pOcaInterface->LockOcaBufAvailable(&mosContext, gpuContextHandle);
101 if (MOS_OCA_INVALID_BUFFER_HANDLE == ocaBufHandle)
102 {
103 OnOcaError(&mosContext, MOS_STATUS_INVALID_HANDLE, __FUNCTION__, __LINE__);
104 return;
105 }
106 auto success = s_hOcaMap.insert(std::make_pair(cmdBuffer.pCmdBase, ocaBufHandle));
107 if (!success.second)
108 {
109 // Should never come to here.
110 MOS_OS_ASSERTMESSAGE("ocaBufHandle has already been assigned to current cmdBuffer!");
111 OnOcaError(&mosContext, MOS_STATUS_INVALID_HANDLE, __FUNCTION__, __LINE__);
112 return;
113 }
114 }
115
116 status = pOcaInterface->On1stLevelBBStart(ocaBase, ocaBufHandle, &mosContext, &cmdBuffer.OsResource, 0, true, 0);
117 if (MOS_FAILED(status))
118 {
119 RemoveOcaBufferHandle(cmdBuffer, mosContext);
120 OnOcaError(&mosContext, status, __FUNCTION__, __LINE__);
121 }
122}
