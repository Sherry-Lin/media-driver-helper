350static bool InitTglMediaWaExt(struct GfxDeviceInfo *devInfo,
351 MediaWaTable *waTable,
352 struct LinuxDriverInfo *drvInfo)
353{
354 if ((devInfo == nullptr) || (waTable == nullptr) || (drvInfo == nullptr))
355 {
356 DEVINFO_ERROR("null ptr is passed\n");
357 return false;
358 }
359
360 MEDIA_WR_WA(waTable, WaForceGlobalGTT, !drvInfo->hasPpgtt);
361 MEDIA_WR_WA(waTable, WaMidBatchPreemption, 0);
362 MEDIA_WR_WA(waTable, WaArbitraryNumMbsInSlice, 1);
363
364 MEDIA_WR_WA(waTable, WaSuperSliceHeaderPacking, 1);
365
366 MEDIA_WR_WA(waTable, WaSFC270DegreeRotation, 0);
367
368 MEDIA_WR_WA(waTable, WaEnableYV12BugFixInHalfSliceChicken7, 1);
369
370 MOS_USER_FEATURE_VALUE_DATA userFeatureData;
371 MOS_ZeroMemory(&userFeatureData, sizeof(userFeatureData));
372 MOS_UserFeature_ReadValue_ID(
373 nullptr,
374 __MEDIA_USER_FEATURE_VALUE_AUX_TABLE_16K_GRANULAR_ID,
375 &userFeatureData,
376 (MOS_CONTEXT_HANDLE)nullptr);
377
378 MEDIA_WR_WA(waTable, WaDummyReference, 1);
379
380 MEDIA_WR_WA(waTable, Wa16KInputHeightNV12Planar420, 1);
381
382 /*software wa to disable calculate the UV offset by gmmlib
383 CPU blt call will add/remove padding on the platform*/
384 MEDIA_WR_WA(waTable, WaDisableGmmLibOffsetInDeriveImage, 1);
385
386 /*software wa to fix some corner cases of HEVC/VP9 SFC and Scalability*/
387 MEDIA_WR_WA(waTable, Wa_14010222001, 1);
388
389 /*software wa to fix some corner hang cases for Scalability*/
390 MEDIA_WR_WA(waTable, Wa_2209620131, 1);
391
392
393 if (drvInfo->devId == 0x9A49 ||
394 drvInfo->devId == 0x9AC9 ||
395 drvInfo->devId == 0x9A60 ||
396 drvInfo->devId == 0x9A68 ||
397 drvInfo->devId == 0x9A70 ||
398 drvInfo->devId == 0x4905 ||
399 drvInfo->devId == 0x4906 ||
400 drvInfo->devId == 0x4907)
401 {
402 /*software wa to use frame based decoding for AV1 decode*/
403 MEDIA_WR_WA(waTable, Wa_1409820462, 1);
404 }
405
406 if (drvInfo->devId == 0x4905 ||
407 drvInfo->devId == 0x4906 ||
408 drvInfo->devId == 0x4907)
409 {
410 /* Turn off MMC for codec, need to remove once turn it on */
411 MEDIA_WR_WA(waTable, WaDisableCodecMmc, 1);
412
413 /* Turn off MMC for VPP, need to remove once turn it on */
414 MEDIA_WR_WA(waTable, WaDisableVPMmc, 1);
415 }
416
417 if (MEDIA_IS_WA(waTable, Wa_14012254246))
418 {
419 MOS_UserFeature_ReadValue_ID(
420 nullptr,
421 __MEDIA_USER_FEATURE_VALUE_DISABLE_TLB_PREFETCH_ID,
422 &userFeatureData,
423 (MOS_CONTEXT_HANDLE)nullptr);
424 // For PVC A0, if TLB pre-fetch disabled, resource could be allocated to system memory and local memory;
425 // otherwise, resource could be allocated to local memory only.
426 if (userFeatureData.u32Data == 0)
427 {
428 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 1);
429 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 0);
430 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
431 }
432 }
433
434 // For DG2
435 if (GFX_IS_DG2_G11_CONFIG(drvInfo->devId))
436 {
437 MEDIA_WR_WA(waTable, Wa_22011549751, 1);
438 }
439
440 MEDIA_WR_WA(waTable, WaEnableVPPCopy, 1);
441 return true;
442}

470static bool InitXehpSDVMediaWa(struct GfxDeviceInfo *devInfo,
471 MediaWaTable *waTable,
472 struct LinuxDriverInfo *drvInfo)
473{
474 if (!InitTglMediaWaExt(devInfo, waTable, drvInfo))
475 {
476 DEVINFO_ERROR("InitMediaWA failed\n");
477 return false;
478 }
479 /*software wa to use frame based decoding for AV1 decode*/
480 MEDIA_WR_WA(waTable, Wa_1409820462, 1);
481
482 /*software wa to add a dummy workload in front of
483 normal workload per batch buffer*/
484 MEDIA_WR_WA(waTable, Wa_1508208842, 1);
485
486 /* Turn off MMC for codec, need to remove once turn it on */
487 MEDIA_WR_WA(waTable, WaDisableCodecMmc, 1);
488
489 /* Turn off MMC for VPP, need to remove once turn it on */
490 MEDIA_WR_WA(waTable, WaDisableVPMmc, 1);
491
492 /* default resource allocate policy for XeHP_SDV*/
493 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 0);
494
495 /* only enable L3 for A step*/
496 if (drvInfo->devRev == 0x0 || drvInfo->devRev == 0x1)
497 {
498 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 1);
499 }
500 else
501 {
502 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 0);
503 }
504 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
505
506 MOS_USER_FEATURE_VALUE_DATA userFeatureData;
507 MOS_ZeroMemory(&userFeatureData, sizeof(userFeatureData));
508 MOS_UserFeature_ReadValue_ID(
509 nullptr,
510 __MEDIA_USER_FEATURE_VALUE_LOCAL_MEMORY_LEVEL_SWITCH_ID,
511 &userFeatureData,
512 (MOS_CONTEXT_HANDLE)nullptr);
513
514 if(userFeatureData.u32Data == 2)
515 {
516 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 1);
517 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 0);
518 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
519 }
520
521 if(userFeatureData.u32Data == 3)
522 {
523 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 0);
524 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 1);
525 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
526 }
527
528 if(userFeatureData.u32Data == 4)
529 {
530 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 0);
531 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 1);
532 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 1);
533 }
534
535 if (char* pLocalMemLevelSwitch = getenv("LOCALMEMLEVELSWITCH"))
536 {
537 int localMemLevelSwitch = atoi(pLocalMemLevelSwitch);
538
539 if(localMemLevelSwitch == 2)
540 {
541 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 1);
542 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 0);
543 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
544 }
545 else if(localMemLevelSwitch == 3)
546 {
547 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 0);
548 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 1);
549 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 0);
550 }
551 else if(localMemLevelSwitch == 4)
552 {
553 MEDIA_WR_WA(waTable, WaForceAllocateLML2, 0);
554 MEDIA_WR_WA(waTable, WaForceAllocateLML3, 1);
555 MEDIA_WR_WA(waTable, WaForceAllocateLML4, 1);
556 }
557 }
558 // XeHP_SDV I as P WA
559 MEDIA_WR_WA(waTable, Wa_22011549751, 1);
560#ifdef IGFX_XEHP_SDV_ENABLE_NON_UPSTREAM
561 MEDIA_WR_WA(waTable, WaHEVCVDEncForceDeltaQpRoiNotSupported, 1);
562#else
563 MEDIA_WR_WA(waTable, WaHEVCVDEncForceDeltaQpRoiNotSupported, 0);
564#endif
565 return true;
566}
