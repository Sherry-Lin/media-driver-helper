65inline int memcpy_s(void *dst, size_t numberOfElements, const void *src, size_t count)
66{
67 if ((dst == nullptr) || (src == nullptr))
68 {
69 return EINVAL;
70 }
71 if (numberOfElements < count)
72 {
73 return ERANGE;
74 }
75 std::memcpy(dst, src, count);
76 return 0;
77}
