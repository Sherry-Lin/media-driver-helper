59VAStatus DdiDecodeJpeg::ParseSliceParams(
60 DDI_MEDIA_CONTEXT *mediaCtx,
61 VASliceParameterBufferJPEGBaseline *slcParam,
62 uint32_t numSlices)
63{
64 CodecDecodeJpegScanParameter *jpegSliceParam =
65 (CodecDecodeJpegScanParameter *)(m_decodeCtx->DecodeParams.m_sliceParams);
66
67 CodecDecodeJpegPicParams *picParam = (CodecDecodeJpegPicParams *)(m_decodeCtx->DecodeParams.m_picParams);
68
69 if ((jpegSliceParam == nullptr) ||
70 (picParam == nullptr) ||
71 (slcParam == nullptr))
72 {
73 DDI_CODEC_ASSERTMESSAGE("Invalid Parameter for Parsing JPEG Slice parameter\n");
74 return VA_STATUS_ERROR_INVALID_PARAMETER;
75 }
76
77 jpegSliceParam->NumScans += numSlices;
78 picParam->m_totalScans += numSlices;
79 if (picParam->m_totalScans == 1 && slcParam[0].num_components > 1)
80 {
81 picParam->m_interleavedData = 1;
82 }
83
84 uint32_t j, i;
85 int32_t startIdx = m_numScans;
86 for (j = 0; j < numSlices; j++)
87 {
88 if (j + startIdx >= jpegNumComponent ||
89 slcParam[j].num_components > jpegNumComponent)
90 {
91 return VA_STATUS_ERROR_INVALID_PARAMETER;
92 }
93
94 for (i = 0; i < slcParam[j].num_components; i++)
95 {
96 jpegSliceParam->ScanHeader[j + startIdx].ComponentSelector[i] = slcParam[j].components[i].component_selector;
97 jpegSliceParam->ScanHeader[j + startIdx].DcHuffTblSelector[i] = slcParam[j].components[i].dc_table_selector;
98 jpegSliceParam->ScanHeader[j + startIdx].AcHuffTblSelector[i] = slcParam[j].components[i].ac_table_selector;
99 }
100 jpegSliceParam->ScanHeader[j + startIdx].NumComponents = slcParam[j].num_components;
101 jpegSliceParam->ScanHeader[j + startIdx].RestartInterval = slcParam[j].restart_interval;
102 jpegSliceParam->ScanHeader[j + startIdx].MCUCount = slcParam[j].num_mcus;
103 jpegSliceParam->ScanHeader[j + startIdx].ScanHoriPosition = slcParam[j].slice_horizontal_position;
104 jpegSliceParam->ScanHeader[j + startIdx].ScanVertPosition = slcParam[j].slice_vertical_position;
105 jpegSliceParam->ScanHeader[j + startIdx].DataOffset = slcParam[j].slice_data_offset;
106 jpegSliceParam->ScanHeader[j + startIdx].DataLength = slcParam[j].slice_data_size;
107 }
108
109 return VA_STATUS_SUCCESS;
110}

112VAStatus DdiDecodeJpeg::ParsePicParams(
113 DDI_MEDIA_CONTEXT *mediaCtx,
114 VAPictureParameterBufferJPEGBaseline *picParam)
115{
116 CodecDecodeJpegPicParams *jpegPicParam = (CodecDecodeJpegPicParams *)(m_decodeCtx->DecodeParams.m_picParams);
117
118 if ((jpegPicParam == nullptr) ||
119 (picParam == nullptr))
120 {
121 DDI_CODEC_ASSERTMESSAGE("Null Parameter for Parsing JPEG Picture parameter\n");
122 return VA_STATUS_ERROR_INVALID_PARAMETER;
123 }
124
125 jpegPicParam->m_frameWidth = picParam->picture_width;
126 jpegPicParam->m_frameHeight = picParam->picture_height;
127 jpegPicParam->m_numCompInFrame = picParam->num_components;
128
129 switch (picParam->rotation)
130 {
131 case VA_ROTATION_NONE:
132 jpegPicParam->m_rotation = jpegRotation0;
133 break;
134 case VA_ROTATION_90:
135 jpegPicParam->m_rotation = jpegRotation90;
136 break;
137 case VA_ROTATION_180:
138 jpegPicParam->m_rotation = jpegRotation180;
139 break;
140 case VA_ROTATION_270:
141 jpegPicParam->m_rotation = jpegRotation270;
142 break;
143 default:
144 /* For the other rotation type, the rotation is disabled. */
145 jpegPicParam->m_rotation = jpegRotation0;
146 break;
147 ;
148 }
149
150 if (jpegPicParam->m_numCompInFrame == 1)
151 {
152 jpegPicParam->m_chromaType = jpegYUV400;
153 }
154 else if (jpegPicParam->m_numCompInFrame == 3)
155 {
156 int32_t h1 = picParam->components[0].h_sampling_factor;
157 int32_t h2 = picParam->components[1].h_sampling_factor;
158 int32_t h3 = picParam->components[2].h_sampling_factor;
159 int32_t v1 = picParam->components[0].v_sampling_factor;
160 int32_t v2 = picParam->components[1].v_sampling_factor;
161 int32_t v3 = picParam->components[2].v_sampling_factor;
162
163 if (h1 == 2 && h2 == 1 && h3 == 1 &&
164 v1 == 2 && v2 == 1 && v3 == 1)
165 {
166 jpegPicParam->m_chromaType = jpegYUV420;
167 }
168 else if (h1 == 2 && h2 == 1 && h3 == 1 &&
169 v1 == 1 && v2 == 1 && v3 == 1)
170 {
171 jpegPicParam->m_chromaType = jpegYUV422H2Y;
172 }
173 else if (h1 == 1 && h2 == 1 && h3 == 1 &&
174 v1 == 1 && v2 == 1 && v3 == 1)
175 {
176 switch (picParam->color_space)
177 {
178 case 0: // YUV
179 jpegPicParam->m_chromaType = jpegYUV444;
180 break;
181 case 1: // RGB
182 jpegPicParam->m_chromaType = jpegRGB;
183 break;
184 case 2: // BGR
185 jpegPicParam->m_chromaType = jpegBGR;
186 break;
187 default:
188 /* For the other type, the default YUV444 is used */
189 jpegPicParam->m_chromaType = jpegYUV444;
190 break;
191 ;
192 }
193 }
194 else if (h1 == 4 && h2 == 1 && h3 == 1 &&
195 v1 == 1 && v2 == 1 && v3 == 1)
196 {
197 jpegPicParam->m_chromaType = jpegYUV411;
198 }
199 else if (h1 == 1 && h2 == 1 && h3 == 1 &&
200 v1 == 2 && v2 == 1 && v3 == 1)
201 {
202 jpegPicParam->m_chromaType = jpegYUV422V2Y;
203 }
204 else if (h1 == 2 && h2 == 1 && h3 == 1 &&
205 v1 == 2 && v2 == 2 && v3 == 2)
206 {
207 jpegPicParam->m_chromaType = jpegYUV422H4Y;
208 }
209 else if (h1 == 2 && h2 == 2 && h3 == 2 &&
210 v1 == 2 && v2 == 1 && v3 == 1)
211 {
212 jpegPicParam->m_chromaType = jpegYUV422V4Y;
213 }
214 else
215 {
216 DDI_CODEC_NORMALMESSAGE("Unsupported sampling factor in JPEG Picture parameter\n");
217 return VA_STATUS_ERROR_INVALID_PARAMETER;
218 }
219 }
220
221 memset(jpegPicParam->m_componentIdentifier, 0, jpegNumComponent);
222 memset(jpegPicParam->m_quantTableSelector, 0, jpegNumComponent);
223
224 if (picParam->num_components > jpegNumComponent)
225 {
226 DDI_CODEC_NORMALMESSAGE("Unsupported component num in JPEG Picture parameter\n");
227 return VA_STATUS_ERROR_INVALID_PARAMETER;
228 }
229 for (int32_t i = 0; i < picParam->num_components; i++)
230 {
231 jpegPicParam->m_componentIdentifier[i] = picParam->components[i].component_id;
232 jpegPicParam->m_quantTableSelector[i] = picParam->components[i].quantiser_table_selector;
233 }
234
235 return VA_STATUS_SUCCESS;
236}

238VAStatus DdiDecodeJpeg::ParseHuffmanTbl(
239 DDI_MEDIA_CONTEXT *mediaCtx,
240 VAHuffmanTableBufferJPEGBaseline *huffmanTbl)
241{
242 PCODECHAL_DECODE_JPEG_HUFFMAN_TABLE jpegHuffTbl = (PCODECHAL_DECODE_JPEG_HUFFMAN_TABLE)(m_decodeCtx->DecodeParams.m_huffmanTable);
243 int32_t sumBITS = 0;
244
245 if ((jpegHuffTbl == nullptr) ||
246 (huffmanTbl == nullptr))
247 {
248 DDI_CODEC_ASSERTMESSAGE("Null Parameter for Parsing JPEG Huffman Tableparameter\n");
249 return VA_STATUS_ERROR_INVALID_PARAMETER;
250 }
251
252 memset(jpegHuffTbl, 0, sizeof(CODECHAL_DECODE_JPEG_HUFFMAN_TABLE));
253
254 for (int32_t i = 0; i < DDI_DECODE_JPEG_MAXIMUM_HUFFMAN_TABLE; i++)
255 {
256 if (huffmanTbl->load_huffman_table[i] == BUFFER_LOADED)
257 {
258 sumBITS = 0;
259 for (int32_t j = 0; j < JPEG_NUM_HUFF_TABLE_DC_BITS; j++)
260 {
261 sumBITS += huffmanTbl->huffman_table[i].num_dc_codes[j];
262 }
263
264 if (sumBITS > JPEG_NUM_HUFF_TABLE_DC_HUFFVAL)
265 {
266 DDI_CODEC_ASSERTMESSAGE("Huffman table DC entries number is out of HW limitation.");
267 return VA_STATUS_ERROR_INVALID_PARAMETER;
268 }
269
270 // the size of jpegHuffTbl->HuffTable[i].DC_BITS is 12 (defined in driver)
271 // the size of huffmanTbl->huffman_table[i].num_dc_codes is 16 (defined in libva)
272 // it is using the size of "DC_BITS" for solve the overflow
273 MOS_SecureMemcpy(jpegHuffTbl->HuffTable[i].DC_BITS,
274 sizeof(jpegHuffTbl->HuffTable[i].DC_BITS),
275 huffmanTbl->huffman_table[i].num_dc_codes,
276 sizeof(jpegHuffTbl->HuffTable[i].DC_BITS));
277 MOS_SecureMemcpy(jpegHuffTbl->HuffTable[i].DC_HUFFVAL,
278 sizeof(jpegHuffTbl->HuffTable[i].DC_HUFFVAL),
279 huffmanTbl->huffman_table[i].dc_values,
280 sizeof(huffmanTbl->huffman_table[i].dc_values));
281 MOS_SecureMemcpy(jpegHuffTbl->HuffTable[i].AC_BITS,
282 sizeof(jpegHuffTbl->HuffTable[i].AC_BITS),
283 huffmanTbl->huffman_table[i].num_ac_codes,
284 sizeof(huffmanTbl->huffman_table[i].num_ac_codes));
285 MOS_SecureMemcpy(jpegHuffTbl->HuffTable[i].AC_HUFFVAL,
286 sizeof(jpegHuffTbl->HuffTable[i].AC_HUFFVAL),
287 huffmanTbl->huffman_table[i].ac_values,
288 sizeof(huffmanTbl->huffman_table[i].ac_values));
289 }
290 }
291
292 return VA_STATUS_SUCCESS;
293}

302VAStatus DdiDecodeJpeg::ParseIQMatrix(
303 DDI_MEDIA_CONTEXT *mediaCtx,
304 VAIQMatrixBufferJPEGBaseline *matrix)
305{
306 CodecJpegQuantMatrix *jpegQMatrix = (CodecJpegQuantMatrix *)(m_decodeCtx->DecodeParams.m_iqMatrixBuffer);
307
308 if ((matrix == nullptr) || (jpegQMatrix == nullptr))
309 {
310 DDI_CODEC_ASSERTMESSAGE("Null Parameter for Parsing JPEG IQMatrix \n");
311 return VA_STATUS_ERROR_INVALID_PARAMETER;
312 }
313
314 memset(jpegQMatrix, 0, sizeof(CodecJpegQuantMatrix));
315
316 int32_t idx = 0, idx2 = 0;
317 for (idx = 0; idx < DDI_DECODE_JPEG_MAXIMUM_QMATRIX_NUM; idx++)
318 {
319 if (matrix->load_quantiser_table[idx] == BUFFER_LOADED)
320 {
321 for (idx2 = 0; idx2 < DDI_DECODE_JPEG_MAXIMUM_QMATRIX_ENTRIES; idx2++)
322 {
323 jpegQMatrix->m_quantMatrix[idx][zigzag_order[idx2]] = matrix->quantiser_table[idx][idx2];
324 }
325 }
326 }
327
328 return VA_STATUS_SUCCESS;
329}

331VAStatus DdiDecodeJpeg::SetBufferRendered(VABufferID bufferID)
332{
333 DDI_CODEC_COM_BUFFER_MGR *bufMgr = &(m_decodeCtx->BufMgr);
334
335 if (bufMgr == nullptr)
336 {
337 DDI_CODEC_ASSERTMESSAGE("Null Parameter for Parsing Data buffer for JPEG\n");
338 return VA_STATUS_ERROR_INVALID_PARAMETER;
339 }
340
341 bool renderFlag = false;
342 for (uint32_t i = 0; i < bufMgr->dwNumSliceData; i++)
343 {
344 // Depend on the ID we tracked, if application want to rendered one of them
345 // we set some flags
346 if (bufMgr->pSliceData[i].vaBufferId == bufferID)
347 {
348 if (bufMgr->pSliceData[i].bRendered == false)
349 {
350 // set the rendered flags. which will be used when EndPicture.
351 bufMgr->pSliceData[i].bRendered = true;
352 // calculate the size of a bunch of rendered buffers. for endpicture to allocate appropriate memory size of GPU.
353 bufMgr->dwSizeOfRenderedSliceData += bufMgr->pSliceData[i].uiLength;
354 // keep record the render order, so that we can calculate the offset correctly when endpicture.
355 // in this array we save the buffer index in pSliceData which render in sequence. if application create buffers like: 4,3,5,1,2.
356 // but only render 2,3,5. the content in this array is: [4,1,2], which is the index of created buffer.
357 bufMgr->pRenderedOrder[bufMgr->dwNumOfRenderedSliceData] = i;
358 bufMgr->dwNumOfRenderedSliceData++;
359 }
360 renderFlag = true;
361 break;
362 }
363 }
364
365 if (renderFlag)
366 {
367 return VA_STATUS_SUCCESS;
368 }
369 else
370 {
371 return VA_STATUS_ERROR_INVALID_BUFFER;
372 }

375VAStatus DdiDecodeJpeg::RenderPicture(
376 VADriverContextP ctx,
377 VAContextID context,
378 VABufferID *buffers,
379 int32_t numBuffers)
380{
381 DDI_CODEC_FUNC_ENTER;
382
383 VAStatus va = VA_STATUS_SUCCESS;
384 PDDI_MEDIA_CONTEXT mediaCtx = GetMediaContext(ctx);
385
386 void *data = nullptr;
387 for (int32_t i = 0; i < numBuffers; i++)
388 {
389 if (!buffers || (buffers[i] == VA_INVALID_ID))
390 {
391 return VA_STATUS_ERROR_INVALID_BUFFER;
392 }
393
394 DDI_MEDIA_BUFFER *buf = MediaLibvaCommonNext::GetBufferFromVABufferID(mediaCtx, buffers[i]);
395 if (nullptr == buf)
396 {
397 return VA_STATUS_ERROR_INVALID_BUFFER;
398 }
399
400 uint32_t dataSize = buf->iSize;
401 MediaLibvaInterfaceNext::MapBuffer(ctx, buffers[i], &data);
402
403 if (data == nullptr)
404 {
405 return VA_STATUS_ERROR_INVALID_BUFFER;
406 }
407
408 switch ((int32_t)buf->uiType)
409 {
410 case VASliceDataBufferType:
411 {
412 DDI_CODEC_CHK_RET(SetBufferRendered(buffers[i]),"SetBufferRendered failed!");
413 m_decodeCtx->DecodeParams.m_dataSize += dataSize;
414 break;
415 }
416 case VASliceParameterBufferType:
417 {
418 if (buf->uiNumElements == 0)
419 {
420 return VA_STATUS_ERROR_INVALID_BUFFER;
421 }
422
423 uint32_t numSlices = buf->uiNumElements;
424
425 if ((m_numScans + numSlices) > jpegNumComponent)
426 {
427 DDI_CODEC_NORMALMESSAGE("the total number of JPEG scans are beyond the supported num(4)\n");
428 return VA_STATUS_ERROR_INVALID_PARAMETER;
429 }
430 DDI_CODEC_CHK_RET(AllocSliceParamContext(numSlices),"AllocSliceParamContext failed!");
431 VASliceParameterBufferJPEGBaseline *slcInfo = (VASliceParameterBufferJPEGBaseline *)data;
432 DDI_CODEC_CHK_RET(ParseSliceParams(mediaCtx, slcInfo, numSlices),"ParseSliceParams failed!");
433 m_decodeCtx->BufMgr.pNumOfRenderedSliceParaForOneBuffer[m_decodeCtx->BufMgr.dwNumOfRenderedSlicePara] = numSlices;
434 m_decodeCtx->BufMgr.dwNumOfRenderedSlicePara++;
435
436 m_decodeCtx->DecodeParams.m_numSlices += numSlices;
437 m_numScans += numSlices;
438 m_groupIndex++;
439 break;
440 }
441 case VAIQMatrixBufferType:
442 {
443 VAIQMatrixBufferJPEGBaseline *imxBuf = (VAIQMatrixBufferJPEGBaseline *)data;
444 DDI_CODEC_CHK_RET(ParseIQMatrix(mediaCtx, imxBuf),"ParseIQMatrix failed!");
445 break;
446 }
447 case VAPictureParameterBufferType:
448 {
449 VAPictureParameterBufferJPEGBaseline *picParam = (VAPictureParameterBufferJPEGBaseline *)data;
450 DDI_CODEC_CHK_RET(ParsePicParams(mediaCtx, picParam),"ParsePicParams failed!");
451 break;
452 }
453 case VAHuffmanTableBufferType:
454 {
455 VAHuffmanTableBufferJPEGBaseline *huffTbl = (VAHuffmanTableBufferJPEGBaseline *)data;
456 DDI_CODEC_CHK_RET(ParseHuffmanTbl(mediaCtx, huffTbl),"ParseHuffmanTbl failed!");
457 break;
458 }
459 case VAProcPipelineParameterBufferType:
460 {
461 DDI_CODEC_NORMALMESSAGE("ProcPipeline is not supported for JPEGBaseline decoding\n");
462 break;
463 }
464 case VADecodeStreamoutBufferType:
465 {
466 MediaLibvaCommonNext::MediaBufferToMosResource(buf, &m_decodeCtx->BufMgr.resExternalStreamOutBuffer);
467 m_streamOutEnabled = true;
468 break;
469 }
470 default:
471 va = VA_STATUS_ERROR_UNSUPPORTED_BUFFERTYPE;
472 break;
473 }
474 MediaLibvaInterfaceNext::UnmapBuffer(ctx, buffers[i]);
475 }
476
477 return va;
478}

531VAStatus DdiDecodeJpeg::SetDecodeParams()
532{
533 DDI_CODEC_COM_BUFFER_MGR *bufMgr = &(m_decodeCtx->BufMgr);
534
535 // we do not support mismatched usecase.
536 if ((bufMgr->dwNumOfRenderedSlicePara != bufMgr->dwNumOfRenderedSliceData) ||
537 (bufMgr->dwNumOfRenderedSlicePara == 0))
538 {
539 DDI_CODEC_NORMALMESSAGE("DDI: Unsupported buffer mismatch usage!\n");
540 return VA_STATUS_ERROR_INVALID_PARAMETER;
541 }
542
543 // Allocate GPU Buffer description keeper.
544 m_jpegBitstreamBuf = (DDI_MEDIA_BUFFER *)MOS_AllocAndZeroMemory(sizeof(DDI_MEDIA_BUFFER));
545 if (m_jpegBitstreamBuf == nullptr)
546 {
547 DDI_CODEC_ASSERTMESSAGE("DDI: Allocate Jpeg Media Buffer Failed!\n");
548 return VA_STATUS_ERROR_UNKNOWN;
549 }
550
551 m_jpegBitstreamBuf->iSize = bufMgr->dwSizeOfRenderedSliceData;
552 m_jpegBitstreamBuf->uiNumElements = bufMgr->dwNumOfRenderedSliceData;
553 m_jpegBitstreamBuf->uiType = VASliceDataBufferType;
554 m_jpegBitstreamBuf->format = Media_Format_Buffer;
555 m_jpegBitstreamBuf->uiOffset = 0;
556 m_jpegBitstreamBuf->bCFlushReq = false;
557 m_jpegBitstreamBuf->pMediaCtx = m_decodeCtx->pMediaCtx;
558
559 // Create GPU buffer
560 VAStatus va = MediaLibvaUtilNext::CreateBuffer(m_jpegBitstreamBuf, m_decodeCtx->pMediaCtx->pDrmBufMgr);
561 if (va != VA_STATUS_SUCCESS)
562 {
563 MediaLibvaUtilNext::FreeBuffer(m_jpegBitstreamBuf);
564 MOS_FreeMemory(m_jpegBitstreamBuf);
565 m_jpegBitstreamBuf = nullptr;
566 return va;
567 }
568
569 // For the first time you call MediaLibvaUtilNext::LockBuffer for a fresh GPU memory, it will map GPU address to a virtual address.
570 // and then, MediaLibvaUtilNext::LockBuffer is acutally to increase the reference count. so we used here for 2 resaons:
571 //
572 // 1. Map GPU address to virtual at the beginning when we combine the CPU -> GPU.
573 uint8_t *mappedBuf = (uint8_t *)MediaLibvaUtilNext::LockBuffer(m_jpegBitstreamBuf, MOS_LOCKFLAG_WRITEONLY);
574
575 if (mappedBuf == nullptr)
576 {
577 MediaLibvaUtilNext::FreeBuffer(m_jpegBitstreamBuf);
578 MOS_FreeMemory(m_jpegBitstreamBuf);
579 m_jpegBitstreamBuf = nullptr;
580 return VA_STATUS_ERROR_ALLOCATION_FAILED;
581 }
582
583 // get the JPEG Slice Header Params for offset recaculated.
584 CodecDecodeJpegScanParameter *sliceParam =
585 (CodecDecodeJpegScanParameter *)(m_decodeCtx->DecodeParams.m_sliceParams);
586
587 uint32_t bufOffset = 0;
588 int32_t orderSlicePara = 0;
589 int32_t orderSliceData = 0;
590 for (uint32_t i = 0; i < bufMgr->dwNumOfRenderedSliceData; i++)
591 {
592 // get the rendered slice data index in rendered order.
593 int32_t renderedBufIdx = bufMgr->pRenderedOrder[i];
594 if (bufMgr->pSliceData[renderedBufIdx].bRendered)
595 {
596 MOS_SecureMemcpy((void *)(mappedBuf + bufOffset),
597 bufMgr->pSliceData[renderedBufIdx].uiLength,
598 bufMgr->pSliceData[renderedBufIdx].pBaseAddress,
599 bufMgr->pSliceData[renderedBufIdx].uiLength);
600
601 // since we assume application must make sure ONE slice parameter buffer ONE slice data buffer, so we recaculate header offset here.
602 for (int32_t j = 0; j < bufMgr->pNumOfRenderedSliceParaForOneBuffer[orderSliceData]; j++)
603 {
604 sliceParam->ScanHeader[orderSlicePara].DataOffset += bufOffset;
605 orderSlicePara++;
606 }
607 bufOffset += bufMgr->pSliceData[renderedBufIdx].uiLength;
608 bufMgr->pNumOfRenderedSliceParaForOneBuffer[orderSliceData] = 0;
609 orderSliceData++;
610 bufMgr->pSliceData[renderedBufIdx].bRendered = false;
611 }
612 }
613 MediaLibvaUtilNext::UnlockBuffer(m_jpegBitstreamBuf);
614 MediaLibvaCommonNext::MediaBufferToMosResource(m_jpegBitstreamBuf, &(bufMgr->resBitstreamBuffer));
615 bufMgr->dwNumOfRenderedSliceData = 0;
616 bufMgr->dwNumOfRenderedSlicePara = 0;
617 bufMgr->dwSizeOfRenderedSliceData = 0;
618
619 m_destSurface.dwOffset = 0;
620 m_destSurface.Format = Format_NV12;
621
622 CodecDecodeJpegPicParams *jpegPicParam = (CodecDecodeJpegPicParams *)(m_decodeCtx->DecodeParams.m_picParams);
623 if ((m_decodeCtx->RTtbl.pCurrentRT->format == Media_Format_NV12)
624 &&(jpegPicParam->m_chromaType == jpegYUV444))
625 {
626 UnRegisterRTSurfaces(&(m_decodeCtx->RTtbl), m_decodeCtx->RTtbl.pCurrentRT);
627 m_decodeCtx->RTtbl.pCurrentRT = MediaLibvaCommonNext::ReplaceSurfaceWithNewFormat(m_decodeCtx->RTtbl.pCurrentRT, Media_Format_444P);
628 RegisterRTSurfaces(&(m_decodeCtx->RTtbl), m_decodeCtx->RTtbl.pCurrentRT);
629 }
630 if (m_decodeCtx->RTtbl.pCurrentRT != nullptr)
631 {
632 MediaLibvaCommonNext::MediaSurfaceToMosResource((&(m_decodeCtx->RTtbl))->pCurrentRT, &(m_destSurface.OsResource));
633 }
634
635 (&m_decodeCtx->DecodeParams)->m_destSurface = &m_destSurface;
636
637 (&m_decodeCtx->DecodeParams)->m_deblockSurface = nullptr;
638
639 (&m_decodeCtx->DecodeParams)->m_dataBuffer = &bufMgr->resBitstreamBuffer;
640 (&m_decodeCtx->DecodeParams)->m_bitStreamBufData = bufMgr->pBitstreamBuffer;
641 Mos_Solo_OverrideBufferSize((&m_decodeCtx->DecodeParams)->m_dataSize, (&m_decodeCtx->DecodeParams)->m_dataBuffer);
642
643 (&m_decodeCtx->DecodeParams)->m_bitplaneBuffer = nullptr;
644
645 if (m_streamOutEnabled)
646 {
647 (&m_decodeCtx->DecodeParams)->m_streamOutEnabled = true;
648 (&m_decodeCtx->DecodeParams)->m_externalStreamOutBuffer = &bufMgr->resExternalStreamOutBuffer;

699VAStatus DdiDecodeJpeg::AllocBsBuffer(
700 DDI_CODEC_COM_BUFFER_MGR *bufMgr,
701 DDI_MEDIA_BUFFER *buf)
702{
703 // Allocate JPEG slice data memory from CPU.
704 uint8_t *bsAddr = nullptr;
705 uint32_t index = 0;
706
707 index = bufMgr->dwNumSliceData;
708
709 /* the pSliceData needs to be reallocated in order to contain more SliceDataBuf */
710 if (index >= bufMgr->m_maxNumSliceData)
711 {
712 /* In theroy it can resize the m_maxNumSliceData one by one. But in order to
713 * avoid calling realloc frequently, it will try to allocate 10 to hold more
714 * SliceDataBuf. This is only for the optimized purpose.
715 */
716 int32_t reallocSize = bufMgr->m_maxNumSliceData + 10;
717
718 bufMgr->pSliceData = (DDI_CODEC_BITSTREAM_BUFFER_INFO *)realloc(bufMgr->pSliceData, sizeof(bufMgr->pSliceData[0]) * reallocSize);
719
720 if (bufMgr->pSliceData == nullptr)
721 {
722 DDI_CODEC_ASSERTMESSAGE("fail to reallocate pSliceData for JPEG\n.");
723 return VA_STATUS_ERROR_ALLOCATION_FAILED;
724 }
725 memset((void *)(bufMgr->pSliceData + bufMgr->m_maxNumSliceData), 0,
726 sizeof(bufMgr->pSliceData[0]) * 10);
727
728 bufMgr->m_maxNumSliceData += 10;
729 }
730
731 bsAddr = (uint8_t*)MOS_AllocAndZeroMemory(buf->iSize);
732 if (bsAddr == 0)
733 {
734 return VA_STATUS_ERROR_ALLOCATION_FAILED;
735 }
736
737 buf->pData = bsAddr;
738 buf->format = Media_Format_CPU;
739 buf->bCFlushReq = false;
740 buf->uiOffset = 0;
741 bufMgr->pSliceData[index].uiLength = buf->iSize;
742 bufMgr->pSliceData[index].uiOffset = buf->uiOffset;
743 bufMgr->pSliceData[index].pBaseAddress = buf->pData;
744 bufMgr->dwNumSliceData ++;
745 return VA_STATUS_SUCCESS;
746}

749 DDI_MEDIA_BUFFER *buf)
750{
751 DDI_CODEC_COM_BUFFER_MGR *bufMgr = nullptr;
752 uint32_t availSize = 0;
753 uint32_t newSize = 0;

758 if (availSize < buf->uiNumElements)

766 MOS_ZeroMemory(bufMgr->Codec_Param.Codec_Param_JPEG.pVASliceParaBufJPEG + m_sliceCtrlBufNum, sizeof(VASliceParameterBufferJPEGBaseline) * (buf->uiNumElements - availSize));
767 m_sliceCtrlBufNum = m_sliceCtrlBufNum - availSize + buf->uiNumElements;
