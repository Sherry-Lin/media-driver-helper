145int32_t Linux_GetCommandBuffer(
146 PMOS_CONTEXT osContext,
147 PMOS_COMMAND_BUFFER cmdBuffer,
148 int32_t size)
149{
150 MOS_LINUX_BO *cmd_bo = nullptr;
151
152 if ( osContext == nullptr ||
153 cmdBuffer == nullptr)
154 {
155 MOS_OS_ASSERTMESSAGE("Linux_GetCommandBuffer:osContext == nullptr || cmdBuffer == NULL");
156 return false;
157 }
158
159 // Allocate the command buffer from GEM
160 cmd_bo = mos_bo_alloc(osContext->bufmgr,"MOS CmdBuf",size,4096, MOS_MEMPOOL_SYSTEMMEMORY); // Align to page boundary
161 MOS_OS_CHK_NULL_RETURN_VALUE(cmd_bo, false);
162
163 //MOS_OS_NORMALMESSAGE("alloc CMB, bo is 0x%x.", cmd_bo);
164
165 // Map command buffer to user virtual address
166 if (mos_bo_map(cmd_bo,1) != 0) // Write enable set
167 {
168 MOS_OS_ASSERTMESSAGE("Mapping of command buffer failed.");
169 mos_bo_unreference(cmd_bo);
170 return false;
171 }
172
173 Mos_ResetResource(&cmdBuffer->OsResource);
174
175 // Fill in resource information
176 cmdBuffer->OsResource.Format = Format_Buffer;
177 cmdBuffer->OsResource.iWidth = cmd_bo->size;
178 cmdBuffer->OsResource.iHeight = 1;
179 cmdBuffer->OsResource.iPitch = cmd_bo->size;
180 cmdBuffer->OsResource.iSize = cmdBuffer->OsResource.iPitch * cmdBuffer->OsResource.iHeight;
181 cmdBuffer->OsResource.iCount = 1;
182 cmdBuffer->OsResource.pData = (uint8_t*)cmd_bo->virt;
183 cmdBuffer->OsResource.TileType = MOS_TILE_LINEAR;
184 cmdBuffer->OsResource.bo = cmd_bo;
185 cmdBuffer->OsResource.bMapped = true;
186
187 // for MOS wrapper to avoid memory leak
188 cmdBuffer->OsResource.bConvertedFromDDIResource = true;
189
190 cmdBuffer->pCmdBase = (uint32_t*)cmd_bo->virt;
191 cmdBuffer->pCmdPtr = (uint32_t*)cmd_bo->virt;
192 cmdBuffer->iOffset = 0;
193 cmdBuffer->iRemaining = cmd_bo->size;
194 cmdBuffer->iCmdIndex = -1;
195 cmdBuffer->iVdboxNodeIndex = MOS_VDBOX_NODE_INVALID;
196 cmdBuffer->iVeboxNodeIndex = MOS_VEBOX_NODE_INVALID;
197 cmdBuffer->is1stLvlBB = true;
198 MOS_ZeroMemory(cmdBuffer->pCmdBase, cmd_bo->size);
199 cmdBuffer->iSubmissionType = SUBMISSION_TYPE_SINGLE_PIPE;
200 MOS_ZeroMemory(&cmdBuffer->Attributes, sizeof(cmdBuffer->Attributes));
201
202 return true;
203}

316MOS_STATUS Linux_WaitAndReleaseCmdBuffer(
317 PMOS_CONTEXT osContext,
318 int32_t index)
319{
320 MOS_LINUX_BO *cmd_bo;
321
322 MOS_OS_FUNCTION_ENTER;
323
324 MOS_OS_CHK_NULL_RETURN(osContext);
325
326 if (index < 0 || index >= MAX_CMD_BUF_NUM)
327 {
328 return MOS_STATUS_INVALID_PARAMETER;
329 }
330
331 // According to the logic of CmdBufferPool now, the command buffer is used in a circular way.
332 // The input index always points to the next(oldest) buffer after the latest(newest) buffer.
333 // If the next buffer is not empty (!= nullptr), all the buffers in the pool will also be not empty.
334 // So it's not necessary to check all buffers to see whether there's empty buffer.
335 cmd_bo = osContext->CmdBufferPool.pCmd_bo[index];
336 if (cmd_bo != nullptr)
337 {
338 mos_bo_wait_rendering(cmd_bo);
339 mos_bo_unreference(cmd_bo);
340 osContext->CmdBufferPool.pCmd_bo[index] = nullptr;
341 }
342
343 return MOS_STATUS_SUCCESS;
344}

432MOS_STATUS Linux_InitGPUStatus(
433 PMOS_CONTEXT osContext)
434{
435 MOS_LINUX_BO *bo = nullptr;
436
437 MOS_OS_CHK_NULL_RETURN(osContext);
438
439 osContext->pGPUStatusBuffer =
440 (MOS_RESOURCE*)MOS_AllocAndZeroMemory(sizeof(MOS_RESOURCE) * MOS_GPU_CONTEXT_MAX);
441
442 MOS_OS_CHK_NULL_RETURN(osContext->pGPUStatusBuffer);
443
444 // Allocate the command buffer from GEM
445 bo = mos_bo_alloc(osContext->bufmgr,"GPU Status Buffer", sizeof(MOS_GPU_STATUS_DATA) * MOS_GPU_CONTEXT_MAX, 4096, MOS_MEMPOOL_SYSTEMMEMORY); // Align to page boundary
446 if (bo == nullptr)
447 {
448 MOS_OS_ASSERTMESSAGE("Allocation of GPU Status Buffer failed.");
449 MOS_FreeMemAndSetNull(osContext->pGPUStatusBuffer);
450 return MOS_STATUS_NO_SPACE;
451 }
452
453 // Map command buffer to user virtual address
454 if (mos_bo_map(bo, 1) != 0) // Write enable set
455 {
456 MOS_OS_ASSERTMESSAGE("Mapping of GPU Status Buffer failed.");
457 mos_bo_unreference(bo);
458 MOS_FreeMemAndSetNull(osContext->pGPUStatusBuffer);
459 return MOS_STATUS_INVALID_HANDLE;
460 }
461
462 Mos_ResetResource(osContext->pGPUStatusBuffer);
463
464 // Fill in resource information
465 osContext->pGPUStatusBuffer->Format = Format_Buffer;
466 osContext->pGPUStatusBuffer->iWidth = bo->size;
467 osContext->pGPUStatusBuffer->iHeight = 1;
468 osContext->pGPUStatusBuffer->iPitch = bo->size;
469 osContext->pGPUStatusBuffer->iCount = 1;
470 osContext->pGPUStatusBuffer->pData = (uint8_t*)bo->virt;
471 osContext->pGPUStatusBuffer->TileType = MOS_TILE_LINEAR;
472 osContext->pGPUStatusBuffer->bo = bo;
473 osContext->pGPUStatusBuffer->bMapped = true;
474
475 MOS_ZeroMemory(osContext->pGPUStatusBuffer->pData, bo->size);
476
477 return MOS_STATUS_SUCCESS;
478}

610MOS_STATUS Linux_InitContext(
611 MOS_OS_CONTEXT *context,
612 PMOS_CONTEXT osDriverContext)
613{
614 MOS_STATUS eStatus;
615 int32_t i = -1;
616
617 MOS_OS_FUNCTION_ENTER;
618
619 eStatus = MOS_STATUS_SUCCESS;
620
621 Linux_InitCmdBufferPool(context);
622
623 // Initialize GPU Status Buffer
624 eStatus = Linux_InitGPUStatus(context);
625 MOS_OS_CHK_STATUS_RETURN(eStatus);
626
627 for (i = 0; i < MOS_GPU_CONTEXT_MAX; i++)
628 {
629 context->OsGpuContext[i].pStartCB = nullptr;
630 context->OsGpuContext[i].pCurrentCB = nullptr;
631 context->OsGpuContext[i].bCBFlushed = true;
632 context->OsGpuContext[i].uiCommandBufferSize = COMMAND_BUFFER_SIZE;
633 context->OsGpuContext[i].pCB =
634 (PMOS_COMMAND_BUFFER)MOS_AllocAndZeroMemory(sizeof(MOS_COMMAND_BUFFER));
635
636 if (nullptr == context->OsGpuContext[i].pCB)
637 {
638 MOS_OS_ASSERTMESSAGE("No More Avaliable Memory");
639 Mos_Specific_ClearGpuContext(context);
640 return MOS_STATUS_NO_SPACE;
641 }
642
643 // each thread has its own GPU context, so do not need any lock as guarder here
644 context->OsGpuContext[i].pAllocationList =
645 (ALLOCATION_LIST*)MOS_AllocAndZeroMemory(sizeof(ALLOCATION_LIST) * ALLOCATIONLIST_SIZE);
646 if (nullptr == context->OsGpuContext[i].pAllocationList)
647 {
648 MOS_OS_ASSERTMESSAGE("context->OsGpuContext[%d].pAllocationList malloc failed.", i);
649 Mos_Specific_ClearGpuContext(context);
650 return MOS_STATUS_NO_SPACE;
651 }
652 context->OsGpuContext[i].uiMaxNumAllocations = ALLOCATIONLIST_SIZE;
653
654 context->OsGpuContext[i].pPatchLocationList =
655 (PATCHLOCATIONLIST*)MOS_AllocAndZeroMemory(sizeof(PATCHLOCATIONLIST) * PATCHLOCATIONLIST_SIZE);
656 if (nullptr == context->OsGpuContext[i].pPatchLocationList)
657 {
658 MOS_OS_ASSERTMESSAGE("context->OsGpuContext[%d].pPatchLocationList malloc failed.", i);
659 Mos_Specific_ClearGpuContext(context);
660 return MOS_STATUS_NO_SPACE;
661 }
662 context->OsGpuContext[i].uiMaxPatchLocationsize = PATCHLOCATIONLIST_SIZE;
663
664 context->OsGpuContext[i].pResources =
665 (PMOS_RESOURCE)MOS_AllocAndZeroMemory(sizeof(MOS_RESOURCE) * ALLOCATIONLIST_SIZE);
666 if (nullptr == context->OsGpuContext[i].pResources)
667 {
668 MOS_OS_ASSERTMESSAGE("context->OsGpuContext[%d].pResources malloc failed.", i);
669 Mos_Specific_ClearGpuContext(context);
670 return MOS_STATUS_NO_SPACE;
671 }
672
673 context->OsGpuContext[i].pbWriteMode =
674 (int32_t*)MOS_AllocAndZeroMemory(sizeof(int32_t) * ALLOCATIONLIST_SIZE);
675 if (nullptr == context->OsGpuContext[i].pbWriteMode)
676 {
677 MOS_OS_ASSERTMESSAGE("context->OsGpuContext[%d].pbWriteMode malloc failed.", i);
678 Mos_Specific_ClearGpuContext(context);
679 return MOS_STATUS_NO_SPACE;
680 }
681
682 context->OsGpuContext[i].uiGPUStatusTag = 1;
683 }
684
685
686 context->pTranscryptedKernels = nullptr;
687 context->uiTranscryptedKernelsSize = 0;
688
689 // For Media Memory compression
690 context->ppMediaMemDecompState = osDriverContext->ppMediaMemDecompState;
691 context->pfnMemoryDecompress = osDriverContext->pfnMemoryDecompress;
692 context->pfnMediaMemoryCopy = osDriverContext->pfnMediaMemoryCopy;
693 context->pfnMediaMemoryCopy2D = osDriverContext->pfnMediaMemoryCopy2D;
694
695 // Set interface functions
696 return eStatus;
697}

878MOS_STATUS Mos_DestroyInterface(PMOS_INTERFACE osInterface)
879{
880 MOS_OS_CHK_NULL_RETURN(osInterface);
881
882 MOS_STREAM_HANDLE streamState = osInterface->osStreamState;
883 MOS_OS_CHK_NULL_RETURN(streamState);
884
885 auto deviceContext = streamState->osDeviceContext;
886 MOS_OS_CHK_NULL_RETURN(deviceContext);
887 auto gpuContextMgr = deviceContext->GetGpuContextMgr();
888
889 if (!Mos_Solo_IsEnabled((PMOS_CONTEXT)streamState->perStreamParameters) && gpuContextMgr)
890 {
891 for (uint32_t i = 0; i < MOS_GPU_CONTEXT_MAX; i++)
892 {

899 continue;
900 }

902 osInterface->m_GpuContextHandleMap[i] = MOS_GPU_CONTEXT_INVALID_HANDLE;
903 }

905 }

907 if (osInterface->osCpInterface)

909 Delete_MosCpInterface(osInterface->osCpInterface);
910 osInterface->osCpInterface = nullptr;
911 }
912

914
915 if (perStreamParameters && perStreamParameters->bFreeContext)

917 perStreamParameters->m_skuTable.reset();

919 Mos_Specific_ClearGpuContext(perStreamParameters);
920
921 if (perStreamParameters->contextOffsetList.size())
922 {

924 perStreamParameters->contextOffsetList.shrink_to_fit();
925 }

928 {
929 Linux_ReleaseCmdBufferPool(perStreamParameters);
930 PCOMMAND_BUFFER pCurrCB = nullptr;
931 PCOMMAND_BUFFER pNextCB = nullptr;
932 for (uint32_t i = 0; i < MOS_GPU_CONTEXT_MAX; i++)
933 {
934 MOS_FreeMemAndSetNull(perStreamParameters->OsGpuContext[i].pCB);
935
936 pCurrCB = perStreamParameters->OsGpuContext[i].pStartCB;
937 for (; (pCurrCB); pCurrCB = pNextCB)
938 {
939 pNextCB = pCurrCB->pNext;
940 MOS_FreeMemAndSetNull(pCurrCB);
941 }
942 }
943 Linux_ReleaseGPUStatus(perStreamParameters);
944 }
945 if (perStreamParameters->intel_context)
946 {
947 if (perStreamParameters->intel_context->vm_id != INVALID_VM)
948 {
949 mos_vm_destroy(perStreamParameters->intel_context->bufmgr, perStreamParameters->intel_context->vm_id);
950 perStreamParameters->intel_context->vm_id = INVALID_VM;
951 }
952 mos_context_destroy(perStreamParameters->intel_context);
953 perStreamParameters->intel_context = nullptr;
954 }
955 MOS_Delete(perStreamParameters);
956 streamState->perStreamParameters = nullptr;
957 }
958
959 MosInterface::DestroyVirtualEngineState(streamState);
960 MOS_FreeMemAndSetNull(osInterface->pVEInterf);
961
962 MOS_OS_CHK_STATUS_RETURN(MosInterface::DestroyOsStreamState(streamState));
963 osInterface->osStreamState = nullptr;

1286void *Mos_Specific_LockResource(
1287 PMOS_INTERFACE osInterface,
1288 PMOS_RESOURCE osResource,
1289 PMOS_LOCK_PARAMS lockFlags)
1290{
1291 MOS_OS_CONTEXT *context = nullptr;
1292
1293 MOS_OS_FUNCTION_ENTER;
1294 MOS_OS_CHK_NULL_RETURN_VALUE(osInterface, nullptr);
1295 MOS_OS_CHK_NULL_RETURN_VALUE(osResource, nullptr);
1296 if(!Mos_Solo_IsEnabled(nullptr))
1297 {
1298 return MosInterface::LockMosResource(osInterface->osStreamState, osResource, lockFlags);
1299 }
1300
1301 context = osInterface->pOsContext;
1302 MOS_OS_CHK_NULL_RETURN_VALUE(context, nullptr);
1303 MOS_OS_CHK_NULL_RETURN_VALUE(osResource->bo, nullptr);
1304 MOS_OS_CHK_NULL_RETURN_VALUE(osResource->pGmmResInfo, nullptr);
1305
1306 {
1307 MOS_LINUX_BO *bo = osResource->bo;
1308 GMM_RESOURCE_FLAG gmmFlags = {};
1309
1310 gmmFlags = osResource->pGmmResInfo->GetResFlags();
1311
1312 // Do decompression for a compressed surface before lock
1313 if (!lockFlags->NoDecompress &&
1314 (((gmmFlags.Gpu.MMC || gmmFlags.Gpu.CCS) && gmmFlags.Gpu.UnifiedAuxSurface)||
1315 osResource->pGmmResInfo->IsMediaMemoryCompressed(0)))
1316 {
1317 if(context->pfnMemoryDecompress)
1318 {
1319 context->pfnMemoryDecompress(context, osResource);
1320 }
1321 }
1322 if(osResource->bMapped)
1323 {
1324 return osResource->pData;
1325 }
1326
1327 if (osResource->TileType != MOS_TILE_LINEAR && !lockFlags->TiledAsTiled)
1328 {
1329 if (context->bUseSwSwizzling)
1330 {
1331 mos_bo_map(bo, (OSKM_LOCKFLAG_WRITEONLY&lockFlags->WriteOnly));
1332 osResource->MmapOperation = MOS_MMAP_OPERATION_MMAP;
1333 if (osResource->pSystemShadow == nullptr)
1334 {
1335 osResource->pSystemShadow = (uint8_t *)MOS_AllocMemory(bo->size);
1336 MOS_OS_CHECK_CONDITION((osResource->pSystemShadow == nullptr), "Failed to allocate shadow surface", nullptr);
1337 }
1338 if (osResource->pSystemShadow)
1339 {
1340 int32_t flags = context->bTileYFlag ? 0 : 1;
1341 MOS_OS_CHECK_CONDITION((osResource->TileType != MOS_TILE_Y), "Unsupported tile type", nullptr);
1342 MOS_OS_CHECK_CONDITION((bo->size <= 0 || osResource->iPitch <= 0), "Invalid BO size or pitch", nullptr);
1343 Mos_SwizzleData((uint8_t*)bo->virt, osResource->pSystemShadow,
1344 MOS_TILE_Y, MOS_TILE_LINEAR, bo->size / osResource->iPitch, osResource->iPitch, flags);
1345 }
1346 }
1347 else
1348 {
1349 mos_bo_map_gtt(bo);
1350 osResource->MmapOperation = MOS_MMAP_OPERATION_MMAP_GTT;
1351 }
1352 }
1353 else if (lockFlags->Uncached)
1354 {
1355 mos_bo_map_wc(bo);
1356 osResource->MmapOperation = MOS_MMAP_OPERATION_MMAP_WC;
1357 }
1358 else
1359 {
1360 mos_bo_map(bo, (OSKM_LOCKFLAG_WRITEONLY&lockFlags->WriteOnly));
1361 osResource->MmapOperation = MOS_MMAP_OPERATION_MMAP;
1362 }
1363 osResource->pData = osResource->pSystemShadow ? osResource->pSystemShadow : (uint8_t*)bo->virt;
1364 osResource->bMapped = true;
1365 }
1366 return osResource->pData;
1367}
