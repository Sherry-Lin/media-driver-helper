69void DdiMediaUtil_MediaPrintFps()
70{
71 struct timeval tv2;
72
73 if (0 == vaFpsSampleSize)
74 {
75 return;
76 }
77 gettimeofday(&tv2, 0);
78
79 pthread_mutex_lock(&fpsMutex);
80 if (-1 == frameCountFps)
81 {
82 gettimeofday(&tv1, 0);
83 }
84
85 if (++frameCountFps >= vaFpsSampleSize)
86 {
87 char fpsFileName[LENGTH_OF_FPS_FILE_NAME];
88 FILE *fp = nullptr;
89 char temp[LENGTH_OF_FPS_FILE_NAME];
90
91 int64_t diff = (tv2.tv_sec - tv1.tv_sec)*1000000 + tv2.tv_usec - tv1.tv_usec;
92 float fps = frameCountFps / (diff / 1000000.0);
93 DDI_NORMALMESSAGE("FPS:%6.4f, Interval:%11lu.", fps,((uint64_t)tv2.tv_sec)*1000 + (tv2.tv_usec/1000));
94 sprintf(temp,"FPS:%6.4f, Interval:%11lu\n", fps,((uint64_t)tv2.tv_sec)*1000 + (tv2.tv_usec/1000));
95
96 MOS_ZeroMemory(fpsFileName,LENGTH_OF_FPS_FILE_NAME);
97 sprintf(fpsFileName, FPS_FILE_NAME);
98 if ((fp = fopen(fpsFileName, "wb")) == nullptr)
99 {
100 pthread_mutex_unlock(&fpsMutex);
101 DDI_ASSERTMESSAGE("Unable to open fps file.");
102 }
103
104 fwrite(temp, 1, strlen(temp), fp);
105 fclose(fp);
106 frameCountFps = -1;
107 }
108 pthread_mutex_unlock(&fpsMutex);
109}

624 {

626 {

652 {
653 case I915_TILING_Y:

656 if (MEDIA_IS_SKU(&mediaDrvCtx->SkuTable, FtrE2ECompression) &&

660 bMemCompEnable)

691 gmmParams.Flags.Info.MediaCompressed = 0;
692 gmmParams.Flags.Info.RenderCompressed = 0;
693 gmmParams.Flags.Gpu.CCS = 0;
694 gmmParams.Flags.Gpu.UnifiedAuxSurface = 0;

764 pitch = ulPitch;
765 }
