91BufmgrPrelim *BufmgrPrelim::CreatePrelim(int fd)
92{
93 if (fd < 0) {
94 return nullptr;
95 }
96
97 struct drm_i915_query_item item{};
98 BufmgrPrelim *prelim = nullptr;
99
100 memclear(item);
101 item.query_id = PRELIM_DRM_I915_QUERY_MEMORY_REGIONS;
102
103 struct drm_i915_query q;
104
105 memclear(q);
106 q.num_items = 1;
107 q.items_ptr = (uintptr_t)&item;
108
109 if (0 == drmIoctl(fd, DRM_IOCTL_I915_QUERY, &q) && item.length > 0) {
110 prelim = new BufmgrPrelim(fd);
111 if (nullptr != prelim) {
112 m_prelimEnabled = true;
113 }
114 }
115
116 return prelim;

141int BufmgrPrelim::Init(struct mos_bufmgr &bufmgr, bool hasLmem)
142{
143 uint32_t nengine = 8;
144 struct i915_engine_class_instance uengines[nengine];
145 memset(uengines, 0, sizeof(uengines));
146
147 m_tileId = GetTileId();
148
149 bufmgr.get_reserved = &m_tileId;
150 int ret = QueryEngines(hasLmem, I915_ENGINE_CLASS_VIDEO, PRELIM_I915_VIDEO_CLASS_CAPABILITY_VDENC, &nengine, uengines);

155 bufmgr.has_full_vd = true;
156 }
157 else
158 {
159 bufmgr.has_full_vd = false;
160 }
161 }
162
163 return ret;

237uint32_t BufmgrPrelim::GetTileId()
238{
239 uint8_t lmemRegions = 0;
240 uint32_t tileId = 0;
241 lmemRegions = GetLmemRegionCount();
242
243 if (lmemRegions > 0)
244 {
245 // generate an instance index by random
246 struct timespec currentTime;
247 clock_gettime(CLOCK_MONOTONIC, &currentTime);
248 srand(currentTime.tv_nsec);
249 uint8_t instance = rand() % lmemRegions;
250 tileId = instance;
251
252 // get tile setting from environment variable
253 char *tileInstance = getenv("INTEL_TILE_INSTANCE");
254 if (tileInstance != nullptr)
255 {
256 errno = 0;
257 long int instance = strtol(tileInstance, nullptr, TYPE_DECIMAL);
258 /* Check for various possible errors */
259 if ((errno == ERANGE && (instance == LONG_MAX || instance == LONG_MIN)) ||
260 (errno != 0 && instance == 0))
261 {
262 fprintf(stderr, "Invalid INTEL_TILE_INSTANCE setting.(%d)\n", errno);
263 }
264 else
265 {
266 // valid instance value should be 0, 1, ..., lmemRegions-1
267 if (instance >= 0 && instance < lmemRegions)
268 {
269 tileId = instance;
270 }
271 else
272 {
273 fprintf(stderr, "Invalid tile instance provided by user, will use default tile.\n");
274 }
275 }
276 }
277 }
278
279 return tileId;
280}

331 memclear(create);

338 outHandle = create.handle;

371{
372 assert(m_fd >= 0);
373 assert(nullptr != nengine);
374
375 struct drm_i915_query query;
376 struct drm_i915_query_item queryItem;
377 struct prelim_drm_i915_query_engine_info *engines = nullptr;
378 int ret, len;
379
380 memclear(queryItem);
381 queryItem.query_id = PRELIM_DRM_I915_QUERY_ENGINE_INFO;

402 queryItem.length = len;
403 queryItem.data_ptr = (uintptr_t)engines;
404 ret = QueryItems(m_fd, &queryItem, 1);
405
406 *nengine = ret ? 0 : engines->num_engines;
407
408 if (engines)
409 {
410 free(engines);
411 }
412 return ret;

415bool BufmgrPrelim::IsEngineSatisfyCaps(
416 struct prelim_drm_i915_engine_info *engine,
417 uint64_t caps)
418{
419 if (!(engine->flags & PRELIM_I915_ENGINE_INFO_HAS_KNOWN_CAPABILITIES) ||
420 !(engine->known_capabilities & PRELIM_I915_VIDEO_CLASS_CAPABILITY_VDENC))
421 caps &= ~PRELIM_I915_VIDEO_CLASS_CAPABILITY_VDENC;
422
423 return (caps & engine->capabilities) == caps;
424}

426int BufmgrPrelim::QueryEnginesInternal(uint16_t engine_class,
427 uint64_t caps,
428 unsigned int *nengine,
429 struct i915_engine_class_instance *ci,
430 std::map<uint16_t, uint16_t> &logicalMap)
431{
432 if(nengine == nullptr || ci == nullptr)
433 {
434 return -EINVAL;
435 }
436
437 assert(m_fd >= 0);
438
439 struct drm_i915_query query;
440 struct drm_i915_query_item item;
441 struct prelim_drm_i915_query_engine_info *engines = nullptr;
442 int ret, len;
443
444 memclear(item);
445 item.query_id = PRELIM_DRM_I915_QUERY_ENGINE_INFO;
446 item.length = 0;
447
448 GOTO_FINI_IF_NOT_0(ret = QueryItems(m_fd, &item, 1));
449 len = item.length;
450 if (len <= 0)
451 {
452 goto fini;
453 }
454
455 engines = (prelim_drm_i915_query_engine_info *)malloc(len);
456 GOTO_FINI_IF_MALLOC_FAIL(engines);
457
458 memset(engines,0,len);
459 memclear(item);
460 item.query_id = PRELIM_DRM_I915_QUERY_ENGINE_INFO;
461 item.length = len;
462 item.data_ptr = (uintptr_t)engines;
463 GOTO_FINI_IF_NOT_0(ret = QueryItems(m_fd, &item, 1));
464
465 int i, num;
466 for (i = 0, num = 0; i < engines->num_engines; i++)
467 {
468 struct prelim_drm_i915_engine_info *engine = (struct prelim_drm_i915_engine_info *)&engines->engines[i];
469 if (engine_class == engine->engine.engine_class && IsEngineSatisfyCaps(engine, caps))
470 {
471 ci->engine_class = engine_class;
472 ci->engine_instance = engine->engine.engine_instance;
473
481 logicalMap[engine->engine.engine_instance] = engine->logical_instance;
482 ci++;
483 num++;
484 }
485 if (num > *nengine)
486 {
487 fprintf(stderr,"%s: Number of engine instances out of range, %d,%d\n",
488 __FUNCTION__, num, *nengine);
489 goto fini;
490 }
491 }
492 *nengine = num;
493
494fini:
495 if (engines)
496 free(engines);
497 return ret;
498}

500int BufmgrPrelim::QueryEngines(bool hasLmem,
501 uint16_t engineClass,
502 uint64_t caps,
503 unsigned int *nengine,
504 struct i915_engine_class_instance *ci)
505{
506 if((nengine == nullptr) || (ci == nullptr))
507 {
508 return -EINVAL;
509 }
510
511 uint32_t tileId = m_tileId;
512 struct i915_engine_class_instance *ciTemp = nullptr;
513 struct prelim_drm_i915_query_distance_info *distances = nullptr;
514 std::map<uint16_t, uint16_t> logicalMap;
515 logicalMap.clear();
516 unsigned int engineNum;
517
518 ciTemp = (struct i915_engine_class_instance *)malloc((*nengine) * sizeof(struct i915_engine_class_instance));
519 if (!ciTemp)
520 {
521 assert(ciTemp);
522 return -ENOMEM;
523 }
524
525 memset(ciTemp, 0, (*nengine) * sizeof(struct i915_engine_class_instance));
526
527 int ret = QueryEnginesInternal(engineClass,
528 caps,
529 nengine,
530 ciTemp,
531 logicalMap);
532 GOTO_FINI_IF_NOT_0(ret);
533
534 distances = (struct prelim_drm_i915_query_distance_info *)malloc((*nengine) * sizeof(struct prelim_drm_i915_query_distance_info));
535 GOTO_FINI_IF_MALLOC_FAIL(distances);
536
537 memset(distances, 0, (*nengine) * sizeof(struct prelim_drm_i915_query_distance_info));
538 for (int i = 0; i < *nengine; i++)
539 {
540 distances[i].engine = ciTemp[i];
541 distances[i].region.memory_class = hasLmem ? PRELIM_I915_MEMORY_CLASS_DEVICE : PRELIM_I915_MEMORY_CLASS_SYSTEM;
542 distances[i].region.memory_instance = tileId;
543 }
544 GOTO_FINI_IF_NOT_0(ret = QueryEnginesDistance(*nengine, distances));
545
546 //select the engine with 0 distance
547 int i, j;
548 for(i = 0, j = 0; i < *nengine; i++)
549 {
550 if(0 == distances[i].distance)
551 {
552 ci[j] = distances[i].engine;
553 j++;
554 }
555#if (_DEBUG || _RELEASE_INTERNAL)
556 else if (0 > distances[i].distance)
557 {
558 MOS_OS_NORMALMESSAGE("Engine = {engine_class = %hu, engine_instance = %hu}; Distance (error code) == %d",
559 distances[i].engine.engine_class, distances[i].engine.engine_instance, distances[i].distance);
560 }
561#endif
562 }
563 *nengine = j;
564
565 engineNum = *nengine;
566 if (engineClass == I915_ENGINE_CLASS_VIDEO && engineNum > 2)
567 {
568 std::map<uint16_t, uint16_t> reorderMap; //<logical id, uabi id>
569
570 for (i = 0; i < engineNum; i++)
571 {
572 uint16_t index = logicalMap[(ci + i)->engine_instance];
573 reorderMap[index] = (ci + i)->engine_instance;
574 }
575
576 i = 0;
577 for (auto iter = reorderMap.begin(); iter != reorderMap.end(); iter++)
578 {
579 (ci + i)->engine_instance = iter->second;
580 i++;
581 }
582 }
583
584fini:
585 if(ciTemp)
586 {
587 free(ciTemp);
588 }
589 if(distances)
590 {
591 free(distances);
592 }
593
594 return ret;
595}

597int BufmgrPrelim::SetContextParamParallel(struct mos_linux_context *ctx,
598 struct i915_engine_class_instance *ci,
599 unsigned int count)
600{
601 if((ctx == nullptr) || (ci == nullptr) || (count <= 0))
602 {
603 return -EINVAL;
604 }
605
606 int ret = 0;
607 uint32_t size = 0;
608 int n;
609 struct i915_context_engines_parallel_submit* parallel_submit = nullptr;
610 struct i915_context_param_engines* set_engines = nullptr;
611
612 size = sizeof(struct i915_context_engines_parallel_submit) + count * sizeof(*ci);
613 parallel_submit = (struct i915_context_engines_parallel_submit*)malloc(size);
614 GOTO_FINI_IF_MALLOC_FAIL(parallel_submit);
615
616 memset(parallel_submit, 0, size);
617 parallel_submit->base.name = PRELIM_I915_CONTEXT_ENGINES_EXT_PARALLEL2_SUBMIT;
618 parallel_submit->engine_index = 0;
619 parallel_submit->width = count;
620 parallel_submit->num_siblings = 1;
621 for(int i = 0; i < count; i++)
622 {
623 parallel_submit->engines[i] = ci[i];
624 }
625
626 /* I915_DEFINE_CONTEXT_PARAM_ENGINES */
627 size = sizeof(struct i915_context_param_engines) + sizeof(struct i915_engine_class_instance);
628 set_engines = (struct i915_context_param_engines*) malloc(size);
629 GOTO_FINI_IF_MALLOC_FAIL(set_engines);
630
631 set_engines->extensions = (uintptr_t)(parallel_submit);
632 set_engines->engines[0].engine_class = I915_ENGINE_CLASS_INVALID;
633 set_engines->engines[0].engine_instance = I915_ENGINE_CLASS_INVALID_NONE;
634
635 ret = mos_set_context_param(ctx,
636 size,
637 I915_CONTEXT_PARAM_ENGINES,
638 (uintptr_t)set_engines);
639fini:
640 if (set_engines)
641 free(set_engines);
642 if (parallel_submit)
643 free(parallel_submit);
644 return ret;
645}

647int BufmgrPrelim::QueryDeviceBlob(int fd, MEDIA_SYSTEM_INFO* gfxInfo)
648{
649 uint32_t *hwInfo;
650 uint32_t ulength= 0;
651 int ret, i;
652 struct drm_i915_query_item item;
653
654 hwInfo = nullptr;
655
656 memclear(item);
657 item.length = 0;
658 item.query_id = PRELIM_DRM_I915_QUERY_HWCONFIG_TABLE;
659 ret = QueryItems(fd, &item, 1);
660 if (ret != 0 || item.length <= 0)
661 {
662 return (ret != 0) ? ret : -1;
663 }
664
665 hwInfo = (uint32_t*) malloc(item.length);
666 if (hwInfo != nullptr)
667 memset(hwInfo, 0, item.length);
668 else
669 return -ENOMEM;
670 item.data_ptr = (uintptr_t) hwInfo;
671 ret = QueryItems(fd, &item, 1);
672 if (ret != 0 || item.length <= 0)
673 {
674 return (ret != 0) ? ret : -1;
675 }
676 ulength = item.length / sizeof(uint32_t);
677 i = 0;
678 assert(gfxInfo);
679 while (i < ulength) {
680 /* Attribute ID starts with 1 */
681 assert(hwInfo[i] > 0);
682
683 switch (hwInfo[i])
684 {
685 case INTEL_HWCONFIG_MAX_SLICES_SUPPORTED:
686 {
687 assert(hwInfo[i+1] == 1);
688 gfxInfo->SliceCount = hwInfo[i+2];
689 gfxInfo->MaxSlicesSupported = hwInfo[i+2];
690 }
691
692 case INTEL_HWCONFIG_MAX_DUAL_SUBSLICES_SUPPORTED:
693 {
694 assert(hwInfo[i+1] == 1);
695 gfxInfo->SubSliceCount = hwInfo[i+2];
696 gfxInfo->MaxSubSlicesSupported = hwInfo[i+2];
697 }
698
699 case INTEL_HWCONFIG_MAX_NUM_EU_PER_DSS:
700 {
701 assert(hwInfo[i+1] == 1);
702 gfxInfo->MaxEuPerSubSlice = hwInfo[i+2];
703 }
704
705 case INTEL_HWCONFIG_NUM_THREADS_PER_EU:
706 {
707 assert(hwInfo[i+1] == 1);
708 gfxInfo->NumThreadsPerEu = hwInfo[i+2];
709 }
710
711 case INTEL_HWCONFIG_MAX_VECS:
712 {
713 assert(hwInfo[i+1] == 1);
714 gfxInfo->MaxVECS = hwInfo[i+2];
715 }
716
717 default:
718 break;
719 }
720
721 /* Advance to next key */
722 i += hwInfo[i + 1]; // value size
723 i += 2;// KL size
724 }
725
726 if (hwInfo != nullptr)
727 free(hwInfo);
728
729 return ret;
730}
