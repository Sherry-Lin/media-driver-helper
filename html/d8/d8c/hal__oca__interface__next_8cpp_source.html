131void HalOcaInterfaceNext::On1stLevelBBEnd(MOS_COMMAND_BUFFER &cmdBuffer, MOS_INTERFACE &osInterface)
132{
133 if (nullptr == osInterface.pOsContext)
134 {
135 OnOcaError(nullptr, MOS_STATUS_INVALID_PARAMETER, __FUNCTION__, __LINE__);
136 return;
137 }
138 MOS_CONTEXT & mosContext = *osInterface.pOsContext;
139 MosOcaInterface *pOcaInterface = &MosOcaInterfaceSpecific::GetInstance();
140 MOS_OCA_BUFFER_HANDLE ocaBufHandle = 0;
141 MOS_STATUS status = MOS_STATUS_SUCCESS;
142
143 if (nullptr == pOcaInterface || !((MosOcaInterfaceSpecific*)pOcaInterface)->IsOcaEnabled())
144 {
145 // Will come here for UMD_OCA not being enabled case.
146 return;
147 }
148 if ((ocaBufHandle = GetOcaBufferHandle(cmdBuffer, mosContext)) == MOS_OCA_INVALID_BUFFER_HANDLE)
149 {
150 // May come here for workloads not enabling UMD_OCA.
151 return;
152 }
153
154 status = pOcaInterface->On1stLevelBBEnd(ocaBufHandle, osInterface.pOsContext);
155 if (MOS_FAILED(status))
156 {
157 // UnlockOcaBuf and RemoveOcaBufferHandle should also be called
158 // for failure case to avoid oca buffer leak.
159 OnOcaError(osInterface.pOsContext, status, __FUNCTION__, __LINE__);
160 }
161 status = pOcaInterface->UnlockOcaBufferWithDelay(ocaBufHandle);
162 if (MOS_FAILED(status))
163 {
164 OnOcaError(osInterface.pOsContext, status, __FUNCTION__, __LINE__);
165 }
166
167 // remove the oca buffer handle
168 RemoveOcaBufferHandle(cmdBuffer, mosContext);
169}

495void HalOcaInterfaceNext::AddRTLogReource(MOS_COMMAND_BUFFER &cmdBuffer,
496 MOS_CONTEXT &mosContext,
497 MOS_INTERFACE &osInterface)
498{
499 if (!MosOcaRTLogMgr::IsOcaRTLogEnabled())
500 {
501 return;
502 }
503 MOS_STATUS status = MOS_STATUS_SUCCESS;
504 PMOS_RESOURCE osResource = nullptr;
505 uint32_t size = 0;
506 MosInterface::GetRtLogResourceInfo(osInterface.osStreamState, osResource, size);
507
508 MOS_LINUX_BO *bo = cmdBuffer.OsResource.bo;
509 OCA_LOG_SECTION_HEADER *header = (OCA_LOG_SECTION_HEADER *)((uint64_t)bo->virt + bo->size - OCA_LOG_SECTION_SIZE_MAX);
510 if (header->magicNum != OCA_LOG_SECTION_MAGIC_NUMBER)
511 {
512 return;
513 }
514 MHW_RESOURCE_PARAMS ResourceParams = {};
515
516 ResourceParams.dwOffset = 0;
517 uint32_t cmd = 0;
518 ResourceParams.presResource = osResource;
519 ResourceParams.pdwCmd = &cmd;
520 // Align logic in Mhw_AddResourceToCmd
521 ResourceParams.dwLocationInCmd = ((int32_t)((uint8_t*)&header->rtlogPatchAddr - (uint8_t*)cmdBuffer.pCmdBase) - cmdBuffer.iOffset) / sizeof(uint32_t);
522 ResourceParams.HwCommandType = MOS_OCA_RESERVED;
523 ResourceParams.dwSharedMocsOffset = 1 - ResourceParams.dwLocationInCmd;
524
525 if (osInterface.bUsesGfxAddress)
526 {
527 status = Mhw_AddResourceToCmd_GfxAddress(&osInterface, &cmdBuffer, &ResourceParams);
528 }
529 else
530 {
531 status = Mhw_AddResourceToCmd_PatchList(&osInterface, &cmdBuffer, &ResourceParams);
532 }
533
534 if (MOS_FAILED(status))
535 {
536 OnOcaError(&mosContext, status, __FUNCTION__, __LINE__);
537 }
538 HalOcaInterfaceNext::OnIndirectState(cmdBuffer, mosContext, osResource, 0, false, size);
539}

550MOS_OCA_BUFFER_HANDLE HalOcaInterfaceNext::GetOcaBufferHandle(MOS_COMMAND_BUFFER &cmdBuffer, MOS_CONTEXT &mosContext)
551{
552 MosOcaInterface *pOcaInterface = &MosOcaInterfaceSpecific::GetInstance();
553 if(pOcaInterface == nullptr)
554 {
555 OnOcaError(&mosContext, MOS_STATUS_NULL_POINTER, __FUNCTION__, __LINE__);
556 return MOS_OCA_INVALID_BUFFER_HANDLE;
557 }
558 PMOS_MUTEX mutex = pOcaInterface->GetMutex();
559 if (mutex == nullptr)
560 {
561 return MOS_OCA_INVALID_BUFFER_HANDLE;
562 }
563 MosOcaAutoLock autoLock(mutex);
564 std::map<uint32_t*, MOS_OCA_BUFFER_HANDLE>::iterator it = s_hOcaMap.find(cmdBuffer.pCmdBase);
565 if (it == s_hOcaMap.end())
566 {
567 // May come here for workloads not enabling UMD_OCA.
568 return MOS_OCA_INVALID_BUFFER_HANDLE;
569 }
570 if (it->second >= MAX_NUM_OF_OCA_BUF_CONTEXT || it->second < 0)
571 {
572 OnOcaError(&mosContext, MOS_STATUS_NULL_POINTER, __FUNCTION__, __LINE__);
573 return MOS_OCA_INVALID_BUFFER_HANDLE;
574 }
575 return it->second;
576}

612void HalOcaInterfaceNext::On1stLevelBBStart(MOS_COMMAND_BUFFER &cmdBuffer, MOS_CONTEXT &mosContext,
613 uint32_t gpuContextHandle)
614{
615 MosInterface::SetObjectCapture(&cmdBuffer.OsResource);
616 MOS_STATUS status = MOS_STATUS_SUCCESS;
617 MosOcaInterface *pOcaInterface = &MosOcaInterfaceSpecific::GetInstance();
618 MOS_OCA_BUFFER_HANDLE ocaBufHandle = 0;
619 uint64_t ocaBase = 0;
620
621 if (nullptr == pOcaInterface || !((MosOcaInterfaceSpecific*)pOcaInterface)->IsOcaEnabled())
622 {
623 return;
624 }
625 PMOS_MUTEX mutex = pOcaInterface->GetMutex();
626 if (mutex == nullptr)
627 {
628 OnOcaError(&mosContext, MOS_STATUS_NULL_POINTER, __FUNCTION__, __LINE__);
629 return;
630 }
631 ocaBufHandle = GetOcaBufferHandle(cmdBuffer, mosContext);
632 if (ocaBufHandle != MOS_OCA_INVALID_BUFFER_HANDLE)
633 {
634 // will come here when On1stLevelBBStart being called twice without On1stLevelBBEnd being called.
635 OnOcaError(&mosContext, MOS_STATUS_INVALID_PARAMETER, __FUNCTION__, __LINE__);
636 return;
637 }
638 else
639 {
640 MosOcaAutoLock autoLock(mutex);
641 ocaBufHandle = pOcaInterface->LockOcaBufAvailable(&mosContext, gpuContextHandle);
642 if (MOS_OCA_INVALID_BUFFER_HANDLE == ocaBufHandle)
643 {
644 OnOcaError(&mosContext, MOS_STATUS_INVALID_HANDLE, __FUNCTION__, __LINE__);
645 return;
646 }
647 auto success = s_hOcaMap.insert(std::make_pair(cmdBuffer.pCmdBase, ocaBufHandle));
648 if (!success.second)
649 {
650 // Should never come to here.
651 MOS_OS_ASSERTMESSAGE("ocaBufHandle has already been assigned to current cmdBuffer!");
652 OnOcaError(&mosContext, MOS_STATUS_INVALID_HANDLE, __FUNCTION__, __LINE__);
653 return;
654 }
655 }
656
657 status = pOcaInterface->On1stLevelBBStart(ocaBase, ocaBufHandle, &mosContext, &cmdBuffer.OsResource, 0, true, 0);
658 if (MOS_FAILED(status))
659 {
660 RemoveOcaBufferHandle(cmdBuffer, mosContext);
661 OnOcaError(&mosContext, status, __FUNCTION__, __LINE__);
662 }
663}
