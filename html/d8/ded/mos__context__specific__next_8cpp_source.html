62MOS_STATUS OsContextSpecificNext::Init(DDI_DEVICE_CONTEXT ddiDriverContext)
63{
64 uint32_t iDeviceId = 0;
65 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
66 uint32_t value = 0;
67 MediaUserSettingSharedPtr userSettingPtr = nullptr;
68
69 MOS_OS_FUNCTION_ENTER;
70
71 eStatus = MOS_STATUS_SUCCESS;
72
73 PMOS_CONTEXT osDriverContext = (PMOS_CONTEXT)ddiDriverContext;
74
75 if (GetOsContextValid() == false)
76 {
77 m_skuTable.reset();
78 m_waTable.reset();
79 MosUtilities::MosZeroMemory(&m_platformInfo, sizeof(m_platformInfo));
80 MosUtilities::MosZeroMemory(&m_gtSystemInfo, sizeof(m_gtSystemInfo));
81
82 if( nullptr == osDriverContext ||
83 0 > osDriverContext->fd )
84 {
85 MOS_OS_ASSERT(false);
86 return MOS_STATUS_INVALID_HANDLE;
87 }
88 m_fd = osDriverContext->fd;
89
90 userSettingPtr = MosInterface::MosGetUserSettingInstance(osDriverContext);
91
92 m_bufmgr = mos_bufmgr_gem_init(m_fd, BATCH_BUFFER_SIZE, &m_deviceType);
93 if (nullptr == m_bufmgr)
94 {
95 MOS_OS_ASSERTMESSAGE("Not able to allocate buffer manager, fd=0x%d", m_fd);
96 return MOS_STATUS_INVALID_PARAMETER;
97 }
98 mos_bufmgr_enable_reuse(m_bufmgr);
99
100 osDriverContext->bufmgr = m_bufmgr;
101
102 //Latency reducation:replace HWGetDeviceID to get device using ioctl from drm.
103 iDeviceId = mos_bufmgr_get_devid(m_bufmgr);
104 m_isAtomSOC = IS_ATOMSOC(iDeviceId);
105
106 eStatus = NullHwInit((MOS_CONTEXT_HANDLE)osDriverContext);
107 if (!GetNullHwIsEnabled())
108 {
109 eStatus = HWInfo_GetGfxInfo(m_fd, m_bufmgr, &m_platformInfo, &m_skuTable, &m_waTable, &m_gtSystemInfo, userSettingPtr);
110 }
111 else
112 {
113 m_platformInfo = osDriverContext->m_platform;
114 m_skuTable = osDriverContext->m_skuTable;
115 m_waTable = osDriverContext->m_waTable;
116 m_gtSystemInfo = osDriverContext->m_gtSystemInfo;
117 iDeviceId = osDriverContext->iDeviceId;
118 }
119
120 if (eStatus != MOS_STATUS_SUCCESS)
121 {
122 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Sku/Wa/GtSystemInfo initialization");
123 return eStatus;
124 }
125
126 ReadUserSetting(
127 userSettingPtr,
128 value,
129 __MEDIA_USER_FEATURE_VALUE_ENABLE_SOFTPIN,
130 MediaUserSetting::Group::Device);
131
132 if (value)
133 {
134 bool softpin_va1Malign = false;
135 if (MEDIA_IS_SKU(&m_skuTable, Ftr1MGranularAuxTable))
136 {
137 softpin_va1Malign = true;
138 }
139
140 mos_bufmgr_enable_softpin(m_bufmgr, softpin_va1Malign);
141
142 ReadUserSetting(
143 userSettingPtr,
144 value,
145 __MEDIA_USER_FEATURE_VALUE_ENABLE_VM_BIND,
146 MediaUserSetting::Group::Device);
147
148 if (value)
149 {
150 mos_bufmgr_enable_vmbind(m_bufmgr);
151 MOS_OS_NORMALMESSAGE("mos_bufmg_enable_vmbind");
152 }
153 }
154
155 uint64_t isRecoverableContextEnabled = 0;
156 MOS_LINUX_CONTEXT *intel_context = mos_context_create_ext(m_bufmgr, 0, false);
157 int ret = mos_get_context_param(intel_context, 0, DRM_CONTEXT_PARAM_RECOVERABLE, &isRecoverableContextEnabled);
158 if (ret == -EINVAL)
159 {
160 isRecoverableContextEnabled = 1;
161 }
162 if (intel_context)
163 {
164 mos_context_destroy(intel_context);
165 }
166 // set recoverablecontext disabled if want disable object capture
167 if (MEDIA_IS_WA(&m_waTable, WaDisableSetObjectCapture) && isRecoverableContextEnabled)
168 {
169 mos_bufmgr_disable_object_capture(m_bufmgr);
170 }
171
172 if (MEDIA_IS_SKU(&m_skuTable, FtrEnableMediaKernels) == 0)
173 {
174 MEDIA_WR_WA(&m_waTable, WaHucStreamoutOnlyDisable, 0);
175 }
176
177 MosUtilities::MosTraceSetupInfo(
178 (VA_MAJOR_VERSION << 16) | VA_MINOR_VERSION,
179 m_platformInfo.eProductFamily,
180 m_platformInfo.eRenderCoreFamily,
181 (m_platformInfo.usRevId << 16) | m_platformInfo.usDeviceID);
182
183 GMM_SKU_FEATURE_TABLE gmmSkuTable = {};
184 GMM_WA_TABLE gmmWaTable = {};
185 GMM_GT_SYSTEM_INFO gmmGtInfo = {};
186 GMM_ADAPTER_BDF gmmAdapterBDF = {};
187 eStatus = HWInfo_GetGmmInfo(m_bufmgr, &gmmSkuTable, &gmmWaTable, &gmmGtInfo);
188 if (MOS_STATUS_SUCCESS != eStatus)
189 {
190 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Gmm Sku/Wa/GtSystemInfo initialization");
191 return eStatus;
192 }
193
194 eStatus = MosInterface::GetAdapterBDF(osDriverContext, &gmmAdapterBDF);
195 if (MOS_STATUS_SUCCESS != eStatus)
196 {
197 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Gmm Adapter BDF initialization");
198 return eStatus;
199 }
200
201 // Initialize Gmm context
202 GMM_INIT_IN_ARGS gmmInitAgrs = {};
203 GMM_INIT_OUT_ARGS gmmOutArgs = {};
204 gmmInitAgrs.Platform = m_platformInfo;
205 gmmInitAgrs.pSkuTable = &gmmSkuTable;
206 gmmInitAgrs.pWaTable = &gmmWaTable;
207 gmmInitAgrs.pGtSysInfo = &gmmGtInfo;
208 gmmInitAgrs.FileDescriptor = gmmAdapterBDF.Data;
209 gmmInitAgrs.ClientType = (GMM_CLIENT)GMM_LIBVA_LINUX;
210
211 GMM_STATUS status = InitializeGmm(&gmmInitAgrs, &gmmOutArgs);
212 if (status != GMM_SUCCESS)
213 {
214 MOS_OS_ASSERTMESSAGE("Fatal error - InitializeGmm fail.");
215 return MOS_STATUS_INVALID_PARAMETER;
216 }
217 m_gmmClientContext = gmmOutArgs.pGmmClientContext;
218
219 m_auxTableMgr = AuxTableMgr::CreateAuxTableMgr(m_bufmgr, &m_skuTable, m_gmmClientContext);
220
221#if (_DEBUG || _RELEASE_INTERNAL)
222 ReadUserSettingForDebug(
223 userSettingPtr,
224 osDriverContext->bSimIsActive,
225 __MEDIA_USER_FEATURE_VALUE_SIM_ENABLE,
226 MediaUserSetting::Group::Device);
227#endif
228
229 m_useSwSwizzling = osDriverContext->bSimIsActive || MEDIA_IS_SKU(&m_skuTable, FtrUseSwSwizzling);
230
231 m_tileYFlag = MEDIA_IS_SKU(&m_skuTable, FtrTileY);
232
233 m_use64BitRelocs = true;
234
235 if (!GetNullHwIsEnabled())
236 {
237 osDriverContext->iDeviceId = iDeviceId;
238 osDriverContext->m_skuTable = m_skuTable;
239 osDriverContext->m_waTable = m_waTable;
240 osDriverContext->m_gtSystemInfo = m_gtSystemInfo;
241 osDriverContext->m_platform = m_platformInfo;
242 }
243 osDriverContext->pGmmClientContext = m_gmmClientContext;
244 osDriverContext->m_auxTableMgr = m_auxTableMgr;
245 osDriverContext->bUseSwSwizzling = m_useSwSwizzling;
246 osDriverContext->bTileYFlag = m_tileYFlag;
247 osDriverContext->bIsAtomSOC = m_isAtomSOC;
248 osDriverContext->m_osDeviceContext = this;
249
250 m_usesPatchList = true;
251 m_usesGfxAddress = false;
252
253 SetOsContextValid(true);
254 // Prepare the command buffer manager
255 m_cmdBufMgr = CmdBufMgrNext::GetObject();
256 MOS_OS_CHK_NULL_RETURN(m_cmdBufMgr);
257 MOS_OS_CHK_STATUS_RETURN(m_cmdBufMgr->Initialize(this, COMMAND_BUFFER_SIZE/2));
258
259 // Prepare the gpu Context manager
260 m_gpuContextMgr = GpuContextMgrNext::GetObject(this);
261 MOS_OS_CHK_NULL_RETURN(m_gpuContextMgr);
262
263 m_perfData = (PERF_DATA*)MOS_AllocAndZeroMemory(sizeof(PERF_DATA));
264 MOS_OS_CHK_NULL_RETURN(m_perfData);
265 osDriverContext->pPerfData = m_perfData;
266
267 //It must be done with m_gpuContextMgr ready. Insides it will create gpu context.
268#ifdef _MMC_SUPPORTED
269 m_mosDecompression = MOS_New(MosDecompression, osDriverContext);
270 MOS_OS_CHK_NULL_RETURN(m_mosDecompression);
271 osDriverContext->ppMediaMemDecompState = m_mosDecompression->GetMediaMemDecompState();
272 MOS_OS_CHK_NULL_RETURN(osDriverContext->ppMediaMemDecompState);
273 if (*osDriverContext->ppMediaMemDecompState == nullptr)
274 {
275 MOS_OS_NORMALMESSAGE("Decomp state creation failed");
276 }
277#endif
278 m_mosMediaCopy = MOS_New(MosMediaCopy, osDriverContext);
279 if (nullptr == m_mosMediaCopy)
280 {
281 MOS_OS_NORMALMESSAGE("m_mosMediaCopy creation failed");
282 }
283 else
284 {
285 osDriverContext->ppMediaCopyState = (void **)m_mosMediaCopy->GetMediaCopyState();
286 if ((nullptr == osDriverContext->ppMediaCopyState) || (nullptr == *osDriverContext->ppMediaCopyState))
287 {
288 MOS_OS_NORMALMESSAGE("Media Copy state creation failed");
289 }
290 }
291 }
292 MosOcaRTLogMgr::RegisterContext(this, osDriverContext);
293 return eStatus;
294}

296void OsContextSpecificNext::Destroy()
297{
298 MOS_OS_FUNCTION_ENTER;
299
300 if (GetOsContextValid() == true)
301 {
302 if (m_auxTableMgr != nullptr)
303 {
304 MOS_Delete(m_auxTableMgr);
305 m_auxTableMgr = nullptr;
306 }
307
308 m_skuTable.reset();
309 m_waTable.reset();
310
311 mos_bufmgr_destroy(m_bufmgr);
312
313 // Delete Gmm context
314 GMM_INIT_OUT_ARGS gmmOutArgs = {};
315 gmmOutArgs.pGmmClientContext = m_gmmClientContext;
316 GmmAdapterDestroy(&gmmOutArgs);
317 m_gmmClientContext = nullptr;
318
319 SetOsContextValid(false);
320
321 if (m_perfData != nullptr)
322 {
323 MOS_FreeMemory(m_perfData);
324 m_perfData = nullptr;
325 }
326
327 if (m_mosMediaCopy != nullptr)
328 {
329 MOS_Delete(m_mosMediaCopy);
330 }
331 }
