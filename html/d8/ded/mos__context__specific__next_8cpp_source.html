63MOS_STATUS OsContextSpecificNext::Init(DDI_DEVICE_CONTEXT ddiDriverContext)
64{
65 uint32_t iDeviceId = 0;
66 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
67 uint32_t value = 0;
68 MediaUserSettingSharedPtr userSettingPtr = nullptr;
69
70 MOS_OS_FUNCTION_ENTER;
71
72 eStatus = MOS_STATUS_SUCCESS;
73
74 PMOS_CONTEXT osDriverContext = (PMOS_CONTEXT)ddiDriverContext;
75
76 if (GetOsContextValid() == false)
77 {
78 m_skuTable.reset();
79 m_waTable.reset();
80 MosUtilities::MosZeroMemory(&m_platformInfo, sizeof(m_platformInfo));
81 MosUtilities::MosZeroMemory(&m_gtSystemInfo, sizeof(m_gtSystemInfo));
82
83 if( nullptr == osDriverContext ||
84 0 >= osDriverContext->fd )
85 {
86 MOS_OS_ASSERT(false);
87 return MOS_STATUS_INVALID_HANDLE;
88 }
89 m_fd = osDriverContext->fd;
90
91 userSettingPtr = MosInterface::MosGetUserSettingInstance(osDriverContext);
92
93 m_bufmgr = mos_bufmgr_gem_init(m_fd, BATCH_BUFFER_SIZE);
94 if (nullptr == m_bufmgr)
95 {
96 MOS_OS_ASSERTMESSAGE("Not able to allocate buffer manager, fd=0x%d", m_fd);
97 return MOS_STATUS_INVALID_PARAMETER;
98 }
99 mos_bufmgr_gem_enable_reuse(m_bufmgr);
100
101 osDriverContext->bufmgr = m_bufmgr;
102
103 //Latency reducation:replace HWGetDeviceID to get device using ioctl from drm.
104 iDeviceId = mos_bufmgr_gem_get_devid(m_bufmgr);
105 m_isAtomSOC = IS_ATOMSOC(iDeviceId);
106
107 eStatus = NullHW::Init(osDriverContext);
108 if (!NullHW::IsEnabled())
109 {
110 eStatus = HWInfo_GetGfxInfo(m_fd, m_bufmgr, &m_platformInfo, &m_skuTable, &m_waTable, &m_gtSystemInfo, userSettingPtr);
111 }
112 else
113 {
114 m_platformInfo = osDriverContext->platform;
115 m_skuTable = osDriverContext->SkuTable;
116 m_waTable = osDriverContext->WaTable;
117 m_gtSystemInfo = osDriverContext->gtSystemInfo;
118 iDeviceId = osDriverContext->iDeviceId;
119 }
120
121 if (eStatus != MOS_STATUS_SUCCESS)
122 {
123 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Sku/Wa/GtSystemInfo initialization");
124 return eStatus;
125 }
126
127 ReadUserSetting(
128 userSettingPtr,
129 value,
130 __MEDIA_USER_FEATURE_VALUE_ENABLE_SOFTPIN,
131 MediaUserSetting::Group::Device);
132
133 if (value)
134 {
135 bool softpin_va1Malign = false;
136 if (MEDIA_IS_SKU(&m_skuTable, Ftr1MGranularAuxTable))
137 {
138 softpin_va1Malign = true;
139 }
140
141 mos_bufmgr_gem_enable_softpin(m_bufmgr, softpin_va1Malign);
142 }
143
144 if (MEDIA_IS_SKU(&m_skuTable, FtrEnableMediaKernels) == 0)
145 {
146 MEDIA_WR_WA(&m_waTable, WaHucStreamoutOnlyDisable, 0);
147 }
148
149 MosUtilities::MosTraceSetupInfo(
150 (VA_MAJOR_VERSION << 16) | VA_MINOR_VERSION,
151 m_platformInfo.eProductFamily,
152 m_platformInfo.eRenderCoreFamily,
153 (m_platformInfo.usRevId << 16) | m_platformInfo.usDeviceID);
154
155 GMM_SKU_FEATURE_TABLE gmmSkuTable = {};
156 GMM_WA_TABLE gmmWaTable = {};
157 GMM_GT_SYSTEM_INFO gmmGtInfo = {};
158 GMM_ADAPTER_BDF gmmAdapterBDF = {};
159 eStatus = HWInfo_GetGmmInfo(m_fd, &gmmSkuTable, &gmmWaTable, &gmmGtInfo);
160 if (MOS_STATUS_SUCCESS != eStatus)
161 {
162 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Gmm Sku/Wa/GtSystemInfo initialization");
163 return eStatus;
164 }
165
166 eStatus = MosInterface::GetAdapterBDF(osDriverContext, &gmmAdapterBDF);
167 if (MOS_STATUS_SUCCESS != eStatus)
168 {
169 MOS_OS_ASSERTMESSAGE("Fatal error - unsuccesfull Gmm Adapter BDF initialization");
170 return eStatus;
171 }
172
173 // Initialize Gmm context
174 GMM_INIT_IN_ARGS gmmInitAgrs = {};
175 GMM_INIT_OUT_ARGS gmmOutArgs = {};
176 gmmInitAgrs.Platform = m_platformInfo;
177 gmmInitAgrs.pSkuTable = &gmmSkuTable;
178 gmmInitAgrs.pWaTable = &gmmWaTable;
179 gmmInitAgrs.pGtSysInfo = &gmmGtInfo;
180 gmmInitAgrs.FileDescriptor = gmmAdapterBDF.Data;
181 gmmInitAgrs.ClientType = (GMM_CLIENT)GMM_LIBVA_LINUX;
182
183 GMM_STATUS status = InitializeGmm(&gmmInitAgrs, &gmmOutArgs);
184 if (status != GMM_SUCCESS)
185 {
186 MOS_OS_ASSERTMESSAGE("Fatal error - InitializeGmm fail.");
187 return MOS_STATUS_INVALID_PARAMETER;
188 }
189 m_gmmClientContext = gmmOutArgs.pGmmClientContext;
190
191 m_auxTableMgr = AuxTableMgr::CreateAuxTableMgr(m_bufmgr, &m_skuTable, m_gmmClientContext);
192
193#if (_DEBUG || _RELEASE_INTERNAL)
194 ReadUserSettingForDebug(
195 userSettingPtr,
196 osDriverContext->bSimIsActive,
197 __MEDIA_USER_FEATURE_VALUE_SIM_ENABLE,
198 MediaUserSetting::Group::Device);
199#endif
200
201 m_useSwSwizzling = osDriverContext->bSimIsActive || MEDIA_IS_SKU(&m_skuTable, FtrUseSwSwizzling);
202
203 m_tileYFlag = MEDIA_IS_SKU(&m_skuTable, FtrTileY);
204
205 m_use64BitRelocs = true;
206
207 if (!NullHW::IsEnabled())
208 {
209 osDriverContext->iDeviceId = iDeviceId;
210 osDriverContext->SkuTable = m_skuTable;
211 osDriverContext->WaTable = m_waTable;
212 osDriverContext->gtSystemInfo = m_gtSystemInfo;
213 osDriverContext->platform = m_platformInfo;
214 }
215 osDriverContext->pGmmClientContext = m_gmmClientContext;
216 osDriverContext->m_auxTableMgr = m_auxTableMgr;
217 osDriverContext->bUseSwSwizzling = m_useSwSwizzling;
218 osDriverContext->bTileYFlag = m_tileYFlag;
219 osDriverContext->bIsAtomSOC = m_isAtomSOC;
220 osDriverContext->m_osDeviceContext = this;
221
222 m_usesPatchList = true;
223 m_usesGfxAddress = false;
224
225 SetOsContextValid(true);
226 // Prepare the command buffer manager
227 m_cmdBufMgr = CmdBufMgrNext::GetObject();
228 MOS_OS_CHK_NULL_RETURN(m_cmdBufMgr);
229 MOS_OS_CHK_STATUS_RETURN(m_cmdBufMgr->Initialize(this, COMMAND_BUFFER_SIZE/2));
230
231 // Prepare the gpu Context manager
232 m_gpuContextMgr = GpuContextMgrNext::GetObject(this);
233 MOS_OS_CHK_NULL_RETURN(m_gpuContextMgr);
234
235 m_perfData = (PERF_DATA*)MOS_AllocAndZeroMemory(sizeof(PERF_DATA));
236 MOS_OS_CHK_NULL_RETURN(m_perfData);
237 osDriverContext->pPerfData = m_perfData;
238
239 //It must be done with m_gpuContextMgr ready. Insides it will create gpu context.
240#ifdef _MMC_SUPPORTED
241 m_mosDecompression = MOS_New(MosDecompression, osDriverContext);
242 MOS_OS_CHK_NULL_RETURN(m_mosDecompression);
243 osDriverContext->ppMediaMemDecompState = m_mosDecompression->GetMediaMemDecompState();
244 MOS_OS_CHK_NULL_RETURN(osDriverContext->ppMediaMemDecompState);
245 if (*osDriverContext->ppMediaMemDecompState == nullptr)
246 {
247 MOS_OS_NORMALMESSAGE("Decomp state creation failed");
248 }
249#endif
250 m_mosMediaCopy = MOS_New(MosMediaCopy, osDriverContext);
251 if (nullptr == m_mosMediaCopy)
252 {
253 MOS_OS_NORMALMESSAGE("m_mosMediaCopy creation failed");
254 }
255 else
256 {
257 osDriverContext->ppMediaCopyState = (void **)m_mosMediaCopy->GetMediaCopyState();
258 if ((nullptr == osDriverContext->ppMediaCopyState) || (nullptr == *osDriverContext->ppMediaCopyState))
259 {
260 MOS_OS_NORMALMESSAGE("Media Copy state creation failed");
261 }
262 }
263 }
264 MOS_OS_CHK_STATUS_RETURN(MosOcaRTLogMgr::InitMgr(m_ocaRTLogMgr, this));
265 return eStatus;
266}

268void OsContextSpecificNext::Destroy()
269{
270 MOS_OS_FUNCTION_ENTER;
271
272 if (GetOsContextValid() == true)
273 {
274 if (m_auxTableMgr != nullptr)
275 {
276 MOS_Delete(m_auxTableMgr);
277 m_auxTableMgr = nullptr;
278 }
279
280 m_skuTable.reset();
281 m_waTable.reset();
282
283 mos_bufmgr_destroy(m_bufmgr);
284
285 // Delete Gmm context
286 GMM_INIT_OUT_ARGS gmmOutArgs = {};
287 gmmOutArgs.pGmmClientContext = m_gmmClientContext;
288 GmmAdapterDestroy(&gmmOutArgs);
289 m_gmmClientContext = nullptr;
290
291 SetOsContextValid(false);
292
293 if (m_perfData != nullptr)
294 {
295 MOS_FreeMemory(m_perfData);
296 m_perfData = nullptr;
297 }
298
299 if (m_mosMediaCopy != nullptr)
300 {
301 MOS_Delete(m_mosMediaCopy);
302 }
303 }
