46static bool InitBdwMediaSku(struct GfxDeviceInfo *devInfo,
47 MediaFeatureTable *skuTable,
48 struct LinuxDriverInfo *drvInfo,
49 MediaUserSettingSharedPtr userSettingPtr)
50{
51 if ((devInfo == nullptr) || (skuTable == nullptr) || (drvInfo == nullptr))
52 {
53 DEVINFO_ERROR("null ptr is passed\n");
54 return false;
55 }
56
57 if (drvInfo->hasBsd)
58 {
59 LinuxCodecInfo *codecInfo = &bdwCodecInfo;
60
61 MEDIA_WR_SKU(skuTable, FtrAVCVLDLongDecoding, codecInfo->avcDecoding);
62 MEDIA_WR_SKU(skuTable, FtrMPEG2VLDDecoding, codecInfo->mpeg2Decoding);
63 MEDIA_WR_SKU(skuTable, FtrIntelVP8VLDDecoding, codecInfo->vp8Decoding);
64 MEDIA_WR_SKU(skuTable, FtrVC1VLDDecoding, codecInfo->vc1Decoding);
65 MEDIA_WR_SKU(skuTable, FtrIntelJPEGDecoding, codecInfo->jpegDecoding);
66 MEDIA_WR_SKU(skuTable, FtrEncodeAVC, codecInfo->avcEncoding);
67 MEDIA_WR_SKU(skuTable, FtrEncodeMPEG2, codecInfo->mpeg2Encoding);
68 }
69
70 if (devInfo->eGTType == GTTYPE_GT1)
71 {
72 MEDIA_WR_SKU(skuTable, FtrGT1, 1);
73 }
74 else if (devInfo->eGTType == GTTYPE_GT2)
75 {
76 MEDIA_WR_SKU(skuTable, FtrGT2, 1);
77 }
78 else if (devInfo->eGTType == GTTYPE_GT3)
79 {
80 MEDIA_WR_SKU(skuTable, FtrGT3, 1);
81 }
82 else if (devInfo->eGTType == GTTYPE_GT4)
83 {
84 MEDIA_WR_SKU(skuTable, FtrGT4, 1);
85 }
86 else
87 {
88 /* GT1 is by default */
89 MEDIA_WR_SKU(skuTable, FtrGT1, 1);
90 }
91 MEDIA_WR_SKU(skuTable, FtrVcs2, drvInfo->hasBsd2);
92 MEDIA_WR_SKU(skuTable, FtrVERing, drvInfo->hasVebox);
93 MEDIA_WR_SKU(skuTable, FtrPPGTT, drvInfo->hasPpgtt);
94 MEDIA_WR_SKU(skuTable, FtrEDram, devInfo->hasERAM);
95 if ((devInfo->eGTType == GTTYPE_GT3) ||
96 (devInfo->eGTType == GTTYPE_GT4))
97 {
98 MEDIA_WR_SKU(skuTable, FtrSliceShutdown, 1);
99 }
100 else
101 {
102 MEDIA_WR_SKU(skuTable, FtrSingleVeboxSlice, 1);
103 }
104
105 MEDIA_WR_SKU(skuTable, FtrSliceShutdownOverride, 1);
106 MEDIA_WR_SKU(skuTable, FtrTileY, 1);
107
108 MEDIA_WR_SKU(skuTable, FtrUseSwSwizzling, 1);
109
110 return true;
111}
