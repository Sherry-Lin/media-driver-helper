92 GMM_RESCREATE_PARAMS gmmParams;
93 MOS_ZeroMemory(&gmmParams, sizeof(gmmParams));

95 switch (params.m_type)
96 {
97 case MOS_GFXRES_BUFFER:
98 case MOS_GFXRES_SCRATCH:

100 gmmParams.Flags.Gpu.State = true;
101 alignedHeight = 1;
102 break;
103
104 case MOS_GFXRES_2D:

106 gmmParams.Flags.Gpu.Video = true;
107 break;
108
109 case MOS_GFXRES_VOLUME:

133 switch (tileformat)

135 case MOS_TILE_Y:

155 gmmParams.Flags.Info.TiledX = true;

205 case GMM_TILED_Y:
206 tileformat = MOS_TILE_Y;
207 tileFormatLinux = TILING_Y;
208 break;
209 case GMM_NOT_TILED:
210 tileformat = MOS_TILE_LINEAR;

215 tileFormatLinux = TILING_Y;
216 break;

225 uint32_t bufSize = GFX_ULONG_CAST(gmmResourceInfoPtr->GetSizeSurface());

230 char bufName[m_maxBufNameLength];
231 MOS_SecureStrcpy(bufName, m_maxBufNameLength, params.m_name.c_str());

238 memPolicyPar.skuTable = pOsContextSpecific->GetSkuTable();

242 memPolicyPar.preferredMemType = params.m_memType;

251 bufName,
252 params.m_pSystemMemory,
253 tileFormatLinux,
254 bufPitch,
255 bufSize,
256 0);

432void* GraphicsResourceSpecific::Lock(OsContext* osContextPtr, LockParams& params)
433{
434 MOS_OS_FUNCTION_ENTER;
435
436 if (osContextPtr == nullptr)
437 {
438 MOS_OS_ASSERTMESSAGE("Unable to get the active OS context.");
439 return nullptr;
440 }
441
442 if (osContextPtr ->GetOsContextValid() == false)
443 {
444 MOS_OS_ASSERTMESSAGE("The OS context got is not valid.");
445 return nullptr;
446 }
447
448 OsContextSpecific *pOsContextSpecific = static_cast<OsContextSpecific *>(osContextPtr);
449
450 void* dataPtr = nullptr;
451 MOS_LINUX_BO* boPtr = m_bo;
452
453 if (boPtr)
454 {
455 // Do decompression for a compressed surface before lock
456 const auto pGmmResInfo = m_gmmResInfo;
457 MOS_OS_ASSERT(pGmmResInfo);
458 GMM_RESOURCE_FLAG GmmFlags = pGmmResInfo->GetResFlags();
459
460 if (!params.m_noDecompress &&
461 (((GmmFlags.Gpu.MMC || GmmFlags.Gpu.CCS) && GmmFlags.Info.MediaCompressed) ||
462 pGmmResInfo->IsMediaMemoryCompressed(0)))
463 {
464 if ((pOsContextSpecific->m_mediaMemDecompState == nullptr) ||
465 (pOsContextSpecific->m_memoryDecompress == nullptr))
466 {
467 MOS_OS_ASSERTMESSAGE("m_mediaMemDecompState/m_memoryDecompress is not valid.");
468 return nullptr;
469 }
470
471 MOS_RESOURCE mosResource = {};
472 ConvertToMosResource(&mosResource);
473 pOsContextSpecific->m_memoryDecompress(pOsContextSpecific->m_mosContext, &mosResource);
474 }
475
476 if(false == m_mapped)
477 {
478 if (pOsContextSpecific->IsAtomSoc())
479 {
480 mos_bo_map_gtt(boPtr);
481 }
482 else
483 {
484 if (m_tileType != MOS_TILE_LINEAR && !params.m_tileAsTiled)
485 {
486 if (pOsContextSpecific->UseSwSwizzling())
487 {
488 mos_bo_map(boPtr, ( OSKM_LOCKFLAG_WRITEONLY & params.m_writeRequest ));
489 m_mmapOperation = MOS_MMAP_OPERATION_MMAP;
490 if (m_systemShadow == nullptr)
491 {
492 m_systemShadow = (uint8_t *)MOS_AllocMemory(boPtr->size);
493 MOS_OS_CHECK_CONDITION((m_systemShadow == nullptr), "Failed to allocate shadow surface", nullptr);
494 }
495 if (m_systemShadow)
496 {
497 int32_t flags = pOsContextSpecific->GetTileYFlag() ? 0 : 1;
498 uint64_t surfSize = m_gmmResInfo->GetSizeMainSurface();
499 MOS_OS_CHECK_CONDITION((m_tileType != MOS_TILE_Y), "Unsupported tile type", nullptr);
500 MOS_OS_CHECK_CONDITION((boPtr->size <= 0 || m_pitch <= 0), "Invalid BO size or pitch", nullptr);
501 Mos_SwizzleData((uint8_t*)boPtr->virt, m_systemShadow,
502 MOS_TILE_Y, MOS_TILE_LINEAR,
503 (int32_t)(surfSize / m_pitch), m_pitch, flags);
504 }
505 }
506 else
507 {
508 mos_bo_map_gtt(boPtr);
509 m_mmapOperation = MOS_MMAP_OPERATION_MMAP_GTT;
510 }
511 }
512 else if (params.m_uncached)
513 {
514 mos_bo_map_wc(boPtr);
515 m_mmapOperation = MOS_MMAP_OPERATION_MMAP_WC;
516 }
517 else
518 {
519 mos_bo_map(boPtr, ( OSKM_LOCKFLAG_WRITEONLY & params.m_writeRequest ));
520 m_mmapOperation = MOS_MMAP_OPERATION_MMAP;
521 }
522 }
523 m_mapped = true;
524 m_pData = m_systemShadow ? m_systemShadow : (uint8_t *)boPtr->virt;
525 }
526
527 dataPtr = m_pData;
528 }
529
530 MOS_OS_ASSERT(dataPtr);
531 return dataPtr;
532}

534MOS_STATUS GraphicsResourceSpecific::Unlock(OsContext* osContextPtr)
535{
536 MOS_OS_FUNCTION_ENTER;
537
538 if (osContextPtr == nullptr)
539 {
540 MOS_OS_ASSERTMESSAGE("Unable to get the active OS context.");
541 return MOS_STATUS_INVALID_HANDLE;
542 }
543
544 if (osContextPtr ->GetOsContextValid() == false)
545 {
546 MOS_OS_ASSERTMESSAGE("The OS context got is not valid.");
547 return MOS_STATUS_INVALID_HANDLE;
548 }
549
550 OsContextSpecific *pOsContextSpecific = static_cast<OsContextSpecific *>(osContextPtr);
551
552 MOS_LINUX_BO* boPtr = m_bo;
553 if (boPtr)
554 {
555 if (m_mapped)
556 {
557 if (pOsContextSpecific->IsAtomSoc())
558 {
559 mos_bo_unmap_gtt(boPtr);
560 }
561 else
562 {
563 if (m_systemShadow)
564 {
565 int32_t flags = pOsContextSpecific->GetTileYFlag() ? 0 : 1;
566 uint64_t surfSize = m_gmmResInfo->GetSizeMainSurface();
567 Mos_SwizzleData(m_systemShadow, (uint8_t*)boPtr->virt,
568 MOS_TILE_LINEAR, MOS_TILE_Y,
569 (int32_t)(surfSize / m_pitch), m_pitch, flags);
570 MOS_FreeMemory(m_systemShadow);
571 m_systemShadow = nullptr;
572 }
573
574 switch(m_mmapOperation)
575 {
576 case MOS_MMAP_OPERATION_MMAP_GTT:
577 mos_bo_unmap_gtt(boPtr);
578 break;
579 case MOS_MMAP_OPERATION_MMAP_WC:
580 mos_bo_unmap_wc(boPtr);
581 break;
582 case MOS_MMAP_OPERATION_MMAP:
583 mos_bo_unmap(boPtr);
584 break;
585 default:
586 MOS_OS_ASSERTMESSAGE("Invalid mmap operation type");
587 break;
588 }
589 }
590
591 m_mapped = false;
592 m_mmapOperation = MOS_MMAP_OPERATION_NONE;
593
594 boPtr->virt = nullptr;
595 m_bo = boPtr;
596 }
597
598 m_pData = nullptr;
599 }
600
601 return MOS_STATUS_SUCCESS;
602}
