78VAStatus MediaLibvaCapsG11::QueryImageFormats(VAImageFormat *formatList, int32_t *numFormats)
79{
80 DDI_CHK_NULL(formatList, "Null pointer", VA_STATUS_ERROR_INVALID_PARAMETER);
81 DDI_CHK_NULL(numFormats, "Null pointer", VA_STATUS_ERROR_INVALID_PARAMETER);
82 int32_t num = 0;
83 uint32_t maxNum = GetImageFormatsMaxNum();
84
85 memset(formatList, 0, sizeof(m_supportedImageformatsG11));
86 for (uint32_t idx = 0; idx < maxNum; idx++)
87 {
88 formatList[num].fourcc = m_supportedImageformatsG11[idx].fourcc;
89 formatList[num].byte_order = m_supportedImageformatsG11[idx].byte_order;
90 formatList[num].bits_per_pixel = m_supportedImageformatsG11[idx].bits_per_pixel;
91 formatList[num].depth = m_supportedImageformatsG11[idx].depth;
92 formatList[num].red_mask = m_supportedImageformatsG11[idx].red_mask;
93 formatList[num].green_mask = m_supportedImageformatsG11[idx].green_mask;
94 formatList[num].blue_mask = m_supportedImageformatsG11[idx].blue_mask;
95 formatList[num].alpha_mask = m_supportedImageformatsG11[idx].alpha_mask;
96 num++;
97 }
98 *numFormats = num;
99
100 return VA_STATUS_SUCCESS;
101}
