28bool CompareFeatureIDTable(vector<FeatureID> &currentTable, vector<FeatureID> &refTable)
29{
30 if (currentTable.size() != refTable.size())
31 {
32 return false;
33 }
34
35 sort(currentTable.begin(), currentTable.end());
36 sort(refTable.begin() , refTable.end());
37
38 return equal(currentTable.begin(), currentTable.end(), refTable.begin());
39}

41int Test_QueryConfigProfiles(VADriverContextP ctx, vector<FeatureID> &queriedFeatureIDTable)
42{
43 VAProfile profile;
44 VAEntrypoint entrypoints[10];
45 int num_profiles;
46 int num_entrypoint;
47 int max_num_profiles = ctx->max_profiles;
48 VAProfile *profile_list = nullptr;
49
50 profile_list = (VAProfile *)malloc(max_num_profiles * sizeof(VAProfile));
51 if (profile_list == nullptr)
52 {
53 return -1;
54 }
55
56 int ret = ctx->vtable->vaQueryConfigProfiles(ctx, profile_list, &num_profiles);
57 if (ret)
58 {
59 free(profile_list);
60 return -1;
61 }
62
63 for (int i = 0; i < num_profiles; i++)
64 {
65 profile = profile_list[i];
66 ret = ctx->vtable->vaQueryConfigEntrypoints(ctx, profile, entrypoints, &num_entrypoint);
67
68 if (ret == VA_STATUS_ERROR_UNSUPPORTED_PROFILE)
69 {
70 continue;
71 }
72 else if (ret)
73 {
74 free(profile_list);
75 return -1;
76 }
77 else
78 {
79 for (int j = 0; j < num_entrypoint; j++)
80 {
81 queriedFeatureIDTable.push_back({profile, entrypoints[j]});
82 }
83 }
84 }
85
86 free(profile_list);
87
88 return ret;
89}
