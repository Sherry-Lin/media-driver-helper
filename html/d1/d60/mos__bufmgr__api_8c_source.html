154int
155mos_bo_mrb_exec(struct mos_linux_bo *bo, int used,
156 drm_clip_rect_t *cliprects, int num_cliprects, int DR4,
157 unsigned int rings)
158{
159 if (bo->bufmgr->bo_mrb_exec)
160 return bo->bufmgr->bo_mrb_exec(bo, used,
161 cliprects, num_cliprects, DR4,
162 rings);
163
164 switch (rings) {
165 case I915_EXEC_DEFAULT:
166 case I915_EXEC_RENDER:
167 return bo->bufmgr->bo_exec(bo, used,
168 cliprects, num_cliprects, DR4);
169 default:
170 return -ENODEV;
171 }
172}
