412static inline int mos_query_device_type(int fd)
413{
414 int device_type = DEVICE_TYPE_COUNT;
415 drm_version_t version;
416 char name[5] = "";
417
418 memset(&version, 0, sizeof(version));
419 version.name_len = 4;
420 version.name = name;
421
422 if (drmIoctl(fd, DRM_IOCTL_VERSION, &version))
423 {
424 fprintf(stderr, "DRM_IOCTL_VERSION failed: %s\n", strerror(errno));
425 return device_type;
426 }
427
428 if(strcmp("i915", name) == 0)
429 {
430 device_type = DEVICE_TYPE_I915;
431 }
432 else if(strcmp("xe", name) == 0)
433 {
434 device_type = DEVICE_TYPE_XE;
435 }
436 else
437 {
438 fprintf(stderr, "DRM_IOCTL_VERSION, unsupported drm device by media driver: %s\n", name);
439 }
440 return device_type;
441}
