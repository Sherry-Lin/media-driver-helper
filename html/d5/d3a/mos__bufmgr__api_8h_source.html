377static inline int mos_query_device_type(int fd)
378{
379 int device_type = DEVICE_TYPE_COUNT;
380 drm_version_t version;
381 char name[5] = "";
382
383 memset(&version, 0, sizeof(version));
384 version.name_len = 4;
385 version.name = name;
386
387 if (drmIoctl(fd, DRM_IOCTL_VERSION, &version))
388 {
389 fprintf(stderr, "DRM_IOCTL_VERSION failed: %s\n", strerror(errno));
390 return device_type;
391 }
392
393 if(strcmp("i915", name) == 0)
394 {
395 device_type = DEVICE_TYPE_I915;
396 }
397 else if(strcmp("xe", name) == 0)
398 {
399 device_type = DEVICE_TYPE_XE;
400 }
401 else
402 {
403 fprintf(stderr, "DRM_IOCTL_VERSION, unsupported drm device by media driver: %s\n", name);
404 }
405 return device_type;
406}
