197/*
198 * vpgEncodeCreateContext - Create an encode context
199 * dpy: display
200 * config_id: configuration for the encode context
201 * picture_width: encode picture width
202 * picture_height: encode picture height
203 * flag: any combination of the following:
204 * VA_PROGRESSIVE (only progressive frame pictures in the sequence when set)
205 * render_targets: render targets (surfaces) tied to the context
206 * num_render_targets: number of render targets in the above array
207 * context: created context id upon return
208 */
209VAStatus DdiEncode_CreateContext(
210 VADriverContextP ctx,
211 VAConfigID config_id,
212 int32_t picture_width,
213 int32_t picture_height,
214 int32_t flag,
215 VASurfaceID *render_targets,
216 int32_t num_render_targets,
217 VAContextID *context)
218{
219 PERF_UTILITY_AUTO(__FUNCTION__, PERF_ENCODE, PERF_LEVEL_DDI);
220
221 DDI_CHK_NULL(ctx, "nullptr ctx", VA_STATUS_ERROR_INVALID_CONTEXT);
222 DDI_CHK_NULL(ctx->pDriverData, "nullptr ctx->pDriverData", VA_STATUS_ERROR_INVALID_CONTEXT);
223
224 PDDI_MEDIA_CONTEXT mediaDrvCtx = DdiMedia_GetMediaContext(ctx);
225 DDI_CHK_NULL(mediaDrvCtx->m_caps, "nullptr m_caps", VA_STATUS_ERROR_INVALID_CONTEXT);
226
227 VAProfile profile;
228 VAEntrypoint entrypoint;
229 uint32_t rcMode = 0;
230 uint32_t feiFunction = 0;
231 VAStatus vaStatus = mediaDrvCtx->m_caps->GetEncConfigAttr(
232 config_id + DDI_CODEC_GEN_CONFIG_ATTRIBUTES_ENC_BASE,
233 &profile,
234 &entrypoint,
235 &rcMode,
236 &feiFunction);
237 DDI_CHK_RET(vaStatus, "Invalide config_id!");
238
239 vaStatus = mediaDrvCtx->m_caps->CheckEncodeResolution(
240 profile,
241 picture_width,
242 picture_height);
243 if (vaStatus != VA_STATUS_SUCCESS)
244 {
245 return VA_STATUS_ERROR_RESOLUTION_NOT_SUPPORTED;
246 }
247
248 if (num_render_targets > DDI_MEDIA_MAX_SURFACE_NUMBER_CONTEXT)
249 {
250 return VA_STATUS_ERROR_MAX_NUM_EXCEEDED;
251 }
252
253 std::string encodeKey = mediaDrvCtx->m_caps->GetEncodeCodecKey(profile, entrypoint, feiFunction);
254 DdiEncodeBase *ddiEncode = DdiEncodeFactory::CreateCodec(encodeKey);
255 DDI_CHK_NULL(ddiEncode, "nullptr ddiEncode", VA_STATUS_ERROR_UNIMPLEMENTED);
256
257 // first create encoder context
258 ddiEncode->m_encodeCtx = (PDDI_ENCODE_CONTEXT)MOS_AllocAndZeroMemory(sizeof(DDI_ENCODE_CONTEXT));
259 DDI_CHK_NULL(ddiEncode->m_encodeCtx, "nullptr ddiEncode->m_encodeCtx", VA_STATUS_ERROR_ALLOCATION_FAILED);
260
261 PDDI_ENCODE_CONTEXT encCtx = ddiEncode->m_encodeCtx;
262 encCtx->m_encode = ddiEncode;
263
264 //initialize DDI level cp interface
265 MOS_CONTEXT mosCtx = { };
266 encCtx->pCpDdiInterface = Create_DdiCpInterface(mosCtx);
267 if (nullptr == encCtx->pCpDdiInterface)
268 {
269 vaStatus = VA_STATUS_ERROR_ALLOCATION_FAILED;
270 DdiEncodeCleanUp(encCtx);
271 return vaStatus;
272 }
273
274 // Get the buf manager for codechal create
275 mosCtx.bufmgr = mediaDrvCtx->pDrmBufMgr;
276 mosCtx.m_gpuContextMgr = mediaDrvCtx->m_gpuContextMgr;
277 mosCtx.m_cmdBufMgr = mediaDrvCtx->m_cmdBufMgr;
278 mosCtx.fd = mediaDrvCtx->fd;
279 mosCtx.iDeviceId = mediaDrvCtx->iDeviceId;
280 mosCtx.SkuTable = mediaDrvCtx->SkuTable;
281 mosCtx.WaTable = mediaDrvCtx->WaTable;
282 mosCtx.gtSystemInfo = *mediaDrvCtx->pGtSystemInfo;
283 mosCtx.platform = mediaDrvCtx->platform;
284
285 mosCtx.ppMediaMemDecompState = &mediaDrvCtx->pMediaMemDecompState;
286 mosCtx.pfnMemoryDecompress = mediaDrvCtx->pfnMemoryDecompress;
287 mosCtx.pfnMediaMemoryCopy = mediaDrvCtx->pfnMediaMemoryCopy;
288 mosCtx.pfnMediaMemoryCopy2D = mediaDrvCtx->pfnMediaMemoryCopy2D;
289 mosCtx.ppMediaCopyState = &mediaDrvCtx->pMediaCopyState;
290 mosCtx.gtSystemInfo = *mediaDrvCtx->pGtSystemInfo;
291 mosCtx.m_auxTableMgr = mediaDrvCtx->m_auxTableMgr;
292 mosCtx.pGmmClientContext = mediaDrvCtx->pGmmClientContext;
293
294 mosCtx.m_osDeviceContext = mediaDrvCtx->m_osDeviceContext;
295 mosCtx.m_apoMosEnabled = mediaDrvCtx->m_apoMosEnabled;
296
297 mosCtx.m_userSettingPtr = mediaDrvCtx->m_userSettingPtr;
298
299 mosCtx.pPerfData = (PERF_DATA *)MOS_AllocAndZeroMemory(sizeof(PERF_DATA));
300 if (nullptr == mosCtx.pPerfData)
301 {
302 vaStatus = VA_STATUS_ERROR_ALLOCATION_FAILED;
303 DdiEncodeCleanUp(encCtx);
304 return vaStatus;
305 }
306
307 encCtx->vaEntrypoint = entrypoint;
308 encCtx->vaProfile = profile;
309 encCtx->uiRCMethod = rcMode;
310 encCtx->wModeType = mediaDrvCtx->m_caps->GetEncodeCodecMode(profile, entrypoint);
311 encCtx->codecFunction = mediaDrvCtx->m_caps->GetEncodeCodecFunction(profile, entrypoint, feiFunction);
312
313 if (entrypoint == VAEntrypointEncSliceLP)
314 {
315 encCtx->bVdencActive = true;
316 }
317
318 //Both dual pipe and LP pipe should support 10bit for below profiles
319 // - HEVCMain10 profile
320 // - VAProfileVP9Profile2
321 // - VAProfileVP9Profile3
322 if (profile == VAProfileVP9Profile2 ||
323 profile == VAProfileVP9Profile3)
324 {
325 encCtx->m_encode->m_is10Bit = true;
326 }
327
328 if (profile == VAProfileVP9Profile1 ||
329 profile == VAProfileVP9Profile3)
330 {
331 encCtx->m_encode->m_chromaFormat = DdiEncodeBase::yuv444;
332 }
333
334 CODECHAL_STANDARD_INFO standardInfo;
335 MOS_ZeroMemory(&standardInfo, sizeof(CODECHAL_STANDARD_INFO));
336 standardInfo.CodecFunction = encCtx->codecFunction;
337 standardInfo.Mode = encCtx->wModeType;
338 Codechal *pCodecHal = CodechalDevice::CreateFactory(
339 nullptr,
340 &mosCtx,
341 &standardInfo,
342 nullptr);
343 if (pCodecHal == nullptr)
344 {
345 // add anything necessary here to free the resource
346 vaStatus = VA_STATUS_ERROR_ALLOCATION_FAILED;
347 DdiEncodeCleanUp(encCtx);
348 return vaStatus;
349 }
350
351 encCtx->pCodecHal = pCodecHal;
352
353 // Setup some initial data
354 encCtx->dworiFrameWidth = picture_width;
355 encCtx->dworiFrameHeight = picture_height;
356 encCtx->wPicWidthInMB = (uint16_t)(DDI_CODEC_NUM_MACROBLOCKS_WIDTH(picture_width));
357 encCtx->wPicHeightInMB = (uint16_t)(DDI_CODEC_NUM_MACROBLOCKS_HEIGHT(picture_height));
358 encCtx->dwFrameWidth = encCtx->wPicWidthInMB * CODECHAL_MACROBLOCK_WIDTH;
359 encCtx->dwFrameHeight = encCtx->wPicHeightInMB * CODECHAL_MACROBLOCK_HEIGHT;
360 //recoder old resolution for dynamic resolution change
361 encCtx->wContextPicWidthInMB = encCtx->wPicWidthInMB;
362 encCtx->wContextPicHeightInMB = encCtx->wPicHeightInMB;
363 encCtx->wOriPicWidthInMB = encCtx->wPicWidthInMB;
364 encCtx->wOriPicHeightInMB = encCtx->wPicHeightInMB;
365 encCtx->targetUsage = TARGETUSAGE_RT_SPEED;
366 // Attach PMEDIDA_DRIVER_CONTEXT
367 encCtx->pMediaCtx = mediaDrvCtx;
368
369 encCtx->pCpDdiInterface->SetCpFlags(flag);
370 encCtx->pCpDdiInterface->SetCpParams(CP_TYPE_NONE, encCtx->m_encode->m_codechalSettings);
371
372 vaStatus = encCtx->m_encode->ContextInitialize(encCtx->m_encode->m_codechalSettings);
373
374 if (vaStatus != VA_STATUS_SUCCESS)
375 {
376 DdiEncodeCleanUp(encCtx);
377 return vaStatus;
378 }
379
380 MOS_STATUS eStatus = pCodecHal->Allocate(encCtx->m_encode->m_codechalSettings);
381
382#ifdef _MMC_SUPPORTED
383 PMOS_INTERFACE osInterface = pCodecHal->GetOsInterface();
384 if (osInterface != nullptr &&
385 !osInterface->apoMosEnabled &&
386 MEDIA_IS_SKU(osInterface->pfnGetSkuTable(osInterface), FtrMemoryCompression) &&
387 !mediaDrvCtx->pMediaMemDecompState)
388 {
389 mediaDrvCtx->pMediaMemDecompState =
390 static_cast<MediaMemDecompState*>(MmdDevice::CreateFactory(&mosCtx));
391 }
392#endif
393
394 if (eStatus != MOS_STATUS_SUCCESS)
395 {
396 vaStatus = VA_STATUS_ERROR_ALLOCATION_FAILED;
397 DdiEncodeCleanUp(encCtx);
398 return vaStatus;
399 }
400
401 vaStatus = encCtx->m_encode->InitCompBuffer();
402 if (vaStatus != VA_STATUS_SUCCESS)
403 {
404 vaStatus = VA_STATUS_ERROR_ALLOCATION_FAILED;
405 DdiEncodeCleanUp(encCtx);
406 return vaStatus;
407 }
408
409 // register the render target surfaces for this encoder instance
410 // This is a must as driver has the constraint, 127 surfaces per context
411 for (int32_t i = 0; i < num_render_targets; i++)
412 {
413 DDI_MEDIA_SURFACE *surface;
414
415 surface = DdiMedia_GetSurfaceFromVASurfaceID(mediaDrvCtx, render_targets[i]);
416 if (nullptr == surface)
417 {
418 DDI_ASSERTMESSAGE("DDI: invalid render target %d in vpgEncodeCreateContext.", i);
419 vaStatus = VA_STATUS_ERROR_INVALID_SURFACE;
420 DdiEncodeCleanUp(encCtx);
421 return vaStatus;
422 }
423 encCtx->RTtbl.pRT[i] = surface;
424 encCtx->RTtbl.iNumRenderTargets++;
425 }
426
427 // convert PDDI_ENCODE_CONTEXT to VAContextID
428 DdiMediaUtil_LockMutex(&mediaDrvCtx->EncoderMutex);
429 PDDI_MEDIA_VACONTEXT_HEAP_ELEMENT vaContextHeapElmt = DdiMediaUtil_AllocPVAContextFromHeap(mediaDrvCtx->pEncoderCtxHeap);
430 if (nullptr == vaContextHeapElmt)
431 {
432 DdiMediaUtil_UnLockMutex(&mediaDrvCtx->EncoderMutex);
433 vaStatus = VA_STATUS_ERROR_MAX_NUM_EXCEEDED;
434 DdiEncodeCleanUp(encCtx);
435 return vaStatus;
436 }
437
438 vaContextHeapElmt->pVaContext = (void*)encCtx;
439 mediaDrvCtx->uiNumEncoders++;
440 *context = (VAContextID)(vaContextHeapElmt->uiVaContextID + DDI_MEDIA_VACONTEXTID_OFFSET_ENCODER);
441 DdiMediaUtil_UnLockMutex(&mediaDrvCtx->EncoderMutex);
442
443 return vaStatus;
444}

446/*
447 * vaDestroyContext - Destroy a context
448 * dpy: display
449 * context: context to be destroyed
450 */
451VAStatus DdiEncode_DestroyContext(VADriverContextP ctx, VAContextID context)
452{
453 DDI_CHK_NULL(ctx, "nullptr ctx", VA_STATUS_ERROR_INVALID_CONTEXT);
454 DDI_CHK_NULL(ctx->pDriverData, "nullptr ctx->pDriverData", VA_STATUS_ERROR_INVALID_CONTEXT);
455
456 PDDI_MEDIA_CONTEXT mediaCtx = DdiMedia_GetMediaContext(ctx);
457 DDI_CHK_NULL(mediaCtx, "nullptr mediaCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
458
459 // assume the VAContextID is encoder ID
460 PDDI_ENCODE_CONTEXT encCtx = DdiEncode_GetEncContextFromContextID(ctx, context);
461 DDI_CHK_NULL(encCtx, "nullptr encCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
462 DDI_CHK_NULL(encCtx->pCodecHal, "nullptr encCtx->pCodecHal", VA_STATUS_ERROR_INVALID_CONTEXT);
463
464 Codechal *codecHal = encCtx->pCodecHal;
465
466 if (nullptr != encCtx->m_encode)
467 {
468 encCtx->m_encode->FreeCompBuffer();
469 if(nullptr != encCtx->m_encode->m_codechalSettings)
470 {
471 MOS_Delete(encCtx->m_encode->m_codechalSettings);
472 encCtx->m_encode->m_codechalSettings = nullptr;
473 }
474 }
475
476 if (codecHal->GetOsInterface() && codecHal->GetOsInterface()->pOsContext)
477 {
478 MOS_FreeMemory(codecHal->GetOsInterface()->pOsContext->pPerfData);
479 codecHal->GetOsInterface()->pOsContext->pPerfData = nullptr;
480 }
481
482 // destroy codechal
483 codecHal->Destroy();
484 MOS_Delete(codecHal);
485
486 if (encCtx->pCpDdiInterface)
487 {
488 Delete_DdiCpInterface(encCtx->pCpDdiInterface);
489 encCtx->pCpDdiInterface = nullptr;
490 }
491
492 if (nullptr != encCtx->m_encode)
493 {
494 MOS_Delete(encCtx->m_encode);
495 encCtx->m_encode = nullptr;
496 }
497
498 MOS_FreeMemory(encCtx);
499 encCtx = nullptr;
500
501 uint32_t encIndex = (uint32_t)context;
502 encIndex &= DDI_MEDIA_MASK_VACONTEXTID;
503
504 DdiMediaUtil_LockMutex(&mediaCtx->EncoderMutex);
505 DdiMediaUtil_ReleasePVAContextFromHeap(mediaCtx->pEncoderCtxHeap, encIndex);
506 mediaCtx->uiNumEncoders--;
507 DdiMediaUtil_UnLockMutex(&mediaCtx->EncoderMutex);
508
509 return VA_STATUS_SUCCESS;
510}

512/*
513 * Creates a buffer for "num_elements" elements of "size" bytes and
514 * initalize with "data".
515 * if "data" is null, then the contents of the buffer data store
516 * are undefined.
517 * Basically there are two ways to get buffer data to the server side. One is
518 * to call vaCreateBuffer() with a non-null "data", which results the data being
519 * copied to the data store on the server side. A different method that
520 * eliminates this copy is to pass null as "data" when calling vaCreateBuffer(),
521 * and then use vaMapBuffer() to map the data store from the server side to the
522 * client address space for access.
523 * Note: image buffers are created by the library, not the client. Please see
524 * vaCreateImage on how image buffers are managed.
525 */
526
527VAStatus DdiEncode_CreateBuffer(
528 VADriverContextP ctx,
529 VAContextID context,
530 VABufferType type,
531 uint32_t size,
532 uint32_t num_elements,
533 void *data,
534 VABufferID *buf_id)
535{
536 DDI_CHK_NULL(ctx, "nullptr context!", VA_STATUS_ERROR_INVALID_CONTEXT);
537
538 DDI_ENCODE_CONTEXT *encCtx = DdiEncode_GetEncContextFromContextID(ctx, context);
539 DDI_CHK_NULL(encCtx, "nullptr encCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
540 DDI_CHK_NULL(encCtx->m_encode, "nullptr encCtx->m_encode", VA_STATUS_ERROR_INVALID_CONTEXT);
541
542 VAStatus vaStatus = encCtx->m_encode->CreateBuffer(ctx, type, size, num_elements, data, buf_id);
543
544 return vaStatus;
545}

571/*
572 * Send encode buffers to the server.
573 * Buffers are automatically destroyed afterwards
574 */
575VAStatus DdiEncode_RenderPicture(
576 VADriverContextP ctx,
577 VAContextID context,
578 VABufferID *buffers,
579 int32_t num_buffers)
580{
581 VAStatus vaStatus = VA_STATUS_SUCCESS;
582 int32_t numOfBuffers = num_buffers;
583 int32_t priority = 0;
584 int32_t priorityIndexInBuffers = -1;
585 bool updatePriority = false;
586

636VAStatus DdiEncode_MfeSubmit(
637 VADriverContextP ctx,
638 VAMFContextID mfe_context,
639 VAContextID *contexts,
640 int32_t num_contexts
641)
642{
643 PDDI_MEDIA_CONTEXT mediaCtx = DdiMedia_GetMediaContext(ctx);
644 DDI_CHK_NULL(mediaCtx, "nullptr mediaCtx", VA_STATUS_ERROR_INVALID_CONTEXT);
645
646 uint32_t ctxType = DDI_MEDIA_CONTEXT_TYPE_NONE;
647 PDDI_ENCODE_MFE_CONTEXT encodeMfeContext = (PDDI_ENCODE_MFE_CONTEXT)DdiMedia_GetContextFromContextID(ctx, mfe_context, &ctxType);
648 DDI_CHK_NULL(encodeMfeContext, "nullptr encodeMfeContext", VA_STATUS_ERROR_INVALID_CONTEXT);
649
650 std::vector<PDDI_ENCODE_CONTEXT> encodeContexts;
651 PDDI_ENCODE_CONTEXT encodeContext = nullptr;
652 int32_t validContextNumber = 0;
653 // Set mfe encoder params for this submission
654 for (int32_t i = 0; i < num_contexts; i++)
655 {
656 encodeContext = DdiEncode_GetEncContextFromContextID(ctx, contexts[i]);
657 DDI_CHK_NULL(encodeContext, "nullptr encodeContext", VA_STATUS_ERROR_INVALID_CONTEXT);
658 CodechalEncoderState *encoder = dynamic_cast<CodechalEncoderState *>(encodeContext->pCodecHal);
659 DDI_CHK_NULL(encoder, "nullptr codechal encoder", VA_STATUS_ERROR_INVALID_CONTEXT);
660
661 if (!encoder->m_mfeEnabled ||
662 encoder->m_mfeEncodeSharedState != encodeMfeContext->mfeEncodeSharedState)
663 {
664 return VA_STATUS_ERROR_INVALID_CONTEXT;
665 }
666
667 // make sure the context has called BeginPicture&RenderPicture&EndPicture
668 if (encodeContext->RTtbl.pRT[0] == nullptr
669 || encodeContext->dwNumSlices <= 0
670 || encodeContext->EncodeParams.pBSBuffer != encodeContext->pbsBuffer)
671 {
672 return VA_STATUS_ERROR_INVALID_PARAMETER;
673 }
674
675 encoder->m_mfeEncodeParams.submitIndex = i;
676 encoder->m_mfeEncodeParams.submitNumber = num_contexts;
677 encodeContexts.push_back(encodeContext);
678 validContextNumber++;
679 }
680
681 CmDevice *device = encodeMfeContext->mfeEncodeSharedState->pCmDev;
682 CmTask *task = encodeMfeContext->mfeEncodeSharedState->pCmTask;
683 CmQueue *queue = encodeMfeContext->mfeEncodeSharedState->pCmQueue;
684 CodechalEncodeMdfKernelResource *resMbencKernel = encodeMfeContext->mfeEncodeSharedState->resMbencKernel;
685 SurfaceIndex *vmeSurface = encodeMfeContext->mfeEncodeSharedState->vmeSurface;
686 SurfaceIndex *commonSurface = encodeMfeContext->mfeEncodeSharedState->commonSurface;
687
688
689 MOS_ZeroMemory(encodeMfeContext->mfeEncodeSharedState, sizeof(MfeSharedState));
690
691 encodeMfeContext->mfeEncodeSharedState->pCmDev = device;
692 encodeMfeContext->mfeEncodeSharedState->pCmTask = task;
693 encodeMfeContext->mfeEncodeSharedState->pCmQueue = queue;
694 encodeMfeContext->mfeEncodeSharedState->resMbencKernel = resMbencKernel;
695 encodeMfeContext->mfeEncodeSharedState->vmeSurface = vmeSurface;
696 encodeMfeContext->mfeEncodeSharedState->commonSurface = commonSurface;
697
698 encodeMfeContext->mfeEncodeSharedState->encoders.clear();
699
700 // Call Enc functions for all the sub contexts
701 MOS_STATUS status = MOS_STATUS_SUCCESS;
702 for (int32_t i = 0; i < validContextNumber; i++)
703 {
704 encodeContext = encodeContexts[i];
705 if (encodeContext->vaEntrypoint != VAEntrypointFEI )
706 {
707 encodeContext->EncodeParams.ExecCodecFunction = CODECHAL_FUNCTION_ENC;
708 }
709 else
710 {
711 encodeContext->EncodeParams.ExecCodecFunction = CODECHAL_FUNCTION_FEI_ENC;
712 }
713
714 CodechalEncoderState *encoder = dynamic_cast<CodechalEncoderState *>(encodeContext->pCodecHal);
715
716 encodeMfeContext->mfeEncodeSharedState->encoders.push_back(encoder);
717
718 status = encoder->Execute(&encodeContext->EncodeParams);
719 if (MOS_STATUS_SUCCESS != status)
720 {
721 DDI_ASSERTMESSAGE("DDI:Failed in Execute Enc!");
722 return VA_STATUS_ERROR_ENCODING_ERROR;
723 }
724 }
725
726 // Call Pak functions for all the sub contexts
727 for (int32_t i = 0; i < validContextNumber; i++)
728 {
729 encodeContext = encodeContexts[i];
730 if (encodeContext->vaEntrypoint != VAEntrypointFEI )
731 {
732 encodeContext->EncodeParams.ExecCodecFunction = CODECHAL_FUNCTION_PAK;
733 }
734 else
735 {
736 encodeContext->EncodeParams.ExecCodecFunction = CODECHAL_FUNCTION_FEI_PAK;
737 }
738
739 CodechalEncoderState *encoder = dynamic_cast<CodechalEncoderState *>(encodeContext->pCodecHal);
740 status = encoder->Execute(&encodeContext->EncodeParams);
741 if (MOS_STATUS_SUCCESS != status)
742 {
743 DDI_ASSERTMESSAGE("DDI:Failed in Execute Pak!");
744 return VA_STATUS_ERROR_ENCODING_ERROR;
745 }
746 }
747
748 return VA_STATUS_SUCCESS;
749}
