411void GpuContextSpecificNext::Clear()
412{
413 MOS_OS_FUNCTION_ENTER;
414
415 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_START,
416 m_i915Context, sizeof(void *), nullptr, 0);
417 // hanlde the status buf bundled w/ the specified gpucontext
418 if (m_statusBufferResource && m_statusBufferResource->pGfxResourceNext)
419 {
420 if (m_statusBufferResource->pGfxResourceNext->Unlock(m_osContext) != MOS_STATUS_SUCCESS)
421 {
422 MOS_OS_ASSERTMESSAGE("failed to unlock the status buf bundled w/ the specified gpucontext");
423 }
424 m_statusBufferResource->pGfxResourceNext->Free(m_osContext, 0);
425 MOS_Delete(m_statusBufferResource->pGfxResourceNext);
426 }
427 MOS_FreeMemAndSetNull(m_statusBufferResource);
428
429 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
430
431 if (m_cmdBufMgr)
432 {
433 for (auto& curCommandBuffer : m_cmdBufPool)
434 {
435 auto curCommandBufferSpecific = static_cast<CommandBufferSpecificNext *>(curCommandBuffer);
436 if (curCommandBufferSpecific == nullptr)
437 continue;
438 curCommandBufferSpecific->waitReady(); // wait ready and return to comamnd buffer manager.
439 m_cmdBufMgr->ReleaseCmdBuf(curCommandBuffer);
440 }
441 }
442
443 m_cmdBufPool.clear();
444
445 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
446 MosUtilities::MosDestroyMutex(m_cmdBufPoolMutex);
447 m_cmdBufPoolMutex = nullptr;
448 MOS_SafeFreeMemory(m_commandBuffer);
449 MOS_SafeFreeMemory(m_allocationList);
450 MOS_SafeFreeMemory(m_patchLocationList);
451 MOS_SafeFreeMemory(m_attachedResources);
452 MOS_SafeFreeMemory(m_writeModeList);
453 MOS_SafeFreeMemory(m_createOptionEnhanced);
454
455 for (int i=0; i<MAX_ENGINE_INSTANCE_NUM; i++)
456 {
457 if (m_i915Context[i])
458 {
459 mos_gem_context_destroy(m_i915Context[i]);
460 m_i915Context[i] = nullptr;
461 }
462 }
463 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_END,
464 nullptr, 0, nullptr, 0);
465}

467MOS_STATUS GpuContextSpecificNext::RegisterResource(
468 PMOS_RESOURCE osResource,
469 bool writeFlag)
470{
471 MOS_OS_FUNCTION_ENTER;
472
473 MOS_OS_CHK_NULL_RETURN(osResource);
474
475 MOS_OS_CHK_NULL_RETURN(m_attachedResources);
476
477 PMOS_RESOURCE registeredResources = m_attachedResources;
478 uint32_t allocationIndex = 0;
479
480 for ( allocationIndex = 0; allocationIndex < m_resCount; allocationIndex++, registeredResources++)
481 {
482 if (osResource->bo == registeredResources->bo)
483 {
484 break;
485 }
486 }
487
488 // Allocation list to be updated
489 if (allocationIndex < m_maxNumAllocations)
490 {
491 // New buffer
492 if (allocationIndex == m_resCount)
493 {
494 m_resCount++;
495 }
496
497 // Set allocation
498 if (m_gpuContext >= MOS_GPU_CONTEXT_MAX)
499 {
500 MOS_OS_ASSERTMESSAGE("Gpu context exceeds max.");
501 return MOS_STATUS_UNKNOWN;
502 }
503
504 osResource->iAllocationIndex[m_gpuContext] = (allocationIndex);
505 m_attachedResources[allocationIndex] = *osResource;
506 m_writeModeList[allocationIndex] |= writeFlag;
507 m_allocationList[allocationIndex].hAllocation = &m_attachedResources[allocationIndex];
508 m_allocationList[allocationIndex].WriteOperation |= writeFlag;
509 m_numAllocations = m_resCount;
510 }
511 else
512 {
513 MOS_OS_ASSERTMESSAGE("Reached max # registrations.");
514 return MOS_STATUS_UNKNOWN;
515 }
516
517 return MOS_STATUS_SUCCESS;
518}

557MOS_STATUS GpuContextSpecificNext::GetCommandBuffer(
558 PMOS_COMMAND_BUFFER comamndBuffer,
559 uint32_t flags)
560{
561 MOS_OS_FUNCTION_ENTER;
562
563 MOS_OS_CHK_NULL_RETURN(comamndBuffer);
564 MOS_OS_CHK_NULL_RETURN(m_cmdBufMgr);
565 MOS_OS_CHK_NULL_RETURN(m_commandBuffer);
566
567 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
568 CommandBufferNext* cmdBuf = nullptr;
569
570 uint32_t secondaryIdx = flags;
571 bool isPrimaryCmdBuffer = (secondaryIdx == 0);
572 bool hasSecondaryCmdBuffer = (!isPrimaryCmdBuffer &&
573 (m_secondaryCmdBufs.count(secondaryIdx) != 0));
574
575 bool needToAlloc = ((isPrimaryCmdBuffer && m_cmdBufFlushed) ||
576 (!isPrimaryCmdBuffer && !hasSecondaryCmdBuffer));
577
578 if (needToAlloc)
579 {
580 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
581 if (m_cmdBufPool.size() < MAX_CMD_BUF_NUM)
582 {
583 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
584 if (cmdBuf == nullptr)
585 {
586 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
587 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
588 return MOS_STATUS_NULL_POINTER;
589 }
590 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
591 {
592 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
593 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
594 return eStatus;
595 }
596 m_cmdBufPool.push_back(cmdBuf);
597 }
598 else if (m_cmdBufPool.size() == MAX_CMD_BUF_NUM && m_nextFetchIndex < m_cmdBufPool.size())
599 {
600 auto cmdBufOld = m_cmdBufPool[m_nextFetchIndex];
601 auto cmdBufSpecificOld = static_cast<CommandBufferSpecificNext *>(cmdBufOld);
602 if (cmdBufSpecificOld == nullptr)
603 {
604 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
605 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
606 return MOS_STATUS_NULL_POINTER;
607 }
608 cmdBufSpecificOld->waitReady();
609 cmdBufSpecificOld->UnBindToGpuContext();
610 m_cmdBufMgr->ReleaseCmdBuf(cmdBufOld); // here just return old command buffer to available pool
611
612 //pick up new comamnd buffer
613 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
614 if (cmdBuf == nullptr)
615 {
616 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
617 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
618 return MOS_STATUS_NULL_POINTER;
619 }
620 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
621 {
622 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
623 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
624 return eStatus;
625 }
626 m_cmdBufPool[m_nextFetchIndex] = cmdBuf;
627 }
628 else
629 {
630 MOS_OS_ASSERTMESSAGE("Command buffer bool size exceed max.");
631 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
632 return MOS_STATUS_UNKNOWN;
633 }
634 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
635
636 // util now, we got new command buffer from CmdBufMgr, next step to fill in the input command buffer
637 MOS_OS_CHK_STATUS_RETURN(cmdBuf->GetResource()->ConvertToMosResource(&comamndBuffer->OsResource));
638 comamndBuffer->pCmdBase = (uint32_t *)cmdBuf->GetLockAddr();
639 comamndBuffer->pCmdPtr = (uint32_t *)cmdBuf->GetLockAddr();
640 comamndBuffer->iOffset = 0;
641 comamndBuffer->iRemaining = cmdBuf->GetCmdBufSize();
642 comamndBuffer->iCmdIndex = m_nextFetchIndex;
643 comamndBuffer->iVdboxNodeIndex = MOS_VDBOX_NODE_INVALID;
644 comamndBuffer->iVeboxNodeIndex = MOS_VEBOX_NODE_INVALID;
645 comamndBuffer->Attributes.pAttriVe = nullptr;
646 comamndBuffer->is1stLvlBB = true;
647
648 // zero comamnd buffer
649 MosUtilities::MosZeroMemory(comamndBuffer->pCmdBase, comamndBuffer->iRemaining);
650 comamndBuffer->iSubmissionType = SUBMISSION_TYPE_SINGLE_PIPE;
651 MosUtilities::MosZeroMemory(&comamndBuffer->Attributes,sizeof(comamndBuffer->Attributes));
652
653 if (isPrimaryCmdBuffer)
654 {
655 // update command buffer relared filed in GPU context
656 m_cmdBufFlushed = false;
657
658 // keep a copy in GPU context
659 MosUtilities::MosSecureMemcpy(m_commandBuffer, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
660 }
661 else
662 {
663 PMOS_COMMAND_BUFFER tempCmdBuf = (PMOS_COMMAND_BUFFER)MOS_AllocAndZeroMemory(sizeof(MOS_COMMAND_BUFFER));
664 MOS_OS_CHK_NULL_RETURN(tempCmdBuf);
665 m_secondaryCmdBufs[secondaryIdx] = tempCmdBuf;
666 MosUtilities::MosSecureMemcpy(tempCmdBuf, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
667 }
668
669 if (m_ocaLogSectionSupported)
670 {
671 MOS_LINUX_BO *boTemp = ((GraphicsResourceSpecificNext *)cmdBuf->GetResource())->GetBufferObject();
672 MosOcaInterfaceSpecific::InitOcaLogSection(boTemp);
673 }
674
675 // Command buffers are treated as cyclical buffers, the CB after the just submitted one
676 // has the minimal fence value that we should wait
677 m_nextFetchIndex++;
678 if (m_nextFetchIndex >= MAX_CMD_BUF_NUM)
679 {
680 m_nextFetchIndex = 0;
681 }
682 }
683 else
684 {
685 // current command buffer still active, directly copy to comamndBuffer
686 if (isPrimaryCmdBuffer)
687 {
688 MosUtilities::MosSecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_commandBuffer, sizeof(MOS_COMMAND_BUFFER));
689 }
690 else
691 {
692 MosUtilities::MosSecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_secondaryCmdBufs[secondaryIdx], sizeof(MOS_COMMAND_BUFFER));
693 }
694 }
695
696 if (isPrimaryCmdBuffer)
697 {
698 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_commandBuffer->OsResource, false));
699 }
700 else
701 {
702 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_secondaryCmdBufs[secondaryIdx]->OsResource, false));
703 }
704
705 return MOS_STATUS_SUCCESS;
706}

978MOS_STATUS GpuContextSpecificNext::SubmitCommandBuffer(
979 MOS_STREAM_HANDLE streamState,
980 PMOS_COMMAND_BUFFER cmdBuffer,
981 bool nullRendering)
982{
983 MOS_OS_FUNCTION_ENTER;
984
985 MOS_TraceEventExt(EVENT_MOS_BATCH_SUBMIT, EVENT_TYPE_START, nullptr, 0, nullptr, 0);
986
987 MOS_OS_CHK_NULL_RETURN(streamState);
988 auto perStreamParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
989 MOS_OS_CHK_NULL_RETURN(perStreamParameters);
990 MOS_OS_CHK_NULL_RETURN(cmdBuffer);
991 MOS_OS_CHK_NULL_RETURN(m_patchLocationList);
992
993 MOS_GPU_NODE gpuNode = OSKMGetGpuNode(m_gpuContext);
994 uint32_t execFlag = gpuNode;
995 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
996 int32_t ret = 0;
997 bool scalaEnabled = false;
998 auto it = m_secondaryCmdBufs.begin();
999
1000 // Command buffer object DRM pointer
1001 m_cmdBufFlushed = true;
1002 auto cmd_bo = cmdBuffer->OsResource.bo;
1003
1004 // Map Resource to Aux if needed
1005 MapResourcesToAuxTable(cmd_bo);
1006 for(auto it : m_secondaryCmdBufs)
1007 {
1008 MapResourcesToAuxTable(it.second->OsResource.bo);
1009 }
1010
1011 if (m_secondaryCmdBufs.size() >= 2)
1012 {
1013 scalaEnabled = true;
1014 cmdBuffer->iSubmissionType = SUBMISSION_TYPE_MULTI_PIPE_MASTER;
1015 }
1016
1017 std::vector<PMOS_RESOURCE> mappedResList;
1018 std::vector<MOS_LINUX_BO *> skipSyncBoList;
1019
1020 // Now, the patching will be done, based on the patch list.
1021 for (uint32_t patchIndex = 0; patchIndex < m_currentNumPatchLocations; patchIndex++)
1022 {
1023 auto currentPatch = &m_patchLocationList[patchIndex];
1024 MOS_OS_CHK_NULL_RETURN(currentPatch);
1025
1026 auto tempCmdBo = currentPatch->cmdBo == nullptr ? cmd_bo : currentPatch->cmdBo;
1027
1028 // Following are for Nested BB buffer, if it's nested BB, we need to ensure it's locked.
1029 if (tempCmdBo != cmd_bo)
1030 {
1031 bool isSecondaryCmdBuf = false;
1032 it = m_secondaryCmdBufs.begin();
1033 while(it != m_secondaryCmdBufs.end())
1034 {
1035 if (it->second->OsResource.bo == tempCmdBo)
1036 {
1037 isSecondaryCmdBuf = true;
1038 break;
1039 }
1040 it++;
1041 }
1042
1043 for(auto allocIdx = 0; allocIdx < m_numAllocations && (!isSecondaryCmdBuf); allocIdx++)
1044 {
1045 auto tempRes = (PMOS_RESOURCE)m_allocationList[allocIdx].hAllocation;
1046 if (tempCmdBo == tempRes->bo)
1047 {
1048 GraphicsResourceNext::LockParams param;
1049 param.m_writeRequest = true;
1050 tempRes->pGfxResourceNext->Lock(m_osContext, param);
1051 mappedResList.push_back(tempRes);
1052 break;
1053 }
1054 }
1055 }
1056
1057 // This is the resource for which patching will be done
1058 auto resource = (PMOS_RESOURCE)m_allocationList[currentPatch->AllocationIndex].hAllocation;
1059 MOS_OS_CHK_NULL_RETURN(resource);
1060
1061 // For now, we'll assume the system memory's DRM bo pointer
1062 // is NULL. If nullptr is detected, then the resource has been
1063 // placed inside the command buffer's indirect state area.
1064 // We'll simply set alloc_bo to the command buffer's bo pointer.
1065 MOS_OS_ASSERT(resource->bo);
1066
1067 auto alloc_bo = (resource->bo) ? resource->bo : tempCmdBo;
1068
1069 MOS_OS_CHK_STATUS_RETURN(streamState->osCpInterface->PermeatePatchForHM(
1070 tempCmdBo->virt,
1071 currentPatch,
1072 resource));
1073
1074 uint64_t boOffset = alloc_bo->offset64;
1075 if (!mos_gem_bo_is_softpin(alloc_bo))
1076 {
1077 if (alloc_bo != tempCmdBo)
1078 {
1079 auto item_ctx = perStreamParameters->contextOffsetList.begin();
1080 for (; item_ctx != perStreamParameters->contextOffsetList.end(); item_ctx++)
1081 {
1082 if (item_ctx->intel_context == perStreamParameters->intel_context && item_ctx->target_bo == alloc_bo)
1083 {
1084 boOffset = item_ctx->offset64;
1085 break;
1086 }
1087 }
1088 }
1089 }
1090
1091 MOS_OS_CHK_NULL_RETURN(tempCmdBo->virt);
1092 if (perStreamParameters->bUse64BitRelocs)
1093 {
1094 *((uint64_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
1095 boOffset + currentPatch->AllocationOffset;
1096 }
1097 else
1098 {
1099 *((uint32_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
1100 boOffset + currentPatch->AllocationOffset;
1101 }
1102
1103 if (scalaEnabled)
1104 {
1105 it = m_secondaryCmdBufs.begin();
1106 while(it != m_secondaryCmdBufs.end())
1107 {
1108 if (it->second->OsResource.bo == tempCmdBo &&
1109 it->second->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE &&
1110 !mos_gem_bo_is_exec_object_async(alloc_bo))
1111 {
1112 skipSyncBoList.push_back(alloc_bo);
1113 break;
1114 }
1115 it++;
1116 }
1117 }
1118 else if (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE &&
1119 !mos_gem_bo_is_exec_object_async(alloc_bo))
1120 {
1121 skipSyncBoList.push_back(alloc_bo);
1122 }
1123
1124#if (_DEBUG || _RELEASE_INTERNAL)
1125 {
1126 uint32_t evtData[] = {alloc_bo->handle, currentPatch->uiWriteOperation, currentPatch->AllocationOffset};
1127 MOS_TraceEventExt(EVENT_MOS_BATCH_SUBMIT, EVENT_TYPE_INFO,
1128 evtData, sizeof(evtData),
1129 &boOffset, sizeof(boOffset));
1130 }
1131#endif
1132
1133 if(mos_gem_bo_is_softpin(alloc_bo))
1134 {
1135 if (alloc_bo != tempCmdBo)
1136 {
1137 ret = mos_bo_add_softpin_target(tempCmdBo, alloc_bo, currentPatch->uiWriteOperation);
1138 }
1139 }
1140 else
1141 {
1142 // This call will patch the command buffer with the offsets of the indirect state region of the command buffer
1143 ret = mos_bo_emit_reloc2(
1144 tempCmdBo, // Command buffer
1145 currentPatch->PatchOffset, // Offset in the command buffer
1146 alloc_bo, // Allocation object for which the patch will be made.
1147 currentPatch->AllocationOffset, // Offset to the indirect state
1148 I915_GEM_DOMAIN_RENDER, // Read domain
1149 (currentPatch->uiWriteOperation) ? I915_GEM_DOMAIN_RENDER : 0x0, // Write domain
1150 boOffset);
1151 }
1152
1153 if (ret != 0)
1154 {
1155 MOS_OS_ASSERTMESSAGE("Error patching alloc_bo = 0x%x, cmd_bo = 0x%x.",
1156 (uintptr_t)alloc_bo,
1157 (uintptr_t)tempCmdBo);
1158 return MOS_STATUS_UNKNOWN;
1159 }
1160 }
1161
1162 for(auto res: mappedResList)
1163 {
1164 res->pGfxResourceNext->Unlock(m_osContext);
1165 }
1166 mappedResList.clear();
1167
1168 if (scalaEnabled)
1169 {
1170 it = m_secondaryCmdBufs.begin();
1171 while(it != m_secondaryCmdBufs.end())
1172 {
1173 //Add Batch buffer End Command
1174 uint32_t batchBufferEndCmd = MI_BATCHBUFFER_END;
1175 if (MOS_FAILED(Mos_AddCommand(
1176 it->second,
1177 &batchBufferEndCmd,
1178 sizeof(uint32_t))))
1179 {
1180 MOS_OS_ASSERTMESSAGE("Inserting BB_END failed!");
1181 return MOS_STATUS_UNKNOWN;
1182 }
1183 it++;
1184 }
1185 }
1186 else
1187 {
1188 //Add Batch buffer End Command
1189 uint32_t batchBufferEndCmd = MI_BATCHBUFFER_END;
1190 if (MOS_FAILED(Mos_AddCommand(
1191 cmdBuffer,
1192 &batchBufferEndCmd,
1193 sizeof(uint32_t))))
1194 {
1195 MOS_OS_ASSERTMESSAGE("Inserting BB_END failed!");
1196 return MOS_STATUS_UNKNOWN;
1197 }
1198 }
1199
1200 // Now, we can unmap the video command buffer, since we don't need CPU access anymore.
1201 MOS_OS_CHK_NULL_RETURN(cmdBuffer->OsResource.pGfxResourceNext);
1202
1203 cmdBuffer->OsResource.pGfxResourceNext->Unlock(m_osContext);
1204
1205 it = m_secondaryCmdBufs.begin();
1206 while(it != m_secondaryCmdBufs.end())
1207 {
1208 MOS_OS_CHK_NULL_RETURN(it->second->OsResource.pGfxResourceNext);
1209 it->second->OsResource.pGfxResourceNext->Unlock(m_osContext);
1210
1211 it++;
1212 }
1213
1214 int32_t perfData;
1215 if (perStreamParameters->pPerfData != nullptr)
1216 {
1217 perfData = *(int32_t *)(perStreamParameters->pPerfData);
1218 }
1219 else
1220 {
1221 perfData = 0;
1222 }
1223
1224 drm_clip_rect_t *cliprects = nullptr;
1225 int32_t num_cliprects = 0;
1226 int32_t DR4 = perStreamParameters->uEnablePerfTag ? perfData : 0;
1227
1228 //Since CB2 command is not supported, remove it and set cliprects to nullprt as default.
1229 if ((gpuNode == MOS_GPU_NODE_VIDEO || gpuNode == MOS_GPU_NODE_VIDEO2) &&
1230 (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_SINGLE_PIPE_MASK))
1231 {
1232 if (perStreamParameters->bKMDHasVCS2)
1233 {
1234 if (perStreamParameters->bPerCmdBufferBalancing)
1235 {
1236 execFlag = GetVcsExecFlag(cmdBuffer, gpuNode);
1237 }
1238 else if (gpuNode == MOS_GPU_NODE_VIDEO)
1239 {
1240 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING1;
1241 }
1242 else if (gpuNode == MOS_GPU_NODE_VIDEO2)
1243 {
1244 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING2;
1245 }
1246 else if ((gpuNode == MOS_GPU_NODE_BLT))
1247 {
1248 execFlag = I915_EXEC_BLT;
1249 }
1250 else
1251 {
1252 MOS_OS_ASSERTMESSAGE("Invalid gpuNode.");
1253 }
1254 }
1255 else
1256 {
1257 execFlag = I915_EXEC_BSD | I915_EXEC_BSD_RING1;
1258 }
1259 }
1260
1261#if (_DEBUG || _RELEASE_INTERNAL)
1262
1263 MOS_LINUX_BO *nop_cmd_bo = nullptr;
1264
1265 if (nullRendering == true)
1266 {
1267 nop_cmd_bo = GetNopCommandBuffer(streamState);
1268
1269 if (nop_cmd_bo)
1270 {
1271 ret = mos_bo_mrb_exec(nop_cmd_bo,
1272 4096,
1273 nullptr,
1274 0,
1275 0,
1276 execFlag);
1277 }
1278 else
1279 {
1280 MOS_OS_ASSERTMESSAGE("Mos_GetNopCommandBuffer_Linux failed!");
1281 }
1282 }
1283
1284#endif //(_DEBUG || _RELEASE_INTERNAL)
1285
1286 if (gpuNode != I915_EXEC_RENDER &&
1287 streamState->osCpInterface->IsTearDownHappen())
1288 {
1289 // skip PAK command when CP tear down happen to avoid of GPU hang
1290 // conditonal batch buffer start PoC is in progress
1291 }
1292 else if (nullRendering == false)
1293 {
1294 UnlockPendingOcaBuffers(cmdBuffer, perStreamParameters);
1295 if (streamState->ctxBasedScheduling && m_i915Context[0] != nullptr)
1296 {
1297 if (cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASK)
1298 {
1299 if (scalaEnabled && !streamState->bGucSubmission)
1300 {
1301 uint32_t secondaryIndex = 0;
1302 it = m_secondaryCmdBufs.begin();
1303 while(it != m_secondaryCmdBufs.end())
1304 {
1305 if (it->second->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_SLAVE)
1306 {
1307 if(execFlag == MOS_GPU_NODE_VE)
1308 {
1309 // decode excluded since init in other place
1310 it->second->iSubmissionType |= (secondaryIndex << SUBMISSION_TYPE_MULTI_PIPE_SLAVE_INDEX_SHIFT);
1311 secondaryIndex++;
1312 }
1313 }
1314 ret = SubmitPipeCommands(it->second,
1315 it->second->OsResource.bo,
1316 perStreamParameters,
1317 skipSyncBoList,
1318 execFlag,
1319 DR4);
1320 it++;
1321 }
1322 }
1323 else if(scalaEnabled && streamState->bGucSubmission)
1324 {
1325 ret = ParallelSubmitCommands(m_secondaryCmdBufs,
1326 perStreamParameters,
1327 execFlag,
1328 DR4);
1329 }
1330 else
1331 {
1332 ret = SubmitPipeCommands(cmdBuffer,
1333 cmd_bo,
1334 perStreamParameters,
1335 skipSyncBoList,
1336 execFlag,
1337 DR4);
1338 }
1339 }
1340 else
1341 {
1342 ret = mos_gem_bo_context_exec2(cmd_bo,
1343 m_commandBufferSize,
1344 m_i915Context[0],
1345 cliprects,
1346 num_cliprects,
1347 DR4,
1348 m_i915ExecFlag,
1349 nullptr);
1350 }
1351 }
