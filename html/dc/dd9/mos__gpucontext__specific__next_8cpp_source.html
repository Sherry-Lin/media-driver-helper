150MOS_STATUS GpuContextSpecificNext::PatchGPUContextProtection(MOS_STREAM_HANDLE streamState)
151{
152 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
153 MOS_OS_CHK_NULL_RETURN(streamState);
154 auto osParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
155 MOS_OS_CHK_NULL_RETURN(osParameters);
156
157 // clean up clear gem context and create protected gem context if CP is enabled
158 if (streamState->osCpInterface &&
159 streamState->osCpInterface->IsCpEnabled()) // Check if CP is enabled as protected GEM context is only needed when CP is enabled
160 {
161 if (streamState->ctxBasedScheduling)
162 {
163 if (m_bProtectedContext == false) // Check if GEM context is already protected or not
164 {
165 // Context is not protected, recreate it as protected
166 eStatus = RecreateContext(true, streamState);
167 if (eStatus == MOS_STATUS_SUCCESS)
168 {
169 //Register Protected Context
170 streamState->osCpInterface->RegisterAndCheckProtectedGemCtx(true, (void*)this, nullptr);
171 }
172 }
173 //If m_bProtectedContext == true then check if is stale context or not.
174 //If it is stale protected context then recreate another one
175 else
176 {
177 bool bIsContextStale = false;
178 //Check protected context
179 streamState->osCpInterface->RegisterAndCheckProtectedGemCtx(false, (void*)this, &bIsContextStale);
180
181 //Recreate protected context
182 if (bIsContextStale)
183 {
184 eStatus = RecreateContext(true, streamState);
185 if (eStatus == MOS_STATUS_SUCCESS)
186 {
187 //Register Protected Context
188 streamState->osCpInterface->RegisterAndCheckProtectedGemCtx(true, (void*)this, nullptr);
189 }
190 }
191 }
192 }
193 else
194 {
195 if (osParameters->m_protectedGEMContext == false)
196 {
197 // for non context based scheduling protected context is always created as protected during Initialization if needed
198 // If it is not created during Initialization then do nothing and add a comment for Debug purposes
199 MOS_OS_CRITICALMESSAGE("Using Clear GEM context when protected Context is needed");
200 eStatus = MOS_STATUS_SUCCESS;
201 }
202 }
203 }
204
205 // clean up protected gem context and recreate clear gem context if CP is disabled
206 if (streamState->osCpInterface &&
207 !streamState->osCpInterface->IsCpEnabled() &&
208 streamState->ctxBasedScheduling &&
209 m_bProtectedContext == true) // Check if GEM context is protected or not
210 {
211 // Context is protected, recreate it as clear
212 eStatus = RecreateContext(false, streamState);
213 }
214
215 return eStatus;
216}

218MOS_STATUS GpuContextSpecificNext::Init3DCtx(PMOS_CONTEXT osParameters,
219 PMOS_GPUCTX_CREATOPTIONS createOption,
220 unsigned int *nengine,
221 void *engine_map)
222{
223 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
224
225 m_i915Context[0] = mos_context_create_shared(osParameters->bufmgr,
226 osParameters->intel_context,
227 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE,
228 m_bProtectedContext,
229 engine_map,
230 1,
231 *nengine,
232 0);
233 if (m_i915Context[0] == nullptr)
234 {
235 MOS_OS_ASSERTMESSAGE("Failed to create context.\n");
236 return MOS_STATUS_UNKNOWN;
237 }
238 m_i915Context[0]->pOsContext = osParameters;
239
240 __u16 engine_class = I915_ENGINE_CLASS_RENDER;
241 __u64 caps = 0;
242
243 if (mos_query_engines(osParameters->bufmgr, engine_class, caps, nengine, engine_map))
244 {
245 MOS_OS_ASSERTMESSAGE("Failed to query engines.\n");
246 return MOS_STATUS_UNKNOWN;
247 }
248
249 if (mos_set_context_param_load_balance(m_i915Context[0], (struct i915_engine_class_instance *)engine_map, *nengine))
250 {
251 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
252 return MOS_STATUS_UNKNOWN;
253 }
254
255 if (createOption->SSEUValue != 0)
256 {
257 struct drm_i915_gem_context_param_sseu sseu;
258 MosUtilities::MosZeroMemory(&sseu, sizeof(sseu));
259 sseu.flags = I915_CONTEXT_SSEU_FLAG_ENGINE_INDEX;
260 sseu.engine.engine_instance = m_i915ExecFlag;
261
262 if (mos_get_context_param_sseu(m_i915Context[0], &sseu))
263 {
264 MOS_OS_ASSERTMESSAGE("Failed to get sseu configuration.");
265 return MOS_STATUS_UNKNOWN;
266 }
267
268 if (mos_hweight8(m_i915Context[0], sseu.subslice_mask) > createOption->packed.SubSliceCount)
269 {
270 sseu.subslice_mask = mos_switch_off_n_bits(m_i915Context[0], sseu.subslice_mask,
271 mos_hweight8(m_i915Context[0], sseu.subslice_mask)-createOption->packed.SubSliceCount);
272 }
273
274 if (mos_set_context_param_sseu(m_i915Context[0], sseu))
275 {
276 MOS_OS_ASSERTMESSAGE("Failed to set sseu configuration.");
277 return MOS_STATUS_UNKNOWN;
278 }
279 }
280
281 return eStatus;
282}

328MOS_STATUS GpuContextSpecificNext::InitVdVeCtx(PMOS_CONTEXT osParameters,
329 MOS_STREAM_HANDLE streamState,
330 PMOS_GPUCTX_CREATOPTIONS createOption,
331 unsigned int *nengine,
332 void *engine_map,
333 MOS_GPU_NODE gpuNode,
334 bool *isEngineSelectEnable)
335{
336 MOS_OS_FUNCTION_ENTER;
337
338 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
339
340 m_i915Context[0] = mos_context_create_shared(osParameters->bufmgr,
341 osParameters->intel_context,
342 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE,
343 m_bProtectedContext,
344 engine_map,
345 1,
346 *nengine,
347 0);
348 if (m_i915Context[0] == nullptr)
349 {
350 MOS_OS_ASSERTMESSAGE("Failed to create context.\n");
351 return MOS_STATUS_UNKNOWN;
352 }
353 struct i915_engine_class_instance *_engine_map = (struct i915_engine_class_instance *)engine_map;
354 m_i915Context[0]->pOsContext = osParameters;
355
356 __u16 engine_class = (gpuNode == MOS_GPU_NODE_VE)? I915_ENGINE_CLASS_VIDEO_ENHANCE : I915_ENGINE_CLASS_VIDEO;
357 __u64 caps = 0;
358
359 SetEngineQueryFlags(createOption, caps);
360
361 if (mos_query_engines(osParameters->bufmgr, engine_class, caps, nengine, (void *)_engine_map))
362 {
363 MOS_OS_ASSERTMESSAGE("Failed to query engines.\n");
364 return MOS_STATUS_UNKNOWN;
365 }
366
367#if (_DEBUG || _RELEASE_INTERNAL)
368 *isEngineSelectEnable = SelectEngineInstanceByUser((void *)_engine_map, nengine, m_engineInstanceSelect, gpuNode);
369#endif
370 if (mos_set_context_param_load_balance(m_i915Context[0], _engine_map, *nengine))
371 {
372 MOS_OS_ASSERTMESSAGE("Failed to set balancer extension.\n");
373 return MOS_STATUS_UNKNOWN;
374 }
375
376 if (*nengine >= 2 && *nengine <= MAX_ENGINE_INSTANCE_NUM)
377 {
378 int i;
379 //master queue
380 m_i915Context[1] = mos_context_create_shared(osParameters->bufmgr,
381 osParameters->intel_context,
382 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE,
383 m_bProtectedContext,
384 (void *)_engine_map,
385 1,
386 1,
387 0);
388 if (m_i915Context[1] == nullptr)
389 {
390 MOS_OS_ASSERTMESSAGE("Failed to create master context.\n");
391 return MOS_STATUS_UNKNOWN;
392 }
393 m_i915Context[1]->pOsContext = osParameters;
394
395 if (mos_set_context_param_load_balance(m_i915Context[1], _engine_map, 1))
396 {
397 MOS_OS_ASSERTMESSAGE("Failed to set master context bond extension.\n");
398 return MOS_STATUS_UNKNOWN;
399 }
400
401 //slave queue
402 for (i=1; i < *nengine; i++)
403 {
404 m_i915Context[i+1] = mos_context_create_shared(osParameters->bufmgr,
405 osParameters->intel_context,
406 I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE,
407 m_bProtectedContext,
408 (void *)_engine_map,
409 1,
410 1,
411 0);
412 if (m_i915Context[i+1] == nullptr)
413 {
414 MOS_OS_ASSERTMESSAGE("Failed to create slave context.\n");
415 return MOS_STATUS_UNKNOWN;
416 }
417 m_i915Context[i+1]->pOsContext = osParameters;
418
419 if (mos_set_context_param_bond(m_i915Context[i+1], _engine_map[0], &_engine_map[i], 1) != S_SUCCESS)
420 {
421 int err = errno;
422 if (err == ENODEV)
423 {
424 mos_context_destroy(m_i915Context[1]);
425 mos_context_destroy(m_i915Context[i+1]);
426 m_i915Context[i+1] = nullptr;
427 break;
428 }
429 else
430 {
431 MOS_OS_ASSERTMESSAGE("Failed to set slave context bond extension. errno=%d\n",err);
432 return MOS_STATUS_UNKNOWN;
433 }
434 }
435 }
436 if (i == *nengine)
437 {
438 streamState->bParallelSubmission = false;
439 }
440 else
441 {
442 streamState->bParallelSubmission = true;
443 //create context with different width
444 for(i = 1; i < *nengine; i++)
445 {
446 unsigned int ctxWidth = i + 1;
447 m_i915Context[i] = mos_context_create_shared(osParameters->bufmgr,
448 osParameters->intel_context,
449 0, // I915_CONTEXT_CREATE_FLAGS_SINGLE_TIMELINE not allowed for parallel submission
450 m_bProtectedContext,
451 (void *)_engine_map,
452 ctxWidth,
453 1,
454 0);
455 if (mos_set_context_param_parallel(m_i915Context[i], _engine_map, ctxWidth) != S_SUCCESS)
456 {
457 MOS_OS_ASSERTMESSAGE("Failed to set parallel extension since discontinuous logical engine.\n");
458 mos_context_destroy(m_i915Context[i]);
459 m_i915Context[i] = nullptr;
460 break;
461 }
462 }
463 }
464 }
465
466 return eStatus;
467}

511MOS_STATUS GpuContextSpecificNext::Init(OsContextNext *osContext,
512 MOS_STREAM_HANDLE streamState,
513 PMOS_GPUCTX_CREATOPTIONS createOption)
514{
515 MOS_OS_FUNCTION_ENTER;
516
517 MOS_OS_CHK_NULL_RETURN(osContext);
518 MOS_OS_CHK_NULL_RETURN(streamState);
519 MOS_OS_CHK_NULL_RETURN(createOption);
520
521 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
522
523 auto osParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
524 MOS_OS_CHK_NULL_RETURN(osParameters);
525
526 m_osParameters = osParameters;
527
528 MOS_GPU_NODE gpuNode = MOS_GPU_NODE_3D;
529 gpuNode = static_cast<MOS_GPU_NODE>(createOption->gpuNode);
530
531 if (m_cmdBufPoolMutex == nullptr)
532 {
533 m_cmdBufPoolMutex = MosUtilities::MosCreateMutex();
534 }
535
536 MOS_OS_CHK_NULL_RETURN(m_cmdBufPoolMutex);
537
538 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
539
540 m_cmdBufPool.clear();
541
542 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
543
544 if (m_ocaLogSectionSupported)
545 {
546 // increase size for oca log section
547 m_commandBufferSize = MosOcaInterfaceSpecific::IncreaseSize(COMMAND_BUFFER_SIZE);
548 }
549 else
550 {
551 m_commandBufferSize = COMMAND_BUFFER_SIZE;
552 }
553
554 m_nextFetchIndex = 0;
555
556 m_cmdBufFlushed = true;
557
558 m_osContext = osContext;
559
560 MOS_OS_CHK_STATUS_RETURN(AllocateGPUStatusBuf());
561
562 m_commandBuffer = (PMOS_COMMAND_BUFFER)MOS_AllocAndZeroMemory(sizeof(MOS_COMMAND_BUFFER));
563
564 MOS_OS_CHK_NULL_RETURN(m_commandBuffer);
565
566 m_IndirectHeapSize = 0;
567
568 // each thread has its own GPU context, so do not need any lock as guarder here
569 m_allocationList = (ALLOCATION_LIST *)MOS_AllocAndZeroMemory(sizeof(ALLOCATION_LIST) * ALLOCATIONLIST_SIZE);
570 MOS_OS_CHK_NULL_RETURN(m_allocationList);
571 m_maxNumAllocations = ALLOCATIONLIST_SIZE;
572
573 m_patchLocationList = (PATCHLOCATIONLIST *)MOS_AllocAndZeroMemory(sizeof(PATCHLOCATIONLIST) * PATCHLOCATIONLIST_SIZE);
574 MOS_OS_CHK_NULL_RETURN(m_patchLocationList);
575 m_maxPatchLocationsize = PATCHLOCATIONLIST_SIZE;
576
577 m_attachedResources = (PMOS_RESOURCE)MOS_AllocAndZeroMemory(sizeof(MOS_RESOURCE) * ALLOCATIONLIST_SIZE);
578 MOS_OS_CHK_NULL_RETURN(m_attachedResources);
579
580 m_writeModeList = (bool *)MOS_AllocAndZeroMemory(sizeof(bool) * ALLOCATIONLIST_SIZE);
581 MOS_OS_CHK_NULL_RETURN(m_writeModeList);
582
583 m_GPUStatusTag = 1;
584
585 StoreCreateOptions(createOption);
586
587 for (int i=0; i<MAX_ENGINE_INSTANCE_NUM+1; i++)
588 {
589 m_i915Context[i] = nullptr;
590 }
591
592 if (streamState->ctxBasedScheduling)
593 {
594 bool isEngineSelectEnable = false;
595 unsigned int nengine = 0;
596 size_t engine_class_size = 0;
597 void *engine_map = nullptr;
598
599 MOS_TraceEventExt(EVENT_GPU_CONTEXT_CREATE, EVENT_TYPE_START,
600 &gpuNode, sizeof(gpuNode), nullptr, 0);
601
602 m_i915ExecFlag = I915_EXEC_DEFAULT;
603
604 if (mos_query_engines_count(osParameters->bufmgr, &nengine))
605 {
606 MOS_OS_ASSERTMESSAGE("Failed to query engines count.\n");
607 return MOS_STATUS_UNKNOWN;
608 }
609 engine_class_size = mos_get_engine_class_size(osParameters->bufmgr);
610 if (!engine_class_size)
611 {
612 MOS_OS_ASSERTMESSAGE("Failed to get engine class instance size.\n");
613 return MOS_STATUS_UNKNOWN;
614 }
615 engine_map = MOS_AllocAndZeroMemory(nengine * engine_class_size);
616 MOS_OS_CHK_NULL_RETURN(engine_map);
617
618 if (gpuNode == MOS_GPU_NODE_3D)
619 {
620 eStatus = Init3DCtx(osParameters, createOption, &nengine, engine_map);
621 }
622 else if (gpuNode == MOS_GPU_NODE_COMPUTE)
623 {
624 eStatus = InitComputeCtx(osParameters, &nengine, engine_map, gpuNode, &isEngineSelectEnable);
625 }
626 else if (gpuNode == MOS_GPU_NODE_VIDEO || gpuNode == MOS_GPU_NODE_VIDEO2
627 || gpuNode == MOS_GPU_NODE_VE)
628 {
629 eStatus = InitVdVeCtx(osParameters, streamState, createOption, &nengine, engine_map, gpuNode, &isEngineSelectEnable);
630 }
631 else if (gpuNode == MOS_GPU_NODE_BLT)
632 {
633 eStatus = InitBltCtx(osParameters, &nengine, engine_map);
634 }
635 else
636 {
637 MOS_OS_ASSERTMESSAGE("Unknown engine class.\n");
638 MOS_SafeFreeMemory(engine_map);
639 return MOS_STATUS_UNKNOWN;
640 }
641
642 if (eStatus == MOS_STATUS_SUCCESS)
643 {
644 MOS_OS_CHK_STATUS_RETURN(ReportEngineInfo(engine_map, nengine, isEngineSelectEnable));
645 }
646 MOS_SafeFreeMemory(engine_map);
647 MOS_TraceEventExt(EVENT_GPU_CONTEXT_CREATE, EVENT_TYPE_END,
648 m_i915Context, sizeof(void *),
649 &nengine, sizeof(nengine));
650 }
651
652 return eStatus;
653}

655void GpuContextSpecificNext::Clear()
656{
657 MOS_OS_FUNCTION_ENTER;
658
659 MOS_TraceEventExt(EVENT_GPU_CONTEXT_DESTROY, EVENT_TYPE_START,
660 m_i915Context, sizeof(void *), nullptr, 0);
661 // hanlde the status buf bundled w/ the specified gpucontext
662 if (m_statusBufferResource && m_statusBufferResource->pGfxResourceNext)
663 {
664 if (m_statusBufferResource->pGfxResourceNext->Unlock(m_osContext) != MOS_STATUS_SUCCESS)
665 {
666 MOS_OS_ASSERTMESSAGE("failed to unlock the status buf bundled w/ the specified gpucontext");
667 }
668 m_statusBufferResource->pGfxResourceNext->Free(m_osContext, 0);
669 MOS_Delete(m_statusBufferResource->pGfxResourceNext);
670 }
671 MOS_FreeMemAndSetNull(m_statusBufferResource);
672
673 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
674
675 if (m_cmdBufMgr)
676 {
677 for (auto& curCommandBuffer : m_cmdBufPool)
678 {
679 auto curCommandBufferSpecific = static_cast<CommandBufferSpecificNext *>(curCommandBuffer);
680 if (curCommandBufferSpecific == nullptr)
681 continue;
682 curCommandBufferSpecific->waitReady(); // wait ready and return to comamnd buffer manager.
683 m_cmdBufMgr->ReleaseCmdBuf(curCommandBuffer);
684 }
685 }
686
687 m_cmdBufPool.clear();
688
689 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);

805MOS_STATUS GpuContextSpecificNext::GetCommandBuffer(
806 PMOS_COMMAND_BUFFER comamndBuffer,
807 uint32_t flags)
808{
809 MOS_OS_FUNCTION_ENTER;
810
811 MOS_OS_CHK_NULL_RETURN(comamndBuffer);
812 MOS_OS_CHK_NULL_RETURN(m_cmdBufMgr);
813 MOS_OS_CHK_NULL_RETURN(m_commandBuffer);
814
815 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
816 CommandBufferNext* cmdBuf = nullptr;
817
818 uint32_t secondaryIdx = flags;
819 bool isPrimaryCmdBuffer = (secondaryIdx == 0);
820 bool hasSecondaryCmdBuffer = (!isPrimaryCmdBuffer &&
821 (m_secondaryCmdBufs.count(secondaryIdx) != 0));
822
823 bool needToAlloc = ((isPrimaryCmdBuffer && m_cmdBufFlushed) ||
824 (!isPrimaryCmdBuffer && !hasSecondaryCmdBuffer));
825
826 if (needToAlloc)
827 {
828 MosUtilities::MosLockMutex(m_cmdBufPoolMutex);
829 if (m_cmdBufPool.size() < MAX_CMD_BUF_NUM)
830 {
831 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
832 if (cmdBuf == nullptr)
833 {
834 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
835 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
836 return MOS_STATUS_NULL_POINTER;
837 }
838 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
839 {
840 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
841 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
842 return eStatus;
843 }
844 m_cmdBufPool.push_back(cmdBuf);
845 }
846 else if (m_cmdBufPool.size() == MAX_CMD_BUF_NUM && m_nextFetchIndex < m_cmdBufPool.size())
847 {
848 auto cmdBufOld = m_cmdBufPool[m_nextFetchIndex];
849 auto cmdBufSpecificOld = static_cast<CommandBufferSpecificNext *>(cmdBufOld);
850 if (cmdBufSpecificOld == nullptr)
851 {
852 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
853 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
854 return MOS_STATUS_NULL_POINTER;
855 }
856 cmdBufSpecificOld->waitReady();
857 cmdBufSpecificOld->UnBindToGpuContext();
858 m_cmdBufMgr->ReleaseCmdBuf(cmdBufOld); // here just return old command buffer to available pool
859
860 //pick up new comamnd buffer
861 cmdBuf = m_cmdBufMgr->PickupOneCmdBuf(m_commandBufferSize);
862 if (cmdBuf == nullptr)
863 {
864 MOS_OS_ASSERTMESSAGE("Invalid (nullptr) Pointer.");
865 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
866 return MOS_STATUS_NULL_POINTER;
867 }
868 if ((eStatus = cmdBuf->BindToGpuContext(this)) != MOS_STATUS_SUCCESS)
869 {
870 MOS_OS_ASSERTMESSAGE("Invalid status of BindToGpuContext.");
871 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
872 return eStatus;
873 }
874 m_cmdBufPool[m_nextFetchIndex] = cmdBuf;
875 }
876 else
877 {
878 MOS_OS_ASSERTMESSAGE("Command buffer bool size exceed max.");
879 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
880 return MOS_STATUS_UNKNOWN;
881 }
882 MosUtilities::MosUnlockMutex(m_cmdBufPoolMutex);
883
884 // util now, we got new command buffer from CmdBufMgr, next step to fill in the input command buffer
885 MOS_OS_CHK_STATUS_RETURN(cmdBuf->GetResource()->ConvertToMosResource(&comamndBuffer->OsResource));
886 comamndBuffer->pCmdBase = (uint32_t *)cmdBuf->GetLockAddr();
887 comamndBuffer->pCmdPtr = (uint32_t *)cmdBuf->GetLockAddr();
888 comamndBuffer->iOffset = 0;
889 comamndBuffer->iRemaining = cmdBuf->GetCmdBufSize();
890 comamndBuffer->iCmdIndex = m_nextFetchIndex;
891 comamndBuffer->iVdboxNodeIndex = MOS_VDBOX_NODE_INVALID;
892 comamndBuffer->iVeboxNodeIndex = MOS_VEBOX_NODE_INVALID;
893 comamndBuffer->Attributes.pAttriVe = nullptr;
894 comamndBuffer->is1stLvlBB = true;
895
896 // zero comamnd buffer
897 MosUtilities::MosZeroMemory(comamndBuffer->pCmdBase, comamndBuffer->iRemaining);
898 comamndBuffer->iSubmissionType = SUBMISSION_TYPE_SINGLE_PIPE;
899 MosUtilities::MosZeroMemory(&comamndBuffer->Attributes,sizeof(comamndBuffer->Attributes));
900
901 if (isPrimaryCmdBuffer)
902 {
903 // update command buffer relared filed in GPU context
904 m_cmdBufFlushed = false;
905
906 // keep a copy in GPU context
907 MosUtilities::MosSecureMemcpy(m_commandBuffer, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
908 }
909 else
910 {
911 PMOS_COMMAND_BUFFER tempCmdBuf = (PMOS_COMMAND_BUFFER)MOS_AllocAndZeroMemory(sizeof(MOS_COMMAND_BUFFER));
912 MOS_OS_CHK_NULL_RETURN(tempCmdBuf);
913 m_secondaryCmdBufs[secondaryIdx] = tempCmdBuf;
914 MosUtilities::MosSecureMemcpy(tempCmdBuf, sizeof(MOS_COMMAND_BUFFER), comamndBuffer, sizeof(MOS_COMMAND_BUFFER));
915 }
916
917 if (m_ocaLogSectionSupported)
918 {
919 MOS_LINUX_BO *boTemp = ((GraphicsResourceSpecificNext *)cmdBuf->GetResource())->GetBufferObject();
920 MosOcaInterfaceSpecific::InitOcaLogSection(boTemp);
921 }
922
923 // Command buffers are treated as cyclical buffers, the CB after the just submitted one
924 // has the minimal fence value that we should wait
925 m_nextFetchIndex++;
926 if (m_nextFetchIndex >= MAX_CMD_BUF_NUM)
927 {
928 m_nextFetchIndex = 0;
929 }
930 }
931 else
932 {
933 // current command buffer still active, directly copy to comamndBuffer
934 if (isPrimaryCmdBuffer)
935 {
936 MosUtilities::MosSecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_commandBuffer, sizeof(MOS_COMMAND_BUFFER));
937 }
938 else
939 {
940 MosUtilities::MosSecureMemcpy(comamndBuffer, sizeof(MOS_COMMAND_BUFFER), m_secondaryCmdBufs[secondaryIdx], sizeof(MOS_COMMAND_BUFFER));
941 }
942 }
943
944 if (isPrimaryCmdBuffer)
945 {
946 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_commandBuffer->OsResource, false));
947 }
948 else
949 {
950 MOS_OS_CHK_STATUS_RETURN(RegisterResource(&m_secondaryCmdBufs[secondaryIdx]->OsResource, false));
951 }
952
953 return MOS_STATUS_SUCCESS;
954}

1231MOS_STATUS GpuContextSpecificNext::SubmitCommandBuffer(
1232 MOS_STREAM_HANDLE streamState,
1233 PMOS_COMMAND_BUFFER cmdBuffer,
1234 bool nullRendering)
1235{
1236 MOS_OS_FUNCTION_ENTER;
1237
1238 MOS_TraceEventExt(EVENT_MOS_BATCH_SUBMIT, EVENT_TYPE_START, nullptr, 0, nullptr, 0);
1239
1240 MOS_OS_CHK_NULL_RETURN(streamState);
1241 auto perStreamParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
1242 MOS_OS_CHK_NULL_RETURN(perStreamParameters);
1243 MOS_OS_CHK_NULL_RETURN(cmdBuffer);
1244 MOS_OS_CHK_NULL_RETURN(m_patchLocationList);
1245
1246 MOS_GPU_NODE gpuNode = OSKMGetGpuNode(m_gpuContext);
1247 uint32_t execFlag = gpuNode;
1248 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
1249 int32_t ret = 0;
1250 bool scalaEnabled = false;
1251 auto it = m_secondaryCmdBufs.begin();
1252
1253 // Command buffer object DRM pointer
1254 m_cmdBufFlushed = true;
1255 auto cmd_bo = cmdBuffer->OsResource.bo;
1256
1257 // Map Resource to Aux if needed
1258 MapResourcesToAuxTable(cmd_bo);
1259 for(auto it : m_secondaryCmdBufs)
1260 {
1261 MapResourcesToAuxTable(it.second->OsResource.bo);
1262 }
1263
1264 if (m_secondaryCmdBufs.size() >= 2)
1265 {
1266 scalaEnabled = true;
1267 cmdBuffer->iSubmissionType = SUBMISSION_TYPE_MULTI_PIPE_MASTER;
1268 }
1269
1270 std::vector<PMOS_RESOURCE> mappedResList;
1271 std::vector<MOS_LINUX_BO *> skipSyncBoList;
1272
1273 // Now, the patching will be done, based on the patch list.
1274 for (uint32_t patchIndex = 0; patchIndex < m_currentNumPatchLocations; patchIndex++)
1275 {
1276 auto currentPatch = &m_patchLocationList[patchIndex];
1277 MOS_OS_CHK_NULL_RETURN(currentPatch);
1278
1279 auto tempCmdBo = currentPatch->cmdBo == nullptr ? cmd_bo : currentPatch->cmdBo;
1280
1281 // Following are for Nested BB buffer, if it's nested BB, we need to ensure it's locked.
1282 if (tempCmdBo != cmd_bo)
1283 {
1284 bool isSecondaryCmdBuf = false;
1285 it = m_secondaryCmdBufs.begin();
1286 while(it != m_secondaryCmdBufs.end())
1287 {
1288 if (it->second->OsResource.bo == tempCmdBo)
1289 {
1290 isSecondaryCmdBuf = true;
1291 break;
1292 }
1293 it++;
1294 }
1295
1296 for(auto allocIdx = 0; allocIdx < m_numAllocations && (!isSecondaryCmdBuf); allocIdx++)
1297 {
1298 auto tempRes = (PMOS_RESOURCE)m_allocationList[allocIdx].hAllocation;
1299 if (tempCmdBo == tempRes->bo)
1300 {
1301 GraphicsResourceNext::LockParams param;
1302 param.m_writeRequest = true;
1303 tempRes->pGfxResourceNext->Lock(m_osContext, param);
1304 mappedResList.push_back(tempRes);
1305 break;
1306 }
1307 }
1308 }
1309
1310 // This is the resource for which patching will be done
1311 auto resource = (PMOS_RESOURCE)m_allocationList[currentPatch->AllocationIndex].hAllocation;
1312 MOS_OS_CHK_NULL_RETURN(resource);
1313
1314 // For now, we'll assume the system memory's DRM bo pointer
1315 // is NULL. If nullptr is detected, then the resource has been
1316 // placed inside the command buffer's indirect state area.
1317 // We'll simply set alloc_bo to the command buffer's bo pointer.
1318 MOS_OS_ASSERT(resource->bo);
1319
1320 auto alloc_bo = (resource->bo) ? resource->bo : tempCmdBo;
1321
1322 MOS_OS_CHK_STATUS_RETURN(streamState->osCpInterface->PermeatePatchForHM(
1323 tempCmdBo->virt,
1324 currentPatch,
1325 resource));
1326
1327 uint64_t boOffset = alloc_bo->offset64;
1328 if (!mos_bo_is_softpin(alloc_bo))
1329 {
1330 if (alloc_bo != tempCmdBo)
1331 {
1332 auto item_ctx = perStreamParameters->contextOffsetList.begin();
1333 for (; item_ctx != perStreamParameters->contextOffsetList.end(); item_ctx++)
1334 {
1335 if (item_ctx->intel_context == perStreamParameters->intel_context && item_ctx->target_bo == alloc_bo)
1336 {
1337 boOffset = item_ctx->offset64;
1338 break;
1339 }
1340 }
1341 }
1342 }
1343
1344 MOS_OS_CHK_NULL_RETURN(tempCmdBo->virt);
1345 if (perStreamParameters->bUse64BitRelocs)
1346 {
1347 *((uint64_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
1348 boOffset + currentPatch->AllocationOffset;
1349 }
1350 else
1351 {
1352 *((uint32_t *)((uint8_t *)tempCmdBo->virt + currentPatch->PatchOffset)) =
1353 boOffset + currentPatch->AllocationOffset;

1715 MOS_COMMAND_BUFFER *cmdBuffer,
1716 MOS_LINUX_BO *cmdBo,
1717 PMOS_CONTEXT osContext,
1718 const std::vector<MOS_LINUX_BO *> &skipSyncBoList,
1719 uint32_t execFlag,
1720 int32_t dr4)

1722 int32_t ret = 0;
1723 int fence = -1;

1726 MOS_LINUX_CONTEXT *queue = m_i915Context[0];

1756 queue = m_i915Context[cmdBuffer->iVeboxNodeIndex + 1];

1761 mos_bo_set_exec_object_async(cmdBo, bo);

1766 {
1767 //Only master pipe needs fence out flag
1768 fence_flag = I915_EXEC_FENCE_OUT;
1769 queue = m_i915Context[1];
1770 }
1771
1772 ret = mos_bo_context_exec2(cmdBo,
1773 cmdBo->size,
1774 queue,
1775 nullptr,
1776 0,
1777 dr4,
1778 execFlag | fence_flag,
1779 &fence);
1780
1781 if(cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_MASTER)
1782 {
1783 osContext->submit_fence = fence;
1784 }
1785
1786 if(cmdBuffer->iSubmissionType & SUBMISSION_TYPE_MULTI_PIPE_FLAGS_LAST_PIPE)
1787 {
1788 close(fence);
1789 }
1790
1791 return ret;
