354{
355 uint32_t i = 0;
356 MOS_MEMCOMP_STATE resMmcMode = {};
357 if (!m_isOcaEnabled || !m_isInitialized)
358 {
359 return MOS_STATUS_SUCCESS;
360 }
361
362 if (!m_ocaBufContextList[ocaBufHandle].is1stLevelBBStarted)
363 {
364 MOS_OS_ASSERTMESSAGE("AddResourceToDumpList is called before oca buffer initialized!");
365 MosOcaInterfaceSpecific::OnOcaError(mosCtx, MOS_STATUS_UNINITIALIZED, __FUNCTION__, __LINE__);
366 return MOS_STATUS_UNINITIALIZED;
367 }
368
369 if (nullptr == m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList)
370 {
371 MOS_OS_NORMALMESSAGE("OCA resource info dump not enabled.");
372 return MOS_STATUS_SUCCESS;
373 }
374
375 if (nullptr == resource.pGmmResInfo || Mos_ResourceIsNull(&resource))
376 {
377 return MOS_STATUS_SUCCESS;
378 }
379
380 if (hwCmdType >= MOS_HW_COMMAND_MAX || MOS_HW_COMMAND_MAX > 64)
381 {
382 MosOcaInterfaceSpecific::OnOcaError(mosCtx, MOS_STATUS_UNINITIALIZED, __FUNCTION__, __LINE__);
383 return MOS_STATUS_INVALID_PARAMETER;
384 }
385
386 for (i = 0; i < m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount; ++i)
387 {
388 if (m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].allocationHandle == resource.bo->handle)
389 {
390 if (offsetInRes > m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes)
391 {
392 // Only update resource info for the largest offsetInRes case.
393 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].hwCmdType = (uint32_t)hwCmdType;
394 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes = offsetInRes;
395 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].locationInCmd = locationInCmd;
396 }
397 return MOS_STATUS_SUCCESS;
398 }
399 }
400
401 if (m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount >= m_ocaBufContextList[ocaBufHandle].logSection.resInfo.maxResInfoCount)
402 {
403 // Not reture error but record the resource count skipped.
404 ++m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped;
405 MOS_OS_NORMALMESSAGE("The resource is skipped to be dumpped to oca buffer.");
406 return MOS_STATUS_SUCCESS;
407 }
408

410 steamState.osDeviceContext = mosCtx->m_osDeviceContext;
411 if (MOS_FAILED(MosInterface::GetMemoryCompressionMode(&steamState, &resource, resMmcMode)))

416 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeAllocation = resource.pGmmResInfo->GetSizeAllocation();
417 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeSurface = resource.pGmmResInfo->GetSizeSurface();
418 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeSurfacePhy = resource.pGmmResInfo->GetSizeSurface();
419 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].sizeMainSurface = resource.pGmmResInfo->GetSizeMainSurface();
420 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].allocationHandle = resource.bo->handle;
421 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].hwCmdType = (uint32_t)hwCmdType;

423 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].offsetInRes = offsetInRes;
424 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].pitch = (uint32_t)resource.pGmmResInfo->GetRenderPitch();
425 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].width = (uint32_t)resource.pGmmResInfo->GetBaseWidth();
426 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].height = (uint32_t)resource.pGmmResInfo->GetBaseHeight();
427 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].format = (uint32_t)resource.Format;
428 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmFormat = (uint32_t)resource.pGmmResInfo->GetResourceFormat();
429 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmTileMode = (uint32_t)resource.pGmmResInfo->GmmGetTileMode();
430 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmClient = (uint32_t)resource.pGmmResInfo->GetClientType();
431 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList[i].gmmResUsageType = (uint32_t)resource.pGmmResInfo->GetCachePolicyUsage();

451void MosOcaInterfaceSpecific::AddResourceInfoToLogSection(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_CONTEXT mosCtx)
452{
453 if (!m_ocaBufContextList[ocaBufHandle].is1stLevelBBStarted)
454 {
455 return;
456 }
457 if (0 == m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount)
458 {
459 return;
460 }
461 if (!IsLogSectionEnabled(ocaBufHandle))
462 {
463 return;
464 }
465
466 MOS_OCA_LOG_HEADER_RESOURCE_INFO header = {};
467 header.header.type = MOS_OCA_LOG_TYPE_RESOURCE_INFO;
468 header.header.headerSize = sizeof(MOS_OCA_LOG_HEADER_RESOURCE_INFO);
469 header.header.dataSize = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount * sizeof(MOS_OCA_RESOURCE_INFO);
470 header.resCount = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount;
471 header.resCountSkipped = m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped;
472
473 MOS_STATUS status = DumpDataBlock(ocaBufHandle, (PMOS_OCA_LOG_HEADER)&header, m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resInfoList);
474 if (MOS_FAILED(status))
475 {
476 MosOcaInterfaceSpecific::OnOcaError(mosCtx, status, __FUNCTION__, __LINE__);
477 }
478 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount = 0;
479 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped = 0;
480 return;
481}

545MOS_STATUS MosOcaInterfaceSpecific::DumpDataBlock(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_OCA_LOG_HEADER pHeader, void *pData)
546{
547 if (pHeader->headerSize < sizeof(MOS_OCA_LOG_HEADER) ||
548 pHeader->type >= MOS_OCA_LOG_TYPE_COUNT || pHeader->type <= MOS_OCA_LOG_TYPE_INVALID ||
549 (pHeader->dataSize > 0 && nullptr == pData) || (pHeader->dataSize == 0 && pData))
550 {
551 return MOS_STATUS_INVALID_PARAMETER;
552 }
553
554 uint32_t copySize = pHeader->headerSize + pHeader->dataSize;
555 if (m_ocaBufContextList[ocaBufHandle].logSection.offset + copySize > m_ocaLogSectionSizeLimit)
556 {
557 return MOS_STATUS_NOT_ENOUGH_BUFFER;
558 }
559
560 MOS_OS_CHK_STATUS_RETURN(InsertData(ocaBufHandle, (uint8_t *)pHeader, pHeader->headerSize));
561 if (pHeader->dataSize > 0)
562 {
563 MOS_OS_CHK_STATUS_RETURN(InsertData(ocaBufHandle, (uint8_t *)pData, pHeader->dataSize));
564 }
565
566 return MOS_STATUS_SUCCESS;
567}

569MOS_STATUS MosOcaInterfaceSpecific::InsertData(MOS_OCA_BUFFER_HANDLE ocaBufHandle, uint8_t *p, uint32_t size)
570{
571 if (nullptr == p || 0 == size)
572 {
573 return MOS_STATUS_INVALID_PARAMETER;
574 }
575
576 uint32_t sizeCpy = std::min(m_ocaLogSectionSizeLimit - m_ocaBufContextList[ocaBufHandle].logSection.offset, size);
577 MOS_OS_CHK_STATUS_RETURN(MOS_SecureMemcpy((uint8_t *)m_ocaBufContextList[ocaBufHandle].logSection.base + m_ocaBufContextList[ocaBufHandle].logSection.offset, sizeCpy, p, sizeCpy));
578 m_ocaBufContextList[ocaBufHandle].logSection.offset += sizeCpy;
579 return MOS_STATUS_SUCCESS;
580}

597void MosOcaInterfaceSpecific::InitLogSection(MOS_OCA_BUFFER_HANDLE ocaBufHandle, PMOS_RESOURCE resCmdBuf)
598{
599 MOS_LINUX_BO *boCmdBuf = resCmdBuf->bo;
600 if (boCmdBuf->size <= OCA_LOG_SECTION_SIZE_MAX)
601 {
602 MOS_OS_NORMALMESSAGE("1st level BB size is not larger than OCA_LOG_SECTION_SIZE_MAX.");
603 return;
604 }
605
606 uint64_t *logSectionBase = (uint64_t*)((char *)boCmdBuf->virt + boCmdBuf->size - OCA_LOG_SECTION_SIZE_MAX);
607 OCA_LOG_SECTION_HEADER *header = (OCA_LOG_SECTION_HEADER *)logSectionBase;
608
609 if (OCA_LOG_SECTION_MAGIC_NUMBER != header->magicNum)
610 {
611 MOS_OS_NORMALMESSAGE("Log section not exists in current 1st level BB.");
612 return;
613 }
614
615 m_ocaBufContextList[ocaBufHandle].logSection.base = logSectionBase;
616 // Reserve an uint64 for magic number;
617 m_ocaBufContextList[ocaBufHandle].logSection.offset = sizeof(OCA_LOG_SECTION_HEADER);
618 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCount = 0;
619 m_ocaBufContextList[ocaBufHandle].logSection.resInfo.resCountSkipped = 0;
620}

638void MosOcaInterfaceSpecific::OnOcaError(PMOS_CONTEXT mosCtx, MOS_STATUS status, const char *functionName, uint32_t lineNumber)
639{
640 if (MOS_SUCCEEDED(status))
641 {
642 return;
643 }
644
645 MOS_OS_ASSERTMESSAGE("%s failed at line %d with status = %d!", functionName, lineNumber, status);
646
647 MOS_TraceEventExt(EVENT_OCA_ERROR,
648 EVENT_TYPE_INFO,
649 &status,
650 sizeof(uint32_t),
651 nullptr,
652 0);
653
654 if (MOS_SUCCEEDED(s_ocaStatus))
655 {
656 MediaUserSettingSharedPtr userSettingPtr = nullptr;
657
658 s_ocaStatus = status;
659 s_lineNumForOcaErr = lineNumber;
660
661 userSettingPtr = MosInterface::MosGetUserSettingInstance(mosCtx);
662 ReportUserSetting(
663 userSettingPtr,
664 __MEDIA_USER_FEATURE_VALUE_OCA_STATUS,
665 status,
666 MediaUserSetting::Group::Device);
667
668 ReportUserSetting(
669 userSettingPtr,
670 __MEDIA_USER_FEATURE_VALUE_OCA_ERROR_HINT,
671 lineNumber,
672 MediaUserSetting::Group::Device);
673 }
674}

680{
681 if (!s_bOcaStatusExistInReg)

703{

708 // read isOcaEnabled from reg key

714 MediaUserSetting::Group::Device);
715
716 m_isOcaEnabled = value;

720 }

722 m_resInfoPool = MOS_NewArray(MOS_OCA_RESOURCE_INFO, m_config.maxResInfoCount * MAX_NUM_OF_OCA_BUF_CONTEXT);
723 if (nullptr == m_resInfoPool)

728 for (int i = 0; i < MAX_NUM_OF_OCA_BUF_CONTEXT; ++i)
729 {
730 m_ocaBufContextList[i].logSection.resInfo.resInfoList = m_resInfoPool + i * m_config.maxResInfoCount;
731 }
732 if (nullptr != m_ocaMutex)
733 {
734 return;
735 }
736

742 }

745 if (nullptr == m_mutexForOcaBufPool)

747 MOS_DeleteArray(m_resInfoPool);
748 MosUtilities::MosDestroyMutex(m_ocaMutex);
749 m_ocaMutex = nullptr;
750 return;
751 }
752
753 m_isInitialized = true;
754 }
