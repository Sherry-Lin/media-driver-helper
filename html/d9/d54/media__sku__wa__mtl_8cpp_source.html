246 MediaWaTable *waTable,

248{

254
255 MEDIA_WR_WA(waTable, WaForceGlobalGTT, !drvInfo->hasPpgtt);
256 MEDIA_WR_WA(waTable, WaMidBatchPreemption, 0);
257 MEDIA_WR_WA(waTable, WaArbitraryNumMbsInSlice, 1);
258
259 MEDIA_WR_WA(waTable, WaSFC270DegreeRotation, 0);
260
261 MEDIA_WR_WA(waTable, WaEnableYV12BugFixInHalfSliceChicken7, 1);
262
263 MOS_USER_FEATURE_VALUE_DATA userFeatureData;
264 MOS_ZeroMemory(&userFeatureData, sizeof(userFeatureData));
265 MOS_UserFeature_ReadValue_ID(
266 nullptr,
267 __MEDIA_USER_FEATURE_VALUE_AUX_TABLE_16K_GRANULAR_ID,
268 &userFeatureData,
269 (MOS_CONTEXT_HANDLE)nullptr);
270
271 MEDIA_WR_WA(waTable, Wa16KInputHeightNV12Planar420, 1);
272
273 /*software wa to disable calculate the UV offset by gmmlib
274 CPU blt call will add/remove padding on the platform*/
275 MEDIA_WR_WA(waTable, WaDisableGmmLibOffsetInDeriveImage, 1);
276
277
278
279 /* Turn off MMC for codec, need to remove once turn it on */
280 MEDIA_WR_WA(waTable, WaDisableCodecMmc, 0);
281
282 /* Turn off MMC for VPP, need to remove once turn it on */
283 MEDIA_WR_WA(waTable, WaDisableVPMmc, 0);

286
287 MEDIA_WR_WA(waTable, Wa_Vp9UnalignedHeight, 1);
288
289 MEDIA_WR_WA(waTable, Wa_15013355402, 1);
