43CM_RT_API int32_t CmEventRT::GetStatus(CM_STATUS &status)
44{
45 static const uint64_t TIME_OUT = 10000; // 10 microseconds.
46 if (m_status == CM_STATUS_FLUSHED || m_status == CM_STATUS_STARTED)
47 {
48 if (!m_osSignalTriggered)
49 {
50 CM_CHK_NULL_RETURN_CMERROR(m_osData);
51 MOS_LINUX_BO *buffer_object = reinterpret_cast<MOS_LINUX_BO*>(m_osData);
52 int result = mos_bo_wait(buffer_object, TIME_OUT);
53 mos_bo_clear_relocs(buffer_object, 0);
54 m_osSignalTriggered = (result == 0);
55 }
56 if (m_osSignalTriggered)
57 {
58 Query();
59 }
60 }
61
62 m_queue->FlushTaskWithoutSync();
63
64 status = m_status;
65 return CM_SUCCESS;
66}

79CM_RT_API int32_t CmEventRT::WaitForTaskFinished(uint32_t timeOutMs)
80{
81 int32_t result = CM_SUCCESS;
82
83 if( m_status == CM_STATUS_FINISHED )
84 goto finish;
85
86 //Make sure task flushed
87 while ( m_status == CM_STATUS_QUEUED )
88 {
89 m_queue->FlushTaskWithoutSync(); //Flush none if 1st task NOT finished yet
90 }
91
92 CM_ASSERT(m_osData != nullptr);
93
94 //Wait bo finished
95 result = mos_bo_wait((MOS_LINUX_BO*)m_osData, 1000000LL*timeOutMs);
96 mos_bo_clear_relocs((MOS_LINUX_BO*)m_osData, 0);
97 if (result) {
98 result = CM_EXCEED_MAX_TIMEOUT; //translate the drm ecode (-ETIME or potentional variants) to CM ecode.
99 goto finish;
100 }
101
102 //Query status
103 Query();
104 if(m_status != CM_STATUS_FINISHED)
105 {
106 // if bo_wai() returns success but status is not finished in time stamp
107 // it indicates something wrong in KMD, such as gpu reset happens.
108 // need to return error code to application.
109 result = CM_EXCEED_MAX_TIMEOUT;
110 goto finish;
111 }
112
113 //Call flush to pop/destroy finished task and flush task in EnqueueedQueue.
114 m_queue->FlushTaskWithoutSync();
115
116finish:
117 return result;
118}
