50VAStatus MediaLibvaCapsG10Cnl::GetMbProcessingRateEnc(
51 MEDIA_FEATURE_TABLE *skuTable,
52 uint32_t tuIdx,
53 uint32_t codecMode,
54 bool vdencActive,
55 uint32_t *mbProcessingRatePerSec)
56{
57 DDI_CHK_NULL(skuTable, "Null pointer", VA_STATUS_ERROR_INVALID_PARAMETER);
58 DDI_CHK_NULL(mbProcessingRatePerSec, "Null pointer", VA_STATUS_ERROR_INVALID_PARAMETER);
59
60 uint32_t gtIdx = 0;
61
62 if (vdencActive)
63 {
64 // For AVC Vdenc
65 if (codecMode == CODECHAL_ENCODE_MODE_AVC)
66 {
67 if (MEDIA_IS_SKU(skuTable, FtrULX))
68 {
69 const uint32_t mbRate[7] =
70 {
71 // TU7 | TU6 | TU5 | TU4 | TU3 | TU2 | TU1
72 1200000, 1200000, 800000, 800000, 800000, 600000, 600000
73 };
74
75 *mbProcessingRatePerSec = mbRate[tuIdx];
76 }
77 else // same numbers for ULT/ classic
78 {
79 const uint32_t mbRate[7] =
80 {
81 // TU7 | TU6 | TU5 | TU4 | TU3 | TU2 | TU1
82 2200000, 2200000, 1650000, 1650000, 1650000, 1100000, 1100000
83 };
84
85 *mbProcessingRatePerSec = mbRate[tuIdx];
86 }
87 }
88 else if (codecMode == CODECHAL_ENCODE_MODE_HEVC)
89 {
90 if (MEDIA_IS_SKU(skuTable, FtrULX))
91 {
92 const uint32_t mbRate[7] =
93 {
94 // TU7 | TU6 | TU5 | TU4 | TU3 | TU2 | TU1
95 1200000, 1200000, 600000, 600000, 600000, 300000, 300000
96 };
97
98 *mbProcessingRatePerSec = mbRate[tuIdx];
99 }
100 else // same numbers for ULT/ classic
101 {
102 const uint32_t mbRate[7] =
103 {
104 // TU7 | TU6 | TU5 | TU4 | TU3 | TU2 | TU1
105 2200000, 2200000, 1200000, 1200000, 1200000, 550000, 550000
106 };
107
108 *mbProcessingRatePerSec = mbRate[tuIdx];
109 }
110 }
111 }
112 else
113 {
114 DDI_CHK_NULL(m_mediaCtx->pGtSystemInfo, "Null pointer", VA_STATUS_ERROR_INVALID_PARAMETER);
115 //Dual pipe mode
116 if (m_mediaCtx->pGtSystemInfo->EUCount == 16) //FtrGT0_5
117 {
118 gtIdx = 5;
119 }
120 else if (m_mediaCtx->pGtSystemInfo->EUCount == 24) //FtrGT1
121 {
122 gtIdx = 4;
123 }
124 else if (m_mediaCtx->pGtSystemInfo->EUCount == 32) //FtrGT1_5
125 {
126 gtIdx = 3;
127 }
128 else if (m_mediaCtx->pGtSystemInfo->EUCount == 40) //FtrGT2
129 {
130 gtIdx = 2;
131 }
132 else if (m_mediaCtx->pGtSystemInfo->EUCount == 56) //FtrGT3
133 {
134 gtIdx = 1;
135 }
136 else if (m_mediaCtx->pGtSystemInfo->EUCount == 72) //FtrGT4
137 {
138 gtIdx = 0;
139 }
140 else
141 {
142 return VA_STATUS_ERROR_INVALID_PARAMETER;
143 }
144 // AVC Dual pipe mode. Data obtained from regular KBL
145 if (codecMode == CODECHAL_ENCODE_MODE_AVC)
146 {
147 if (MEDIA_IS_SKU(skuTable, FtrULX))
148 {
149 const uint32_t mbRate[7][6] =
150 {
151 // GT4 | GT3 | GT2 | GT1.5 | GT1 | GT0.5
152 { 0, 0, 1029393, 1029393, 676280, 676280 },
153 { 0, 0, 975027, 975027, 661800, 661800 },
154 { 0, 0, 776921, 776921, 640000, 640000 },
155 { 0, 0, 776921, 776921, 640000, 640000 },
156 { 0, 0, 776921, 776921, 640000, 640000 },
157 { 0, 0, 416051, 416051, 317980, 317980 },
158 { 0, 0, 214438, 214438, 180655, 180655 }
159 };
160 if (gtIdx == 0 || gtIdx == 1)
161 {
162 return VA_STATUS_ERROR_INVALID_PARAMETER;
163 }
164 *mbProcessingRatePerSec = mbRate[tuIdx][gtIdx];
165 }
166 else // same numbers for ULT/ classic
167 {
168 const uint32_t mbRate[7][6] =
169 {
170 // GT4 | GT3 | GT2 | GT1.5 | GT1 | GT0.5
171 { 1544090, 1544090, 1544090, 1029393, 676280, 676280 },
172 { 1462540, 1462540, 1462540, 975027, 661800, 661800 },
173 { 1165381, 1165381, 1165381, 776921, 640000, 640000 },
174 { 1165381, 1165381, 1165381, 776921, 640000, 640000 },
175 { 1165381, 1165381, 1165381, 776921, 640000, 640000 },
176 { 624076, 624076, 624076, 416051, 317980, 317980 },
177 { 321657, 321657, 321657, 214438, 180655, 180655 }
178 };
179 *mbProcessingRatePerSec = mbRate[tuIdx][gtIdx];
180 }
181 }
182 else if (codecMode == CODECHAL_ENCODE_MODE_HEVC)
183 {
184 // HEVC dual pipe values temporarily set to 50000 for ULT/ULX all TUs and GT
185 const uint32_t mbRate[7][6] =
186 {
187 // GT4 | GT3 | GT2 | GT1.5 | GT1 | GT0.5
188 { 500000, 500000, 500000, 500000, 500000, 500000 },
189 { 500000, 500000, 500000, 500000, 500000, 500000 },
190 { 250000, 250000, 250000, 250000, 250000, 250000 },
191 { 250000, 250000, 250000, 250000, 250000, 250000 },
192 { 250000, 250000, 250000, 250000, 250000, 250000 },
193 { 125000, 125000, 125000, 125000, 125000, 250000 },
194 { 125000, 125000, 125000, 125000, 125000, 250000 }
195 };
196 *mbProcessingRatePerSec = mbRate[tuIdx][gtIdx];
197 }
198 else
199 {
200 return VA_STATUS_ERROR_INVALID_PARAMETER;
201 }
202 }
203 return VA_STATUS_SUCCESS;
204}
