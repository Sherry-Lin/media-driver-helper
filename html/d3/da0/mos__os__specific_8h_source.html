190static inline MOS_GPU_NODE OSKMGetGpuNode(MOS_GPU_CONTEXT uiGpuContext)
191{
192 switch (uiGpuContext)
193 {
194 case MOS_GPU_CONTEXT_RENDER:
195 case MOS_GPU_CONTEXT_RENDER2:
196 case MOS_GPU_CONTEXT_RENDER3:
197 case MOS_GPU_CONTEXT_RENDER4:
198 case MOS_GPU_CONTEXT_COMPUTE: //change this context mapping to Compute Node instead of 3D node after the node name is defined in linux.
199 return MOS_GPU_NODE_3D;
200 break;
201 case MOS_GPU_CONTEXT_VEBOX:
202 return MOS_GPU_NODE_VE;
203 break;
204 case MOS_GPU_CONTEXT_VIDEO:
205 case MOS_GPU_CONTEXT_VIDEO2:
206 case MOS_GPU_CONTEXT_VIDEO3:
207 case MOS_GPU_CONTEXT_VIDEO4:
208 case MOS_GPU_CONTEXT_VIDEO5:
209 case MOS_GPU_CONTEXT_VIDEO6:
210 case MOS_GPU_CONTEXT_VIDEO7:
211 return MOS_GPU_NODE_VIDEO;
212 break;
213 case MOS_GPU_CONTEXT_VDBOX2_VIDEO:
214 case MOS_GPU_CONTEXT_VDBOX2_VIDEO2:
215 case MOS_GPU_CONTEXT_VDBOX2_VIDEO3:
216 return MOS_GPU_NODE_VIDEO2;
217 break;
218 case MOS_GPU_CONTEXT_BLT:
219 return MOS_GPU_NODE_BLT;
220 break;
221 default:
222 return MOS_GPU_NODE_MAX ;
223 break;
224 }
