54MOS_STATUS GraphicsResourceSpecificNext::Allocate(OsContextNext* osContextPtr, CreateParams& params)
55{
56 MOS_OS_FUNCTION_ENTER;
57
58 if (osContextPtr == nullptr)
59 {
60 MOS_OS_ASSERTMESSAGE("Unable to get the active OS context.");
61 return MOS_STATUS_INVALID_HANDLE;
62 }
63
64 if (osContextPtr->GetOsContextValid() == false)
65 {
66 MOS_OS_ASSERTMESSAGE("The OS context got is not valid.");
67 return MOS_STATUS_INVALID_HANDLE;
68 }
69
70 OsContextSpecificNext *pOsContextSpecific = static_cast<OsContextSpecificNext *>(osContextPtr);
71 if (pOsContextSpecific == nullptr)
72 {
73 MOS_OS_ASSERTMESSAGE("Convert OsContextSpecific failed.");
74 return MOS_STATUS_INVALID_HANDLE;
75 }
76
77 MOS_STATUS status = MOS_STATUS_SUCCESS;
78 uint32_t tileFormatLinux = I915_TILING_NONE;
79 uint32_t alignedHeight = params.m_height;
80 uint32_t bufHeight = params.m_height;
81 GMM_RESOURCE_TYPE resourceType = RESOURCE_2D;
82 int32_t mem_type = MOS_MEMPOOL_VIDEOMEMORY;
83
84 GMM_RESCREATE_PARAMS gmmParams;
85 MosUtilities::MosZeroMemory(&gmmParams, sizeof(gmmParams));
86
87 gmmParams.Usage = params.m_gmmResUsageType;
88
89 switch (params.m_type)
90 {
91 case MOS_GFXRES_BUFFER:
92 case MOS_GFXRES_SCRATCH:
93 gmmParams.Type = RESOURCE_BUFFER;
94 gmmParams.Flags.Gpu.State = true;
95 alignedHeight = 1;
96 break;
97
98 case MOS_GFXRES_2D:
99 gmmParams.Type = RESOURCE_2D;
100 gmmParams.Flags.Gpu.Video = true;
101 break;
102
103 case MOS_GFXRES_VOLUME:
104 gmmParams.Type = RESOURCE_3D;
105 gmmParams.Flags.Gpu.Video = true;
106 gmmParams.Depth = params.m_depth;
107 break;
108
109 default:
110 MOS_OS_ASSERTMESSAGE("Unknown surface type");
111 return MOS_STATUS_UNKNOWN;
112 }
113
114 // Create GmmResourceInfo
115 gmmParams.Format = MosInterface::MosFmtToGmmFmt(params.m_format);
116 if (gmmParams.Format == GMM_FORMAT_INVALID)
117 {
118 MOS_OS_ASSERTMESSAGE("Unsupported format");
119 return MOS_STATUS_UNIMPLEMENTED;
120 }
121 gmmParams.BaseWidth = params.m_width;
122 gmmParams.BaseHeight = alignedHeight;
123 gmmParams.ArraySize = 1;
124
125 MOS_TILE_TYPE tileformat = params.m_tileType;
126 MOS_OS_NORMALMESSAGE("tilemode: tileformat = %d for %s, tileModeByForce = %d", tileformat, params.m_name.c_str(), params.m_tileModeByForce);
127 switch (tileformat)
128 {
129 case MOS_TILE_Y:
130 tileFormatLinux = I915_TILING_Y;
131 if (params.m_isCompressible &&
132 MEDIA_IS_SKU(pOsContextSpecific->GetSkuTable(), FtrE2ECompression) &&
133 MEDIA_IS_SKU(pOsContextSpecific->GetSkuTable(), FtrCompressibleSurfaceDefault))
134 {
135 gmmParams.Flags.Gpu.MMC = 1;
136
137 if (params.m_compressionMode == MOS_MMC_RC)
138 {
139 gmmParams.Flags.Info.MediaCompressed = 0;
140 gmmParams.Flags.Info.RenderCompressed = 1;
141 }
142 else
143 {
144 gmmParams.Flags.Info.MediaCompressed = 1;
145 gmmParams.Flags.Info.RenderCompressed = 0;
146 }
147
148 gmmParams.Flags.Gpu.CCS = 1;
149 gmmParams.Flags.Gpu.UnifiedAuxSurface = 1;
150 gmmParams.Flags.Gpu.RenderTarget = 1;
151
152 if(MEDIA_IS_SKU(pOsContextSpecific->GetSkuTable(), FtrFlatPhysCCS))
153 {
154 gmmParams.Flags.Gpu.UnifiedAuxSurface = 0;
155 }
156 }
157 SetTileModebyForce(gmmParams, params.m_tileModeByForce);
158 break;
159 case MOS_TILE_X:
160 gmmParams.Flags.Info.TiledX = true;
161 tileFormatLinux = I915_TILING_X;
162 break;
163 default:
164 gmmParams.Flags.Info.Linear = true;
165 tileFormatLinux = I915_TILING_NONE;
166 }
167
168 if (nullptr != params.m_pSystemMemory)
169 {
170 // If user provides a system memory pointer, the gfx resource is backed
171 // by the system memory pages. The resource is required to be linear.
172 gmmParams.Flags.Info.Linear = true;
173 gmmParams.Flags.Info.Cacheable = true;
174 gmmParams.NoGfxMemory = true;
175 GMM_RESOURCE_INFO *tmpGmmResInfoPtr = pOsContextSpecific
176 ->GetGmmClientContext()->CreateResInfoObject(&gmmParams);
177 if (tmpGmmResInfoPtr == nullptr)
178 {
179 MOS_OS_ASSERTMESSAGE("Create GmmResInfo failed");
180 return MOS_STATUS_UNKNOWN;
181 }
182
183 gmmParams.ExistingSysMemSize = GFX_ULONG_CAST(tmpGmmResInfoPtr->GetSizeSurface());
184 gmmParams.pExistingSysMem = (GMM_VOIDPTR64)params.m_pSystemMemory;
185 gmmParams.NoGfxMemory = false;
186 gmmParams.Flags.Info.ExistingSysMem = true;
187
188 pOsContextSpecific->GetGmmClientContext()
189 ->DestroyResInfoObject(tmpGmmResInfoPtr);
190 }
191 else
192 {
193 gmmParams.Flags.Info.LocalOnly = MEDIA_IS_SKU(pOsContextSpecific->GetSkuTable(), FtrLocalMemory);
194 }
195
196 GMM_RESOURCE_INFO* gmmResourceInfoPtr = pOsContextSpecific->GetGmmClientContext()->CreateResInfoObject(&gmmParams);
197
198 if (gmmResourceInfoPtr == nullptr)
199 {
200 MOS_OS_ASSERTMESSAGE("Get gmmResourceInfoPtr failed.");
201 return MOS_STATUS_INVALID_PARAMETER;
202 }
203
204 switch (gmmResourceInfoPtr->GetTileType())
205 {
206 case GMM_TILED_X:
207 tileformat = MOS_TILE_X;
208 tileFormatLinux = I915_TILING_X;
209 break;
210 case GMM_TILED_Y:
211 tileformat = MOS_TILE_Y;
212 tileFormatLinux = I915_TILING_Y;
213 break;
214 case GMM_NOT_TILED:
215 tileformat = MOS_TILE_LINEAR;
216 tileFormatLinux = I915_TILING_NONE;
217 break;
218 default:
219 tileformat = MOS_TILE_Y;
220 tileFormatLinux = I915_TILING_Y;
221 break;
222 }
223
224 if (params.m_tileType== MOS_TILE_Y)
225 {
226 gmmResourceInfoPtr->SetMmcMode((GMM_RESOURCE_MMC_INFO)params.m_compressionMode, 0);
227 }
228
229 if(!params.m_pSystemMemory)
230 {
231 MemoryPolicyParameter memPolicyPar;
232 MOS_ZeroMemory(&memPolicyPar, sizeof(MemoryPolicyParameter));
233
234 memPolicyPar.skuTable = pOsContextSpecific->GetSkuTable();
235 memPolicyPar.waTable = pOsContextSpecific->GetWaTable();
236 memPolicyPar.resInfo = gmmResourceInfoPtr;
237 memPolicyPar.resName = params.m_name.c_str();
238 memPolicyPar.preferredMemType = params.m_memType;
239
240 mem_type = MemoryPolicyManager::UpdateMemoryPolicy(&memPolicyPar);
241 }
242
243 uint32_t bufPitch = GFX_ULONG_CAST(gmmResourceInfoPtr->GetRenderPitch());
244 uint32_t bufSize = GFX_ULONG_CAST(gmmResourceInfoPtr->GetSizeSurface());
245 bufHeight = gmmResourceInfoPtr->GetBaseHeight();
246 unsigned long linuxPitch = 0;
247 MOS_LINUX_BO* boPtr = nullptr;
248
249 char bufName[m_maxBufNameLength];
250 MosUtilities::MosSecureStrcpy(bufName, m_maxBufNameLength, params.m_name.c_str());
251
252 MOS_TraceEventExt(EVENT_RESOURCE_ALLOCATE, EVENT_TYPE_START, nullptr, 0, nullptr, 0);
253 if (nullptr != params.m_pSystemMemory)
254 {
255 boPtr = mos_bo_alloc_userptr(pOsContextSpecific->m_bufmgr,
256 bufName,
257 params.m_pSystemMemory,
258 tileFormatLinux,
259 bufPitch,
260 bufSize,
261 0);
262 }
263 // Only Linear and Y TILE supported
264 else if (tileFormatLinux == I915_TILING_NONE)
265 {
266 boPtr = mos_bo_alloc(pOsContextSpecific->m_bufmgr, bufName, bufSize, 4096, mem_type);
267 }
268 else
269 {
270 boPtr = mos_bo_alloc_tiled(pOsContextSpecific->m_bufmgr,
271 bufName,
272 bufPitch,
273 bufSize/bufPitch,
274 1,
275 &tileFormatLinux,
276 &linuxPitch,
277 0,
278 mem_type);
279 bufPitch = (uint32_t)linuxPitch;
280 }
281
282 m_mapped = false;
283 if (boPtr)
284 {
285 m_format = params.m_format;
286 m_width = params.m_width;
287 m_height = bufHeight;
288 m_pitch = bufPitch;
289 m_count = 0;
290 m_bo = boPtr;
291 m_name = params.m_name;
292 m_pData = (uint8_t*) boPtr->virt;
293
294 m_gmmResInfo = gmmResourceInfoPtr;
295 m_mapped = false;
296 m_mmapOperation = MOS_MMAP_OPERATION_NONE;
297
298 m_arraySize = 1;
299 m_depth = MOS_MAX(1, gmmResourceInfoPtr->GetBaseDepth());
300 m_size = (uint32_t)gmmResourceInfoPtr->GetSizeSurface();
301 m_tileType = tileformat;
302 m_tileModeGMM = (MOS_TILE_MODE_GMM)gmmResourceInfoPtr->GetTileModeSurfaceState();
303 m_isGMMTileEnabled = true;
304
305 m_compressible = gmmParams.Flags.Gpu.MMC ?
306 (gmmResourceInfoPtr->GetMmcHint(0) == GMM_MMC_HINT_ON) : false;
307 m_isCompressed = gmmResourceInfoPtr->IsMediaMemoryCompressed(0);
308 m_compressionMode = (MOS_RESOURCE_MMC_MODE)gmmResourceInfoPtr->GetMmcMode(0);
309 m_memObjCtrlState = MosInterface::GetCachePolicyMemoryObject(pOsContextSpecific->GetGmmClientContext(), params.m_mocsMosResUsageType);
310
311 MOS_OS_VERBOSEMESSAGE("Alloc %7d bytes (%d x %d resource), tile encoding %d.", bufSize, params.m_width, bufHeight, m_tileModeGMM);
312
313 struct {
314 uint32_t m_handle;
315 uint32_t m_resFormat;
316 uint32_t m_baseWidth;
317 uint32_t m_baseHeight;
318 uint32_t m_pitch;
319 uint32_t m_size;
320 uint32_t m_resTileType;
321 GMM_RESOURCE_FLAG m_resFlag;
322 uint32_t m_reserve;
323 } eventData;
324
325 eventData.m_handle = boPtr->handle;
326 eventData.m_baseWidth = m_width;
327 eventData.m_baseHeight = m_height;
328 eventData.m_pitch = m_pitch;
329 eventData.m_size = m_size;
330 eventData.m_resFormat = m_format;
331 eventData.m_resTileType = m_tileType;
332 eventData.m_resFlag = gmmResourceInfoPtr->GetResFlags();
333 eventData.m_reserve = 0;
334 MOS_TraceEventExt(EVENT_RESOURCE_ALLOCATE,
335 EVENT_TYPE_INFO,
336 &eventData,
337 sizeof(eventData),
338 params.m_name.c_str(),
339 params.m_name.size() + 1);
340 }
341 else
342 {
343 MOS_OS_ASSERTMESSAGE("Fail to Alloc %7d bytes (%d x %d resource).",bufSize, params.m_width, params.m_height);
344 status = MOS_STATUS_NO_SPACE;
345 }
346 MOS_TraceEventExt(EVENT_RESOURCE_ALLOCATE, EVENT_TYPE_END, &status, sizeof(status), nullptr, 0);
347
348 m_memAllocCounterGfx++;
349 return status;

352void GraphicsResourceSpecificNext::Free(OsContextNext* osContextPtr, uint32_t freeFlag)
353{
354 MOS_OS_FUNCTION_ENTER;
355
356 MOS_UNUSED(osContextPtr);
357 MOS_UNUSED(freeFlag);
358
359 OsContextSpecificNext *pOsContextSpecific = static_cast<OsContextSpecificNext *>(osContextPtr);
360
361 MOS_LINUX_BO* boPtr = m_bo;
362
363 if (boPtr)
364 {
365 AuxTableMgr *auxTableMgr = pOsContextSpecific->GetAuxTableMgr();
366 if (auxTableMgr)
367 {
368 auxTableMgr->UnmapResource(m_gmmResInfo, boPtr);
369 }
370 mos_bo_unreference(boPtr);
371 m_bo = nullptr;
372 if (nullptr != m_gmmResInfo)
373 {
374 pOsContextSpecific->GetGmmClientContext()->DestroyResInfoObject(m_gmmResInfo);
375 m_gmmResInfo = nullptr;
376 m_memAllocCounterGfx--;
377 }
378 }
379 return;

432void* GraphicsResourceSpecificNext::Lock(OsContextNext* osContextPtr, LockParams& params)
433{
434 MOS_OS_FUNCTION_ENTER;
435
436 if (osContextPtr == nullptr)
437 {
438 MOS_OS_ASSERTMESSAGE("Unable to get the active OS context.");
439 return nullptr;
440 }
441
442 if (osContextPtr ->GetOsContextValid() == false)
443 {
444 MOS_OS_ASSERTMESSAGE("The OS context got is not valid.");
445 return nullptr;
446 }
447
448 OsContextSpecificNext *pOsContextSpecific = static_cast<OsContextSpecificNext *>(osContextPtr);
449
450 void* dataPtr = nullptr;
451 MOS_LINUX_BO* boPtr = m_bo;
452
453 if (boPtr)
454 {
455 // Do decompression for a compressed surface before lock
456 const auto pGmmResInfo = m_gmmResInfo;
457 MOS_OS_ASSERT(pGmmResInfo);
458 GMM_RESOURCE_FLAG GmmFlags = pGmmResInfo->GetResFlags();
459
460 if (!params.m_noDecompress &&
461 (((GmmFlags.Gpu.MMC || GmmFlags.Gpu.CCS) && GmmFlags.Info.MediaCompressed) ||
462 pGmmResInfo->IsMediaMemoryCompressed(0)))
463 {
464 MOS_RESOURCE mosResource = {};
465 ConvertToMosResource(&mosResource);
466
467 MosDecompression *mosDecompression = pOsContextSpecific->GetMosDecompression();
468 if (nullptr == mosDecompression)
469 {
470 MOS_OS_ASSERTMESSAGE("mosDecompression is NULL.");
471 return nullptr;
472 }
473 mosDecompression->MemoryDecompress(&mosResource);
474 }
475
476 if(false == m_mapped)
477 {
478 if (pOsContextSpecific->IsAtomSoc())
479 {
480 mos_gem_bo_map_gtt(boPtr);
481 }
482 else
483 {
484 if (m_tileType != MOS_TILE_LINEAR && !params.m_tileAsTiled)
485 {
486 if (pOsContextSpecific->UseSwSwizzling())
487 {
488 mos_bo_map(boPtr, ( OSKM_LOCKFLAG_WRITEONLY & params.m_writeRequest ));
489 m_mmapOperation = MOS_MMAP_OPERATION_MMAP;
490 if (m_systemShadow == nullptr)
491 {
492 m_systemShadow = (uint8_t *)MOS_AllocMemory(boPtr->size);
493 MOS_OS_CHECK_CONDITION((m_systemShadow == nullptr), "Failed to allocate shadow surface", nullptr);
494 }
495 if (m_systemShadow)
496 {
497 int32_t flags = pOsContextSpecific->GetTileYFlag() ? 0 : 1;
498 uint64_t surfSize = m_gmmResInfo->GetSizeMainSurface();
499 MOS_OS_CHECK_CONDITION((m_tileType != MOS_TILE_Y), "Unsupported tile type", nullptr);
500 MOS_OS_CHECK_CONDITION((boPtr->size <= 0 || m_pitch <= 0), "Invalid BO size or pitch", nullptr);
501 MosUtilities::MosSwizzleData((uint8_t*)boPtr->virt, m_systemShadow,
502 MOS_TILE_Y, MOS_TILE_LINEAR,
503 (int32_t)(surfSize / m_pitch), m_pitch, flags);
504 }
505 }
506 else
507 {
508 mos_gem_bo_map_gtt(boPtr);
509 m_mmapOperation = MOS_MMAP_OPERATION_MMAP_GTT;
510 }
511 }
512 else if (params.m_uncached)
513 {
514 mos_gem_bo_map_wc(boPtr);
515 m_mmapOperation = MOS_MMAP_OPERATION_MMAP_WC;
516 }
517 else
518 {
519 mos_bo_map(boPtr, ( OSKM_LOCKFLAG_WRITEONLY & params.m_writeRequest ));
520 m_mmapOperation = MOS_MMAP_OPERATION_MMAP;
521 }
522 }
523 m_mapped = true;
524 m_pData = m_systemShadow ? m_systemShadow : (uint8_t *)boPtr->virt;
525 }
526
527 dataPtr = m_pData;
528 }
529
530 MOS_OS_ASSERT(dataPtr);
531 return dataPtr;
532}

534MOS_STATUS GraphicsResourceSpecificNext::Unlock(OsContextNext* osContextPtr)
535{
536 MOS_OS_FUNCTION_ENTER;
537
538 if (osContextPtr == nullptr)
539 {
540 MOS_OS_ASSERTMESSAGE("Unable to get the active OS context.");
541 return MOS_STATUS_INVALID_HANDLE;
542 }
543
544 if (osContextPtr ->GetOsContextValid() == false)
545 {
546 MOS_OS_ASSERTMESSAGE("The OS context got is not valid.");
547 return MOS_STATUS_INVALID_HANDLE;
548 }
549
550 OsContextSpecificNext *pOsContextSpecific = static_cast<OsContextSpecificNext *>(osContextPtr);
551
552 MOS_LINUX_BO* boPtr = m_bo;
553 if (boPtr)
554 {
555 if (m_mapped)
556 {
557 if (pOsContextSpecific->IsAtomSoc())
558 {
559 mos_gem_bo_unmap_gtt(boPtr);
560 }
561 else
562 {
563
564 if (m_systemShadow)
565 {
566 int32_t flags = pOsContextSpecific->GetTileYFlag() ? 0 : 1;
567 uint64_t surfSize = m_gmmResInfo->GetSizeMainSurface();
568 MosUtilities::MosSwizzleData(m_systemShadow, (uint8_t*)boPtr->virt,
569 MOS_TILE_LINEAR, MOS_TILE_Y,
570 (int32_t)(surfSize / m_pitch), m_pitch, flags);
571 MOS_FreeMemory(m_systemShadow);
572 m_systemShadow = nullptr;
573 }
574
575 switch(m_mmapOperation)
576 {
577 case MOS_MMAP_OPERATION_MMAP_GTT:
578 mos_gem_bo_unmap_gtt(boPtr);
579 break;
580 case MOS_MMAP_OPERATION_MMAP_WC:
581 mos_gem_bo_unmap_wc(boPtr);
582 break;
583 case MOS_MMAP_OPERATION_MMAP:
584 mos_bo_unmap(boPtr);
585 break;
586 default:
587 MOS_OS_ASSERTMESSAGE("Invalid mmap operation type");
588 break;
589 }
590 }
591
592 m_mapped = false;
593 m_mmapOperation = MOS_MMAP_OPERATION_NONE;
594
595 boPtr->virt = nullptr;
596 m_bo = boPtr;
597 }
598
599 m_pData = nullptr;
600 }
601
602 return MOS_STATUS_SUCCESS;
603}

605MOS_STATUS GraphicsResourceSpecificNext::AllocateExternalResource(
606 MOS_STREAM_HANDLE streamState,
607 PMOS_ALLOC_GFXRES_PARAMS params,
608 MOS_RESOURCE_HANDLE& resource)
609{
610 MOS_OS_FUNCTION_ENTER;
611
612 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;
613
614 MOS_OS_CHK_NULL_RETURN(resource);
615 MOS_OS_CHK_NULL_RETURN(streamState);
616 MOS_OS_CHK_NULL_RETURN(streamState->osDeviceContext);
617
618 const char *bufname = params->pBufName;;
619 int32_t iSize = 0;
620 int32_t iPitch = 0;
621 unsigned long ulPitch = 0;
622 MOS_LINUX_BO *bo = nullptr;
623 MOS_TILE_TYPE tileformat = params->TileType;
624 uint32_t tileformat_linux = I915_TILING_NONE;
625 int32_t iHeight = params->dwHeight;
626 int32_t iAlignedHeight = 0;
627 GMM_RESCREATE_PARAMS gmmParams;
628 GMM_RESOURCE_INFO *gmmResourceInfo = nullptr;
629 GMM_RESOURCE_TYPE resourceType = RESOURCE_2D;
630
631 MosUtilities::MosZeroMemory(&gmmParams, sizeof(gmmParams));
632
633 MOS_OS_CHK_NULL_RETURN(streamState->perStreamParameters);
634 auto perStreamParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
635
636 gmmParams.Usage = MosInterface::GetGmmResourceUsageType(params->ResUsageType);
637
638 switch (params->Format)
639 {
640 case Format_Buffer:
641 case Format_RAW:
642 resourceType = RESOURCE_BUFFER;
643 iAlignedHeight = 1;
644 //indicate buffer Restriction is Vertex.
645 gmmParams.Flags.Gpu.State = true;
646 break;
647 case Format_L8:
648 case Format_L16:
649 case Format_STMM:
650 case Format_AI44:
651 case Format_IA44:
652 case Format_R5G6B5:
653 case Format_R8G8B8:
654 case Format_X8R8G8B8:
655 case Format_A8R8G8B8:
656 case Format_X8B8G8R8:
657 case Format_A8B8G8R8:
658 case Format_R32S:
659 case Format_R32F:
660 case Format_V8U8:
661 case Format_YUY2:
662 case Format_UYVY:
663 case Format_P8:
664 case Format_A8:
665 case Format_AYUV:
666 case Format_NV12:
667 case Format_NV21:
668 case Format_YV12:
669 case Format_Buffer_2D:
670 case Format_R32U:
671 case Format_444P:
672 case Format_422H:
673 case Format_422V:
674 case Format_IMC3:
675 case Format_411P:
676 case Format_411R:
677 case Format_RGBP:
678 case Format_BGRP:
679 case Format_R16U:
680 case Format_R8U:
681 case Format_R8UN:
682 case Format_P010:
683 case Format_P016:
684 case Format_Y216:
685 case Format_Y416:
686 case Format_P208:
687 case Format_Y210:
688 case Format_Y410:
689 case Format_R16F:
690 resourceType = RESOURCE_2D;
691 //indicate buffer Restriction is Planar surface restrictions.
692 gmmParams.Flags.Gpu.Video = true;
693 break;
694 default:
695 MOS_OS_ASSERTMESSAGE("Unsupported format");
696 eStatus = MOS_STATUS_UNIMPLEMENTED;
697 return eStatus;
698 }
699
700 // Create GmmResourceInfo
701 gmmParams.BaseWidth = params->dwWidth;
702 gmmParams.BaseHeight = iAlignedHeight;
703 gmmParams.ArraySize = 1;
704 gmmParams.Type = resourceType;
705 gmmParams.Format = MosInterface::MosFmtToGmmFmt(params->Format);
706
707 MOS_OS_CHECK_CONDITION(gmmParams.Format == GMM_FORMAT_INVALID,
708 "Unsupported format",
709 MOS_STATUS_UNKNOWN);
710
711 switch (tileformat)
712 {
713 case MOS_TILE_Y:
714 gmmParams.Flags.Gpu.MMC = params->bIsCompressible;
715 tileformat_linux = I915_TILING_Y;
716 break;
717 case MOS_TILE_X:
718 gmmParams.Flags.Info.TiledX = true;
719 tileformat_linux = I915_TILING_X;
720 break;
721 default:
722 gmmParams.Flags.Info.Linear = true;
723 tileformat_linux = I915_TILING_NONE;
724 }
725 gmmParams.Flags.Info.LocalOnly = MEDIA_IS_SKU(&perStreamParameters->SkuTable, FtrLocalMemory);
726
727 resource->pGmmResInfo = gmmResourceInfo = perStreamParameters->pGmmClientContext->CreateResInfoObject(&gmmParams);
728
729 MOS_OS_CHK_NULL_RETURN(gmmResourceInfo);
730
731 switch (gmmResourceInfo->GetTileType())
732 {
733 case GMM_TILED_X:
734 tileformat = MOS_TILE_X;
735 tileformat_linux = I915_TILING_X;
736 break;
737 case GMM_TILED_Y:
738 tileformat = MOS_TILE_Y;
739 tileformat_linux = I915_TILING_Y;
740 break;
741 case GMM_NOT_TILED:
742 tileformat = MOS_TILE_LINEAR;
743 tileformat_linux = I915_TILING_NONE;
744 break;
745 default:
746 tileformat = MOS_TILE_Y;
747 tileformat_linux = I915_TILING_Y;
748 break;
749 }
750
751 if (params->TileType == MOS_TILE_Y)
752 {
753 gmmResourceInfo->SetMmcMode((GMM_RESOURCE_MMC_INFO)params->CompressionMode, 0);
754 }
755
756 iPitch = GFX_ULONG_CAST(gmmResourceInfo->GetRenderPitch());
757 iSize = GFX_ULONG_CAST(gmmResourceInfo->GetSizeSurface());
758 iHeight = gmmResourceInfo->GetBaseHeight();
759
760 // Only Linear and Y TILE supported
761 if (tileformat_linux == I915_TILING_NONE)
762 {
763 bo = mos_bo_alloc(perStreamParameters->bufmgr, bufname, iSize, 4096, MOS_MEMPOOL_VIDEOMEMORY);
764 }
765 else
766 {
767 bo = mos_bo_alloc_tiled(perStreamParameters->bufmgr,
768 bufname,
769 iPitch,
770 iSize / iPitch,
771 1,
772 &tileformat_linux,
773 &ulPitch,
774 0,
775 MOS_MEMPOOL_VIDEOMEMORY);
776 iPitch = (int32_t)ulPitch;
777 }
778
779 resource->bMapped = false;
780 if (bo)
781 {
782 resource->Format = params->Format;
783 resource->iWidth = params->dwWidth;
784 resource->iHeight = iHeight;
785 resource->iPitch = iPitch;
786 resource->iCount = 0;
787 resource->bufname = bufname;
788 resource->bo = bo;
789 resource->TileType = tileformat;
790 resource->TileModeGMM = (MOS_TILE_MODE_GMM)gmmResourceInfo->GetTileModeSurfaceState();
791 resource->bGMMTileEnabled = true;
792 resource->pData = (uint8_t *)bo->virt; //It is useful for batch buffer to fill commands
793 if (params->ResUsageType == MOS_CODEC_RESOURCE_USAGE_BEGIN_CODEC ||
794 params->ResUsageType >= MOS_HW_RESOURCE_USAGE_MEDIA_BATCH_BUFFERS)
795 {
796 resource->memObjCtrlState = MosInterface::GetCachePolicyMemoryObject(perStreamParameters->pGmmClientContext, MOS_MP_RESOURCE_USAGE_DEFAULT);
797 resource->mocsMosResUsageType = MOS_MP_RESOURCE_USAGE_DEFAULT;
798 }
799 else
800 {
801 resource->memObjCtrlState = MosInterface::GetCachePolicyMemoryObject(perStreamParameters->pGmmClientContext, params->ResUsageType);
802 resource->mocsMosResUsageType = params->ResUsageType;
803 }
804 MOS_OS_VERBOSEMESSAGE("Alloc %7d bytes (%d x %d resource), tile encoding.", iSize, params->dwWidth, iHeight, resource->TileModeGMM);
805 }
806 else
807 {
808 MOS_OS_ASSERTMESSAGE("Fail to Alloc %7d bytes (%d x %d resource).", iSize, params->dwWidth, params->dwHeight);
809 eStatus = MOS_STATUS_NO_SPACE;
810 }
811
812 return eStatus;
813}

822 MOS_STATUS eStatus = MOS_STATUS_SUCCESS;

824 MOS_OS_CHK_NULL_RETURN(resource);

861 item_ctx = pOsCtx->contextOffsetList.erase(item_ctx);
862 }

864 {
865 item_ctx++;
866 }
867 }
868 }

870 resource->bo = nullptr;
871 }

876void* GraphicsResourceSpecificNext::LockExternalResource(
877 MOS_STREAM_HANDLE streamState,
878 MOS_RESOURCE_HANDLE resource,
879 PMOS_LOCK_PARAMS flags)
880{
881 MOS_OS_FUNCTION_ENTER;
882
883 void *pData = nullptr;
884 if (nullptr == streamState)
885 {
886 MOS_OS_ASSERTMESSAGE("input parameter streamState is NULL.");
887 return nullptr;
888 }
889
890 if (nullptr == resource)
891 {
892 MOS_OS_ASSERTMESSAGE("input parameter resource is NULL.");
893 return nullptr;
894 }
895
896 if (streamState->perStreamParameters == nullptr)
897 {
898 MOS_OS_ASSERTMESSAGE("perStreamParameters is nullptr, skip lock");
899 return nullptr;
900 }
901 auto perStreamParameters = (PMOS_CONTEXT)streamState->perStreamParameters;
902 if (resource && resource->bo && resource->pGmmResInfo)
903 {
904 MOS_LINUX_BO *bo = resource->bo;
905 GMM_RESOURCE_FLAG GmmFlags = resource->pGmmResInfo->GetResFlags();
906
907 // Do decompression for a compressed surface before lock
908 if (!flags->NoDecompress &&
909 (((GmmFlags.Gpu.MMC || GmmFlags.Gpu.CCS) && GmmFlags.Info.MediaCompressed) ||
910 resource->pGmmResInfo->IsMediaMemoryCompressed(0)))
911 {
912 OsContextNext *osCtx = streamState->osDeviceContext;
913 if (nullptr == osCtx)
914 {
915 MOS_OS_ASSERTMESSAGE("osCtx is NULL.");
916 return nullptr;
917 }
918 MosDecompression *mosDecompression = osCtx->GetMosDecompression();
919 if (nullptr == mosDecompression)
920 {
921 MOS_OS_ASSERTMESSAGE("mosDecompression is NULL.");
922 return nullptr;
923 }
924 mosDecompression->MemoryDecompress(resource);
925 }
926
927 if (false == resource->bMapped)
928 {
929 if (perStreamParameters->bIsAtomSOC)
930 {
931 mos_gem_bo_map_gtt(bo);
932 }
933 else
934 {
935 if (resource->TileType != MOS_TILE_LINEAR && !flags->TiledAsTiled)
936 {
937 if (perStreamParameters->bUseSwSwizzling)
938 {
939 mos_bo_map(bo, (OSKM_LOCKFLAG_WRITEONLY & flags->WriteOnly));
940 resource->MmapOperation = MOS_MMAP_OPERATION_MMAP;
941 if (resource->pSystemShadow == nullptr)
942 {
943 resource->pSystemShadow = (uint8_t *)MOS_AllocMemory(bo->size);
944 MOS_OS_CHECK_CONDITION((resource->pSystemShadow == nullptr), "Failed to allocate shadow surface", nullptr);
945 }
946 if (resource->pSystemShadow)
947 {
948 int32_t swizzleflags = perStreamParameters->bTileYFlag ? 0 : 1;
949 MOS_OS_CHECK_CONDITION((resource->TileType != MOS_TILE_Y), "Unsupported tile type", nullptr);
950 MOS_OS_CHECK_CONDITION((bo->size <= 0 || resource->iPitch <= 0), "Invalid BO size or pitch", nullptr);
951 MosUtilities::MosSwizzleData((uint8_t *)bo->virt, resource->pSystemShadow, MOS_TILE_Y, MOS_TILE_LINEAR, bo->size / resource->iPitch, resource->iPitch, swizzleflags);
952 }
953 }
954 else
955 {
956 mos_gem_bo_map_gtt(bo);
957 resource->MmapOperation = MOS_MMAP_OPERATION_MMAP_GTT;
958 }
959 }
960 else if (flags->Uncached)
961 {
962 mos_gem_bo_map_wc(bo);
963 resource->MmapOperation = MOS_MMAP_OPERATION_MMAP_WC;
964 }
965 else
966 {
967 mos_bo_map(bo, (OSKM_LOCKFLAG_WRITEONLY & flags->WriteOnly));
968 resource->MmapOperation = MOS_MMAP_OPERATION_MMAP;
969 }
970 }
971 resource->pData = resource->pSystemShadow ? resource->pSystemShadow : (uint8_t *)bo->virt;
972 resource->bMapped = true;
973 }
974
975 pData = resource->pData;
976 }
977
978 MOS_OS_ASSERT(pData);
979 return pData;
980}

982MOS_STATUS GraphicsResourceSpecificNext::UnlockExternalResource(

988 MOS_OS_CHK_NULL_RETURN(resource);

994
995 if (resource->bo)

1002 }
1003 else

1027 }
1028 }
1029 resource->bo->virt = nullptr;
1030 resource->bMapped = false;
1031 }
1032 resource->pData = nullptr;
1033 }
1034
1035 return MOS_STATUS_SUCCESS;
